55 // state update table for hybrid astar
56 void AstarSearch::createStateUpdateTable()
57 {
58  // Vehicle moving for each angle
59  state_update_table_.resize(theta_size_);
60  double dtheta = 2.0 * M_PI / theta_size_;
61 
62  // Minimum moving distance with one state update
63  // arc = r * theta
64  double step = minimum_turning_radius_ * dtheta;
65 
66  for (int i = 0; i < theta_size_; i++)
67  {
68  double theta = dtheta * i;
69 
70  // Calculate right and left circle
71  // Robot moves along these circles
72  double right_circle_center_x = minimum_turning_radius_ * std::sin(theta);
73  double right_circle_center_y = minimum_turning_radius_ * -std::cos(theta);
74  double left_circle_center_x = -right_circle_center_x;
75  double left_circle_center_y = -right_circle_center_y;
76 
77  // Calculate x and y shift to next state
78  NodeUpdate nu;
79 
80  // forward
81  nu.shift_x = step * std::cos(theta);
82  nu.shift_y = step * std::sin(theta);
83  nu.rotation = 0.0;
84  nu.index_theta = 0;
85  nu.step = step;
86  nu.curve = false;
87  nu.back = false;
88  state_update_table_[i].emplace_back(nu);
89 
90  // forward right
91  nu.shift_x = right_circle_center_x + minimum_turning_radius_ * std::cos(M_PI_2 + theta - dtheta);
92  nu.shift_y = right_circle_center_y + minimum_turning_radius_ * std::sin(M_PI_2 + theta - dtheta);
93  nu.rotation = -dtheta;
94  nu.index_theta = -1;
95  nu.step = step;
96  nu.curve = true;
97  nu.back = false;
98  state_update_table_[i].emplace_back(nu);
99 
100  // forward left
101  nu.shift_x = left_circle_center_x + minimum_turning_radius_ * std::cos(-M_PI_2 + theta + dtheta);
102  nu.shift_y = left_circle_center_y + minimum_turning_radius_ * std::sin(-M_PI_2 + theta + dtheta);
103  nu.rotation = dtheta;
104  nu.index_theta = 1;
105  nu.step = step;
106  nu.curve = true;
107  nu.back = false;
108  state_update_table_[i].emplace_back(nu);
109 
110  if (use_back_)
111  {
112  // backward
113  nu.shift_x = step * std::cos(theta) * -1.0;
114  nu.shift_y = step * std::sin(theta) * -1.0;
115  nu.rotation = 0;
116  nu.index_theta = 0;
117  nu.step = step;
118  nu.curve = false;
119  nu.back = true;
120  state_update_table_[i].emplace_back(nu);
121 
122  // backward right
123  nu.shift_x = right_circle_center_x + minimum_turning_radius_ * std::cos(M_PI_2 + theta + dtheta);
124  nu.shift_y = right_circle_center_y + minimum_turning_radius_ * std::sin(M_PI_2 + theta + dtheta);
125  nu.rotation = dtheta;
126  nu.index_theta = 1;
127  nu.step = step;
128  nu.curve = true;
129  nu.back = true;
130  state_update_table_[i].emplace_back(nu);
131 
132  // backward left
133  nu.shift_x = left_circle_center_x + minimum_turning_radius_ * std::cos(-1.0 * M_PI_2 + theta - dtheta);
134  nu.shift_y = left_circle_center_y + minimum_turning_radius_ * std::sin(-1.0 * M_PI_2 + theta - dtheta);
135  nu.rotation = dtheta * -1.0;
136  nu.index_theta = -1;
137  nu.step = step;
138  nu.curve = true;
139  nu.back = true;
140  state_update_table_[i].emplace_back(nu);
141  }
142  }

145 void AstarSearch::initialize(const nav_msgs::OccupancyGrid& costmap)
146 {
147  costmap_ = costmap;
148 
149  int height = costmap_.info.height;
150  int width = costmap_.info.width;
151 
152  // size initialization
153  nodes_.resize(height);
154  for (int i = 0; i < height; i++)
155  {
156  nodes_[i].resize(width);
157  }
158  for (int i = 0; i < height; i++)
159  {
160  for (int j = 0; j < width; j++)
161  {
162  nodes_[i][j].resize(theta_size_);
163  }
164  }
165 
166  // cost initialization
167  for (int i = 0; i < height; i++)
168  {
169  for (int j = 0; j < width; j++)
170  {
171  // Index of subscribing OccupancyGrid message
172  int og_index = i * width + j;
173  int cost = costmap_.data[og_index];
174 
175  // hc is set to be 0 when reset()
176  if (cost == 0)
177  {
178  continue;
179  }
180 
181  // obstacle or unknown area
182  if (cost < 0 || obstacle_threshold_ <= cost)
183  {
184  nodes_[i][j][0].status = STATUS::OBS;
185  }
186 
187  // the cost more than threshold is regarded almost same as an obstacle
188  // because of its very high cost
189  if (use_potential_heuristic_)
190  {
191  nodes_[i][j][0].hc = cost * potential_weight_;
192  }
193  }

214 bool AstarSearch::setStartNode(const geometry_msgs::Pose& start_pose)
215 {
216  // Get index of start pose
217  int index_x, index_y, index_theta;
218  start_pose_local_.pose = start_pose;
219  poseToIndex(start_pose_local_.pose, &index_x, &index_y, &index_theta);
220  SimpleNode start_sn(index_x, index_y, index_theta, 0, 0);
221 
222  // Check if start is valid
223  if (isOutOfRange(index_x, index_y) || detectCollision(start_sn))
224  {
225  return false;
226  }
227 
228  // Set start node
229  AstarNode& start_node = nodes_[index_y][index_x][index_theta];
230  start_node.x = start_pose_local_.pose.position.x;
231  start_node.y = start_pose_local_.pose.position.y;
232  start_node.theta = 2.0 * M_PI / theta_size_ * index_theta;
233  start_node.gc = 0;
234  start_node.move_distance = 0;
235  start_node.back = false;
236  start_node.status = STATUS::OPEN;
237  start_node.parent = NULL;
238 
239  // set euclidean distance heuristic cost
240  if (!use_wavefront_heuristic_ && !use_potential_heuristic_)
241  {
242  start_node.hc = calcDistance(start_pose_local_.pose.position.x, start_pose_local_.pose.position.y,
243  goal_pose_local_.pose.position.x, goal_pose_local_.pose.position.y) *
244  distance_heuristic_weight_;
245  }
246  else if (use_potential_heuristic_)
247  {
248  start_node.gc += start_node.hc;
249  start_node.hc += calcDistance(start_pose_local_.pose.position.x, start_pose_local_.pose.position.y,
250  goal_pose_local_.pose.position.x, goal_pose_local_.pose.position.y) +
251  distance_heuristic_weight_;
252  }
253 
254  // Push start node to openlist
255  start_sn.cost = start_node.gc + start_node.hc;
256  openlist_.push(start_sn);
257 
258  return true;

335 bool AstarSearch::search()
336 {
337  ros::WallTime begin = ros::WallTime::now();
338 
339  // Start A* search
340  // If the openlist is empty, search failed
341  while (!openlist_.empty())
342  {
343  // Check time and terminate if the search reaches the time limit
344  ros::WallTime now = ros::WallTime::now();
345  double msec = (now - begin).toSec() * 1000.0;
346  if (msec > time_limit_)
347  {
348  // ROS_WARN("Exceed time limit of %lf [ms]", time_limit_);
349  return false;
350  }
351 
352  // Pop minimum cost node from openlist
353  SimpleNode top_sn = openlist_.top();
354  openlist_.pop();
355 
356  // Expand nodes from this node
357  AstarNode* current_an = &nodes_[top_sn.index_y][top_sn.index_x][top_sn.index_theta];
358  current_an->status = STATUS::CLOSED;
359 
360  // Goal check
361  if (isGoal(current_an->x, current_an->y, current_an->theta))
362  {
363  // ROS_INFO("Search time: %lf [msec]", (now - begin).toSec() * 1000.0);
364  setPath(top_sn);
365  return true;
366  }
367 
368  // Expand nodes
369  for (const auto& state : state_update_table_[top_sn.index_theta])
370  {
371  // Next state
372  double next_x = current_an->x + state.shift_x;
373  double next_y = current_an->y + state.shift_y;
374  double next_theta = modifyTheta(current_an->theta + state.rotation);
375  double move_cost = state.step;
376  double move_distance = current_an->move_distance + state.step;
377 
378  // Increase reverse cost
379  if (state.back != current_an->back)
380  move_cost *= reverse_weight_;
381 
382  // Calculate index of the next state
383  SimpleNode next_sn;
384  geometry_msgs::Point next_pos;
385  next_pos.x = next_x;
386  next_pos.y = next_y;
387  pointToIndex(next_pos, &next_sn.index_x, &next_sn.index_y);
388  next_sn.index_theta = top_sn.index_theta + state.index_theta;
389 
390  // Avoid invalid index
391  next_sn.index_theta = (next_sn.index_theta + theta_size_) % theta_size_;
392 
393  // Check if the index is valid
394  if (isOutOfRange(next_sn.index_x, next_sn.index_y) || detectCollision(next_sn))
395  {
396  continue;
397  }
398 
399  AstarNode* next_an = &nodes_[next_sn.index_y][next_sn.index_x][next_sn.index_theta];
400  double next_gc = current_an->gc + move_cost;
401  double next_hc = nodes_[next_sn.index_y][next_sn.index_x][0].hc; // wavefront or distance transform heuristic
402 
403  // increase the cost with euclidean distance
404  if (use_potential_heuristic_)
405  {
406  next_gc += nodes_[next_sn.index_y][next_sn.index_x][0].hc;
407  next_hc += calcDistance(next_x, next_y, goal_pose_local_.pose.position.x, goal_pose_local_.pose.position.y) *
408  distance_heuristic_weight_;
409  }
410 
411  // increase the cost with euclidean distance
412  if (!use_wavefront_heuristic_ && !use_potential_heuristic_)
413  {
414  next_hc = calcDistance(next_x, next_y, goal_pose_local_.pose.position.x, goal_pose_local_.pose.position.y) *
415  distance_heuristic_weight_;
416  }
417 
418  // NONE
419  if (next_an->status == STATUS::NONE)
420  {
421  next_an->status = STATUS::OPEN;
422  next_an->x = next_x;
423  next_an->y = next_y;
424  next_an->theta = next_theta;
425  next_an->gc = next_gc;
426  next_an->hc = next_hc;
427  next_an->move_distance = move_distance;
428  next_an->back = state.back;
429  next_an->parent = current_an;
430  next_sn.cost = next_an->gc + next_an->hc;
431  openlist_.push(next_sn);
432  continue;
433  }
434 
435  // OPEN or CLOSED
436  if (next_an->status == STATUS::OPEN || next_an->status == STATUS::CLOSED)
437  {
438  if (next_gc < next_an->gc)
439  {
440  next_an->status = STATUS::OPEN;
441  next_an->x = next_x;
442  next_an->y = next_y;
443  next_an->theta = next_theta;
444  next_an->gc = next_gc;
445  next_an->hc = next_hc; // already calculated ?
446  next_an->move_distance = move_distance;
447  next_an->back = state.back;
448  next_an->parent = current_an;
449  next_sn.cost = next_an->gc + next_an->hc;
450  openlist_.push(next_sn);
451  continue;
452  }
453  }
454  } // state update
455  }
456 
457  // Failed to find path
458  // ROS_INFO("Open list is empty...");
459  return false;
460 }

462 void AstarSearch::setPath(const SimpleNode& goal)
463 {
464  std_msgs::Header header;
465  header.stamp = ros::Time::now();
466  header.frame_id = costmap_.header.frame_id;
467  path_.header = header;
468 
469  // From the goal node to the start node
470  AstarNode* node = &nodes_[goal.index_y][goal.index_x][goal.index_theta];
471 
472  while (node != NULL)
473  {
474  // Set tf pose
475  tf::Vector3 origin(node->x, node->y, 0);
476  tf::Pose tf_pose;
477  tf_pose.setOrigin(origin);
478  tf_pose.setRotation(tf::createQuaternionFromYaw(node->theta));
479 
480  // Set path as ros message
481  geometry_msgs::PoseStamped ros_pose;
482  tf::poseTFToMsg(tf_pose, ros_pose.pose);
483  ros_pose.header = header;
484  path_.poses.push_back(ros_pose);
485 
486  // To the next node
487  node = node->parent;
488  }
489 
490  // Reverse the vector to be start to goal order
491  std::reverse(path_.poses.begin(), path_.poses.end());

494 // Check if the next state is the goal
495 // Check lateral offset, longitudinal offset and angle
496 bool AstarSearch::isGoal(double x, double y, double theta)
497 {
498  // To reduce computation time, we use square value for distance
499  static const double lateral_goal_range =
500  lateral_goal_range_ / 2.0; // [meter], divide by 2 means we check left and right
501  static const double longitudinal_goal_range =
502  longitudinal_goal_range_ / 2.0; // [meter], check only behind of the goal
503  static const double goal_angle = M_PI * (angle_goal_range_ / 2.0) / 180.0; // degrees -> radian
504 
505  // Calculate the node coordinate seen from the goal point
506  tf::Point p(x, y, 0);
507  geometry_msgs::Point relative_node_point = calcRelativeCoordinate(goal_pose_local_.pose, p);
508 
509  // Check Pose of goal
510  if (relative_node_point.x < 0 && // shoud be behind of goal
511  std::fabs(relative_node_point.x) < longitudinal_goal_range &&
512  std::fabs(relative_node_point.y) < lateral_goal_range)
513  {
514  // Check the orientation of goal
515  if (calcDiffOfRadian(goal_yaw_, theta) < goal_angle)
516  {
517  return true;
518  }
519  }
520 
521  return false;
522 }

534 bool AstarSearch::detectCollision(const SimpleNode& sn)
535 {
536  // Define the robot as rectangle
537  static double left = -1.0 * robot_base2back_;
538  static double right = robot_length_ - robot_base2back_;
539  static double top = robot_width_ / 2.0;
540  static double bottom = -1.0 * robot_width_ / 2.0;
541  static double resolution = costmap_.info.resolution;
542 
543  // Coordinate of base_link in OccupancyGrid frame
544  static double one_angle_range = 2.0 * M_PI / theta_size_;
545  double base_x = sn.index_x * resolution;
546  double base_y = sn.index_y * resolution;
547  double base_theta = sn.index_theta * one_angle_range;
548 
549  // Calculate cos and sin in advance
550  double cos_theta = std::cos(base_theta);
551  double sin_theta = std::sin(base_theta);
552 
553  // Convert each point to index and check if the node is Obstacle
554  for (double x = left; x < right; x += resolution)
555  {
556  for (double y = top; y > bottom; y -= resolution)
557  {
558  // 2D point rotation
559  int index_x = (x * cos_theta - y * sin_theta + base_x) / resolution;
560  int index_y = (x * sin_theta + y * cos_theta + base_y) / resolution;
561 
562  if (isOutOfRange(index_x, index_y))
563  {
564  return true;
565  }
566  else if (nodes_[index_y][index_x][0].status == STATUS::OBS)
567  {
568  return true;
569  }
570  }
571  }
572 
573  return false;
574 }

576 bool AstarSearch::calcWaveFrontHeuristic(const SimpleNode& sn)
577 {
578  // Set start point for wavefront search
579  // This is goal for Astar search
580  nodes_[sn.index_y][sn.index_x][0].hc = 0;
581  WaveFrontNode wf_node(sn.index_x, sn.index_y, 1e-10);
582  std::queue<WaveFrontNode> qu;
583  qu.push(wf_node);
584 
585  // State update table for wavefront search
586  // Nodes are expanded for each neighborhood cells (moore neighborhood)
587  double resolution = costmap_.info.resolution;
588  static std::vector<WaveFrontNode> updates = {
589  getWaveFrontNode(0, 1, resolution),
590  getWaveFrontNode(-1, 0, resolution),
591  getWaveFrontNode(1, 0, resolution),
592  getWaveFrontNode(0, -1, resolution),
593  getWaveFrontNode(-1, 1, std::hypot(resolution, resolution)),
594  getWaveFrontNode(1, 1, std::hypot(resolution, resolution)),
595  getWaveFrontNode(-1, -1, std::hypot(resolution, resolution)),
596  getWaveFrontNode(1, -1, std::hypot(resolution, resolution)),
597  };
598 
599  // Get start index
600  int start_index_x;
601  int start_index_y;
602  int start_index_theta;
603  poseToIndex(start_pose_local_.pose, &start_index_x, &start_index_y, &start_index_theta);
604 
605  // Whether the robot can reach goal
606  bool reachable = false;
607 
608  // Start wavefront search
609  while (!qu.empty())
610  {
611  WaveFrontNode ref = qu.front();
612  qu.pop();
613 
614  WaveFrontNode next;
615  for (const auto& u : updates)
616  {
617  next.index_x = ref.index_x + u.index_x;
618  next.index_y = ref.index_y + u.index_y;
619 
620  // out of range OR already visited OR obstacle node
621  if (isOutOfRange(next.index_x, next.index_y) || nodes_[next.index_y][next.index_x][0].hc > 0 ||
622  nodes_[next.index_y][next.index_x][0].status == STATUS::OBS)
623  {
624  continue;
625  }
626 
627  // Take the size of robot into account
