98  void updateOdometry(const double vx, const double vth, const ros::Time &cur_time)
99  {
100  if (stamp.sec == 0 && stamp.nsec == 0)
101  {
102  stamp = cur_time;
103  }
104  double dt = (cur_time - stamp).toSec();
105  double delta_x = (vx * cos(th)) * dt;
106  double delta_y = (vx * sin(th)) * dt;
107  double delta_th = vth * dt;
108 
109  ROS_INFO("dt : %f delta (x y th) : (%f %f %f %f)", dt, delta_x, delta_y, delta_th);
110 
111  x += delta_x;
112  y += delta_y;
113  th += delta_th;
114  stamp = cur_time;
115  }
116 };
