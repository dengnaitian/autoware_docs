47 std::vector < RectClassScore<float> > SSDDetector::Detect(const cv::Mat& img)
48 {
49  caffe::Blob<float> *input_layer = net_->input_blobs()[0];
50  input_layer->Reshape(1, num_channels_, input_geometry_.height,
51  input_geometry_.width);
52  /* Forward dimension change to all layers. */
53  net_->Reshape();
54 
55  std::vector <cv::Mat> input_channels;
56  WrapInputLayer(&input_channels);
57 
58  Preprocess(img, &input_channels);
59 
60  net_->Forward();
61 
62  /* Copy the output layer to a std::vector */
63  caffe::Blob<float> *result_blob = net_->output_blobs()[0];
64  const float *result = result_blob->cpu_data();
65  const int num_det = result_blob->height();
66  std::vector <RectClassScore<float> > detections;
67  for (int k = 0; k < num_det; ++k)
68  {
69  if (result[0] == -1)
70  {
71  // Skip invalid detection.
72  result += 7;
73  continue;
74  }
75  // Detection format: [image_id (0), label(1), score(2), xmin(3), ymin(4), xmax(5), ymax(6)].
76  RectClassScore<float> detection;
77  detection.class_type = static_cast<int>(result[1]);
78  detection.score = result[2];
79  detection.x = result[3] * img.cols;
80  detection.y = result[4] * img.rows;
81  detection.w = result[5] * img.cols - detection.x;
82  detection.h = result[6] * img.rows - detection.y;
83 
84  detection.enabled = true;
85 
86  detections.push_back(detection);
87  result += 7;
88  }
89  return detections;
90 }

98 /* Wrap the input layer of the network in separate cv::Mat objects
99  * (one per channel). This way we save one memcpy operation and we
100  * don't need to rely on cudaMemcpy2D. The last preprocessing
101  * operation will write the separate channels directly to the input
102  * layer. */
103 void SSDDetector::WrapInputLayer(std::vector<cv::Mat>* input_channels)
104 {
105  caffe::Blob<float> *input_layer = net_->input_blobs()[0];
106 
107  int width = input_layer->width();
108  int height = input_layer->height();
109  float *input_data = input_layer->mutable_cpu_data();
110  for (int i = 0; i < input_layer->channels(); ++i)
111  {
112  cv::Mat channel(height, width, CV_32FC1, input_data);
113  input_channels->push_back(channel);
114  input_data += width * height;
115  }
116 }

118 void SSDDetector::Preprocess(const cv::Mat& img,
119  std::vector<cv::Mat>* input_channels)
120 {
121  /* Convert the input image to the input image format of the network. */
122  cv::Mat sample;
123  if (img.channels() == 3 && num_channels_ == 1)
124  cv::cvtColor(img, sample, cv::COLOR_BGR2GRAY);
125  else if (img.channels() == 4 && num_channels_ == 1)
126  cv::cvtColor(img, sample, cv::COLOR_BGRA2GRAY);
127  else if (img.channels() == 4 && num_channels_ == 3)
128  cv::cvtColor(img, sample, cv::COLOR_BGRA2BGR);
129  else if (img.channels() == 1 && num_channels_ == 3)
130  cv::cvtColor(img, sample, cv::COLOR_GRAY2BGR);
131  else
132  sample = img;
133 
134  cv::Mat sample_resized;
135  if (sample.size() != input_geometry_)
136  cv::resize(sample, sample_resized, input_geometry_);
137  else
138  sample_resized = sample;
139 
140  cv::Mat sample_float;
141  cv::Mat mean_img;
142  if (num_channels_ == 3)
143  {
144  sample_resized.convertTo(sample_float, CV_32FC3);
145  mean_img = cv::Mat(input_geometry_, CV_32FC3, mean_);
146  } else
147  {
148  sample_resized.convertTo(sample_float, CV_32FC1);
149  mean_img = cv::Mat(input_geometry_, CV_32FC1, mean_);
150  }
151 
152  cv::Mat sample_normalized;
153 
154  cv::subtract(sample_float, mean_img, sample_normalized);
155 
156  /* This operation will write the separate BGR planes directly to the
157  * input layer of the network because it is wrapped by the cv::Mat
158  * objects in input_channels. */
159  cv::split(sample_normalized, *input_channels);
160 
161  CHECK(reinterpret_cast<float *>(input_channels->at(0).data)
162  == net_->input_blobs()[0]->cpu_data())
163  << "Input channels are not wrapping the input layer of the network.";
164 }
