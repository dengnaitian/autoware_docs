110 void SSCInterface::callbackFromSSCFeedbacks(const automotive_platform_msgs::VelocityAccelConstPtr& msg_velocity,
111  const automotive_platform_msgs::CurvatureFeedbackConstPtr& msg_curvature,
112  const automotive_platform_msgs::ThrottleFeedbackConstPtr& msg_throttle,
113  const automotive_platform_msgs::BrakeFeedbackConstPtr& msg_brake,
114  const automotive_platform_msgs::GearFeedbackConstPtr& msg_gear,
115  const pacmod_msgs::WheelSpeedRptConstPtr& msg_wheel_speed,
116  const pacmod_msgs::SystemRptFloatConstPtr& msg_steering_wheel)
117 {
118  ros::Time stamp = msg_velocity->header.stamp;
119 
120  // current speed
121  double speed =
122  !use_rear_wheel_speed_ ?
123  (msg_velocity->velocity) :
124  (msg_wheel_speed->rear_left_wheel_speed + msg_wheel_speed->rear_right_wheel_speed) * tire_radius_ / 2.;
125 
126  // update adaptive gear ratio (avoiding zero divizion)
127  adaptive_gear_ratio_ =
128  std::max(1e-5, agr_coef_a_ + agr_coef_b_ * speed * speed - agr_coef_c_ * msg_steering_wheel->output);
129  // current steering curvature
130  double curvature = !use_adaptive_gear_ratio_ ?
131  (msg_curvature->curvature) :
132  std::tan(msg_steering_wheel->output / adaptive_gear_ratio_) / wheel_base_;
133 
134  // as_current_velocity (geometry_msgs::TwistStamped)
135  geometry_msgs::TwistStamped twist;
136  twist.header.frame_id = BASE_FRAME_ID;
137  twist.header.stamp = stamp;
138  twist.twist.linear.x = speed; // [m/s]
139  twist.twist.angular.z = curvature * speed; // [rad/s]
140  current_twist_pub_.publish(twist);
141 
142  // vehicle_status (autoware_msgs::VehicleStatus)
143  autoware_msgs::VehicleStatus vehicle_status;
144  vehicle_status.header.frame_id = BASE_FRAME_ID;
145  vehicle_status.header.stamp = stamp;
146 
147  // drive/steeringmode
148  vehicle_status.drivemode = (module_states_.state == "active") ? autoware_msgs::VehicleStatus::MODE_AUTO :
149  autoware_msgs::VehicleStatus::MODE_MANUAL;
150  vehicle_status.steeringmode = vehicle_status.drivemode;
151 
152  // speed [km/h]
153  vehicle_status.speed = speed * 3.6;
154 
155  // drive/brake pedal [0,1000] (TODO: Scaling)
156  vehicle_status.drivepedal = (int)(1000 * msg_throttle->throttle_pedal);
157  vehicle_status.brakepedal = (int)(1000 * msg_brake->brake_pedal);
158 
159  // steering angle [rad]
160  vehicle_status.angle = std::atan(curvature * wheel_base_);
161 
162  // gearshift
163  if (msg_gear->current_gear.gear == automotive_platform_msgs::Gear::NONE)
164  {
165  vehicle_status.gearshift = 0;
166  }
167  else if (msg_gear->current_gear.gear == automotive_platform_msgs::Gear::PARK)
168  {
169  vehicle_status.gearshift = 3;
170  }
171  else if (msg_gear->current_gear.gear == automotive_platform_msgs::Gear::REVERSE)
172  {
173  vehicle_status.gearshift = 2;
174  }
175  else if (msg_gear->current_gear.gear == automotive_platform_msgs::Gear::NEUTRAL)
176  {
177  vehicle_status.gearshift = 4;
178  }
179  else if (msg_gear->current_gear.gear == automotive_platform_msgs::Gear::DRIVE)
180  {
181  vehicle_status.gearshift = 1;
182  }
183 
184  // lamp/light cannot be obtain from SSC
185  // vehicle_status.lamp
186  // vehicle_status.light
187 
188  vehicle_status_pub_.publish(vehicle_status);
189 }

191 void SSCInterface::publishCommand()
192 {
193  if (!command_initialized_)
194  {
195  return;
196  }
197 
198  ros::Time stamp = ros::Time::now();
199 
200  // Desired values
201  // Driving mode (If autonomy mode should be active, mode = 1)
202  unsigned char desired_mode = engage_ ? 1 : 0;
203 
204  // Speed for SSC speed_model
205  double desired_speed = vehicle_cmd_.ctrl_cmd.linear_velocity;
206 
207  // Curvature for SSC steer_model
208  double desired_steering_angle = !use_adaptive_gear_ratio_ ?
209  vehicle_cmd_.ctrl_cmd.steering_angle :
210  vehicle_cmd_.ctrl_cmd.steering_angle * ssc_gear_ratio_ / adaptive_gear_ratio_;
211  double desired_curvature = std::tan(desired_steering_angle) / wheel_base_;
212 
213  // Gear (TODO: Use vehicle_cmd.gear)
214  unsigned char desired_gear = engage_ ? automotive_platform_msgs::Gear::DRIVE : automotive_platform_msgs::Gear::NONE;
215 
216  // Turn signal
217  unsigned char desired_turn_signal = automotive_platform_msgs::TurnSignalCommand::NONE;
218 
219  if (vehicle_cmd_.lamp_cmd.l == 0 && vehicle_cmd_.lamp_cmd.r == 0)
220  {
221  desired_turn_signal = automotive_platform_msgs::TurnSignalCommand::NONE;
222  }
223  else if (vehicle_cmd_.lamp_cmd.l == 1 && vehicle_cmd_.lamp_cmd.r == 0)
224  {
225  desired_turn_signal = automotive_platform_msgs::TurnSignalCommand::LEFT;
226  }
227  else if (vehicle_cmd_.lamp_cmd.l == 0 && vehicle_cmd_.lamp_cmd.r == 1)
228  {
229  desired_turn_signal = automotive_platform_msgs::TurnSignalCommand::RIGHT;
230  }
231  else if (vehicle_cmd_.lamp_cmd.l == 1 && vehicle_cmd_.lamp_cmd.r == 1)
232  {
233  // NOTE: HAZARD signal cannot be used in automotive_platform_msgs::TurnSignalCommand
234  }
235 
236  // Override desired speed to ZERO by emergency/timeout
237  bool emergency = (vehicle_cmd_.emergency == 1);
238  bool timeouted = (((ros::Time::now() - command_time_).toSec() * 1000) > command_timeout_);
239 
240  if (emergency || timeouted)
241  {
242  ROS_ERROR("Emergency Stopping, emergency = %d, timeouted = %d", emergency, timeouted);
243  desired_speed = 0.0;
244  }
245 
246  // speed command
247  automotive_platform_msgs::SpeedMode speed_mode;
248  speed_mode.header.frame_id = BASE_FRAME_ID;
249  speed_mode.header.stamp = stamp;
250  speed_mode.mode = desired_mode;
251  speed_mode.speed = desired_speed;
252  speed_mode.acceleration_limit = acceleration_limit_;
253  speed_mode.deceleration_limit = deceleration_limit_;
254 
255  // steer command
256  automotive_platform_msgs::SteerMode steer_mode;
257  steer_mode.header.frame_id = BASE_FRAME_ID;
258  steer_mode.header.stamp = stamp;
259  steer_mode.mode = desired_mode;
260  steer_mode.curvature = desired_curvature;
261  steer_mode.max_curvature_rate = max_curvature_rate_;
262 
263  // turn signal command
264  automotive_platform_msgs::TurnSignalCommand turn_signal;
265  turn_signal.header.frame_id = BASE_FRAME_ID;
266  turn_signal.header.stamp = stamp;
267  turn_signal.mode = desired_mode;
268  turn_signal.turn_signal = desired_turn_signal;
269 
270  // gear command
271  automotive_platform_msgs::GearCommand gear_cmd;
272  gear_cmd.header.frame_id = BASE_FRAME_ID;
273  gear_cmd.header.stamp = stamp;
274  gear_cmd.command.gear = desired_gear;
275 
276  // publish
277  speed_mode_pub_.publish(speed_mode);
278  steer_mode_pub_.publish(steer_mode);
279  turn_signal_pub_.publish(turn_signal);
280  gear_pub_.publish(gear_cmd);
281 
282  ROS_INFO_STREAM("Mode: " << (int)desired_mode << ", "
283  << "Speed: " << speed_mode.speed << ", "
284  << "Curvature: " << steer_mode.curvature << ", "
285  << "Gear: " << (int)gear_cmd.command.gear << ", "
286  << "TurnSignal: " << (int)turn_signal.turn_signal);
287 }
