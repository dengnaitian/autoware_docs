153  void cbPing(const vssp::header &header, const std::chrono::microseconds &delayRead)
154  {
155  ros::Time now = ros::Time::now() - ros::Duration(delayRead.count() * 0.001 * 0.001);
156  ros::Duration delay = ((now - timePing)
157  - ros::Duration(header.send_time_ms * 0.001 - header.received_time_ms * 0.001)) * 0.5;
158  ros::Time base = timePing + delay - ros::Duration(header.received_time_ms * 0.001);
159  if(timestampBase == ros::Time(0)) timestampBase = base;
160  else timestampBase += (base - timestampBase) * 0.01;
161  }
162  void cbAux(
163  const vssp::header &header,
164  const vssp::aux_header &aux_header,
165  const boost::shared_array<vssp::aux> &auxs,
166  const std::chrono::microseconds &delayRead)
167  {
168  if(timestampBase == ros::Time(0)) return;
169  ros::Time stamp = timestampBase + ros::Duration(aux_header.timestamp_ms * 0.001);
170 
171  if((aux_header.data_bitfield & (vssp::AX_MASK_ANGVEL | vssp::AX_MASK_LINACC))
172  == (vssp::AX_MASK_ANGVEL | vssp::AX_MASK_LINACC))
173  {
174  imu.header.frame_id = frame_id;
175  imu.header.stamp = stamp;
176  for(int i = 0; i < aux_header.data_count; i ++)
177  {
178  imu.orientation_covariance[0] = -1.0;
179  imu.angular_velocity.x = auxs[i].ang_vel.x;
180  imu.angular_velocity.y = auxs[i].ang_vel.y;
181  imu.angular_velocity.z = auxs[i].ang_vel.z;
182  imu.linear_acceleration.x = auxs[i].lin_acc.x;
183  imu.linear_acceleration.y = auxs[i].lin_acc.y;
184  imu.linear_acceleration.z = auxs[i].lin_acc.z;
185  pubImu.publish(imu);
186  imu.header.stamp += ros::Duration(aux_header.data_ms * 0.001);
187  }
188  }
189  if((aux_header.data_bitfield & vssp::AX_MASK_MAG) == vssp::AX_MASK_MAG )
190  {
191  mag.header.frame_id = frame_id;
192  mag.header.stamp = stamp;
193  for(int i = 0; i < aux_header.data_count; i ++)
194  {
195  mag.magnetic_field.x = auxs[i].mag.x;
196  mag.magnetic_field.y = auxs[i].mag.y;
197  mag.magnetic_field.z = auxs[i].mag.z;
198  pubMag.publish(mag);
199  mag.header.stamp += ros::Duration(aux_header.data_ms * 0.001);
200  }
201  }
