35 void Cluster2D::traverse(Node *x)
36 {
37  std::vector<Node *> p;
38  p.clear();
39  while (x->traversed == 0)
40  {
41  p.push_back(x);
42  x->traversed = 2;
43  x = x->center_node;
44  }
45  if (x->traversed == 2)
46  {
47  for (int i = static_cast<int>(p.size()) - 1; i >= 0 && p[i] != x; i--)
48  {
49  p[i]->is_center = true;
50  }
51  x->is_center = true;
52  }
53  for (size_t i = 0; i < p.size(); i++)
54  {
55  Node *y = p[i];
56  y->traversed = 1;
57  y->parent = x->parent;
58  }
59 }

61 void Cluster2D::cluster(const caffe::Blob<float> &category_pt_blob,
62  const caffe::Blob<float> &instance_pt_blob,
63  const pcl::PointCloud<pcl::PointXYZI>::Ptr &pc_ptr,
64  const pcl::PointIndices &valid_indices,
65  float objectness_thresh, bool use_all_grids_for_clustering)
66 {
67  const float *category_pt_data = category_pt_blob.cpu_data();
68  const float *instance_pt_x_data = instance_pt_blob.cpu_data();
69  const float *instance_pt_y_data =
70  instance_pt_blob.cpu_data() + instance_pt_blob.offset(0, 1);
71 
72  pc_ptr_ = pc_ptr;
73  std::vector<std::vector<Node>> nodes(rows_,
74  std::vector<Node>(cols_, Node()));
75 
76  // map points into grids
77  size_t tot_point_num = pc_ptr_->size();
78  valid_indices_in_pc_ = &(valid_indices.indices);
79  CHECK_LE(valid_indices_in_pc_->size(), tot_point_num);
80  point2grid_.assign(valid_indices_in_pc_->size(), -1);
81 
82  for (size_t i = 0; i < valid_indices_in_pc_->size(); ++i)
83  {
84  int point_id = valid_indices_in_pc_->at(i);
85  CHECK_GE(point_id, 0);
86  CHECK_LT(point_id, static_cast<int>(tot_point_num));
87  const auto &point = pc_ptr_->points[point_id];
88  // * the coordinates of x and y have been exchanged in feature generation
89  // step,
90  // so we swap them back here.
91  int pos_x = F2I(point.y, range_, inv_res_x_); // col
92  int pos_y = F2I(point.x, range_, inv_res_y_); // row
93  if (IsValidRowCol(pos_y, pos_x))
94  {
95  // get grid index and count point number for corresponding node
96  point2grid_[i] = RowCol2Grid(pos_y, pos_x);
97  nodes[pos_y][pos_x].point_num++;
98  }
99  }
100 
101  // construct graph with center offset prediction and objectness
102  for (int row = 0; row < rows_; ++row)
103  {
104  for (int col = 0; col < cols_; ++col)
105  {
106  int grid = RowCol2Grid(row, col);
107  Node *node = &nodes[row][col];
108  DisjointSetMakeSet(node);
109  node->is_object =
110  (use_all_grids_for_clustering || nodes[row][col].point_num > 0) &&
111  (*(category_pt_data + grid) >= objectness_thresh);
112  int center_row = std::round(row + instance_pt_x_data[grid] * scale_);
113  int center_col = std::round(col + instance_pt_y_data[grid] * scale_);
114  center_row = std::min(std::max(center_row, 0), rows_ - 1);
115  center_col = std::min(std::max(center_col, 0), cols_ - 1);
116  node->center_node = &nodes[center_row][center_col];
117  }
118  }
119 
120  // traverse nodes
121  for (int row = 0; row < rows_; ++row)
122  {
123  for (int col = 0; col < cols_; ++col)
124  {
125  Node *node = &nodes[row][col];
126  if (node->is_object && node->traversed == 0)
127  {
128  traverse(node);
129  }
130  }
131  }
132  for (int row = 0; row < rows_; ++row)
133  {
134  for (int col = 0; col < cols_; ++col)
135  {
136  Node *node = &nodes[row][col];
137  if (!node->is_center)
138  {
139  continue;
140  }
141  for (int row2 = row - 1; row2 <= row + 1; ++row2)
142  {
143  for (int col2 = col - 1; col2 <= col + 1; ++col2)
144  {
145  if ((row2 == row || col2 == col) && IsValidRowCol(row2, col2))
146  {
147  Node *node2 = &nodes[row2][col2];
148  if (node2->is_center)
149  {
150  DisjointSetUnion(node, node2);
151  }
152  }
153  }
154  }
155  }
156  }
157 
158  int count_obstacles = 0;
159  obstacles_.clear();
160  id_img_.assign(grids_, -1);
161  for (int row = 0; row < rows_; ++row)
162  {
163  for (int col = 0; col < cols_; ++col)
164  {
165  Node *node = &nodes[row][col];
166  if (!node->is_object)
167  {
168  continue;
169  }
170  Node *root = DisjointSetFind(node);
171  if (root->obstacle_id < 0)
172  {
173  root->obstacle_id = count_obstacles++;
174  CHECK_EQ(static_cast<int>(obstacles_.size()), count_obstacles - 1);
175  obstacles_.push_back(Obstacle());
176  }
177  int grid = RowCol2Grid(row, col);
178  CHECK_GE(root->obstacle_id, 0);
179  id_img_[grid] = root->obstacle_id;
180  obstacles_[root->obstacle_id].grids.push_back(grid);
181  }
182  }
183  CHECK_EQ(static_cast<size_t>(count_obstacles), obstacles_.size());
184 }

186 void Cluster2D::filter(const caffe::Blob<float> &confidence_pt_blob,
187  const caffe::Blob<float> &height_pt_blob)
188 {
189  const float *confidence_pt_data = confidence_pt_blob.cpu_data();
190  const float *height_pt_data = height_pt_blob.cpu_data();
191  for (size_t obstacle_id = 0; obstacle_id < obstacles_.size();
192  obstacle_id++)
193  {
194  Obstacle *obs = &obstacles_[obstacle_id];
195  CHECK_GT(obs->grids.size(), 0);
196  double score = 0.0;
197  double height = 0.0;
198  for (int grid : obs->grids)
199  {
200  score += static_cast<double>(confidence_pt_data[grid]);
201  height += static_cast<double>(height_pt_data[grid]);
202  }
203  obs->score = score / static_cast<double>(obs->grids.size());
204  obs->height = height / static_cast<double>(obs->grids.size());
205  obs->cloud_ptr.reset(new pcl::PointCloud<pcl::PointXYZI>);
206  }
207 }

209 void Cluster2D::classify(const caffe::Blob<float> &classify_pt_blob)
210 {
211  const float *classify_pt_data = classify_pt_blob.cpu_data();
212  int num_classes = classify_pt_blob.channels();
213  CHECK_EQ(num_classes, MAX_META_TYPE);
214  for (size_t obs_id = 0; obs_id < obstacles_.size(); obs_id++)
215  {
216  Obstacle *obs = &obstacles_[obs_id];
217  for (size_t grid_id = 0; grid_id < obs->grids.size(); grid_id++)
218  {
219  int grid = obs->grids[grid_id];
220  for (int k = 0; k < num_classes; k++)
221  {
222  obs->meta_type_probs[k] += classify_pt_data[k * grids_ + grid];
223  }
224  }
225  int meta_type_id = 0;
226  for (int k = 0; k < num_classes; k++)
227  {
228  obs->meta_type_probs[k] /= obs->grids.size();
229  if (obs->meta_type_probs[k] > obs->meta_type_probs[meta_type_id])
230  {
231  meta_type_id = k;
232  }
233  }
234  obs->meta_type = static_cast<MetaType>(meta_type_id);
235  }
236 }

257  float min_z = std::numeric_limits<float>::max();
258  float max_z = -std::numeric_limits<float>::max();
259  float average_x = 0, average_y = 0, average_z = 0;
260  float length, width, height;
261 
262  pcl::PointXYZ min_point;
263  pcl::PointXYZ max_point;
264  pcl::PointXYZ average_point;
265  pcl::PointXYZ centroid;
266 
267  for (auto pit = in_cluster.points.begin(); pit != in_cluster.points.end(); ++pit)
268  {
269  average_x += pit->x;
270  average_y += pit->y;
271  average_z += pit->z;
272  centroid.x += pit->x;
273  centroid.y += pit->y;
274  centroid.z += pit->z;
275 
276  if (pit->x < min_x)
277  min_x = pit->x;
278  if (pit->y < min_y)
279  min_y = pit->y;
280  if (pit->z < min_z)
281  min_z = pit->z;
282  if (pit->x > max_x)
283  max_x = pit->x;
284  if (pit->y > max_y)
285  max_y = pit->y;
286  if (pit->z > max_z)
287  max_z = pit->z;
288  }
289  // min, max points
290  min_point.x = min_x;
291  min_point.y = min_y;
292  min_point.z = min_z;
293  max_point.x = max_x;
294  max_point.y = max_y;
295  max_point.z = max_z;
296 
297  if (in_cluster.points.size() > 0)
298  {
299  centroid.x /= in_cluster.points.size();
300  centroid.y /= in_cluster.points.size();
301  centroid.z /= in_cluster.points.size();
302  }
303  length = max_point.x - min_point.x;
304  width = max_point.y - min_point.y;
305  height = max_point.z - min_point.z;
306 
307  resulting_object.pose.position.x = min_point.x + length / 2;
308  resulting_object.pose.position.y = min_point.y + width / 2;
309  resulting_object.pose.position.z = min_point.z + height / 2;
310  resulting_object.dimensions.x = ((length < 0) ? -1 * length : length);
311  resulting_object.dimensions.y = ((width < 0) ? -1 * width : width);
312  resulting_object.dimensions.z = ((height < 0) ? -1 * height : height);
313 
314  std::vector<cv::Point2f> points;
315  for (unsigned int i = 0; i < in_cluster.points.size(); i++)
316  {
317  cv::Point2f pt;
318  pt.x = in_cluster.points[i].x;
319  pt.y = in_cluster.points[i].y;
320  points.push_back(pt);
321  }
322 
323  resulting_object.space_frame = in_header.frame_id;
324 
325  return resulting_object;
326 }

328 void Cluster2D::getObjects(const float confidence_thresh,
329  const float height_thresh,
330  const int min_pts_num,
331  autoware_msgs::DetectedObjectArray &objects,
332  const std_msgs::Header &in_header)
333 {

335 
336  for (size_t i = 0; i < point2grid_.size(); ++i)
337  {
338  int grid = point2grid_[i];
339  if (grid < 0)
340  {
341  continue;
342  }
343 
344  CHECK_GE(grid, 0);
345  CHECK_LT(grid, grids_);
346  int obstacle_id = id_img_[grid];
347 
348  int point_id = valid_indices_in_pc_->at(i);
349  CHECK_GE(point_id, 0);
350  CHECK_LT(point_id, static_cast<int>(pc_ptr_->size()));
351 
352  if (obstacle_id >= 0 &&
353  obstacles_[obstacle_id].score >= confidence_thresh)
354  {
355  if (height_thresh < 0 ||
356  pc_ptr_->points[point_id].z <=
357  obstacles_[obstacle_id].height + height_thresh)
358  {
359  obstacles_[obstacle_id].cloud_ptr->push_back(pc_ptr_->points[point_id]);
360  }
361  }
362  }
363 
364  for (size_t obstacle_id = 0; obstacle_id < obstacles_.size();
365  obstacle_id++)
366  {
367  Obstacle *obs = &obstacles_[obstacle_id];
368  if (static_cast<int>(obs->cloud_ptr->size()) < min_pts_num)
369  {
370  continue;
371  }
372 
373  autoware_msgs::DetectedObject out_obj = obstacleToObject(*obs, in_header);
374 
375  objects.objects.push_back(out_obj);
376  }
377 }
