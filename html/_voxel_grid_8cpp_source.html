260 void VoxelGrid<PointSourceType>::computeCentroidAndCovariance()
261 {
262  for (int idx = real_min_bx_; idx <= real_max_bx_; idx++)
263  for (int idy = real_min_by_; idy <= real_max_by_; idy++)
264  for (int idz = real_min_bz_; idz <= real_max_bz_; idz++) {
265  int i = voxelId(idx, idy, idz, min_b_x_, min_b_y_, min_b_z_, vgrid_x_, vgrid_y_, vgrid_z_);
266  int ipoint_num = (*points_id_)[i].size();
267  double point_num = static_cast<double>(ipoint_num);
268  Eigen::Vector3d pt_sum = (*tmp_centroid_)[i];
269 
270  if (ipoint_num > 0) {
271  (*centroid_)[i] = pt_sum / point_num;
272  }
273 
274  Eigen::Matrix3d covariance;
275 
276  if (ipoint_num >= min_points_per_voxel_) {
277  covariance = ((*tmp_cov_)[i] - 2.0 * (pt_sum * (*centroid_)[i].transpose())) / point_num + (*centroid_)[i] * (*centroid_)[i].transpose();
278  covariance *= (point_num - 1.0) / point_num;
279 
280  SymmetricEigensolver3x3 sv(covariance);
281 
282  sv.compute();
283  Eigen::Matrix3d evecs = sv.eigenvectors();
284  Eigen::Matrix3d evals = sv.eigenvalues().asDiagonal();
285 
286  if (evals(0, 0) < 0 || evals(1, 1) < 0 || evals(2, 2) <= 0) {
287  (*points_per_voxel_)[i] = -1;
288  continue;
289  }
290 
291  double min_cov_eigvalue = evals(2, 2) * 0.01;
292 
293  if (evals(0, 0) < min_cov_eigvalue) {
294  evals(0, 0) = min_cov_eigvalue;
295 
296  if (evals(1, 1) < min_cov_eigvalue) {
297  evals(1, 1) = min_cov_eigvalue;
298  }
299 
300  covariance = evecs * evals * evecs.inverse();
301  }
302 
303  (*icovariance_)[i] = covariance.inverse();
304  }
305  }

414  int max_id_z = static_cast<int>(floor((t_z + radius) / voxel_z_));

419 
420  /* Find intersection of the cube containing
421  * the NN sphere of the point and the voxel grid
422  */
423  max_id_x = (max_id_x > real_max_bx_) ? real_max_bx_ : max_id_x;
424  max_id_y = (max_id_y > real_max_by_) ? real_max_by_ : max_id_y;
425  max_id_z = (max_id_z > real_max_bz_) ? real_max_bz_ : max_id_z;
426

428  min_id_y = (min_id_y < real_min_by_) ? real_min_by_ : min_id_y;
429  min_id_z = (min_id_z < real_min_bz_) ? real_min_bz_ : min_id_z;
430  int nn = 0;
431 
432  for (int idx = min_id_x; idx <= max_id_x && nn < max_nn; idx++) {
433  for (int idy = min_id_y; idy <= max_id_y && nn < max_nn; idy++) {
434  for (int idz = min_id_z; idz <= max_id_z && nn < max_nn; idz++) {

436  min_b_x_, min_b_y_, min_b_z_,
437  vgrid_x_, vgrid_y_, vgrid_z_);

440  double cx = (*centroid_)[vid](0) - static_cast<double>(t_x);

678 {

681  }
682 
683  float new_max_x, new_max_y, new_max_z;
684  float new_min_x, new_min_y, new_min_z;

688 
689  /* Update current boundaries of the voxel grid
690  * Also allocate buffer for new voxel grid and

693  */
694  updateBoundaries(new_max_x, new_max_y, new_max_z, new_min_x, new_min_y, new_min_z);
695 
696  /* Update changed voxels (voxels that contains new points).
697  * Update centroids of voxels and their covariance matrixes
698  * as well as inverse covariance matrixes */
699  updateVoxelContent(new_cloud);
700 
701  /* Update octree */
702  std::vector<Eigen::Vector3i> new_voxel_id(new_cloud->points.size());
703

718 template <typename PointSourceType>
719 void VoxelGrid<PointSourceType>::updateVoxelContent(typename pcl::PointCloud<PointSourceType>::Ptr new_cloud)
720 {
721  int total_points_num = source_cloud_->points.size();
722 
723  for (int i = 0; i < new_cloud->points.size(); i++) {
724  PointSourceType p = new_cloud->points[i];
725  Eigen::Vector3d p3d(p.x, p.y, p.z);
726  int vx = static_cast<int>(floor(p.x / voxel_x_));
727  int vy = static_cast<int>(floor(p.y / voxel_y_));
728  int vz = static_cast<int>(floor(p.z / voxel_z_));
729  int vid = voxelId(vx, vy, vz, min_b_x_, min_b_y_, min_b_z_, vgrid_x_, vgrid_y_, vgrid_z_);
730  std::vector<int> &tmp_pid = (*points_id_)[vid];
731 
732  if (tmp_pid.size() == 0) {
733  (*centroid_)[vid].setZero();
734  (*tmp_cov_)[vid].setIdentity();
735  (*tmp_centroid_)[vid].setZero();
736  }
737 
738  (*tmp_centroid_)[vid] += p3d;
739  (*tmp_cov_)[vid] += p3d * p3d.transpose();
740  (*points_id_)[vid].push_back(total_points_num + i);
741 
742  // Update centroids
743  int ipoint_num = (*points_id_)[vid].size();
744  (*centroid_)[vid] = (*tmp_centroid_)[vid] / static_cast<double>(ipoint_num);
745  (*points_per_voxel_)[vid] = ipoint_num;
746 
747 
748  // Update covariance
749  double point_num = static_cast<double>(ipoint_num);
750  Eigen::Vector3d pt_sum = (*tmp_centroid_)[vid];
751 
752  // Update centroids
753  (*centroid_)[vid] = (*tmp_centroid_)[vid] / point_num;
754 
755  if (ipoint_num >= min_points_per_voxel_) {
756  Eigen::Matrix3d covariance;
757  covariance = ((*tmp_cov_)[vid] - 2.0 * (pt_sum * (*centroid_)[vid].transpose())) / point_num + (*centroid_)[vid] * (*centroid_)[vid].transpose();
758  covariance *= (point_num - 1.0) / point_num;
759 
760  SymmetricEigensolver3x3 sv(covariance);
761 
762  sv.compute();
763  Eigen::Matrix3d evecs = sv.eigenvectors();
764  Eigen::Matrix3d evals = sv.eigenvalues().asDiagonal();
765 
766  if (evals(0, 0) < 0 || evals(1, 1) < 0 || evals(2, 2) <= 0) {
767  (*points_per_voxel_)[vid] = -1;
768  continue;
769  }
770 
771  double min_cov_eigvalue = evals(2, 2) * 0.01;
772 
773  if (evals(0, 0) < min_cov_eigvalue) {
774  evals(0, 0) = min_cov_eigvalue;
775 
776  if (evals(1, 1) < min_cov_eigvalue) {
777  evals(1, 1) = min_cov_eigvalue;
778  }
779 
780  covariance = evecs * evals * evecs.inverse();
781  }
782 
783  (*icovariance_)[vid] = covariance.inverse();
784  }
785  }
786 }
