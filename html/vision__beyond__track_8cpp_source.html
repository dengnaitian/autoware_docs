206  double BeyondTracker::get_3d2d_score(Detection cd, Detection pd)
207  {
208  Paths solution;
209  Clipper c;
210  c.AddPaths(pd.pd_2d_convhull_clip_, ptSubject, true);
211  c.AddPaths(cd.cd_2d_convhull_clip_, ptClip, true);
212  c.Execute(ctIntersection, solution, pftNonZero, pftNonZero);
213  if (solution.size() > 0)
214  {
215  c.Execute(ctUnion, solution, pftNonZero, pftNonZero);
216  auto area_in = Area(solution[0]);
217  double area_target = cd.cd_area_;
218  return area_in / area_target;
219  } else
220  {
221  return MAX_VALUE;
222  }
223  }

263 void BeyondTrackerNode::detection_to_objects(const std::vector<beyondtrack::Detection> &in_objects,
264  autoware_msgs::DetectedObjectArray &out_message)
265 {
266  for (unsigned int i = 0; i < in_objects.size(); ++i)
267  {
268  autoware_msgs::DetectedObject obj;
269 
270  obj.x = (in_objects[i].bbox_[0]);
271  obj.y = (in_objects[i].bbox_[1]);
272  obj.width = (in_objects[i].bbox_[2] - in_objects[i].bbox_[0]);
273  obj.height = (in_objects[i].bbox_[3] - in_objects[i].bbox_[1]);
274  if (obj.x < 0)
275  obj.x = 0;
276  if (obj.y < 0)
277  obj.y = 0;
278  if (obj.width < 0)
279  obj.width = 0;
280  if (obj.height < 0)
281  obj.height = 0;
282 
283  size_t color_index = in_objects[i].object_id_ % colors_.size();
284  obj.color.r = colors_[color_index].val[0];
285  obj.color.g = colors_[color_index].val[1];
286  obj.color.b = colors_[color_index].val[2];
287  obj.color.a = 1.0f;
288 
289  obj.score = 0.0;
290  obj.label = in_objects[i].class_type_;
291  obj.valid = true;
292 
293  obj.id = in_objects[i].object_id_;
294  obj.image_frame = out_message.header.frame_id;
295 
296  // set bounding box direction
297  tf::Quaternion quat = tf::createQuaternionFromRPY(0.0, 0.0, in_objects[i].yaw_);
298  tf::quaternionTFToMsg(quat, obj.pose.orientation);
299 
300  out_message.objects.push_back(obj);
301  }
