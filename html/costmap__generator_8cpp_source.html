74 void CostmapGenerator::run()

77  pub_occupancy_grid_ = nh_.advertise<nav_msgs::OccupancyGrid>("/semantics/costmap_generator/occupancy_grid", 1);

90 void CostmapGenerator::objectsCallback(const autoware_msgs::DetectedObjectArray::ConstPtr& in_objects)
91 {
92  if(use_objects_box_)
93  {
94  const bool use_convex_hull = !use_objects_box_;
95  costmap_[OBJECTS_BOX_COSTMAP_LAYER_] = generateObjectsCostmap(in_objects, use_convex_hull);
96  }
97 
98  if(use_objects_convex_hull_)
99  {
100  costmap_[OBJECTS_CONVEX_HULL_COSTMAP_LAYER_] = generateObjectsCostmap(in_objects, use_objects_convex_hull_);
101  }
102  costmap_[VECTORMAP_COSTMAP_LAYER_] = generateVectormapCostmap();
103  costmap_[COMBINED_COSTMAP_LAYER_] = generateCombinedCostmap();
104 
105  std_msgs::Header in_header = in_objects->header;
106  publishRosMsg(costmap_, in_header);
107 }

173 grid_map::Matrix CostmapGenerator::generateCombinedCostmap()
174 {
175  // assuming combined_costmap is calculated by element wise max operation
176  grid_map::GridMap combined_costmap = costmap_;
177  combined_costmap[COMBINED_COSTMAP_LAYER_].setConstant(grid_min_value_);
178  combined_costmap[COMBINED_COSTMAP_LAYER_] =
179  combined_costmap[COMBINED_COSTMAP_LAYER_].cwiseMax(combined_costmap[SENSOR_POINTS_COSTMAP_LAYER_]);
180  combined_costmap[COMBINED_COSTMAP_LAYER_] =
181  combined_costmap[COMBINED_COSTMAP_LAYER_].cwiseMax(combined_costmap[VECTORMAP_COSTMAP_LAYER_]);
182  combined_costmap[COMBINED_COSTMAP_LAYER_] =
183  combined_costmap[COMBINED_COSTMAP_LAYER_].cwiseMax(combined_costmap[OBJECTS_BOX_COSTMAP_LAYER_]);
184  combined_costmap[COMBINED_COSTMAP_LAYER_] =
185  combined_costmap[COMBINED_COSTMAP_LAYER_].cwiseMax(combined_costmap[OBJECTS_CONVEX_HULL_COSTMAP_LAYER_]);
186  return combined_costmap[COMBINED_COSTMAP_LAYER_];
187 }
