115 std::vector<std::string> explode(const std::string& text, const std::string& separators)
116 {
117  std::vector<std::string> words;
118  size_t n = text.length ();
119  size_t start = 0;
120  size_t stop = text.find_first_of (separators);
121  if (stop > n) stop = n;
122 
123  while (start <= n)
124  {
125  words.push_back (text.substr (start, stop-start));
126  start = stop+1;
127  stop = text.find_first_of (separators, start);
128  if (stop > n) stop = n;
129  }
130  return words;
131 }

133 uint32_t nmeaHourToUnixTime(const std::string& nmea_sentence)
134 {
135  std::vector< std::string > words = explode(nmea_sentence,",");
136  if(words.size() < 10 )
137  return -1;
138  uint32_t hour = atoi(words[1].substr(0,2).c_str());
139 
140  uint32_t day = atoi(words[9].substr(0,2).c_str());
141  uint32_t mon = atoi(words[9].substr(2,2).c_str());
142  uint32_t year = atoi(words[9].substr(4,2).c_str());
143 
144  time_t hour_time;
145  tm time_info;
146  memset(&time_info, 0, sizeof(time_info));
147  // Assumes year is between 2000 and 2099
148  time_info.tm_year = year + 100;
149  time_info.tm_mon = mon - 1;
150  time_info.tm_mday = day;
151  time_info.tm_hour = hour;
152  hour_time = timegm(&time_info);
153 
154  return hour_time;
155 }

157 //-----------------------------------------------------------------------------
158 bool GpsImuDriver::handlePacket(velodyne_packet_structs::VelodynePositioningPacketRaw &vppr)
159 {
160 
161  velodyne_packet_structs::VelodynePositioningPacket vpp;
162 
163  for( int i=0; i<3; i++ )
164  {
165  vpp.gyro_temp_accel[i].gyro = convert12bit2int16(vppr.gyro_temp_accel[i].gyro) * 0.09766 * M_PI / 180.0;
166  vpp.gyro_temp_accel[i].temp = convert12bit2int16(vppr.gyro_temp_accel[i].temp) * 0.1453+25;
167  vpp.gyro_temp_accel[i].accel_x = convert12bit2int16(vppr.gyro_temp_accel[i].accel_x) * 0.001221;
168  vpp.gyro_temp_accel[i].accel_y = convert12bit2int16(vppr.gyro_temp_accel[i].accel_y) * 0.001221;
169  }
170 
171  const float earth_gravity = 9.80665;
172 
173  vpp.gps_timestamp = vppr.gps_timestamp * microseconds_to_seconds;
174  vpp.gyro_temp_accel_xyz[0].gyro = vpp.gyro_temp_accel[1].gyro;
175  vpp.gyro_temp_accel_xyz[0].temp = vpp.gyro_temp_accel[1].temp;
176  vpp.gyro_temp_accel_xyz[0].accel_x = -vpp.gyro_temp_accel[0].accel_y * earth_gravity;
177  vpp.gyro_temp_accel_xyz[0].accel_y = vpp.gyro_temp_accel[2].accel_x * earth_gravity;
178 
179  vpp.gyro_temp_accel_xyz[1].gyro = -vpp.gyro_temp_accel[0].gyro;
180  vpp.gyro_temp_accel_xyz[1].temp = vpp.gyro_temp_accel[0].temp;
181  vpp.gyro_temp_accel_xyz[1].accel_x = -vpp.gyro_temp_accel[1].accel_y * earth_gravity;
182  vpp.gyro_temp_accel_xyz[1].accel_y = -vpp.gyro_temp_accel[2].accel_y * earth_gravity;
183 
184  vpp.gyro_temp_accel_xyz[2].gyro = -vpp.gyro_temp_accel[2].gyro;
185  vpp.gyro_temp_accel_xyz[2].temp = vpp.gyro_temp_accel[2].temp;
186  vpp.gyro_temp_accel_xyz[2].accel_x = -vpp.gyro_temp_accel[0].accel_x * earth_gravity;
187  vpp.gyro_temp_accel_xyz[2].accel_y = -vpp.gyro_temp_accel[1].accel_x * earth_gravity;
188 
189  nmea_msgs::Sentence nmea_sentence_msg;
190  std::string nmea_sentence(vppr.nmea_sentence);
191 
192  /* This section re-constructs the timestamp of the packet based on the
193  * GPS NMEA time and the packet's microsecond counter. The microsecond counter
194  * contains the time from the beginning of the GPS hour, so to reconstruct the
195  * packet's scan time, the counter value needs to be added to the time of the
196  * beginning of the hour.
197  */
198  // Find the current hour using the NMEA sentence from the Velodyne sensor message
199  double nmeaHourUnixTime = nmeaHourToUnixTime(nmea_sentence);
200  // If the time is un-initialized, or if the seconds counter in the counter is
201  // greater than 2, set the hour time (the 2 second delay allows for the NMEA
202  // string to be updated as it is delayed by several hundred milliseconds
203  // from the GPS PPS event)
204  if (hour_time_ == 0 || vpp.gps_timestamp > 2.0) {
205  hour_time_ = nmeaHourUnixTime;
206  }
207  // The seconds counter has rolled over - increment the hour.
208  else if (vppr.gps_timestamp < last_gps_timestamp_) {
209  hour_time_ += 3600;
210  }
211  last_gps_timestamp_ = vppr.gps_timestamp;
212 
213  // Publish all topics with the same ROS time stamp.
214  ros::Time topic_publish_time = ros::Time::now();
215  // === Time Reference Message ===
216  // Set the TimeReference time_ref with the re-constructed sensor time
217  double sensor_scan_time = (double)hour_time_ + vpp.gps_timestamp ;
218  sensor_msgs::TimeReference gpstime_msg;
219  gpstime_msg.header.stamp = topic_publish_time;
220  gpstime_msg.time_ref = ros::Time(sensor_scan_time);
221  gpstime_msg.source = std::string("Sensor On-Board Clock");
222  gpstime_pub.publish(gpstime_msg);
223 
224  // === NMEA Sentence ===
225  nmea_sentence = nmea_sentence.substr(0,nmea_sentence.length()-2);
226  nmea_sentence_msg.sentence = nmea_sentence;
227  nmea_sentence_msg.header.stamp = topic_publish_time;
228  nmea_sentence_msg.header.frame_id = "/gps";
229  nmea_pub.publish(nmea_sentence_msg);
230 
231  // === IMU Message ===
232  sensor_msgs::Imu imumsg;
233  imumsg.header.frame_id="/velodyne:"+devip_;
234  imumsg.header.stamp = topic_publish_time;
235 
236  imumsg.linear_acceleration.x = (vpp.gyro_temp_accel_xyz[0].accel_x+vpp.gyro_temp_accel_xyz[0].accel_y)/2.0;
237  imumsg.linear_acceleration.y = (vpp.gyro_temp_accel_xyz[1].accel_x+vpp.gyro_temp_accel_xyz[1].accel_y)/2.0;
238  imumsg.linear_acceleration.z = (vpp.gyro_temp_accel_xyz[2].accel_x+vpp.gyro_temp_accel_xyz[2].accel_y)/2.0;
239  imumsg.angular_velocity.x = vpp.gyro_temp_accel_xyz[0].gyro;
240  imumsg.angular_velocity.y = vpp.gyro_temp_accel_xyz[1].gyro;
241  imumsg.angular_velocity.z = vpp.gyro_temp_accel_xyz[2].gyro;
242  imumsg.orientation_covariance[0] = -1;
243  imu_pub.publish(imumsg);
244 
245  // === Temperature Messages ===
246  for( std::size_t i=0; i<3; i++ )
247  {
248  sensor_msgs::Temperature tmpmsg;
249  tmpmsg.header.stamp = topic_publish_time;
250  std::stringstream ss;
251  ss << "/velodyne:" << devip_ << i;
252  tmpmsg.header.frame_id = ss.str();
253  tmpmsg.header.stamp = ros::Time::now();
254  tmpmsg.temperature = vpp.gyro_temp_accel_xyz[i].temp;
255  temperature_pub.publish(tmpmsg);
256  }
257 }
