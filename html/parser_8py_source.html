56 def convert_time(nmea_utc):
57  # Get current time in UTC for date information
58  utc_struct = time.gmtime() # immutable, so cannot modify this one
59  utc_list = list(utc_struct)
60  # If one of the time fields is empty, return NaN seconds
61  if not nmea_utc[0:2] or not nmea_utc[2:4] or not nmea_utc[4:6]:
62  return float('NaN')
63  else:
64  hours = int(nmea_utc[0:2])
65  minutes = int(nmea_utc[2:4])
66  seconds = int(nmea_utc[4:6])
67  utc_list[3] = hours
68  utc_list[4] = minutes
69  utc_list[5] = seconds
70  unix_time = calendar.timegm(tuple(utc_list))
71  return unix_time

120 def parse_nmea_sentence(nmea_sentence):
121  # Check for a valid nmea sentence
122  if not re.match('^\$GP.*\*[0-9A-Fa-f]{2}$', nmea_sentence):
123  if not re.match('^\$GN.*\*[0-9A-Fa-f]{2}$', nmea_sentence):
124  logger.debug("Regex didn't match, sentence not valid NMEA? Sentence was: %s"
125  % repr(nmea_sentence))
126  return False
127 
128  fields = [field.strip(',') for field in nmea_sentence.split(',')]
129 
130  # Ignore the $ and talker ID portions (e.g. GP)
131  sentence_type = fields[0][3:]
132 
133  if not sentence_type in parse_maps:
134  logger.debug("Sentence type %s not in parse map, ignoring."
135  % repr(sentence_type))
136  return False
137 
138  parse_map = parse_maps[sentence_type]
139 
140  parsed_sentence = {}
141  for entry in parse_map:
142  parsed_sentence[entry[0]] = entry[1](fields[entry[2]])
143 
144  return {sentence_type: parsed_sentence}
