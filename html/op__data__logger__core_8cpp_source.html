151 
152  for(unsigned int i = 0 ; i <msg->objects.size(); i++)
153  {
154  if(msg->objects.at(i).id > 0)
155  {
156  PlannerHNS::ROSHelpers::ConvertFromAutowareDetectedObjectToOpenPlannerDetectedObject(msg->objects.at(i), obj);
157 
158 
159  obj.behavior_state = GetBehStateFromNumber(msg->objects.at(i).behavior_state);
160  int i_best_trajectory = -1;
161  double max_cost = 0;
162  for(unsigned int k=0; k < obj.predTrajectories.size(); k++)
163  {
164  if(obj.predTrajectories.at(k).size() > 0)
165  {
166  //std::cout << "Fins Max Traj Cost .............. " << obj.predTrajectories.at(k).at(0).collisionCost << std::endl;
167 
168  if(obj.predTrajectories.at(k).at(0).collisionCost == 1)
169  {
170 
171  max_cost = obj.predTrajectories.at(k).at(0).collisionCost;
172  i_best_trajectory = k;
173  }
174  }
175  }
176 
177  if(i_best_trajectory >= 0 && i_best_trajectory < obj.predTrajectories.size())
178  {
179  std::vector<PlannerHNS::WayPoint> pred_traj = obj.predTrajectories.at(i_best_trajectory);
180  obj.predTrajectories.clear();
181  obj.predTrajectories.push_back(pred_traj);
182  }
183  else
184  obj.predTrajectories.clear();
185 
186  m_PredictedObjects.push_back(obj);
187  }
188  }

273 void OpenPlannerDataLogger::CompareAndLog(VehicleDataContainer& ground_truth, PlannerHNS::DetectedObject& predicted)
274 {
275  //difference between actual and prediction
276  //time diff
277  //distance , orientation, velocity diff
278  //RMS diff (make both same density first) , -1 if no prediction
279  //behavior state matching (1 if match , zero if not)
280 
281  double t_diff = fabs(ground_truth.path_time.toSec() - m_pred_time.toSec());
282  double d_diff = hypot(ground_truth.pose.pos.y - predicted.center.pos.y, ground_truth.pose.pos.x - predicted.center.pos.x);
283  double o_diff = UtilityHNS::UtilityH::AngleBetweenTwoAnglesPositive(ground_truth.pose.pos.a, predicted.center.pos.a);
284  double v_diff = fabs(ground_truth.pose.v - predicted.center.v);
285  double rms = -1;
286  int beh_state_diff = -1;
287 
288  if(predicted.predTrajectories.size() > 0)
289  {
290  rms = CalculateRMS(ground_truth.path, predicted.predTrajectories.at(0));
291  }
292 
293  if(ground_truth.beh.state == predicted.behavior_state)
294  beh_state_diff = 1;
295  else
296  beh_state_diff = 0;
297 
298  beh_state_diff = predicted.behavior_state;
299 
300 
301  std::ostringstream dataLine;
302  dataLine << t_diff << "," << d_diff << "," << o_diff << "," << v_diff << "," << rms << "," << beh_state_diff << ",";
303  m_LogData.at(ground_truth.id -1).push_back(dataLine.str());
304 
305  //std::cout << "Predicted Behavior: " << predicted.behavior_state << std::endl;
306 }

308 double OpenPlannerDataLogger::CalculateRMS(std::vector<PlannerHNS::WayPoint>& path1, std::vector<PlannerHNS::WayPoint>& path2)
309 {
310  if(path1.size() == 0 || path2.size() == 0) return -1;
311 
312  //PlannerHNS::PlanningHelpers::FixPathDensity(path1, 0.1);
313  //PlannerHNS::PlanningHelpers::FixPathDensity(path2, 0.1);
314 
315  int min_size = path1.size();
316  if(path2.size() < min_size)
317  min_size = path2.size();
318 
319  double rms_sum = 0;
320  PlannerHNS::RelativeInfo info;
321  for(unsigned int i=0; i < min_size - 1 ; i++)
322  {
323  PlannerHNS::PlanningHelpers::GetRelativeInfo(path1, path2.at(i), info);
324  rms_sum += fabs(info.perp_distance);
325  //rms_sum += hypot(path1.at(i).pos.y - path2.at(i).pos.y, path1.at(i).pos.x - path2.at(i).pos.x);
326  }
327 
328  return rms_sum / (double)min_size;
329 }
