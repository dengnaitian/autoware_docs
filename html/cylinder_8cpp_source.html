28 bool CylinderModel::estimate(const pcl::PointCloud<pcl::PointXYZ>& cluster, autoware_msgs::DetectedObject& output)
29 {
30  // calc centroid point for cylinder height(z)
31  pcl::PointXYZ centroid;
32  centroid.x = 0;
33  centroid.y = 0;
34  centroid.z = 0;
35  for (const auto& pcl_point : cluster)
36  {
37  centroid.x += pcl_point.x;
38  centroid.y += pcl_point.y;
39  centroid.z += pcl_point.z;
40  }
41  centroid.x = centroid.x / (double)cluster.size();
42  centroid.y = centroid.y / (double)cluster.size();
43  centroid.z = centroid.z / (double)cluster.size();
44 
45  // calc min and max z for cylinder length
46  double min_z = 0;
47  double max_z = 0;
48  for (size_t i = 0; i < cluster.size(); ++i)
49  {
50  if (cluster.at(i).z < min_z || i == 0)
51  min_z = cluster.at(i).z;
52  if (max_z < cluster.at(i).z || i == 0)
53  max_z = cluster.at(i).z;
54  }
55 
56  // calc circumscribed circle on x-y plane
57  cv::Mat_<float> cv_points((int)cluster.size(), 2);
58  for (size_t i = 0; i < cluster.size(); ++i)
59  {
60  cv_points(i, 0) = cluster.at(i).x; // x
61  cv_points(i, 1) = cluster.at(i).y; // y
62  }
63  cv::Point2f center;
64  float radius;
65  cv::minEnclosingCircle(cv::Mat(cv_points).reshape(2), center, radius);
66  constexpr double ep = 0.001;
67  radius = std::max(radius, (float)ep);
68  output.pose.position.x = center.x;
69  output.pose.position.y = center.y;
70  output.pose.position.z = centroid.z;
71  output.dimensions.x = (double)radius * 2.0;
72  output.dimensions.y = (double)radius * 2.0;
73  output.dimensions.z = std::max((max_z - min_z), ep);
74  output.pose_reliable = true;
75  return true;
76 }
