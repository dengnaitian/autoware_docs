77 template<typename PointSource, typename PointTarget> void
78 pcl_omp::NormalDistributionsTransform<PointSource, PointTarget>::computeTransformation (PointCloudSource &output, const Eigen::Matrix4f &guess)
79 {
80  nr_iterations_ = 0;
81  converged_ = false;
82 
83  double gauss_c1, gauss_c2, gauss_d3;
84 
85  // Initializes the guassian fitting parameters (eq. 6.8) [Magnusson 2009]
86  gauss_c1 = 10 * (1 - outlier_ratio_);
87  gauss_c2 = outlier_ratio_ / pow (resolution_, 3);
88  gauss_d3 = -log (gauss_c2);
89  gauss_d1_ = -log ( gauss_c1 + gauss_c2 ) - gauss_d3;
90  gauss_d2_ = -2 * log ((-log ( gauss_c1 * exp ( -0.5 ) + gauss_c2 ) - gauss_d3) / gauss_d1_);
91 
92  if (guess != Eigen::Matrix4f::Identity ())
93  {
94  // Initialise final transformation to the guessed one
95  final_transformation_ = guess;
96  // Apply guessed transformation prior to search for neighbours
97  transformPointCloud (output, output, guess);
98  }
99 
100  // Initialize Point Gradient and Hessian
101  point_gradient_.setZero ();
102  point_gradient_.block<3, 3>(0, 0).setIdentity ();
103  point_hessian_.setZero ();
104 
105  Eigen::Transform<float, 3, Eigen::Affine, Eigen::ColMajor> eig_transformation;
106  eig_transformation.matrix () = final_transformation_;
107 
108  // Convert initial guess matrix to 6 element transformation vector
109  Eigen::Matrix<double, 6, 1> p, delta_p, score_gradient;
110  Eigen::Vector3f init_translation = eig_transformation.translation ();
111  Eigen::Vector3f init_rotation = eig_transformation.rotation ().eulerAngles (0, 1, 2);
112  p << init_translation (0), init_translation (1), init_translation (2),
113  init_rotation (0), init_rotation (1), init_rotation (2);
114 
115  Eigen::Matrix<double, 6, 6> hessian;
116 
117  double score = 0;
118  double delta_p_norm;
119 
120  // Calculate derivates of initial transform vector, subsequent derivative calculations are done in the step length determination.
121  score = computeDerivatives (score_gradient, hessian, output, p);
122 
123  while (!converged_)
124  {
125  // Store previous transformation
126  previous_transformation_ = transformation_;
127 
128  // Solve for decent direction using newton method, line 23 in Algorithm 2 [Magnusson 2009]
129  Eigen::JacobiSVD<Eigen::Matrix<double, 6, 6> > sv (hessian, Eigen::ComputeFullU | Eigen::ComputeFullV);
130  // Negative for maximization as opposed to minimization
131  delta_p = sv.solve (-score_gradient);
132 
133  //Calculate step length with guarnteed sufficient decrease [More, Thuente 1994]
134  delta_p_norm = delta_p.norm ();
135 
136  if (delta_p_norm == 0 || delta_p_norm != delta_p_norm)
137  {
138  trans_probability_ = score / static_cast<double> (input_->points.size ());
139  converged_ = delta_p_norm == delta_p_norm;
140  return;
141  }
142 
143  delta_p.normalize ();
144  delta_p_norm = computeStepLengthMT (p, delta_p, delta_p_norm, step_size_, transformation_epsilon_ / 2, score, score_gradient, hessian, output);
145  delta_p *= delta_p_norm;
146 
147 
148  transformation_ = (Eigen::Translation<float, 3> (static_cast<float> (delta_p (0)), static_cast<float> (delta_p (1)), static_cast<float> (delta_p (2))) *
149  Eigen::AngleAxis<float> (static_cast<float> (delta_p (3)), Eigen::Vector3f::UnitX ()) *
150  Eigen::AngleAxis<float> (static_cast<float> (delta_p (4)), Eigen::Vector3f::UnitY ()) *
151  Eigen::AngleAxis<float> (static_cast<float> (delta_p (5)), Eigen::Vector3f::UnitZ ())).matrix ();
152 
153 
154  p = p + delta_p;
155 
156  // Update Visualizer (untested)
157  if (update_visualizer_ != 0)
158  update_visualizer_ (output, std::vector<int>(), *target_, std::vector<int>() );
159 
160  if (nr_iterations_ > max_iterations_ ||
161  (nr_iterations_ && (std::fabs (delta_p_norm) < transformation_epsilon_)))
162  {
163  converged_ = true;
164  }
165 
166  nr_iterations_++;
167 
168  }
169 
170  // Store transformation probability. The realtive differences within each scan registration are accurate
171  // but the normalization constants need to be modified for it to be globally accurate
172  trans_probability_ = score / static_cast<double> (input_->points.size ());
173 }

177 template<typename PointSource, typename PointTarget> double
178 pcl_omp::NormalDistributionsTransform<PointSource, PointTarget>::computeDerivatives (Eigen::Matrix<double, 6, 1> &score_gradient,
179  Eigen::Matrix<double, 6, 6> &hessian,
180  PointCloudSource &trans_cloud,
181  Eigen::Matrix<double, 6, 1> &p,
182  bool compute_hessian)
183 {
184  // Original Point and Transformed Point
185  PointSource x_pt, x_trans_pt;
186  // Original Point and Transformed Point (for math)
187  Eigen::Vector3d x, x_trans;
188  // Occupied Voxel
189  TargetGridLeafConstPtr cell;
190  // Inverse Covariance of Occupied Voxel
191  Eigen::Matrix3d c_inv;
192 
193  score_gradient.setZero ();
194  hessian.setZero ();
195  double score = 0;
196 
197  // Precompute Angular Derivatives (eq. 6.19 and 6.21)[Magnusson 2009]
198  computeAngleDerivatives (p);
199 
200 #ifdef _OPENMP
201  int num_threads = omp_get_max_threads() * 2;
202  Eigen::Matrix<double, 6, 1> score_gradient_i[num_threads];
203  Eigen::Matrix<double, 6, 6> hessian_i[num_threads];
204  for (int i = 0; i < num_threads; ++i) {
205  score_gradient_i[i].setZero ();
206  hessian_i[i].setZero ();
207  }
208 
209  omp_set_nested(1);
210  omp_set_dynamic(1);
211 #pragma omp parallel for num_threads(num_threads) reduction(+:score) private(x_pt, x_trans_pt, x, x_trans, cell, c_inv)
212 #endif
213 
214  // Update gradient and hessian for each point, line 17 in Algorithm 2 [Magnusson 2009]
215  for (size_t idx = 0; idx < input_->points.size (); idx++)
216  {
217  x_trans_pt = trans_cloud.points[idx];
218 
219  // Find nieghbors (Radius search has been experimentally faster than direct neighbor checking.
220  std::vector<TargetGridLeafConstPtr> neighborhood;
221  std::vector<float> distances;
222  target_cells_.radiusSearch (x_trans_pt, resolution_, neighborhood, distances);
223 
224  for (typename std::vector<TargetGridLeafConstPtr>::iterator neighborhood_it = neighborhood.begin (); neighborhood_it != neighborhood.end (); neighborhood_it++)
225  {
226  cell = *neighborhood_it;
227  x_pt = input_->points[idx];
228  x = Eigen::Vector3d (x_pt.x, x_pt.y, x_pt.z);
229 
230  x_trans = Eigen::Vector3d (x_trans_pt.x, x_trans_pt.y, x_trans_pt.z);
231 
232  // Denorm point, x_k' in Equations 6.12 and 6.13 [Magnusson 2009]
233  x_trans -= cell->getMean ();
234  // Uses precomputed covariance for speed.
235  c_inv = cell->getInverseCov ();
236 
237  // Compute derivative of transform function w.r.t. transform vector, J_E and H_E in Equations 6.18 and 6.20 [Magnusson 2009]
238  computePointDerivatives (x);
239  // Update score, gradient and hessian, lines 19-21 in Algorithm 2, according to Equations 6.10, 6.12 and 6.13, respectively [Magnusson 2009]
240  //score += updateDerivatives (score_gradient, hessian, x_trans, c_inv, compute_hessian);
241  {
242  Eigen::Vector3d cov_dxd_pi;
243  // e^(-d_2/2 * (x_k - mu_k)^T Sigma_k^-1 (x_k - mu_k)) Equation 6.9 [Magnusson 2009]
244  double e_x_cov_x = exp (-gauss_d2_ * x_trans.dot (c_inv * x_trans) / 2);
245  // Calculate probability of transtormed points existance, Equation 6.9 [Magnusson 2009]
246  double score_inc = -gauss_d1_ * e_x_cov_x;
247 
248  e_x_cov_x = gauss_d2_ * e_x_cov_x;
249 
250  // Error checking for invalid values.
251  if (e_x_cov_x > 1 || e_x_cov_x < 0 || e_x_cov_x != e_x_cov_x) {
252  score_inc = 0;
253  } else {
254  // Reusable portion of Equation 6.12 and 6.13 [Magnusson 2009]
255  e_x_cov_x *= gauss_d1_;
256 
257 
258  for (int i = 0; i < 6; i++)
259  {
260  // Sigma_k^-1 d(T(x,p))/dpi, Reusable portion of Equation 6.12 and 6.13 [Magnusson 2009o
261  cov_dxd_pi = c_inv * point_gradient_.col (i);
262 
263  // Update gradient, Equation 6.12 [Magnusson 2009]
264 #ifdef _OPENMP
265  score_gradient_i[omp_get_thread_num()] (i) += x_trans.dot (cov_dxd_pi) * e_x_cov_x;
266 #else
267  score_gradient (i) += x_trans.dot (cov_dxd_pi) * e_x_cov_x;
268 #endif
269  if (compute_hessian)
270  {
271  for (int j = 0; j < hessian.cols (); j++)
272  {
273  // Update hessian, Equation 6.13 [Magnusson 2009]
274 #ifdef _OPENMP
275  hessian_i[omp_get_thread_num()] (i, j) += e_x_cov_x * (-gauss_d2_ * x_trans.dot (cov_dxd_pi) * x_trans.dot (c_inv * point_gradient_.col (j)) +
276  x_trans.dot (c_inv * point_hessian_.block<3, 1>(3 * i, j)) +
277  point_gradient_.col (j).dot (cov_dxd_pi) );
278 #else
279  hessian (i, j) += e_x_cov_x * (-gauss_d2_ * x_trans.dot (cov_dxd_pi) * x_trans.dot (c_inv * point_gradient_.col (j)) +
280  x_trans.dot (c_inv * point_hessian_.block<3, 1>(3 * i, j)) +
281  point_gradient_.col (j).dot (cov_dxd_pi) );
282 #endif
283  }
284  }
285  }
286  }
287  score += score_inc;
288  }
289  }
290  }
291 #ifdef _OPENMP
292  for (int i = 0; i < num_threads; ++i) {
293  score_gradient += score_gradient_i[i];
294  hessian += hessian_i[i];
295  }
296 #endif
297 
298  return (score);
299 }

302 template<typename PointSource, typename PointTarget> void
303 pcl_omp::NormalDistributionsTransform<PointSource, PointTarget>::computeAngleDerivatives (Eigen::Matrix<double, 6, 1> &p, bool compute_hessian)
304 {
305  // Simplified math for near 0 angles
306  double cx, cy, cz, sx, sy, sz;
307  if (fabs (p (3)) < 10e-5)
308  {
309  //p(3) = 0;
310  cx = 1.0;
311  sx = 0.0;
312  }
313  else
314  {
315  cx = cos (p (3));
316  sx = sin (p (3));
317  }
318  if (fabs (p (4)) < 10e-5)
319  {
320  //p(4) = 0;
321  cy = 1.0;
322  sy = 0.0;
323  }
324  else
325  {
326  cy = cos (p (4));
327  sy = sin (p (4));
328  }
329 
330  if (fabs (p (5)) < 10e-5)
331  {
332  //p(5) = 0;
333  cz = 1.0;
334  sz = 0.0;
335  }
336  else
337  {
338  cz = cos (p (5));
339  sz = sin (p (5));
340  }
341 
342  // Precomputed angular gradiant components. Letters correspond to Equation 6.19 [Magnusson 2009]
343  j_ang_a_ << (-sx * sz + cx * sy * cz), (-sx * cz - cx * sy * sz), (-cx * cy);
344  j_ang_b_ << (cx * sz + sx * sy * cz), (cx * cz - sx * sy * sz), (-sx * cy);
345  j_ang_c_ << (-sy * cz), sy * sz, cy;
346  j_ang_d_ << sx * cy * cz, (-sx * cy * sz), sx * sy;
347  j_ang_e_ << (-cx * cy * cz), cx * cy * sz, (-cx * sy);
348  j_ang_f_ << (-cy * sz), (-cy * cz), 0;
349  j_ang_g_ << (cx * cz - sx * sy * sz), (-cx * sz - sx * sy * cz), 0;
350  j_ang_h_ << (sx * cz + cx * sy * sz), (cx * sy * cz - sx * sz), 0;
351 
352  if (compute_hessian)
353  {
354  // Precomputed angular hessian components. Letters correspond to Equation 6.21 and numbers correspond to row index [Magnusson 2009]
355  h_ang_a2_ << (-cx * sz - sx * sy * cz), (-cx * cz + sx * sy * sz), sx * cy;
356  h_ang_a3_ << (-sx * sz + cx * sy * cz), (-cx * sy * sz - sx * cz), (-cx * cy);
357 
358  h_ang_b2_ << (cx * cy * cz), (-cx * cy * sz), (cx * sy);
359  h_ang_b3_ << (sx * cy * cz), (-sx * cy * sz), (sx * sy);
360 
361  h_ang_c2_ << (-sx * cz - cx * sy * sz), (sx * sz - cx * sy * cz), 0;
362  h_ang_c3_ << (cx * cz - sx * sy * sz), (-sx * sy * cz - cx * sz), 0;
363 
364  h_ang_d1_ << (-cy * cz), (cy * sz), (sy);
365  h_ang_d2_ << (-sx * sy * cz), (sx * sy * sz), (sx * cy);
366  h_ang_d3_ << (cx * sy * cz), (-cx * sy * sz), (-cx * cy);
367 
368  h_ang_e1_ << (sy * sz), (sy * cz), 0;
369  h_ang_e2_ << (-sx * cy * sz), (-sx * cy * cz), 0;
370  h_ang_e3_ << (cx * cy * sz), (cx * cy * cz), 0;
371 
372  h_ang_f1_ << (-cy * cz), (cy * sz), 0;
373  h_ang_f2_ << (-cx * sz - sx * sy * cz), (-cx * cz + sx * sy * sz), 0;
374  h_ang_f3_ << (-sx * sz + cx * sy * cz), (-cx * sy * sz - sx * cz), 0;
375  }
376 }

379 template<typename PointSource, typename PointTarget> void
380 pcl_omp::NormalDistributionsTransform<PointSource, PointTarget>::computePointDerivatives (Eigen::Vector3d &x, bool compute_hessian)
381 {
382  // Calculate first derivative of Transformation Equation 6.17 w.r.t. transform vector p.
383  // Derivative w.r.t. ith element of transform vector corresponds to column i, Equation 6.18 and 6.19 [Magnusson 2009]
384  point_gradient_ (1, 3) = x.dot (j_ang_a_);
385  point_gradient_ (2, 3) = x.dot (j_ang_b_);
386  point_gradient_ (0, 4) = x.dot (j_ang_c_);
387  point_gradient_ (1, 4) = x.dot (j_ang_d_);
388  point_gradient_ (2, 4) = x.dot (j_ang_e_);
389  point_gradient_ (0, 5) = x.dot (j_ang_f_);
390  point_gradient_ (1, 5) = x.dot (j_ang_g_);
391  point_gradient_ (2, 5) = x.dot (j_ang_h_);
392 
393  if (compute_hessian)
394  {
395  // Vectors from Equation 6.21 [Magnusson 2009]
396  Eigen::Vector3d a, b, c, d, e, f;
397 
398  a << 0, x.dot (h_ang_a2_), x.dot (h_ang_a3_);
399  b << 0, x.dot (h_ang_b2_), x.dot (h_ang_b3_);
400  c << 0, x.dot (h_ang_c2_), x.dot (h_ang_c3_);
401  d << x.dot (h_ang_d1_), x.dot (h_ang_d2_), x.dot (h_ang_d3_);
402  e << x.dot (h_ang_e1_), x.dot (h_ang_e2_), x.dot (h_ang_e3_);
403  f << x.dot (h_ang_f1_), x.dot (h_ang_f2_), x.dot (h_ang_f3_);
404 
405  // Calculate second derivative of Transformation Equation 6.17 w.r.t. transform vector p.
406  // Derivative w.r.t. ith and jth elements of transform vector corresponds to the 3x1 block matrix starting at (3i,j), Equation 6.20 and 6.21 [Magnusson 2009]
407  point_hessian_.block<3, 1>(9, 3) = a;
408  point_hessian_.block<3, 1>(12, 3) = b;
409  point_hessian_.block<3, 1>(15, 3) = c;
410  point_hessian_.block<3, 1>(9, 4) = b;
411  point_hessian_.block<3, 1>(12, 4) = d;
412  point_hessian_.block<3, 1>(15, 4) = e;
413  point_hessian_.block<3, 1>(9, 5) = c;
414  point_hessian_.block<3, 1>(12, 5) = e;
415  point_hessian_.block<3, 1>(15, 5) = f;
416  }
417 }

420 template<typename PointSource, typename PointTarget> double
421 pcl_omp::NormalDistributionsTransform<PointSource, PointTarget>::updateDerivatives (Eigen::Matrix<double, 6, 1> &score_gradient,
422  Eigen::Matrix<double, 6, 6> &hessian,
423  Eigen::Vector3d &x_trans, Eigen::Matrix3d &c_inv,
424  bool compute_hessian)
425 {
426  Eigen::Vector3d cov_dxd_pi;
427  // e^(-d_2/2 * (x_k - mu_k)^T Sigma_k^-1 (x_k - mu_k)) Equation 6.9 [Magnusson 2009]
428  double e_x_cov_x = exp (-gauss_d2_ * x_trans.dot (c_inv * x_trans) / 2);
429  // Calculate probability of transtormed points existance, Equation 6.9 [Magnusson 2009]
430  double score_inc = -gauss_d1_ * e_x_cov_x;
431 
432  e_x_cov_x = gauss_d2_ * e_x_cov_x;
433 
434  // Error checking for invalid values.
435  if (e_x_cov_x > 1 || e_x_cov_x < 0 || e_x_cov_x != e_x_cov_x)
436  return (0);
437 
438  // Reusable portion of Equation 6.12 and 6.13 [Magnusson 2009]
439  e_x_cov_x *= gauss_d1_;
440 
441 
442  for (int i = 0; i < 6; i++)
443  {
444  // Sigma_k^-1 d(T(x,p))/dpi, Reusable portion of Equation 6.12 and 6.13 [Magnusson 2009]
445  cov_dxd_pi = c_inv * point_gradient_.col (i);
446 
447  // Update gradient, Equation 6.12 [Magnusson 2009]
448  score_gradient (i) += x_trans.dot (cov_dxd_pi) * e_x_cov_x;
449 
450  if (compute_hessian)
451  {
452  for (int j = 0; j < hessian.cols (); j++)
453  {
454  // Update hessian, Equation 6.13 [Magnusson 2009]
455  hessian (i, j) += e_x_cov_x * (-gauss_d2_ * x_trans.dot (cov_dxd_pi) * x_trans.dot (c_inv * point_gradient_.col (j)) +
456  x_trans.dot (c_inv * point_hessian_.block<3, 1>(3 * i, j)) +
457  point_gradient_.col (j).dot (cov_dxd_pi) );
458  }
459  }
460  }
461 
462  return (score_inc);
463 }

466 template<typename PointSource, typename PointTarget> void
467 pcl_omp::NormalDistributionsTransform<PointSource, PointTarget>::computeHessian (Eigen::Matrix<double, 6, 6> &hessian,
468  PointCloudSource &trans_cloud, Eigen::Matrix<double, 6, 1> &)
469 {
470  // Original Point and Transformed Point
471  PointSource x_pt, x_trans_pt;
472  // Original Point and Transformed Point (for math)
473  Eigen::Vector3d x, x_trans;
474  // Occupied Voxel
475  TargetGridLeafConstPtr cell;
476  // Inverse Covariance of Occupied Voxel
477  Eigen::Matrix3d c_inv;
478 
479  hessian.setZero ();
480 
481  // Precompute Angular Derivatives unessisary because only used after regular derivative calculation
482 
483  // Update hessian for each point, line 17 in Algorithm 2 [Magnusson 2009]
484  for (size_t idx = 0; idx < input_->points.size (); idx++)
485  {
486  x_trans_pt = trans_cloud.points[idx];
487 
488  // Find nieghbors (Radius search has been experimentally faster than direct neighbor checking.
489  std::vector<TargetGridLeafConstPtr> neighborhood;
490  std::vector<float> distances;
491  target_cells_.radiusSearch (x_trans_pt, resolution_, neighborhood, distances);
492 
493  for (typename std::vector<TargetGridLeafConstPtr>::iterator neighborhood_it = neighborhood.begin (); neighborhood_it != neighborhood.end (); neighborhood_it++)
494  {
495  cell = *neighborhood_it;
496 
497  {
498  x_pt = input_->points[idx];
499  x = Eigen::Vector3d (x_pt.x, x_pt.y, x_pt.z);
500 
501  x_trans = Eigen::Vector3d (x_trans_pt.x, x_trans_pt.y, x_trans_pt.z);

518 template<typename PointSource, typename PointTarget> void
519 pcl_omp::NormalDistributionsTransform<PointSource, PointTarget>::updateHessian (Eigen::Matrix<double, 6, 6> &hessian, Eigen::Vector3d &x_trans, Eigen::Matrix3d &c_inv)
520 {
521  Eigen::Vector3d cov_dxd_pi;
522  // e^(-d_2/2 * (x_k - mu_k)^T Sigma_k^-1 (x_k - mu_k)) Equation 6.9 [Magnusson 2009]
523  double e_x_cov_x = gauss_d2_ * exp (-gauss_d2_ * x_trans.dot (c_inv * x_trans) / 2);
524 
525  // Error checking for invalid values.
526  if (e_x_cov_x > 1 || e_x_cov_x < 0 || e_x_cov_x != e_x_cov_x)
527  return;
528 
529  // Reusable portion of Equation 6.12 and 6.13 [Magnusson 2009]
530  e_x_cov_x *= gauss_d1_;
531 
532  for (int i = 0; i < 6; i++)
533  {
534  // Sigma_k^-1 d(T(x,p))/dpi, Reusable portion of Equation 6.12 and 6.13 [Magnusson 2009]
535  cov_dxd_pi = c_inv * point_gradient_.col (i);
536 
537  for (int j = 0; j < hessian.cols (); j++)
538  {
539  // Update hessian, Equation 6.13 [Magnusson 2009]
540  hessian (i, j) += e_x_cov_x * (-gauss_d2_ * x_trans.dot (cov_dxd_pi) * x_trans.dot (c_inv * point_gradient_.col (j)) +
541  x_trans.dot (c_inv * point_hessian_.block<3, 1>(3 * i, j)) +
542  point_gradient_.col (j).dot (cov_dxd_pi) );
543  }
544  }

590 template<typename PointSource, typename PointTarget> double
591 pcl_omp::NormalDistributionsTransform<PointSource, PointTarget>::trialValueSelectionMT (double a_l, double f_l, double g_l,
592  double a_u, double f_u, double g_u,
593  double a_t, double f_t, double g_t)
594 {
595  // Case 1 in Trial Value Selection [More, Thuente 1994]
596  if (f_t > f_l)
597  {
598  // Calculate the minimizer of the cubic that interpolates f_l, f_t, g_l and g_t
599  // Equation 2.4.52 [Sun, Yuan 2006]
600  double z = 3 * (f_t - f_l) / (a_t - a_l) - g_t - g_l;
601  double w = std::sqrt (z * z - g_t * g_l);
602  // Equation 2.4.56 [Sun, Yuan 2006]
603  double a_c = a_l + (a_t - a_l) * (w - g_l - z) / (g_t - g_l + 2 * w);
604 
605  // Calculate the minimizer of the quadratic that interpolates f_l, f_t and g_l
606  // Equation 2.4.2 [Sun, Yuan 2006]
607  double a_q = a_l - 0.5 * (a_l - a_t) * g_l / (g_l - (f_l - f_t) / (a_l - a_t));
608 
609  if (std::fabs (a_c - a_l) < std::fabs (a_q - a_l))
610  return (a_c);
611  else
612  return (0.5 * (a_q + a_c));
613  }
614  // Case 2 in Trial Value Selection [More, Thuente 1994]
615  else
616  if (g_t * g_l < 0)
617  {
618  // Calculate the minimizer of the cubic that interpolates f_l, f_t, g_l and g_t
619  // Equation 2.4.52 [Sun, Yuan 2006]
620  double z = 3 * (f_t - f_l) / (a_t - a_l) - g_t - g_l;
621  double w = std::sqrt (z * z - g_t * g_l);
622  // Equation 2.4.56 [Sun, Yuan 2006]
623  double a_c = a_l + (a_t - a_l) * (w - g_l - z) / (g_t - g_l + 2 * w);
624 
625  // Calculate the minimizer of the quadratic that interpolates f_l, g_l and g_t
626  // Equation 2.4.5 [Sun, Yuan 2006]
627  double a_s = a_l - (a_l - a_t) / (g_l - g_t) * g_l;
628 
629  if (std::fabs (a_c - a_t) >= std::fabs (a_s - a_t))
630  return (a_c);
631  else
632  return (a_s);
633  }
634  // Case 3 in Trial Value Selection [More, Thuente 1994]
635  else
636  if (std::fabs (g_t) <= std::fabs (g_l))
637  {
638  // Calculate the minimizer of the cubic that interpolates f_l, f_t, g_l and g_t
639  // Equation 2.4.52 [Sun, Yuan 2006]
640  double z = 3 * (f_t - f_l) / (a_t - a_l) - g_t - g_l;
641  double w = std::sqrt (z * z - g_t * g_l);
642  double a_c = a_l + (a_t - a_l) * (w - g_l - z) / (g_t - g_l + 2 * w);
643 
644  // Calculate the minimizer of the quadratic that interpolates g_l and g_t
645  // Equation 2.4.5 [Sun, Yuan 2006]
646  double a_s = a_l - (a_l - a_t) / (g_l - g_t) * g_l;
647 
648  double a_t_next;
649 
650  if (std::fabs (a_c - a_t) < std::fabs (a_s - a_t))
651  a_t_next = a_c;
652  else
653  a_t_next = a_s;
654 
655  if (a_t > a_l)
656  return (std::min (a_t + 0.66 * (a_u - a_t), a_t_next));
657  else
658  return (std::max (a_t + 0.66 * (a_u - a_t), a_t_next));
659  }
660  // Case 4 in Trial Value Selection [More, Thuente 1994]
661  else
662  {
663  // Calculate the minimizer of the cubic that interpolates f_u, f_t, g_u and g_t
664  // Equation 2.4.52 [Sun, Yuan 2006]
665  double z = 3 * (f_t - f_u) / (a_t - a_u) - g_t - g_u;
666  double w = std::sqrt (z * z - g_t * g_u);
667  // Equation 2.4.56 [Sun, Yuan 2006]
668  return (a_u + (a_t - a_u) * (w - g_u - z) / (g_t - g_u + 2 * w));
669  }
670 }

673 template<typename PointSource, typename PointTarget> double
674 pcl_omp::NormalDistributionsTransform<PointSource, PointTarget>::computeStepLengthMT (const Eigen::Matrix<double, 6, 1> &x, Eigen::Matrix<double, 6, 1> &step_dir, double step_init, double step_max,
675  double step_min, double &score, Eigen::Matrix<double, 6, 1> &score_gradient, Eigen::Matrix<double, 6, 6> &hessian,
676  PointCloudSource &trans_cloud)
677 {
678  // Set the value of phi(0), Equation 1.3 [More, Thuente 1994]
679  double phi_0 = -score;
680  // Set the value of phi'(0), Equation 1.3 [More, Thuente 1994]
681  double d_phi_0 = -(score_gradient.dot (step_dir));
682 
683  Eigen::Matrix<double, 6, 1> x_t;
684 
685  if (d_phi_0 >= 0)
686  {
687  // Not a decent direction
688  if (d_phi_0 == 0)
689  return 0;
690  else
691  {
692  // Reverse step direction and calculate optimal step.
693  d_phi_0 *= -1;
694  step_dir *= -1;
695 
696  }
697  }
698 
699  // The Search Algorithm for T(mu) [More, Thuente 1994]
700 
701  int max_step_iterations = 10;
702  int step_iterations = 0;
703 
704  // Sufficient decreace constant, Equation 1.1 [More, Thuete 1994]
705  double mu = 1.e-4;
706  // Curvature condition constant, Equation 1.2 [More, Thuete 1994]
707  double nu = 0.9;
708 
709  // Initial endpoints of Interval I,
710  double a_l = 0, a_u = 0;
711 
712  // Auxiliary function psi is used until I is determined ot be a closed interval, Equation 2.1 [More, Thuente 1994]
713  double f_l = auxilaryFunction_PsiMT (a_l, phi_0, phi_0, d_phi_0, mu);
714  double g_l = auxilaryFunction_dPsiMT (d_phi_0, d_phi_0, mu);
715 
716  double f_u = auxilaryFunction_PsiMT (a_u, phi_0, phi_0, d_phi_0, mu);
717  double g_u = auxilaryFunction_dPsiMT (d_phi_0, d_phi_0, mu);
718 
719  // Check used to allow More-Thuente step length calculation to be skipped by making step_min == step_max
720  bool interval_converged = (step_max - step_min) > 0, open_interval = true;
721 
722  double a_t = step_init;
723  a_t = std::min (a_t, step_max);
724  a_t = std::max (a_t, step_min);
725 
726  x_t = x + step_dir * a_t;
727 
728  final_transformation_ = (Eigen::Translation<float, 3>(static_cast<float> (x_t (0)), static_cast<float> (x_t (1)), static_cast<float> (x_t (2))) *
729  Eigen::AngleAxis<float> (static_cast<float> (x_t (3)), Eigen::Vector3f::UnitX ()) *
730  Eigen::AngleAxis<float> (static_cast<float> (x_t (4)), Eigen::Vector3f::UnitY ()) *
731  Eigen::AngleAxis<float> (static_cast<float> (x_t (5)), Eigen::Vector3f::UnitZ ())).matrix ();
732 
733  // New transformed point cloud
734  transformPointCloud (*input_, trans_cloud, final_transformation_);
735 
736  // Updates score, gradient and hessian. Hessian calculation is unessisary but testing showed that most step calculations use the
737  // initial step suggestion and recalculation the reusable portions of the hessian would intail more computation time.
738  score = computeDerivatives (score_gradient, hessian, trans_cloud, x_t, true);
739 
740  // Calculate phi(alpha_t)
741  double phi_t = -score;
742  // Calculate phi'(alpha_t)
743  double d_phi_t = -(score_gradient.dot (step_dir));
744 
745  // Calculate psi(alpha_t)
746  double psi_t = auxilaryFunction_PsiMT (a_t, phi_t, phi_0, d_phi_0, mu);
747  // Calculate psi'(alpha_t)
748  double d_psi_t = auxilaryFunction_dPsiMT (d_phi_t, d_phi_0, mu);
749 
750  // Iterate until max number of iterations, interval convergance or a value satisfies the sufficient decrease, Equation 1.1, and curvature condition, Equation 1.2 [More, Thuente 1994]
751  while (!interval_converged && step_iterations < max_step_iterations && !(psi_t <= 0 /*Sufficient Decrease*/ && d_phi_t <= -nu * d_phi_0 /*Curvature Condition*/))
752  {
753  // Use auxilary function if interval I is not closed
754  if (open_interval)
755  {
756  a_t = trialValueSelectionMT (a_l, f_l, g_l,
757  a_u, f_u, g_u,
758  a_t, psi_t, d_psi_t);
759  }
760  else
761  {
762  a_t = trialValueSelectionMT (a_l, f_l, g_l,
763  a_u, f_u, g_u,
764  a_t, phi_t, d_phi_t);
765  }
766 
767  a_t = std::min (a_t, step_max);
768  a_t = std::max (a_t, step_min);
769 
770  x_t = x + step_dir * a_t;
771 
772  final_transformation_ = (Eigen::Translation<float, 3> (static_cast<float> (x_t (0)), static_cast<float> (x_t (1)), static_cast<float> (x_t (2))) *
773  Eigen::AngleAxis<float> (static_cast<float> (x_t (3)), Eigen::Vector3f::UnitX ()) *
774  Eigen::AngleAxis<float> (static_cast<float> (x_t (4)), Eigen::Vector3f::UnitY ()) *
775  Eigen::AngleAxis<float> (static_cast<float> (x_t (5)), Eigen::Vector3f::UnitZ ())).matrix ();
776 
777  // New transformed point cloud
778  // Done on final cloud to prevent wasted computation
779  transformPointCloud (*input_, trans_cloud, final_transformation_);
780 
781  // Updates score, gradient. Values stored to prevent wasted computation.
782  score = computeDerivatives (score_gradient, hessian, trans_cloud, x_t, false);
783 
784  // Calculate phi(alpha_t+)
785  phi_t = -score;
786  // Calculate phi'(alpha_t+)
787  d_phi_t = -(score_gradient.dot (step_dir));
788 
789  // Calculate psi(alpha_t+)
790  psi_t = auxilaryFunction_PsiMT (a_t, phi_t, phi_0, d_phi_0, mu);
791  // Calculate psi'(alpha_t+)
792  d_psi_t = auxilaryFunction_dPsiMT (d_phi_t, d_phi_0, mu);
793 
794  // Check if I is now a closed interval
795  if (open_interval && (psi_t <= 0 && d_psi_t >= 0))
796  {
797  open_interval = false;
798 
799  // Converts f_l and g_l from psi to phi
800  f_l = f_l + phi_0 - mu * d_phi_0 * a_l;
801  g_l = g_l + mu * d_phi_0;
802 
803  // Converts f_u and g_u from psi to phi
804  f_u = f_u + phi_0 - mu * d_phi_0 * a_u;
805  g_u = g_u + mu * d_phi_0;
806  }
807 
808  if (open_interval)
809  {
810  // Update interval end points using Updating Algorithm [More, Thuente 1994]
811  interval_converged = updateIntervalMT (a_l, f_l, g_l,
812  a_u, f_u, g_u,
813  a_t, psi_t, d_psi_t);
814  }
815  else
816  {
817  // Update interval end points using Modified Updating Algorithm [More, Thuente 1994]
818  interval_converged = updateIntervalMT (a_l, f_l, g_l,
819  a_u, f_u, g_u,
820  a_t, phi_t, d_phi_t);
821  }
822 
823  step_iterations++;
824  }
825 
826  // If inner loop was run then hessian needs to be calculated.
827  // Hessian is unnessisary for step length determination but gradients are required
828  // so derivative and transform data is stored for the next iteration.
829  if (step_iterations)
830  computeHessian (hessian, trans_cloud, x_t);
831 
832  return (a_t);
833 }
