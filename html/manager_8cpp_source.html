363  a->setData(*data);
364  }
365 
366  // Datensatz wird nicht mehr benoetigt, loeschen
367  delete data;
368  data = NULL;
369  }
370 
371  //
372  // Naechsten Aufruf vorbereiten
373  //
374  endThread = false;
375  if (m_sourceBuffer.getBufferSize() > 0)

377  // Es sind bereits jetzt neue Daten da, also nur kurz unterbrechen
