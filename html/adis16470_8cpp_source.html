229 int Adis16470::update_burst(void)
230 {
231  unsigned char buff[64] = {0};
232  // 0x6800: Burst read function
233  buff[0] = 0x61;
234  buff[1] = 0x68;
235  buff[2] = 0x00;
236  int size = write(fd_, buff, 24);
237  if (size != 24)
238  {
239  perror("update_burst");
240  return -1;
241  }
242  if (tcdrain(fd_) < 0)
243  {
244  perror("update_burst");
245  return -1;
246  }
247  size = read(fd_, buff, 30);
248  if (size != 30)
249  {
250  perror("update_burst");
251  return -1;
252  }
253  int16_t diag_stat = big_endian_to_short(&buff[3]);
254  if (diag_stat != 0)
255  {
256  fprintf(stderr, "diag_stat error: %04x\n", (uint16_t)diag_stat);
257  return -1;
258  }
259  // X_GYRO_OUT
260  gyro[0] = big_endian_to_short(&buff[5]) * M_PI / 180 / 10.0;
261  // Y_GYRO_OUT
262  gyro[1] = big_endian_to_short(&buff[7]) * M_PI / 180 / 10.0;
263  // Z_GYRO_OUT
264  gyro[2] = big_endian_to_short(&buff[9]) * M_PI / 180 / 10.0;
265  // X_ACCL_OUT
266  accl[0] = big_endian_to_short(&buff[11]) * M_PI / 180 / 10.0;
267  // Y_ACCL_OUT
268  accl[1] = big_endian_to_short(&buff[13]) * M_PI / 180 / 10.0;
269  // Z_ACCL_OUT
270  accl[2] = big_endian_to_short(&buff[15]) * M_PI / 180 / 10.0;
271 
272  return 0;
273 }

292  read_register(0x18, accl_out[1]);
293  read_register(0x1a, accl_low[2]);
294  read_register(0x00, accl_out[2]);
295 
296  // 32bit convert
297  for (int i=0; i < 3; i++)
298  {
299  gyro[i] = ((int32_t(gyro_out[i]) << 16) + int32_t(gyro_low[i])) * M_PI / 180.0 / 655360.0;
300  accl[i] = ((int32_t(accl_out[i]) << 16) + int32_t(accl_low[i])) * 9.8 / 52428800.0;
301  }
302  return 0;
