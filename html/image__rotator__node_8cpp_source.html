46  void ImageCallback(const sensor_msgs::Image& in_image_sensor)
47  {
48  //Receive Image, convert it to OpenCV Mat
49  cv_bridge::CvImagePtr cv_image = cv_bridge::toCvCopy(in_image_sensor, "bgr8");
50  cv::Mat image = cv_image->image;
51 
52  if(mirror_vertical_)
53  {cv::flip(image, image, 0);}
54  if(mirror_horizontal_)
55  {cv::flip(image, image, 1);}
56 
57  switch(rotation_angle_)
58  {
59  case 0:
60  break;
61  case 90:
62  transpose(image, image);
63  flip(image, image, 1);
64  break;
65  case 180:
66  flip(image, image, -1);
67  break;
68  case 270:
69  transpose(image, image);
70  flip(image, image, 0);
71  break;
72  default:
73  cv::Mat rotation_matrix = cv::getRotationMatrix2D( cv::Point(image.cols/2, image.rows/2), (rotation_angle_ ), 1 );
74  cv::warpAffine( image, image, rotation_matrix, image.size() );
75  }
76 
77  cv_bridge::CvImage out_msg;
78  out_msg.header = in_image_sensor.header; // Same timestamp and tf frame as input image
79  out_msg.encoding = sensor_msgs::image_encodings::BGR8;
80  out_msg.image = image; // Your cv::Mat
81 
82  publisher_image_rotated_.publish(out_msg.toImageMsg());
83 
84  }
