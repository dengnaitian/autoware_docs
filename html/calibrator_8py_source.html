107 def _get_outside_corners(corners, board):
108  """
109  Return the four corners of the board as a whole, as (up_left, up_right, down_right, down_left).
110  """
111  xdim = board.n_cols
112  ydim = board.n_rows
113 
114  if corners.shape[1] * corners.shape[0] != xdim * ydim:
115  raise Exception("Invalid number of corners! %d corners. X: %d, Y: %d" % (corners.shape[1] * corners.shape[0],

117

119  up_right = corners[xdim - 1, 0]

121  down_left = corners[-xdim, 0]

136  def angle(a, b, c):
137  """
138  Return angle between lines ab, bc
139  """
140  ab = a - b
141  cb = c - b
142  return math.acos(numpy.dot(ab, cb) / (numpy.linalg.norm(ab) * numpy.linalg.norm(cb)))
143

148 def _get_area(corners, board):
149  """
150  Get 2d image area of the detected checkerboard.
151  The projected checkerboard is assumed to be a convex quadrilateral, and the area computed as
152  |p X q|/2; see http://mathworld.wolfram.com/Quadrilateral.html.
153  """
154  (up_left, up_right, down_right, down_left) = _get_outside_corners(corners, board)
155  a = up_right - up_left
156  b = down_right - up_right
157  c = down_left - down_right
158  p = b + c
159  q = a + b
160  return abs(p[0] * q[1] - p[1] * q[0]) / 2.

163 def _get_corners(img, board, detection, refine=False, checkerboard_flags=0):
164  """
165  Get corners for a particular chessboard for an image
166  """
167  h = img.shape[0]
168  w = img.shape[1]
169  if len(img.shape) == 3 and img.shape[2] == 3:
170  mono = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
171  else:
172  mono = img
173 
174  if detection == "cv2":
175  (ok, corners) = cv2.findChessboardCorners(mono, (board.n_cols, board.n_rows),
176  flags=cv2.CALIB_CB_ADAPTIVE_THRESH |
177  cv2.CALIB_CB_NORMALIZE_IMAGE | checkerboard_flags)
178  elif detection == "matlab":
179  cv2.imwrite("temp_image.jpg", img)
180  out = strio.StringIO()
181  err = strio.StringIO()
182  (mlcorners, boardsize, ok) = matlabeng.detectCheckerboardPoints("temp_image.jpg", nargout=3,stdout=out,stderr=err)
183  ml_corners = numpy.array(mlcorners)
184  if ml_corners.size != (2 * board.n_rows * board.n_cols):
185  ok = 0
186  return (ok, ml_corners)
187  ml_corners = ml_corners.astype(numpy.float32).reshape((board.n_rows * board.n_cols, 1, 2))
188  corners = ml_corners.copy()
189  cv_idx = 0
190  for i in range(0, board.n_rows):
191  for j in range(0, board.n_cols):
192  corners[cv_idx, :, :] = ml_corners[board.n_rows * j + i, :, :]
193  cv_idx = cv_idx + 1
194 
195  if not ok:
196  return (ok, corners)
197 
198  # If any corners are within BORDER pixels of the screen edge, reject the detection by setting ok to false
199  # NOTE: This may cause problems with very low-resolution cameras, where 8 pixels is a non-negligible fraction
200  # of the image size. See http://answers.ros.org/question/3155/how-can-i-calibrate-low-resolution-cameras
201 
202  BORDER = 8
203  if not all([(BORDER < corners[i, 0, 0] < (w - BORDER)) and (BORDER < corners[i, 0, 1] < (h - BORDER)) for i in
204  range(corners.shape[0])]):
205  ok = False
206 
207  # Ensure that all corner-arrays are going from top to bottom.
208  if board.n_rows != board.n_cols:
209  if corners[0, 0, 1] > corners[-1, 0, 1]:
210  corners = numpy.copy(numpy.flipud(corners))
211  else:
212  direction_corners = (corners[-1] - corners[0]) >= numpy.array([[0.0, 0.0]])
213 
214  if not numpy.all(direction_corners):
215  if not numpy.any(direction_corners):
216  corners = numpy.copy(numpy.flipud(corners))
217  elif direction_corners[0][0]:
218  corners = numpy.rot90(corners.reshape(board.n_rows, board.n_cols, 2)).reshape(
219  board.n_cols * board.n_rows, 1, 2)
220  else:
221  corners = numpy.rot90(corners.reshape(board.n_rows, board.n_cols, 2), 3).reshape(
222  board.n_cols * board.n_rows, 1, 2)
223  if refine and ok:
224  # Use a radius of half the minimum distance between corners. This should be large enough to snap to the
225  # correct corner, but not so large as to include a wrong corner in the search window.
226  min_distance = float("inf")
227  for row in range(board.n_rows):
228  for col in range(board.n_cols - 1):
229  index = row * board.n_rows + col
230  min_distance = min(min_distance, _pdist(corners[index, 0], corners[index + 1, 0]))
231  for row in range(board.n_rows - 1):
232  for col in range(board.n_cols):
233  index = row * board.n_rows + col
234  min_distance = min(min_distance, _pdist(corners[index, 0], corners[index + board.n_cols, 0]))
235  radius = int(math.ceil(min_distance * 0.5))
236  cv2.cornerSubPix(mono, corners, (radius, radius), (-1, -1),
237  (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER, 30, 0.1))
238 
239  return (ok, corners)

242 def _get_circles(img, board, pattern):
243  """
244  Get circle centers for a symmetric or asymmetric grid
245  """
246  h = img.shape[0]
247  w = img.shape[1]

249  mono = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

251  mono = img

255  flag = cv2.CALIB_CB_ASYMMETRIC_GRID

257  (ok, corners) = cv2.findCirclesGrid(mono_arr, (board.n_cols, board.n_rows), flags=flag)
258

260  # again with dimensions swapped - not so efficient.
261  # TODO Better to add as second board? Corner ordering will change.

264 
265  return (ok, corners)

335  def get_parameters(self, corners, board, size):
336  """
337  Return list of parameters [X, Y, size, skew] describing the checkerboard view.
338  """
339  (width, height) = size
340  Xs = corners[:, :, 0]
341  Ys = corners[:, :, 1]
342  area = _get_area(corners, board)
343  border = math.sqrt(area)
344  # For X and Y, we "shrink" the image all around by approx. half the board size.
345  # Otherwise large boards are penalized because you can't get much X/Y variation.
346  p_x = min(1.0, max(0.0, (numpy.mean(Xs) - border / 2) / (width - border)))
347  p_y = min(1.0, max(0.0, (numpy.mean(Ys) - border / 2) / (height - border)))
348  p_size = math.sqrt(area / (width * height))
349  skew = _get_skew(corners, board)
350  params = [p_x, p_y, p_size, skew]
351  return params

371  def compute_goodenough(self):
372  if not self.db:
373  return None
374 
375  # Find range of checkerboard poses covered by samples in database
376  all_params = [sample[0] for sample in self.db]
377  min_params = all_params[0]
378  max_params = all_params[0]
379  for params in all_params[1:]:
380  min_params = lmin(min_params, params)
381  max_params = lmax(max_params, params)
382  # Don't reward small size or skew
383  min_params = [min_params[0], min_params[1], 0., 0.]
384 
385  # For each parameter, judge how much progress has been made toward adequate variation
386  progress = [min((hi - lo) / r, 1.0) for (lo, hi, r) in zip(min_params, max_params, self.param_ranges)]
387  # If we have lots of samples, allow calibration even if not all parameters are green
388  # TODO Awkward that we update self.goodenough instead of returning it
389  self.goodenough = (len(self.db) >= self.min_good_enough) or all([p == 1.0 for p in progress])
390 
391  return list(zip(self._param_names, min_params, max_params, progress))

429  def downsample_and_detect(self, img):
430  """
431  Downsample the input image to approximately VGA resolution and detect the
432  calibration target corners in the full-size image.
433 
434  Combines these apparently orthogonal duties as an optimization. Checkerboard
435  detection is too expensive on large images, so it's better to do detection on
436  the smaller display image and scale the corners back up to the correct size.
437 
438  Returns (scrib, corners, downsampled_corners, board, (x_scale, y_scale)).
439  """
440  # Scale the input image down to ~VGA size
441  height = img.shape[0]
442  width = img.shape[1]
443  scale = math.sqrt((width * height) / (640. * 480.))
444  if scale > 1.0:
445  scrib = cv2.resize(img, (int(width / scale), int(height / scale)))
446  else:
447  scrib = img
448  # Due to rounding, actual horizontal/vertical scaling may differ slightly
449  x_scale = float(width) / scrib.shape[1]
450  y_scale = float(height) / scrib.shape[0]
451 
452  if self.pattern == Patterns.Chessboard:
453  # Detect checkerboard
454  (ok, downsampled_corners, board) = self.get_corners(scrib, refine=True)
455 
456  # Scale corners back to full size image
457  corners = None
458  if ok:
459  if scale > 1.0:
460  # Refine up-scaled corners in the original full-res image
461  # TODO Does this really make a difference in practice?
462  corners_unrefined = downsampled_corners.copy()
463  corners_unrefined[:, :, 0] *= x_scale
464  corners_unrefined[:, :, 1] *= y_scale
465  radius = int(math.ceil(scale))
466  if len(img.shape) == 3 and img.shape[2] == 3:
467  mono = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
468  else:
469  mono = img
470  cv2.cornerSubPix(mono, corners_unrefined, (radius, radius), (-1, -1),
471  (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER, 30, 0.1))
472  corners = corners_unrefined
473  else:
474  corners = downsampled_corners
475  else:
476  # Circle grid detection is fast even on large images
477  (ok, corners, board) = self.get_corners(img)
478  # Scale corners to downsampled image for display
479  downsampled_corners = None
480  if ok:
481  if scale > 1.0:
482  downsampled_corners = corners.copy()
483  downsampled_corners[:, :, 0] /= x_scale
484  downsampled_corners[:, :, 1] /= y_scale
485  else:
486  downsampled_corners = corners
487 
488  return (scrib, corners, downsampled_corners, board, (x_scale, y_scale))

775  def pt2line(x0, y0, x1, y1, x2, y2):
776  """ point is (x0, y0), line is (x1, y1, x2, y2) """
777  return abs((x2 - x1) * (y1 - y0) - (x1 - x0) * (y2 - y1)) / math.sqrt((x2 - x1) ** 2 + (y2 - y1) ** 2)

785  for i in range(1, cc - 1):
786  (x0, y0) = corners[(cc * r) + i, 0]
787  errors.append(pt2line(x0, y0, x1, y1, x2, y2))
788  if errors:
789  return math.sqrt(sum([e ** 2 for e in errors]) / len(errors))
