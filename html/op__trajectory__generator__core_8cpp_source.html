193 void TrajectoryGen::callbackGetGlobalPlannerPath(const autoware_msgs::LaneArrayConstPtr& msg)
194 {
195  //std::cout << "test" << std::endl;
196  if(msg->lanes.size() > 0)
197  {
198  bool bOldGlobalPath = m_GlobalPaths.size() == msg->lanes.size();
199 
200  m_GlobalPaths.clear();
201 
202  //each lanes in msg represent a possbile path toward goal point
203  for(unsigned int i = 0 ; i < msg->lanes.size(); i++)
204  {
205  //simply convert and copy waypoints in autoware_msgs::Lane to a vector of PlannerHNS::WayPoint
206  PlannerHNS::ROSHelpers::ConvertFromAutowareLaneToLocalLane(msg->lanes.at(i), m_temp_path);
207 
208  //For each waypoint in m_temp_path, the angle of waypoint m is defined as m.pos.a = atan2(m,m-1);
209  //The cost of waypoint m is defined as: cost(m) = sum_{i=0}^{i=m} distance(p_i,p_{i-1});
210  PlannerHNS::PlanningHelpers::CalcAngleAndCost(m_temp_path);
211 
212  m_GlobalPaths.push_back(m_temp_path);
213 
214  if(bOldGlobalPath)
215  {
216  bOldGlobalPath = PlannerHNS::PlanningHelpers::CompareTrajectories(m_temp_path, m_GlobalPaths.at(i));
217  }
218  }
219 
220  if(!bOldGlobalPath)
221  {
222  bWayGlobalPath = true;
223  std::cout << "Received New Global Path Generator ! " << std::endl;
224  }
225  else
226  {
227  std::cout << "clear generator " << std::endl;
228  m_GlobalPaths.clear();
229  }
230  }
