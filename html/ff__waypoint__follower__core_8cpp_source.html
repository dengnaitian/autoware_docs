254 void FFSteerControl::callbackGetCurrentTrajectory(const autoware_msgs::LaneConstPtr &msg)
255 {
256  m_State.m_Path.clear();
257  for(unsigned int i = 0 ; i < msg->waypoints.size(); i++)
258  {
259  PlannerHNS::WayPoint wp(msg->waypoints.at(i).pose.pose.position.x,
260  msg->waypoints.at(i).pose.pose.position.y, msg->waypoints.at(i).pose.pose.position.z,
261  tf::getYaw(msg->waypoints.at(i).pose.pose.orientation));
262  wp.v = msg->waypoints.at(i).twist.twist.linear.x;
263 
264  if(msg->waypoints.at(i).twist.twist.linear.z == 0)
265  wp.bDir = PlannerHNS::FORWARD_DIR;
266  else if(msg->waypoints.at(i).twist.twist.linear.z == 1)
267  wp.bDir = PlannerHNS::FORWARD_LEFT_DIR;
268  else if(msg->waypoints.at(i).twist.twist.linear.z == 2)
269  wp.bDir = PlannerHNS::FORWARD_RIGHT_DIR;
270  else if(msg->waypoints.at(i).twist.twist.linear.z == 3)
271  wp.bDir = PlannerHNS::BACKWARD_DIR;
272  else if(msg->waypoints.at(i).twist.twist.linear.z == 4)
273  wp.bDir = PlannerHNS::BACKWARD_LEFT_DIR;
274  else if(msg->waypoints.at(i).twist.twist.linear.z == 5)
275  wp.bDir = PlannerHNS::BACKWARD_RIGHT_DIR;
276  else if(msg->waypoints.at(i).twist.twist.linear.z == 6)
277  wp.bDir = PlannerHNS::STANDSTILL_DIR;
278  else
279  wp.bDir = PlannerHNS::STANDSTILL_DIR;
280 
281  m_State.m_Path.push_back(wp);
282  }
283 
284 // cout << "### Current Trajectory CallBaclk -> " << m_State.m_Path.size() << endl;
285 
286  bNewTrajectory = true;

375 PlannerHNS::BehaviorState FFSteerControl::ConvertBehaviorStateFromAutowareToPlannerH(const geometry_msgs::TwistStampedConstPtr& msg)
376 {
377  PlannerHNS::BehaviorState behavior;
378  behavior.followDistance = msg->twist.linear.x;
379  behavior.stopDistance = msg->twist.linear.y;
380  behavior.followVelocity = msg->twist.angular.x;
381  behavior.maxVelocity = msg->twist.angular.y;
382 
383 
384  if(msg->twist.linear.z == PlannerHNS::LIGHT_INDICATOR::INDICATOR_LEFT)
385  behavior.indicator = PlannerHNS::LIGHT_INDICATOR::INDICATOR_LEFT;
386  else if(msg->twist.linear.z == PlannerHNS::LIGHT_INDICATOR::INDICATOR_RIGHT)
387  behavior.indicator = PlannerHNS::LIGHT_INDICATOR::INDICATOR_RIGHT;
388  else if(msg->twist.linear.z == PlannerHNS::LIGHT_INDICATOR::INDICATOR_BOTH)
389  behavior.indicator = PlannerHNS::LIGHT_INDICATOR::INDICATOR_BOTH;
390  else if(msg->twist.linear.z == PlannerHNS::LIGHT_INDICATOR::INDICATOR_NONE)
391  behavior.indicator = PlannerHNS::LIGHT_INDICATOR::INDICATOR_NONE;
392 
393  if(msg->twist.angular.z == PlannerHNS::INITIAL_STATE)
394  behavior.state = PlannerHNS::INITIAL_STATE;
395  else if(msg->twist.angular.z == PlannerHNS::WAITING_STATE)
396  behavior.state = PlannerHNS::WAITING_STATE;
397  else if(msg->twist.angular.z == PlannerHNS::FORWARD_STATE)
398  behavior.state = PlannerHNS::FORWARD_STATE;
399  else if(msg->twist.angular.z == PlannerHNS::STOPPING_STATE)
400  behavior.state = PlannerHNS::STOPPING_STATE;
401  else if(msg->twist.angular.z == PlannerHNS::EMERGENCY_STATE)
402  behavior.state = PlannerHNS::EMERGENCY_STATE;
403  else if(msg->twist.angular.z == PlannerHNS::TRAFFIC_LIGHT_STOP_STATE)
404  behavior.state = PlannerHNS::TRAFFIC_LIGHT_STOP_STATE;
405  else if(msg->twist.angular.z == PlannerHNS::STOP_SIGN_STOP_STATE)
406  behavior.state = PlannerHNS::STOP_SIGN_STOP_STATE;
407  else if(msg->twist.angular.z == PlannerHNS::STOP_SIGN_WAIT_STATE)
408  behavior.state = PlannerHNS::STOP_SIGN_WAIT_STATE;
409  else if(msg->twist.angular.z == PlannerHNS::FOLLOW_STATE)
410  behavior.state = PlannerHNS::FOLLOW_STATE;
411  else if(msg->twist.angular.z == PlannerHNS::LANE_CHANGE_STATE)
412  behavior.state = PlannerHNS::LANE_CHANGE_STATE;
413  else if(msg->twist.angular.z == PlannerHNS::OBSTACLE_AVOIDANCE_STATE)
414  behavior.state = PlannerHNS::OBSTACLE_AVOIDANCE_STATE;
415  else if(msg->twist.angular.z == PlannerHNS::FINISH_STATE)
416  behavior.state = PlannerHNS::FINISH_STATE;
417 
418 
419  return behavior;
420

423 void FFSteerControl::PlannerMainLoop()
424 {
425 
426  ros::Rate loop_rate(100);
427 #ifdef ENABLE_ZMP_LIBRARY_LINK
428  if(m_pComm)
429  m_pComm->GoLive(true);
430 #endif
431 
432  vector<PlannerHNS::WayPoint> path;
433  PlannerHNS::WayPoint p2;
434  double totalDistance = 0;
435 
436  while (ros::ok())
437  {
438  ros::spinOnce();
439 
440  PlannerHNS::BehaviorState currMessage = m_CurrentBehavior;
441  double dt = UtilityHNS::UtilityH::GetTimeDiffNow(m_PlanningTimer);
442  UtilityHNS::UtilityH::GetTickCount(m_PlanningTimer);
443 
444  if(currMessage.state != PlannerHNS::INITIAL_STATE && (bInitPos || bNewCurrentPos))
445  {
450  if(m_CmdParams.statusSource == CONTROL_BOX_STATUS)
451  {
452  //Read StateData From Control Box
453 #ifdef ENABLE_ZMP_LIBRARY_LINK
454  if(m_pComm && m_pComm->IsAuto())
455  {
456  m_CurrVehicleStatus.steer = m_pComm->GetCurrentSteerAngle();
457  m_CurrVehicleStatus.speed = m_pComm->GetCurrentSpeed();
458  m_CurrVehicleStatus.shift = m_pComm->GetCurrentShift();
459 
460  //Send status over message to planner
461  nav_msgs::Odometry control_box_status;
462  control_box_status.header.stamp = ros::Time::now();
463  control_box_status.twist.twist.angular.z = m_CurrVehicleStatus.steer;
464  control_box_status.twist.twist.linear.x = m_CurrVehicleStatus.speed;
465  control_box_status.twist.twist.linear.z = (int)m_CurrVehicleStatus.shift;
466 
467  control_box_status.pose.pose.position.x = m_CurrentPos.pos.x;
468  control_box_status.pose.pose.position.y = m_CurrentPos.pos.y;
469  control_box_status.pose.pose.position.z = m_CurrentPos.pos.z;
470  control_box_status.pose.pose.orientation = tf::createQuaternionMsgFromYaw(UtilityHNS::UtilityH::SplitPositiveAngle(m_CurrentPos.pos.a));
471 
472  pub_ControlBoxOdom.publish(control_box_status);
473 
474  cout << "Read Live Car Info .. " << endl;
475  }
476  else
477  {
478  cout << ">>> Error, Disconnected from Car Control Box !" << endl;
479  }
480 #endif
481  }
482  else if(m_CmdParams.statusSource == SIMULATION_STATUS)
483  {
484  //m_CurrVehicleStatus = m_PrevStepTargetStatus;
485  //m_State.SimulateOdoPosition(dt, m_CurrVehicleStatus);
486 
487  m_State.SimulateOdoPosition(dt, m_PrevStepTargetStatus);
488  m_CurrVehicleStatus.steer = m_State.m_CurrentSteering;
489  m_CurrVehicleStatus.speed = m_State.m_CurrentVelocity;
490  m_CurrVehicleStatus.shift = m_PrevStepTargetStatus.shift;
491  m_CurrentPos = m_State.state;
