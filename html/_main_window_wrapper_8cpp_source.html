197 void MainWindowWrapper::SimuReshape(int width, int height)
198 {
199  glutPositionWindow(m_params.simu_window.x, m_params.simu_window.y);
200  glutReshapeWindow(m_params.simu_window.w, m_params.simu_window.h);
201 
202  glViewport(0,0, m_params.simu_window.w, m_params.simu_window.h);
203 
204  //this should rely on the map not the window
205  double y_max = sqrt(pow(m_params.simu_window.w,2) + pow(m_params.simu_window.h,2))/2.0;
206  if(y_max > 0.05 && y_max < (m_params.simu_window.w*m_params.simu_window.h))
207  m_DisplayParam.prespective_z = y_max;
208 
209  m_DisplayParam.prespective_z = 10000;
210 
211  glMatrixMode(GL_PROJECTION);
212  glLoadIdentity();
213  double aspect_ratio = (double)m_params.simu_window.w/(double)m_params.simu_window.h;
214 
215  gluPerspective(m_DisplayParam.prespective_fov,aspect_ratio,0.05,m_DisplayParam.prespective_z);
216  glMatrixMode(GL_MODELVIEW);
217  glLoadIdentity();
218 
219  if(m_DisplayParam.bDisplayMode == DISPLAY_FOLLOW && m_DrawAndControl)
220  {
221  m_DisplayParam.eye[0] = m_DisplayParam.centerRotX;
222  m_DisplayParam.eye[1] = m_DisplayParam.centerRotY;
223  m_DisplayParam.at[0] = m_DisplayParam.centerRotX;
224  m_DisplayParam.at[1] = m_DisplayParam.centerRotY;
225  }
226 
227 
228  gluLookAt(m_DisplayParam.eye[0], m_DisplayParam.eye[1], m_DisplayParam.zoom,
229  m_DisplayParam.at[0], m_DisplayParam.at[1], m_DisplayParam.at[2],
230  m_DisplayParam.up[0], m_DisplayParam.up[1],m_DisplayParam.up[2]);
231 
232  InitLighting();
233 
234 }

236 void MainWindowWrapper::FromScreenToModelCoordinate(int sx, int sy, double& modelX, double& modelY)
237 {
238  double whole = (double)(m_params.simu_window.w + m_params.simu_window.h)/2.0;
239  double actualViewX = tan(m_DisplayParam.prespective_fov*DEG2RAD) * m_DisplayParam.zoom / whole;
240 
241  modelX = sx * actualViewX ;
242  modelY = sy * actualViewX ;
243 }
