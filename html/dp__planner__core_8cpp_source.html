360 {
361  autoware_msgs::CloudCluster cluster;
362  cluster.centroid_point.point = centerPose.point;
363  cluster.dimensions.x = x_rand;
364  cluster.dimensions.y = y_rand;
365  cluster.dimensions.z = z_rand;
366  pcl::PointCloud<pcl::PointXYZ> point_cloud;
367 
368  timespec t;
369  for(int i=1; i < nPoints; i++)
370  {
371  UtilityHNS::UtilityH::GetTickCount(t);
372  pcl::PointXYZ p;
373  srand(t.tv_nsec/i);
374  double x = (double)(rand()%100)/100.0 - 0.5;
375 
376  srand(t.tv_nsec/i*i);

378 
379  srand(t.tv_nsec);

381 
382  p.x = centerPose.point.x + x*x_rand;
383  p.y = centerPose.point.y + y*y_rand;
384  p.z = centerPose.point.z + z*z_rand;
