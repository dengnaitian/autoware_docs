105  int obj_id = -1;

107  double actual_steering = 0;
108  int indicator = 3;
109 
110  if(msg.poses.size() > 0 )
111  {
112  obj_id = msg.poses.at(0).position.x;
113  actual_speed = msg.poses.at(0).position.y;

115  }
116 
117  if(msg.poses.size() == 4)

119  indicator = msg.poses.at(3).orientation.w;
120  }
121 
122  if(obj_id < 0)
123  return;
124 
125  int index = -1;
126  for(unsigned int i = 0; i < m_ObjClustersArray.clusters.size() ; i++ )
127  {
128  if((int)m_ObjClustersArray.clusters.at(i).id == obj_id)
129  {
130  index = i;
131  break;
132  }
133  }
134 
135  timespec t;
136  UtilityHNS::UtilityH::GetTickCount(t);
137  srand(t.tv_nsec);
138  int nPoints = m_DecParams.nPointsPerObj + (rand()%POINT_CLOUD_ADDTIONAL_ERR_NUM - POINT_CLOUD_ADDTIONAL_ERR_NUM/2);
139 
140  autoware_msgs::CloudCluster c = GenerateSimulatedObstacleCluster(msg.poses.at(2).position.y, msg.poses.at(2).position.x, msg.poses.at(2).position.z, nPoints, msg.poses.at(1));
141  c.id = obj_id;
142  c.score = actual_speed;
143  c.indicator_state = indicator;
144 
145  if(index >= 0) // update existing
146  {
147  m_ObjClustersArray.clusters.at(index) = c;
148  m_keepTime.at(index).second = OBJECT_KEEP_TIME;
149  }
150  else
151  {
152  m_ObjClustersArray.clusters.push_back(c);
153  m_keepTime.push_back(std::make_pair(c.id, OBJECT_KEEP_TIME));
154  }

158 autoware_msgs::CloudCluster OpenPlannerSimulatorPerception::GenerateSimulatedObstacleCluster(const double& width, const double& length, const double& height, const int& nPoints, const geometry_msgs::Pose& centerPose)
159 {
160  //std::cout << "GenerateSimulatedObstacleCluster" << std::endl;
161  autoware_msgs::CloudCluster cluster;
162 
163  timespec t;
164  UtilityHNS::UtilityH::GetTickCount(t);
165  srand(t.tv_nsec);
166 
167  ENG eng(t.tv_nsec);
168  NormalDIST dist_x(0, m_DecParams.errFactor);
169  VariatGEN gen_x(eng, dist_x);
170 
171  cluster.centroid_point.point.x = centerPose.position.x + gen_x();
172  cluster.centroid_point.point.y = centerPose.position.y + gen_x();
173  cluster.centroid_point.point.z = centerPose.position.z;
174 
175  cluster.avg_point.point.x = centerPose.position.x;
176  cluster.avg_point.point.y = centerPose.position.y;
177  cluster.avg_point.point.z = centerPose.position.z;
178 
179  double yaw_angle = tf::getYaw(centerPose.orientation);
180  cluster.estimated_angle = yaw_angle;
181 
182  cluster.dimensions.x = width;
183  cluster.dimensions.y = length;
184  cluster.dimensions.z = height;
185  pcl::PointCloud<pcl::PointXYZI> point_cloud;
186 
187  PlannerHNS::Mat3 rotationMat(yaw_angle);
188  PlannerHNS::Mat3 translationMat(cluster.avg_point.point.x, cluster.avg_point.point.y);
189 
190  for(int i=1; i < nPoints; i++)
191  {
192  UtilityHNS::UtilityH::GetTickCount(t);
193  PlannerHNS::WayPoint center_p;
194  srand(t.tv_nsec);
195 
196  center_p.pos.x = ((double)(rand()%100)/100.0 - CONTOUR_DISTANCE_ERROR);
197  center_p.pos.x *= width;
198 
199  srand(t.tv_nsec/i);
200  center_p.pos.y = ((double)(rand()%100)/100.0 - CONTOUR_DISTANCE_ERROR);
201  center_p.pos.y *= length;
202 
203  srand(t.tv_nsec/i*i);
204  center_p.pos.z = ((double)(rand()%100)/100.0 - CONTOUR_DISTANCE_ERROR)* height;
205 
206  center_p.pos = rotationMat*center_p.pos;
207  center_p.pos = translationMat*center_p.pos;
208 
209  pcl::PointXYZI p;
210  p.x = center_p.pos.x;
211  p.y = center_p.pos.y;
212  p.z = center_p.pos.z;
213 
214  point_cloud.points.push_back(p);
215  }
216 
217  pcl::toROSMsg(point_cloud, cluster.cloud);
218  return cluster;
219 }

221 void OpenPlannerSimulatorPerception::MainLoop()
222 {
223 
224  ros::Rate loop_rate(15);
225 
226  while (ros::ok())
227  {
228  ros::spinOnce();
229 
230  //clean old data
231  for(unsigned int i = 0 ; i < m_keepTime.size(); i++)
232  {
233  if(m_keepTime.at(i).second <= 0)
234  {
235  m_keepTime.erase(m_keepTime.begin()+i);
236  m_ObjClustersArray.clusters.erase(m_ObjClustersArray.clusters.begin()+i);
237  i--;
238  }
239  else
240  m_keepTime.at(i).second -= 1;
241  }
242 
243  if(m_bSetSimulatedObj)
244  {
245  m_AllObjClustersArray = m_ObjClustersArray;
246  m_AllObjClustersArray.clusters.push_back(m_SimulatedCluter);
247  pub_DetectedObjects.publish(m_AllObjClustersArray);
248  m_SimulatedCluter.cloud.header.frame_id = "map";
249  pub_PointClouds.publish(m_SimulatedCluter.cloud);
250  }
251  else
252  {
253  pub_DetectedObjects.publish(m_ObjClustersArray);
254  }
255 
256  loop_rate.sleep();
257  }
258 }
