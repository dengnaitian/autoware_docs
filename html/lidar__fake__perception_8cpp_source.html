114 bool LidarFakePerception::updateFakeObject()
115 {
116  // check lifetime
117  if (object_lifetime_ > 0)
118  {
119  if ((ros::Time::now() - fake_object_initial_time_).toSec() > object_lifetime_)
120  {
121  return false;
122  }
123  }
124 
125  // obtain world -> sensor frame
126  tf::StampedTransform global2local;
127  try
128  {
129  tf_listener_.waitForTransform(global_frame_, object_frame_, ros::Time(0), ros::Duration(1.0));
130  tf_listener_.lookupTransform(global_frame_, object_frame_, ros::Time(0), global2local);
131  }
132  catch (tf::TransformException ex)
133  {
134  ROS_ERROR("%s", ex.what());
135  return false;
136  }
137 
138  // update pose
139  geometry_msgs::Twist twist;
140  if (use_fake_twist_)
141  {
142  // update pose by subscribed twist
143  twist = fake_object_twist_;
144  }
145  else
146  {
147  // update pose by rosparam
148  twist.linear.x = object_velocity_;
149  twist.angular.z = object_angular_velocity_;
150  }
151  updatePose(1. / publish_rate_, twist, fake_object_pose_);
152 
153  // fix height
154  tf::Vector3 objpos = fake_object_pose_.getOrigin();
155  fake_object_pose_.setOrigin(tf::Vector3(objpos.x(), objpos.y(), object_z_offset_ + object_height_ / 2.));
156 
157  // update msg
158  fake_object_.header.stamp = ros::Time::now();
159  fake_object_.header.frame_id = object_frame_;
160  fake_object_.id = fake_object_id_;
161  fake_object_.label = object_label_;
162  fake_object_.dimensions.x = object_length_;
163  fake_object_.dimensions.y = object_width_;
164  fake_object_.dimensions.z = object_height_;
165  fake_object_.velocity = twist;
166  fake_object_.acceleration = geometry_msgs::Twist(); // NOTE: by constant velocity model
167  tf::poseTFToMsg(global2local.inverse() * fake_object_pose_, fake_object_.pose); // local
168 
169  return true;
170 }

172 void LidarFakePerception::updatePose(const double& dt, const geometry_msgs::Twist& twist, tf::Transform& tf)

192 void LidarFakePerception::convertObjectToPoints(const autoware_msgs::DetectedObject& obj, PointCloudT& points)

203  {
