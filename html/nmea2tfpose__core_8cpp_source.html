89 void Nmea2TFPoseNode::convert(std::vector<std::string> nmea, ros::Time current_stamp)
90 {
91  try
92  {
93  if (nmea.at(0).compare(0, 2, "QQ") == 0)
94  {
95  orientation_time_ = stod(nmea.at(3));
96  roll_ = stod(nmea.at(4)) * M_PI / 180.;
97  pitch_ = -1 * stod(nmea.at(5)) * M_PI / 180.;
98  yaw_ = -1 * stod(nmea.at(6)) * M_PI / 180. + M_PI / 2;
99  orientation_stamp_ = current_stamp;
100  ROS_INFO("QQ is subscribed.");
101  }
102  else if (nmea.at(0) == "$PASHR")
103  {
104  orientation_time_ = stod(nmea.at(1));
105  roll_ = stod(nmea.at(4)) * M_PI / 180.;
106  pitch_ = -1 * stod(nmea.at(5)) * M_PI / 180.;
107  yaw_ = -1 * stod(nmea.at(2)) * M_PI / 180. + M_PI / 2;
108  ROS_INFO("PASHR is subscribed.");
109  }
110  else if(nmea.at(0).compare(3, 3, "GGA") == 0)
111  {
112  position_time_ = stod(nmea.at(1));
113  double lat = stod(nmea.at(2));
114  double lon = stod(nmea.at(4));
115  double h = stod(nmea.at(9));
116  geo_.set_llh_nmea_degrees(lat, lon, h);
117  ROS_INFO("GGA is subscribed.");
118  }
119  else if(nmea.at(0) == "$GPRMC")
120  {
121  position_time_ = stoi(nmea.at(1));
122  double lat = stod(nmea.at(3));
123  double lon = stod(nmea.at(5));
124  double h = 0.0;
125  geo_.set_llh_nmea_degrees(lat, lon, h);
126  ROS_INFO("GPRMC is subscribed.");
127  }
128  }catch (const std::exception &e)
129  {
130  ROS_WARN_STREAM("Message is invalid : " << e.what());
131  }
132 }

134 void Nmea2TFPoseNode::callbackFromNmeaSentence(const nmea_msgs::Sentence::ConstPtr &msg)
135 {
136  current_time_ = msg->header.stamp;
137  convert(split(msg->sentence), msg->header.stamp);
138 
139  double timeout = 10.0;
140  if (fabs(orientation_stamp_.toSec() - msg->header.stamp.toSec()) > timeout)
141  {
142  double dt = sqrt(pow(geo_.x() - last_geo_.x(), 2) + pow(geo_.y() - last_geo_.y(), 2));
143  double threshold = 0.2;
144  if (dt > threshold)
145  {
146  ROS_INFO("QQ is not subscribed. Orientation is created by atan2");
147  createOrientation();
148  publishPoseStamped();
149  publishTF();
150  last_geo_ = geo_;
151  }
152  return;
153  }
154 
155  double e = 1e-2;
156  if (fabs(orientation_time_ - position_time_) < e)
157  {
158  publishPoseStamped();
159  publishTF();
160  return;
161  }
162 }
