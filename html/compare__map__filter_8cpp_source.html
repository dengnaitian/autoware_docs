114 void CompareMapFilter::sensorPointsCallback(const sensor_msgs::PointCloud2::ConstPtr& sensorTF_cloud_msg_ptr)
115 {
116  const ros::Time sensor_time = sensorTF_cloud_msg_ptr->header.stamp;
117  const std::string sensor_frame = sensorTF_cloud_msg_ptr->header.frame_id;
118 
119  pcl::PointCloud<pcl::PointXYZI>::Ptr sensorTF_cloud_ptr(new pcl::PointCloud<pcl::PointXYZI>);
120  pcl::fromROSMsg(*sensorTF_cloud_msg_ptr, *sensorTF_cloud_ptr);
121 
122  pcl::PointCloud<pcl::PointXYZI>::Ptr sensorTF_clipping_height_cloud_ptr(new pcl::PointCloud<pcl::PointXYZI>);
123  sensorTF_clipping_height_cloud_ptr->header = sensorTF_cloud_ptr->header;
124  for (size_t i = 0; i < sensorTF_cloud_ptr->points.size(); ++i)
125  {
126  if (sensorTF_cloud_ptr->points[i].z > min_clipping_height_ &&
127  sensorTF_cloud_ptr->points[i].z < max_clipping_height_)
128  {
129  sensorTF_clipping_height_cloud_ptr->points.push_back(sensorTF_cloud_ptr->points[i]);
130  }
131  }
132 
133  pcl::PointCloud<pcl::PointXYZI>::Ptr mapTF_cloud_ptr(new pcl::PointCloud<pcl::PointXYZI>);
134  try
135  {
136  tf_listener_->waitForTransform(map_frame_, sensor_frame, sensor_time, ros::Duration(3.0));
137  pcl_ros::transformPointCloud(map_frame_, sensor_time, *sensorTF_clipping_height_cloud_ptr, sensor_frame,
138  *mapTF_cloud_ptr, *tf_listener_);
139  }
140  catch (tf::TransformException& ex)
141  {
142  ROS_ERROR("Transform error: %s", ex.what());
143  return;
144  }
145 
146  pcl::PointCloud<pcl::PointXYZI>::Ptr mapTF_match_cloud_ptr(new pcl::PointCloud<pcl::PointXYZI>);
147  pcl::PointCloud<pcl::PointXYZI>::Ptr mapTF_unmatch_cloud_ptr(new pcl::PointCloud<pcl::PointXYZI>);

187 void CompareMapFilter::searchMatchingCloud(const pcl::PointCloud<pcl::PointXYZI>::Ptr in_cloud_ptr,
188  pcl::PointCloud<pcl::PointXYZI>::Ptr match_cloud_ptr,
189  pcl::PointCloud<pcl::PointXYZI>::Ptr unmatch_cloud_ptr)
190 {
191  match_cloud_ptr->points.clear();
192  unmatch_cloud_ptr->points.clear();
193 
194  match_cloud_ptr->points.reserve(in_cloud_ptr->points.size());
195  unmatch_cloud_ptr->points.reserve(in_cloud_ptr->points.size());
196 
197  std::vector<int> nn_indices(1);
198  std::vector<float> nn_dists(1);
199  const double squared_distance_threshold = distance_threshold_ * distance_threshold_;
200 
201  for (size_t i = 0; i < in_cloud_ptr->points.size(); ++i)
202  {
203  tree_.nearestKSearch(in_cloud_ptr->points[i], 1, nn_indices, nn_dists);
204  if (nn_dists[0] <= squared_distance_threshold)
205  {
206  match_cloud_ptr->points.push_back(in_cloud_ptr->points[i]);
207  }
208  else
209  {
210  unmatch_cloud_ptr->points.push_back(in_cloud_ptr->points[i]);
211  }
212  }
213 }
