68 class DisplayThread(threading.Thread):
69  """
70  Thread that displays the current images
71  It is its own thread so that all display can be done
72  in one thread to overcome imshow limitations and
73  https://github.com/ros-perception/image_pipeline/issues/85
74  """
75  def __init__(self, queue, opencv_calibration_node):
76  threading.Thread.__init__(self)
77  self.queue = queue
78  self.opencv_calibration_node = opencv_calibration_node
79 
80  def run(self):
81  cv2.namedWindow("display", cv2.WINDOW_NORMAL)
82  cv2.setMouseCallback("display", self.opencv_calibration_node.on_mouse)
83  cv2.createTrackbar("scale", "display", 0, 100, self.opencv_calibration_node.on_scale)
84  while True:
85  # wait for an image (could happen at the very beginning when the queue is still empty)
86  while len(self.queue) == 0:
87  time.sleep(0.1)
88  im = self.queue[0]
89  cv2.imshow("display", im)
90  k = cv2.waitKey(6) & 0xFF
91  if k in [27, ord('q')]:
92  rospy.signal_shutdown('Quit')
93  elif k == ord('s'):
94  self.opencv_calibration_node.screendump(im)

317  def redraw_monocular(self, drawable):
318  height = drawable.scrib.shape[0]
319  width = drawable.scrib.shape[1]
320 
321  display = numpy.ones((max(480, height), width + 100, 3), dtype=numpy.uint8)
322  display[0:height, 0:width,:] = drawable.scrib
323  display[0:height, width:width+100,:].fill(255)
324 
325  self.buttons(display)
326  if not self.c.calibrated:
327  if drawable.params:
328  for i, (label, lo, hi, progress) in enumerate(drawable.params):
329  (w,_) = self.getTextSize(label)
330  self.putText(display, label, (width + (100 - w) // 2, self.y(i)))
331  color = (0,255,0)
332  if progress < 1.0:
333  color = (0, int(progress*255.), 255)
334  cv2.line(display,
335  (int(width + lo * 100), self.y(i) + 20),
336  (int(width + hi * 100), self.y(i) + 20),
337  color, 4)
338 
339  else:
340  self.putText(display, "lin.", (width, self.y(0)))
341  linerror = drawable.linear_error
342  if linerror < 0:
343  msg = "?"
344  else:
345  msg = "%.2f" % linerror
