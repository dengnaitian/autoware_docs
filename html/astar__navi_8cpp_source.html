96 void AstarNavi::run()
97 {
98  ros::Rate rate(update_rate_);
99 
100  nav_msgs::Path empty_path;
101  empty_path.header.stamp = ros::Time::now();
102  empty_path.header.frame_id = costmap_.header.frame_id;
103 
104  while (ros::ok())
105  {
106  ros::spinOnce();
107 
108  if (!costmap_initialized_ || !current_pose_initialized_ || !goal_pose_initialized_)
109  {
110  rate.sleep();
111  continue;
112  }
113 
114  // initialize vector for A* search, this runs only once
115  astar_.initialize(costmap_);
116 
117  // update local goal pose
118  goalPoseCallback(goal_pose_global_);
119 
120  // execute astar search
121  ros::WallTime start = ros::WallTime::now();
122  bool result = astar_.makePlan(current_pose_local_.pose, goal_pose_local_.pose);
123  ros::WallTime end = ros::WallTime::now();
124 
125  ROS_INFO("Astar planning: %f [s]", (end - start).toSec());
126 
127  if (result)
128  {
129  ROS_INFO("Found GOAL!");
130  publishWaypoints(astar_.getPath(), waypoints_velocity_);
131  }
132  else
133  {
134  ROS_INFO("Can't find goal...");
135  publishStopWaypoints();
136  }
137 
138  astar_.reset();
139  rate.sleep();
140  }
141 }

143 void AstarNavi::publishWaypoints(const nav_msgs::Path& path, const double& velocity)
144 {
145  autoware_msgs::Lane lane;
146  lane.header.frame_id = "map";
147  lane.header.stamp = path.header.stamp;
148  lane.increment = 0;
149 
150  for (const auto& pose : path.poses)
151  {
152  autoware_msgs::Waypoint wp;
153  wp.pose.header = lane.header;
154  wp.pose.pose = transformPose(pose.pose, getTransform(lane.header.frame_id, pose.header.frame_id));
155  wp.pose.pose.position.z = current_pose_global_.pose.position.z; // height = const
156  wp.twist.twist.linear.x = velocity / 3.6; // velocity = const
157  lane.waypoints.push_back(wp);
158  }
159 
160  autoware_msgs::LaneArray lane_array;
161  lane_array.lanes.push_back(lane);
162  lane_pub_.publish(lane_array);
163 }
