245 //
246 // Returns the type of the field.
247 // The internal type and the added field should match!
248 //
249 const FieldDescription::FieldType FieldParameter::getFieldType() const
250 {
251  if (m_field == NULL)
252  {
253  // There is no associated field data structure, so we do not have a field type.
254  return FieldDescription::Undefined;
255  }
256  else
257  {
258  // Check if the internal datatype and the field type match
259  if ((m_field->getFieldType() == datatypes::FieldDescription::Rectangle) &&
260  (getFieldTypeIntern() == FieldTypeIntern_RECTANGLE))
261  {
262  return FieldDescription::Rectangle;
263  }
264  if ((m_field->getFieldType() == datatypes::FieldDescription::Segmented) &&
265  (getFieldTypeIntern() == FieldTypeIntern_SEGMENTED))
266  {
267  return FieldDescription::Segmented;
268  }
269 
270  // No match!
271  printError("FieldParameter::getFieldType(): The internal field type and the associated field structure do no match, returning UNDEFINED!");
272  }
273 
274  return FieldDescription::Undefined;
275 }
