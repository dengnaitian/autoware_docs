56  kalmanFilter1D()
57  {

60  kalmanFilter1D(double MovCov, double MeasureCov, double p, double intial_value)
61  {
62  result.MovCov = MovCov;
63  result.MeasureCov = MeasureCov;
64  result.p = p;
65  result.x = intial_value;
66  }
67 
68  Kalman1dState Update(double measurement)
69  {
70  //prediction update
71  result.p = result.p + result.MovCov;
72 
73  //measurement update
74  result.k = result.p / (result.p + result.MeasureCov);
75  result.x = result.x + result.k * (measurement - result.x);
76  result.p = (1 - result.k) * result.p;
77 
78  return result;
79  }

193  double currV = 0;
194  double currAccel = 0;
195 
196  if(m_iLife > 1)
197  {
198  currV = sqrt(vx*vx+vy*vy);
199 
200  double diff_y = predObj.center.pos.y - prev_y;
201  double diff_x = predObj.center.pos.x - prev_x;
202  if(hypot(diff_y, diff_x) > 0.2)
203  {
204  currA = atan2(diff_y, diff_x);
205  }
206  else
207  {
208  currA = prev_a;
209  }
210  }
211 
212 
213  if(m_iLife > MinAppearanceCount)

218 // predObj.acceleration_raw = (currV - prev_v)/_dt;

224  }

245  predObj.bVelocity = false;
246  }
247 
248  if(predObj.centers_list.size() > PREV_TRACK_SIZE)
249  predObj.centers_list.erase(predObj.centers_list.begin()+0);
250

252  {

254  double diff_x = predObj.center.pos.x - predObj.centers_list.at(predObj.centers_list.size()-1).pos.x;

273  }

277  }
278 
279 
280  m_filter.predict();
281  m_filter.statePre.copyTo(m_filter.statePost);
282  m_filter.errorCovPre.copyTo(m_filter.errorCovPost);
283 
284  prev_a = currA;
