38 void NodeStatusPublisher::publishStatus() {
39  ros::Rate rate = ros::Rate(autoware_health_checker::UPDATE_RATE);
40  while (ros_ok_) {
41  mtx_.lock();
42  autoware_system_msgs::NodeStatus status;
43  status.node_activated = node_activated_;
44  ros::Time now = ros::Time::now();
45  status.header.stamp = now;
46  status.node_name = ros::this_node::getName();
47  std::vector<std::string> checker_keys = getRateCheckerKeys();
48  // iterate Rate checker and publish rate_check result
49  for (auto key_itr = checker_keys.begin(); key_itr != checker_keys.end();
50  key_itr++) {
51  autoware_system_msgs::DiagnosticStatusArray diag_array;
52  autoware_system_msgs::DiagnosticStatus diag;
53  diag.type = autoware_system_msgs::DiagnosticStatus::RATE_IS_SLOW;
54  std::pair<uint8_t, double> result =
55  rate_checkers_[*key_itr]->getErrorLevelAndRate();
56  diag.level = result.first;
57  diag.key = *key_itr;
58  diag.value = doubeToJson(result.second);
59  diag.description = rate_checkers_[*key_itr]->description;
60  diag.header.stamp = now;
61  diag_array.status.push_back(diag);
62  status.status.push_back(diag_array);
63  }
64  // iterate Diagnostic Buffer and publish all diagnostic data
65  std::vector<std::string> keys = getKeys();
66  for (auto key_itr = keys.begin(); key_itr != keys.end(); key_itr++) {
67  status.status.push_back(diag_buffers_[*key_itr]->getAndClearData());
68  }
69  status_pub_.publish(status);
70  mtx_.unlock();
71  rate.sleep();
72  }
73  return;
74 }

81 std::vector<std::string> NodeStatusPublisher::getKeys() {
82  std::vector<std::string> keys;
83  std::vector<std::string> checker_keys = getRateCheckerKeys();
84  std::pair<std::string, std::shared_ptr<DiagBuffer>> buf_itr;
85  BOOST_FOREACH (buf_itr, diag_buffers_) {
86  bool matched = false;
87  for (auto checker_key_itr = checker_keys.begin();
88  checker_key_itr != checker_keys.end(); checker_key_itr++) {
89  if (*checker_key_itr == buf_itr.first) {
90  matched = true;
91  }
92  }
93  if (!matched) {
94  keys.push_back(buf_itr.first);
95  }
96  }
97  return keys;
98 }

127 uint8_t NodeStatusPublisher::CHECK_MIN_VALUE(std::string key, double value,
128  double warn_value,

132  addNewBuffer(key, autoware_system_msgs::DiagnosticStatus::OUT_OF_RANGE,
133  description);
134  autoware_system_msgs::DiagnosticStatus new_status;
135  new_status.type = autoware_system_msgs::DiagnosticStatus::OUT_OF_RANGE;
136  if (value < fatal_value) {
137  new_status.level = autoware_system_msgs::DiagnosticStatus::FATAL;
138  } else if (value < error_value) {
139  new_status.level = autoware_system_msgs::DiagnosticStatus::ERROR;
140  } else if (value < warn_value) {
141  new_status.level = autoware_system_msgs::DiagnosticStatus::WARN;
142  } else {
143  new_status.level = autoware_system_msgs::DiagnosticStatus::OK;
144  }
145  new_status.description = description;
146  new_status.value = doubeToJson(value);
147  new_status.header.stamp = ros::Time::now();
148  diag_buffers_[key]->addDiag(new_status);
149  return new_status.level;
150 }
