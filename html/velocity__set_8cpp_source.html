203 // obstacle detection for crosswalk
204 EControl crossWalkDetection(const pcl::PointCloud<pcl::PointXYZ>& points, const CrossWalk& crosswalk,
205  const geometry_msgs::PoseStamped& localizer_pose, const int points_threshold,
206  ObstaclePoints* obstacle_points)
207 {
208  int crosswalk_id = crosswalk.getDetectionCrossWalkID();
209  double search_radius = crosswalk.getDetectionPoints(crosswalk_id).width / 2;
210  // std::vector<int> crosswalk_ids crosswalk.getDetectionCrossWalkIDs();
211 
212  // Search each calculated points in the crosswalk
213  for (const auto& c_id : crosswalk.getDetectionCrossWalkIDs())
214  {
215  for (const auto& p : crosswalk.getDetectionPoints(c_id).points)
216  {
217  geometry_msgs::Point detection_point = calcRelativeCoordinate(p, localizer_pose.pose);
218  tf::Vector3 detection_vector = point2vector(detection_point);
219  detection_vector.setZ(0.0);
220 
221  int stop_count = 0; // the number of points in the detection area
222  for (const auto& p : points)
223  {
224  tf::Vector3 point_vector(p.x, p.y, 0.0);
225  double distance = tf::tfDistance(point_vector, detection_vector);
226  if (distance < search_radius)
227  {
228  stop_count++;
229  geometry_msgs::Point point_temp;
230  point_temp.x = p.x;
231  point_temp.y = p.y;
232  point_temp.z = p.z;
233  obstacle_points->setStopPoint(calcAbsoluteCoordinate(point_temp, localizer_pose.pose));
234  }
235  if (stop_count > points_threshold)
236  return EControl::STOP;
237  }
238  }
239 
240  obstacle_points->clearStopPoints();
241  if (!crosswalk.isMultipleDetection())
242  break;
243  }
244  return EControl::KEEP; // find no obstacles
245 }

247 int detectStopObstacle(const pcl::PointCloud<pcl::PointXYZ>& points, const int closest_waypoint,
248  const autoware_msgs::Lane& lane, const CrossWalk& crosswalk, double stop_range,
249  double points_threshold, const geometry_msgs::PoseStamped& localizer_pose,
250  ObstaclePoints* obstacle_points, EObstacleType* obstacle_type,
251  const int wpidx_detection_result_by_other_nodes)
252 {
253  int stop_obstacle_waypoint = -1;
254  *obstacle_type = EObstacleType::NONE;
255  // start search from the closest waypoint
256  for (int i = closest_waypoint; i < closest_waypoint + STOP_SEARCH_DISTANCE; i++)
257  {
258  // reach the end of waypoints
259  if (i >= static_cast<int>(lane.waypoints.size()))
260  break;
261 
262  // detection another nodes
263  if (wpidx_detection_result_by_other_nodes >= 0 &&
264  lane.waypoints.at(i).gid == wpidx_detection_result_by_other_nodes)
265  {
266  stop_obstacle_waypoint = i;
267  *obstacle_type = EObstacleType::STOPLINE;
268  obstacle_points->setStopPoint(lane.waypoints.at(i).pose.pose.position); // for vizuialization
269  break;
270  }
271 
272  // Detection for cross walk
273  if (i == crosswalk.getDetectionWaypoint())
274  {
275  // found an obstacle in the cross walk
276  if (crossWalkDetection(points, crosswalk, localizer_pose, points_threshold, obstacle_points) == EControl::STOP)
277  {
278  stop_obstacle_waypoint = i;
279  *obstacle_type = EObstacleType::ON_CROSSWALK;
280  break;
281  }
282  }
283 
284  // waypoint seen by localizer
285  geometry_msgs::Point waypoint = calcRelativeCoordinate(lane.waypoints[i].pose.pose.position, localizer_pose.pose);
286  tf::Vector3 tf_waypoint = point2vector(waypoint);
287  tf_waypoint.setZ(0);
288 
289  int stop_point_count = 0;
290  for (const auto& p : points)
291  {
292  tf::Vector3 point_vector(p.x, p.y, 0);
293 
294  // 2D distance between waypoint and points (obstacle)
295  double dt = tf::tfDistance(point_vector, tf_waypoint);
296  if (dt < stop_range)
297  {
298  stop_point_count++;
299  geometry_msgs::Point point_temp;
300  point_temp.x = p.x;
301  point_temp.y = p.y;
302  point_temp.z = p.z;
303  obstacle_points->setStopPoint(calcAbsoluteCoordinate(point_temp, localizer_pose.pose));
304  }
305  }
306 
307  // there is an obstacle if the number of points exceeded the threshold
308  if (stop_point_count > points_threshold)
309  {
310  stop_obstacle_waypoint = i;
311  *obstacle_type = EObstacleType::ON_WAYPOINTS;
312  break;
313  }
314 
315  obstacle_points->clearStopPoints();
316 
317  // check next waypoint...
318  }
319 
320  return stop_obstacle_waypoint;
321 }

323 int detectDecelerateObstacle(const pcl::PointCloud<pcl::PointXYZ>& points, const int closest_waypoint,
324  const autoware_msgs::Lane& lane, const double stop_range, const double deceleration_range,
325  const double points_threshold, const geometry_msgs::PoseStamped& localizer_pose,
326  ObstaclePoints* obstacle_points)
327 {
328  int decelerate_obstacle_waypoint = -1;
329  // start search from the closest waypoint
330  for (int i = closest_waypoint; i < closest_waypoint + DECELERATION_SEARCH_DISTANCE; i++)
331  {
332  // reach the end of waypoints
333  if (i >= static_cast<int>(lane.waypoints.size()))
334  break;
335 
336  // waypoint seen by localizer
337  geometry_msgs::Point waypoint = calcRelativeCoordinate(lane.waypoints[i].pose.pose.position, localizer_pose.pose);
338  tf::Vector3 tf_waypoint = point2vector(waypoint);
339  tf_waypoint.setZ(0);
340 
341  int decelerate_point_count = 0;
342  for (const auto& p : points)
343  {
344  tf::Vector3 point_vector(p.x, p.y, 0);
345 
346  // 2D distance between waypoint and points (obstacle)
347  double dt = tf::tfDistance(point_vector, tf_waypoint);
348  if (dt > stop_range && dt < stop_range + deceleration_range)
349  {
350  decelerate_point_count++;
351  geometry_msgs::Point point_temp;
352  point_temp.x = p.x;
353  point_temp.y = p.y;
354  point_temp.z = p.z;
355  obstacle_points->setDeceleratePoint(calcAbsoluteCoordinate(point_temp, localizer_pose.pose));
356  }
357  }
358 
359  // there is an obstacle if the number of points exceeded the threshold
360  if (decelerate_point_count > points_threshold)
361  {
362  decelerate_obstacle_waypoint = i;
363  break;
364  }
365 
366  obstacle_points->clearDeceleratePoints();
367 
368  // check next waypoint...
369  }
370 
371  return decelerate_obstacle_waypoint;
372 }

374 // Detect an obstacle by using pointcloud
375 EControl pointsDetection(const pcl::PointCloud<pcl::PointXYZ>& points, const int closest_waypoint,
376  const autoware_msgs::Lane& lane, const CrossWalk& crosswalk, const VelocitySetInfo& vs_info,
377  int* obstacle_waypoint, ObstaclePoints* obstacle_points)
378 {
379  // no input for detection || no closest waypoint
380  if ((points.empty() == true && vs_info.getDetectionResultByOtherNodes() == -1) || closest_waypoint < 0)
381  return EControl::KEEP;
382 
383  EObstacleType obstacle_type = EObstacleType::NONE;
384  int stop_obstacle_waypoint =
385  detectStopObstacle(points, closest_waypoint, lane, crosswalk, vs_info.getStopRange(),
386  vs_info.getPointsThreshold(), vs_info.getLocalizerPose(),
387  obstacle_points, &obstacle_type, vs_info.getDetectionResultByOtherNodes());
388 
389  // skip searching deceleration range
390  if (vs_info.getDecelerationRange() < 0.01)
391  {
392  *obstacle_waypoint = stop_obstacle_waypoint;
393  if (stop_obstacle_waypoint < 0)
394  return EControl::KEEP;
395  else if (obstacle_type == EObstacleType::ON_WAYPOINTS || obstacle_type == EObstacleType::ON_CROSSWALK)
396  return EControl::STOP;
397  else if (obstacle_type == EObstacleType::STOPLINE)
398  return EControl::STOPLINE;
399  else
400  return EControl::OTHERS;
401  }
402 
403  int decelerate_obstacle_waypoint =
404  detectDecelerateObstacle(points, closest_waypoint, lane, vs_info.getStopRange(), vs_info.getDecelerationRange(),
405  vs_info.getPointsThreshold(), vs_info.getLocalizerPose(), obstacle_points);
406 
407  // stop obstacle was not found
408  if (stop_obstacle_waypoint < 0)
409  {
410  *obstacle_waypoint = decelerate_obstacle_waypoint;
411  return decelerate_obstacle_waypoint < 0 ? EControl::KEEP : EControl::DECELERATE;
412  }
413 
414  // stop obstacle was found but decelerate obstacle was not found
415  if (decelerate_obstacle_waypoint < 0)
416  {
417  *obstacle_waypoint = stop_obstacle_waypoint;
418  return EControl::STOP;
419  }
420 
421  // about 5.0 meter
422  double waypoint_interval =
423  getPlaneDistance(lane.waypoints[0].pose.pose.position, lane.waypoints[1].pose.pose.position);
424  int stop_decelerate_threshold = 5 / waypoint_interval;
425 
426  // both were found
427  if (stop_obstacle_waypoint - decelerate_obstacle_waypoint > stop_decelerate_threshold)
428  {
429  *obstacle_waypoint = decelerate_obstacle_waypoint;
430  return EControl::DECELERATE;
431  }
432  else
433  {
434  *obstacle_waypoint = stop_obstacle_waypoint;
435  return EControl::STOP;
436  }
437 }

439 EControl obstacleDetection(int closest_waypoint, const autoware_msgs::Lane& lane, const CrossWalk& crosswalk,
440  const VelocitySetInfo vs_info, const ros::Publisher& detection_range_pub,
441  const ros::Publisher& obstacle_pub, int* obstacle_waypoint)
442 {
443  ObstaclePoints obstacle_points;
444  EControl detection_result = pointsDetection(vs_info.getPoints(), closest_waypoint, lane, crosswalk, vs_info,
445  obstacle_waypoint, &obstacle_points);
446  displayDetectionRange(lane, crosswalk, closest_waypoint, detection_result, *obstacle_waypoint, vs_info.getStopRange(),
447  vs_info.getDecelerationRange(), detection_range_pub);
448 
449  static int false_count = 0;
450  static EControl prev_detection = EControl::KEEP;
451  static int prev_obstacle_waypoint = -1;
452 
453  // stop or decelerate because we found obstacles
454  if (detection_result == EControl::STOP || detection_result == EControl::STOPLINE || detection_result == EControl::DECELERATE)
455  {
456  displayObstacle(detection_result, obstacle_points, obstacle_pub);
457  prev_detection = detection_result;
458  false_count = 0;
459  prev_obstacle_waypoint = *obstacle_waypoint;
460  return detection_result;
461  }
462 
463  // there are no obstacles, but wait a little for safety
464  if (prev_detection == EControl::STOP || prev_detection == EControl::STOPLINE || prev_detection == EControl::DECELERATE)
465  {
466  false_count++;
467 
468  if (false_count < LOOP_RATE / 2)
469  {
470  *obstacle_waypoint = prev_obstacle_waypoint;
471  displayObstacle(EControl::OTHERS, obstacle_points, obstacle_pub);
472  return prev_detection;
473  }
474  }
475 
476  // there are no obstacles, so we move forward
477  *obstacle_waypoint = -1;
478  false_count = 0;
479  prev_detection = EControl::KEEP;
480  return detection_result;
481 }

483 void changeWaypoints(const VelocitySetInfo& vs_info, const EControl& detection_result, int closest_waypoint,
484  int obstacle_waypoint, const ros::Publisher& final_waypoints_pub, VelocitySetPath* vs_path)
485 {
486  if (detection_result == EControl::STOP || detection_result == EControl::STOPLINE)
487  { // STOP for obstacle/stopline
488  // stop_waypoint is about stop_distance meter away from obstacles/stoplines
489  int stop_distance = (detection_result == EControl::STOP)
490  ? vs_info.getStopDistanceObstacle() : vs_info.getStopDistanceStopline();
491  double deceleration = (detection_result == EControl::STOP)
492  ? vs_info.getDecelerationObstacle() : vs_info.getDecelerationStopline();
493  int stop_waypoint =
494  calcWaypointIndexReverse(vs_path->getPrevWaypoints(), obstacle_waypoint, stop_distance);
495  // change waypoints to stop by the stop_waypoint
496  vs_path->changeWaypointsForStopping(stop_waypoint, obstacle_waypoint, closest_waypoint, deceleration);
497  vs_path->avoidSuddenAcceleration(deceleration, closest_waypoint);
498  vs_path->avoidSuddenDeceleration(vs_info.getVelocityChangeLimit(), deceleration, closest_waypoint);
499  vs_path->setTemporalWaypoints(vs_info.getTemporalWaypointsSize(), closest_waypoint, vs_info.getControlPose());
500  final_waypoints_pub.publish(vs_path->getTemporalWaypoints());
501  }
502  else if (detection_result == EControl::DECELERATE)
503  { // DECELERATE for obstacles
504  vs_path->initializeNewWaypoints();
505  vs_path->changeWaypointsForDeceleration(vs_info.getDecelerationObstacle(), closest_waypoint, obstacle_waypoint);
506  vs_path->avoidSuddenDeceleration(vs_info.getVelocityChangeLimit(), vs_info.getDecelerationObstacle(), closest_waypoint);
507  vs_path->avoidSuddenAcceleration(vs_info.getDecelerationObstacle(), closest_waypoint);
508  vs_path->setTemporalWaypoints(vs_info.getTemporalWaypointsSize(), closest_waypoint, vs_info.getControlPose());
509  final_waypoints_pub.publish(vs_path->getTemporalWaypoints());
510  }
511  else
512  { // ACCELERATE or KEEP
513  vs_path->initializeNewWaypoints();
514  vs_path->avoidSuddenAcceleration(vs_info.getDecelerationObstacle(), closest_waypoint);
515  vs_path->avoidSuddenDeceleration(vs_info.getVelocityChangeLimit(), vs_info.getDecelerationObstacle(), closest_waypoint);
516  vs_path->setTemporalWaypoints(vs_info.getTemporalWaypointsSize(), closest_waypoint, vs_info.getControlPose());
517  final_waypoints_pub.publish(vs_path->getTemporalWaypoints());
518  }
519 }
