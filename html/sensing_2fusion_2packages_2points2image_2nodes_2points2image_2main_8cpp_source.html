48 static POINTS2IMAGEPtr points_to_image(sensor_msgs::PointCloud2ConstPtr velodyneData, cv::Mat cameraExtrinsicMat,
49  cv::Mat cameraMat, cv::Mat distCoeff, cv::Size imageSize)
50 {
51  POINTS2IMAGEPtr result(new POINTS2IMAGE);
52  result->header = velodyneData->header;
53  result->height = imageSize.height;
54  result->width = imageSize.width;
55  POINT2IMAGE tmpdata;
56  tmpdata.intensity = 0;
57  tmpdata.distance = 0;
58  result->points.resize(result->height * result->width, tmpdata);
59 
60  cv::Mat invR = cameraExtrinsicMat(cv::Rect(0, 0, 3, 3)).t();
61  cv::Mat invT = -invR * (cameraExtrinsicMat(cv::Rect(3, 0, 1, 3)));
62 
63  int n = velodyneData->height;
64  int m = velodyneData->width;
65  char* data = (char*)(velodyneData->data.data());
66  for (int i = 0; i < n; i++)
67  {
68  for (int j = 0; j < m; j++)
69  {
70  int id = i * m + j;
71  float* dataptr = (float*)(data + id * velodyneData->point_step);
72 
73  cv::Mat point(1, 3, CV_64F);
74  point.at<double>(0) = double(dataptr[0]);
75  point.at<double>(1) = double(dataptr[1]);
76  point.at<double>(2) = double(dataptr[2]);
77  point = point * invR.t() + invT.t();
78 
79  if (point.at<double>(2) <= 0)
80  {
81  continue;
82  }
83 
84  double tmpx = point.at<double>(0) / point.at<double>(2);
85  double tmpy = point.at<double>(1) / point.at<double>(2);
86  double r2 = tmpx * tmpx + tmpy * tmpy;
87  double tmpdist =
88  1 + distCoeff.at<double>(0) * r2 + distCoeff.at<double>(1) * r2 * r2 + distCoeff.at<double>(4) * r2 * r2 * r2;
89 
90  cv::Point2d imagepoint;
91  imagepoint.x =
92  tmpx * tmpdist + 2 * distCoeff.at<double>(2) * tmpx * tmpy + distCoeff.at<double>(3) * (r2 + 2 * tmpx * tmpx);
93  imagepoint.y =
94  tmpy * tmpdist + distCoeff.at<double>(2) * (r2 + 2 * tmpy * tmpy) + 2 * distCoeff.at<double>(3) * tmpx * tmpy;
95  imagepoint.x = cameraMat.at<double>(0, 0) * imagepoint.x + cameraMat.at<double>(0, 2);
96  imagepoint.y = cameraMat.at<double>(1, 1) * imagepoint.y + cameraMat.at<double>(1, 2);
97  if (imagepoint.x >= 0 && imagepoint.x < result->width && imagepoint.y >= 0 && imagepoint.y < result->height)
98  {
99  int px = int(imagepoint.x + 0.5);
100  int py = int(imagepoint.y + 0.5);
101  int pid = py * result->width + px;
102  if (result->points[pid].distance == 0 || (result->points[pid].distance / 100.0) > point.at<double>(2))
103  {
104  result->points[pid].distance = ushort(point.at<double>(2) * 100 + 0.5);
105  result->points[pid].intensity = uchar(dataptr[4]);
106  }
107  }
108  }
109  }
110  return result;
111 }
