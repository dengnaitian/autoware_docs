175  void LocalPlannerH::LocalizeMe(const double& dt)
176 {
177  //calculate the new x, y ,
178  WayPoint currPose = state;
179 
180  if(m_CurrentShift == SHIFT_POS_DD)
181  {
182  m_OdometryState.pos.x += m_CurrentVelocity * dt * cos(currPose.pos.a);
183  m_OdometryState.pos.y += m_CurrentVelocity * dt * sin(currPose.pos.a);
184  m_OdometryState.pos.a += m_CurrentVelocity * dt * tan(m_CurrentSteering) / m_CarInfo.wheel_base;
185 
186  }
187  else if(m_CurrentShift == SHIFT_POS_RR )
188  {
189  m_OdometryState.pos.x += -m_CurrentVelocity * dt * cos(currPose.pos.a);
190  m_OdometryState.pos.y += -m_CurrentVelocity * dt * sin(currPose.pos.a);
191  m_OdometryState.pos.a += -m_CurrentVelocity * dt * tan(m_CurrentSteering);
192  }
193 
194  m_OdometryState.pos.a = atan2(sin(m_OdometryState.pos.a), cos(m_OdometryState.pos.a));
195  m_OdometryState.pos.a = UtilityH::FixNegativeAngle(m_OdometryState.pos.a);
196 
197  state.pos.a = m_OdometryState.pos.a;
198  state.pos.x = m_OdometryState.pos.x - (m_CurrentVelocity*dt* (m_CarInfo.wheel_base) * cos (m_OdometryState.pos.a));
199  state.pos.y = m_OdometryState.pos.y - (m_CurrentVelocity*dt* (m_CarInfo.wheel_base/2.0) * sin (m_OdometryState.pos.a));

620  double LocalPlannerH::UpdateVelocityDirectlyToTrajectory(const BehaviorState& beh, const VehicleState& CurrStatus, const double& dt)
621  {
622  if(m_TotalOriginalPath.size() ==0 ) return 0;
623 
624  RelativeInfo info, total_info;
625  PlanningHelpers::GetRelativeInfo(m_TotalOriginalPath.at(m_iCurrentTotalPathId), state, total_info);
626  PlanningHelpers::GetRelativeInfo(m_Path, state, info);
627  double average_braking_distance = -pow(CurrStatus.speed, 2)/(m_CarInfo.max_deceleration) + m_params.additionalBrakingDistance;
628  double max_velocity = PlannerHNS::PlanningHelpers::GetVelocityAhead(m_TotalOriginalPath.at(m_iCurrentTotalPathId), total_info, m_PrevBrakingWayPoint, average_braking_distance);
629 
630  unsigned int point_index = 0;
631  double critical_long_front_distance = 2.0;
632 
633  if(m_Path.size() <= 5)
634  {
635  double target_velocity = 0;
636  for(unsigned int i = 0; i < m_Path.size(); i++)
637  m_Path.at(i).v = target_velocity;
638  }
639  else if(beh.state == TRAFFIC_LIGHT_STOP_STATE || beh.state == STOP_SIGN_STOP_STATE || beh.state == STOP_SIGN_WAIT_STATE || beh.state == TRAFFIC_LIGHT_WAIT_STATE)
640  {
641  PlanningHelpers::GetFollowPointOnTrajectory(m_Path, info, beh.stopDistance - critical_long_front_distance, point_index);
642 
643  double e = -beh.stopDistance;
644  double desiredVelocity = m_pidStopping.getPID(e);
645 
646  for(unsigned int i = info.iBack; i < point_index; i++)
647  {
648  if(i < m_Path.size() && i >= 0)
649  m_Path.at(i).v = desiredVelocity;
650  }
651 
652  return desiredVelocity;
653  }
654  else if(beh.state == FOLLOW_STATE)
655  {
656  double targe_acceleration = -pow(CurrStatus.speed, 2)/(2.0*(beh.followDistance - critical_long_front_distance));
657  if(targe_acceleration <= 0 && targe_acceleration > m_CarInfo.max_deceleration/2.0)
658  {
659  double target_velocity = (targe_acceleration * dt) + CurrStatus.speed;
660 
661  double e = target_velocity - CurrStatus.speed;
662  double desiredVelocity = m_pidVelocity.getPID(e);
663 
664  for(unsigned int i = info.iBack; i < m_Path.size(); i++)
665  {
666  if(i < m_Path.size() && i >= 0)
667  m_Path.at(i).v = desiredVelocity;
668  }
669 
670  return desiredVelocity;
671  //cout << "Accelerate -> Target V: " << target_velocity << ", Brake D: " << average_braking_distance << ", Acceleration: " << targe_acceleration << endl;
672  }
673  else
674  {
675  WayPoint pursuite_point = PlanningHelpers::GetFollowPointOnTrajectory(m_Path, info, beh.followDistance - critical_long_front_distance, point_index);
676 
677  double e = beh.followDistance - m_params.minFollowingDistance;
678  double desiredVelocity = m_pidStopping.getPID(e);
679 
680  for(unsigned int i = info.iBack; i < point_index; i++)
681  {
682  if(i < m_Path.size() && i >= 0)
683  m_Path.at(i).v = desiredVelocity;
684  }
685 
686  return desiredVelocity;
687  //cout << "Decelerate -> Target V: " << target_velocity << ", Brake D: " << average_braking_distance << ", Start I" << info.iBack << endl;
688  }
689 
690  }
691  else if(beh.state == FORWARD_STATE || beh.state == OBSTACLE_AVOIDANCE_STATE )
692  {
693  double target_velocity = max_velocity;
694  if(m_pCurrentBehaviorState->GetCalcParams()->iCurrSafeTrajectory != m_pCurrentBehaviorState->GetCalcParams()->iCentralTrajectory)
695  target_velocity*=AVOIDANCE_SPEED_FACTOR;
696 
697  double e = target_velocity - CurrStatus.speed;
698  double desiredVelocity = m_pidVelocity.getPID(e);
699 
700  for(unsigned int i = info.iBack; i < m_Path.size(); i++)
701  {
702  m_Path.at(i).v = desiredVelocity;
703  }
704 
705  return desiredVelocity;
706  }
707  else
708  {
709  double target_velocity = 0;
710  for(unsigned int i = info.iBack; i < m_Path.size(); i++)
711  m_Path.at(i).v = target_velocity;
712 
713  return target_velocity;
714  }
715 
716  return max_velocity;
717  }
