216 void WaypointVelocityVisualizer::controlCallback(const geometry_msgs::PoseStamped::ConstPtr& current_pose_msg,
217  const geometry_msgs::TwistStamped::ConstPtr& current_twist_msg,
218  const geometry_msgs::TwistStamped::ConstPtr& command_twist_msg)
219 {
220  // buffers are reset when time goes back, e.g. playback rosbag
221  ros::Time current_time = ros::Time::now();
222  if (previous_time_ > current_time)
223  {
224  ROS_WARN("Detected jump back in time of %.3fs. Clearing markers and buffers.",
225  (previous_time_ - current_time).toSec());
226  deleteMarkers(); // call 'DELETEALL'
227  resetBuffers(); // clear circular buffers
228  }
229  previous_time_ = current_time;
230  // if plot_metric_interval <= 0, velocity is plotted by each callback.
231  if (plot_metric_interval_ > 0 && current_pose_buf_.size() > 0)
232  {
233  tf::Vector3 p1, p2;
234  tf::pointMsgToTF(current_pose_buf_.back().pose.position, p1);
235  tf::pointMsgToTF(current_pose_msg->pose.position, p2);
236  if (!(p1.distance(p2) > plot_metric_interval_))
237  return; // skipping plot
238  }
239  current_pose_buf_.push_back(*current_pose_msg);
240  current_twist_buf_.push_back(*current_twist_msg);
241  command_twist_buf_.push_back(*command_twist_msg);
242  current_twist_marker_array_.markers.clear();
243  command_twist_marker_array_.markers.clear();
244  createVelocityMarker(current_pose_buf_, current_twist_buf_, "current_velocity", current_twist_color_,
245  current_twist_marker_array_);
246  createVelocityMarker(current_pose_buf_, command_twist_buf_, "twist_cmd", command_twist_color_,
247  command_twist_marker_array_);
248  publishVelocityMarker();
249 }

313 void WaypointVelocityVisualizer::createVelocityBarMarker(const std::vector<nav_msgs::Odometry>& waypoints,
314  const std::string& ns, const std_msgs::ColorRGBA& color,
315  visualization_msgs::MarkerArray& markers)
316 {
317  visualization_msgs::Marker marker;
318  marker.header.frame_id = "map";
319  marker.header.stamp = ros::Time();
320  marker.ns = ns + "/bar";
321  marker.type = visualization_msgs::Marker::CYLINDER;
322  marker.action = visualization_msgs::Marker::ADD;
323  marker.scale.x = 0.1;
324  marker.scale.y = 0.1;
325  marker.color = color;
326 
327  unsigned int count = 0;
328  for (const auto& wp : waypoints)
329  {
330  double h = plot_height_ratio_ * wp.twist.twist.linear.x;
331  marker.id = count++;
332  marker.pose = wp.pose.pose;
333  marker.pose.position.z += h / 2.0;
334  // When the the cylinder height is 0 or less, a warning occurs in RViz.
335  marker.scale.z = fabs(h) + 1e-6;
336  markers.markers.push_back(marker);
337  }
338 }
