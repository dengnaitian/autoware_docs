70  ~CameraNodelet()
71  {
72  // signal gmsl camera to stop before nodelet exit
73  // note: we are not using a signal handler: just using destructor to stop camera(s)
74  // on the programme termination
75  if(camera && running)
76  {
77  try
78  {
79  NODELET_INFO("shutting down camera thread");
80  running = false;
81  camera->shutdown();
82  delete camera;
83  NODELET_INFO("camera stopped");
84  }
85  catch(std::runtime_error& e)
86  {
87  NODELET_ERROR("%s", e.what());
88  }
89 
90  }
91  }
