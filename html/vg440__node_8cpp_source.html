73 * ARGUMENTS: queue_ptr: is pointer to queue to process

78 int process_xbow_packet(QUEUE_TYPE *queue_ptr, XBOW_PACKET *result)
79 {
80  unsigned short myCRC = 0, packetCRC = 0, numToPop=0, counter=0;
81 // unsigned short packet_type = 0;
82 // char packet[100], tempchar, dataLength;
83  char dataLength;
84  if(Empty(queue_ptr))
85  {
86  return 0; /* empty buffer */
87  }
88  /* find header */
89  for(numToPop=0; numToPop+1<Size(queue_ptr) ;numToPop+=1)
90  {
91  if(0x5555==peekWord(queue_ptr, numToPop)) break;
92  }
93  Pop(queue_ptr, numToPop);
94  if(Size(queue_ptr) <= 0)
95  {
96  /* header was not found */
97  return 0;
98  }
99  /* make sure we can read through minimum length packet */
100  if(Size(queue_ptr)<7)
101  {
102  return 0;
103  }
104  /* get data length (5th byte of packet) */
105  dataLength = peekByte(queue_ptr, 4);
106  /* make sure we can read through entire packet */
107  if(Size(queue_ptr) < 7+dataLength)
108  {
109  return 0;
110  }
111  /* check CRC */
112  myCRC = calcCRC(queue_ptr, 2,dataLength+3);
113  packetCRC = peekWord(queue_ptr, dataLength+5);
114  if(myCRC != packetCRC)
115  {
116  Pop(queue_ptr, dataLength+7);
117  return 0;
118  }
119  /* fill out result of parsing in structure */
120  result->packet_type = peekWord(queue_ptr, 2);
121  result->length = peekByte(queue_ptr, 4);
122  result->crc = packetCRC;
123  for(counter=0; counter < result->length; counter++)
124  {
125  result->data[counter] = peekByte(queue_ptr, 5+counter);
126  }
127  Pop(queue_ptr, dataLength+7);
128  return 1;

135 * ARGUMENTS: queue_ptr is pointer to queue holding area to be CRCed

141 *******************************************************************************/
142 unsigned short calcCRC(QUEUE_TYPE *queue_ptr, unsigned int startIndex, unsigned int num) {
143  unsigned int i=0, j=0;

145  for (i=0; i<num; i+=1) {
