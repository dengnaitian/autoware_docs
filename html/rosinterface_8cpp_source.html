203 void ROSTFSub::receiveTFSlot()
204 {
205  if(ros::ok()&&receiveflag)
206  {
207  lock.lockForWrite();
208  tf::StampedTransform transform;
209  try
210  {
211  listener.lookupTransform(destinationframe.toStdString(),originalframe.toStdString(),ros::Time(0),transform);
212  }
213  catch(tf::TransformException & ex)
214  {
215  //qDebug()<<QString(ex.what());
216  lock.unlock();
217  return;
218  }
219  if(!lastflag||lasttf.stamp_.sec!=transform.stamp_.sec||lasttf.stamp_.nsec!=transform.stamp_.nsec)
220  {
221  tfs.push_back(transform);
222  lasttf=transform;
223  lastflag=1;
224  lock.unlock();
225  emit receiveTFSignal();
226  }
227  else
228  {
229  lock.unlock();
230  }
231  }
232 }
