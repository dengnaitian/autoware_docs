49 void tfRegistration(const cv::Mat &camExtMat, const ros::Time &timeStamp)
50 {
51  tf::Matrix3x3 rotation_mat;
52  double roll = 0, pitch = 0, yaw = 0;
53  tf::Quaternion quaternion;
54  tf::Transform transform;
55  static tf::TransformBroadcaster broadcaster;
56 
57  rotation_mat.setValue(camExtMat.at<double>(0, 0), camExtMat.at<double>(0, 1), camExtMat.at<double>(0, 2),
58  camExtMat.at<double>(1, 0), camExtMat.at<double>(1, 1), camExtMat.at<double>(1, 2),
59  camExtMat.at<double>(2, 0), camExtMat.at<double>(2, 1), camExtMat.at<double>(2, 2));
60 
61  rotation_mat.getRPY(roll, pitch, yaw, 1);
62 
63  quaternion.setRPY(roll, pitch, yaw);
64 
65  transform.setOrigin(
66  tf::Vector3(camExtMat.at<double>(0, 3), camExtMat.at<double>(1, 3), camExtMat.at<double>(2, 3)));
67 
68  transform.setRotation(quaternion);
69 
70  broadcaster.sendTransform(tf::StampedTransform(transform, timeStamp, target_frame_, camera_frame_));
71 }
