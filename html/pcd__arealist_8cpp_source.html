71 int calc_area(const std::string& path, struct Area *area)
72 {
73  pcl::PointCloud<pcl::PointXYZRGB> pcd;
74 
75  if (pcl::io::loadPCDFile(path.c_str(), pcd) == -1) {
76  std::cerr << "load failed " << path << std::endl;
77  return -1;
78  }
79  struct Area a;
80  a.path = path;
81 
82  pcl::PointCloud<pcl::PointXYZRGB>::iterator it = pcd.begin();
83  pcl::PointCloud<pcl::PointXYZRGB>::iterator end = pcd.end();
84  a.x_min = a.x_max = it->x;
85  a.y_min = a.y_max = it->y;
86  a.z_min = a.z_max = it->z;
87 
88  for (it++; it != end; it++) {
89  double x = it->x;
90  double y = it->y;
91  double z = it->z;
92  if (x < a.x_min) a.x_min = x;
93  if (x > a.x_max) a.x_max = x;
94  if (y < a.y_min) a.y_min = y;
95  if (y > a.y_max) a.y_max = y;
96  if (z < a.z_min) a.z_min = z;
97  if (z > a.z_max) a.z_max = z;
98  }
99  *area = a;
100  return 0;
101 }
