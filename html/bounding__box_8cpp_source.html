34 bool BoundingBoxModel::estimate(const pcl::PointCloud<pcl::PointXYZ>& cluster, autoware_msgs::DetectedObject& output)
35 {
36  // calc centroid point for cylinder height(z)
37  pcl::PointXYZ centroid;
38  centroid.x = 0;
39  centroid.y = 0;
40  centroid.z = 0;
41  for (const auto& pcl_point : cluster)
42  {
43  centroid.x += pcl_point.x;
44  centroid.y += pcl_point.y;
45  centroid.z += pcl_point.z;
46  }
47  centroid.x = centroid.x / (double)cluster.size();
48  centroid.y = centroid.y / (double)cluster.size();
49  centroid.z = centroid.z / (double)cluster.size();
50 
51  // calc min and max z for cylinder length
52  double min_z = 0;
53  double max_z = 0;
54  for (size_t i = 0; i < cluster.size(); ++i)
55  {
56  if (cluster.at(i).z < min_z || i == 0)
57  min_z = cluster.at(i).z;
58  if (max_z < cluster.at(i).z || i == 0)
59  max_z = cluster.at(i).z;
60  }
61 
62  // calc circumscribed circle on x-y plane
63  cv::Mat_<float> cv_points((int)cluster.size(), 2);
64  for (size_t i = 0; i < cluster.size(); ++i)
65  {
66  cv_points(i, 0) = cluster.at(i).x; // x
67  cv_points(i, 1) = cluster.at(i).y; // y
68  }

70  /*
71  * Paper : IV2017, Efficient L-Shape Fitting for Vehicle Detection Using Laser Scanners
72  * Authors : Xio Zhang, Wenda Xu, Chiyu Dong and John M. Dolan
73  */
74 
75  // Paper : Algo.2 Search-Based Rectangle Fitting
76  std::vector<std::pair<double /*theta*/, double /*q*/>> Q;
77  const double max_angle = M_PI / 2.0;
78  const double angle_reso = M_PI / 180.0;
79  for (double theta = 0; theta < max_angle; theta += angle_reso)
80  {
81  Eigen::Vector2d e_1;
82  e_1 << std::cos(theta), std::sin(theta); // col.3, Algo.2
83  Eigen::Vector2d e_2;
84  e_2 << -std::sin(theta), std::cos(theta); // col.4, Algo.2
85  std::vector<double> C_1; // col.5, Algo.2
86  std::vector<double> C_2; // col.6, Algo.2
87  for (const auto& point : cluster)
88  {
89  C_1.push_back(point.x * e_1.x() + point.y * e_1.y());
90  C_2.push_back(point.x * e_2.x() + point.y * e_2.y());
91  }
92  double q = calcClosenessCriterion(C_1, C_2); // col.7, Algo.2
93  Q.push_back(std::make_pair(theta, q)); // col.8, Algo.2
94  }
95 
96  double theta_star; // col.10, Algo.2
97  double max_q;
98  for (size_t i = 0; i < Q.size(); ++i)
99  {
100  if (max_q < Q.at(i).second || i == 0)
101  {
102  max_q = Q.at(i).second;
103  theta_star = Q.at(i).first;
104  }
105  }
106 
107  Eigen::Vector2d e_1_star; // col.11, Algo.2
108  Eigen::Vector2d e_2_star;
109  e_1_star << std::cos(theta_star), std::sin(theta_star);
110  e_2_star << -std::sin(theta_star), std::cos(theta_star);
111  std::vector<double> C_1_star; // col.11, Algo.2
112  std::vector<double> C_2_star; // col.11, Algo.2
113  for (const auto& point : cluster)
114  {
115  C_1_star.push_back(point.x * e_1_star.x() + point.y * e_1_star.y());
116  C_2_star.push_back(point.x * e_2_star.x() + point.y * e_2_star.y());
117  }
118 
119  // col.12, Algo.2
120  const double min_C_1_star = *std::min_element(C_1_star.begin(), C_1_star.end());
121  const double max_C_1_star = *std::max_element(C_1_star.begin(), C_1_star.end());
122  const double min_C_2_star = *std::min_element(C_2_star.begin(), C_2_star.end());
123  const double max_C_2_star = *std::max_element(C_2_star.begin(), C_2_star.end());
124 
125  const double a_1 = std::cos(theta_star);
126  const double b_1 = std::sin(theta_star);
127  const double c_1 = min_C_1_star;
128  const double a_2 = -1.0 * std::sin(theta_star);
129  const double b_2 = std::cos(theta_star);
130  const double c_2 = min_C_2_star;
131  const double a_3 = std::cos(theta_star);
132  const double b_3 = std::sin(theta_star);
133  const double c_3 = max_C_1_star;
134  const double a_4 = -1.0 * std::sin(theta_star);
135  const double b_4 = std::cos(theta_star);
136  const double c_4 = max_C_2_star;
137 
138  // calc center of bounding box
139  double intersection_x_1 = (b_1 * c_2 - b_2 * c_1) / (a_2 * b_1 - a_1 * b_2);
140  double intersection_y_1 = (a_1 * c_2 - a_2 * c_1) / (a_1 * b_2 - a_2 * b_1);
141  double intersection_x_2 = (b_3 * c_4 - b_4 * c_3) / (a_4 * b_3 - a_3 * b_4);
142  double intersection_y_2 = (a_3 * c_4 - a_4 * c_3) / (a_3 * b_4 - a_4 * b_3);
143 
144  // calc dimention of bounding box
145  Eigen::Vector2d e_x;
146  Eigen::Vector2d e_y;
147  e_x << a_1 / (std::sqrt(a_1 * a_1 + b_1 * b_1)), b_1 / (std::sqrt(a_1 * a_1 + b_1 * b_1));
148  e_y << a_2 / (std::sqrt(a_2 * a_2 + b_2 * b_2)), b_2 / (std::sqrt(a_2 * a_2 + b_2 * b_2));
149  Eigen::Vector2d diagonal_vec;
150  diagonal_vec << intersection_x_1 - intersection_x_2, intersection_y_1 - intersection_y_2;
151 
152  // calc yaw
153  tf2::Quaternion quat;
154  quat.setEuler(/* roll */ 0, /* pitch */ 0, /* yaw */ std::atan2(e_1_star.y(), e_1_star.x()));
155 
156  output.pose.position.x = (intersection_x_1 + intersection_x_2) / 2.0;
157  output.pose.position.y = (intersection_y_1 + intersection_y_2) / 2.0;
158  output.pose.position.z = centroid.z;
159  output.pose.orientation = tf2::toMsg(quat);
160  constexpr double ep = 0.001;
161  output.dimensions.x = std::fabs(e_x.dot(diagonal_vec));
162  output.dimensions.y = std::fabs(e_y.dot(diagonal_vec));
163  output.dimensions.z = std::max((max_z - min_z), ep);
164  output.pose_reliable = true;
165 
166  // check wrong output
167  if (output.dimensions.x < ep && output.dimensions.y < ep)
168  return false;
169  output.dimensions.x = std::max(output.dimensions.x, ep);
170  output.dimensions.y = std::max(output.dimensions.y, ep);
171  return true;
172 }

291 double BoundingBoxModel::calcClosenessCriterion(const std::vector<double>& C_1, const std::vector<double>& C_2)
292 {
293  // Paper : Algo.4 Closeness Criterion
294  const double min_c_1 = *std::min_element(C_1.begin(), C_1.end()); // col.2, Algo.4
295  const double max_c_1 = *std::max_element(C_1.begin(), C_1.end()); // col.2, Algo.4
296  const double min_c_2 = *std::min_element(C_2.begin(), C_2.end()); // col.3, Algo.4
297  const double max_c_2 = *std::max_element(C_2.begin(), C_2.end()); // col.3, Algo.4
298 
299  std::vector<double> D_1; // col.4, Algo.4
300  for (const auto& c_1_element : C_1)
301  {
302  const double v = std::min(max_c_1 - c_1_element, c_1_element - min_c_1);
303  D_1.push_back(std::fabs(v));
304  }
305 
306  std::vector<double> D_2; // col.5, Algo.4
307  for (const auto& c_2_element : C_2)
308  {
309  const double v = std::min(max_c_2 - c_2_element, c_2_element - min_c_2);
310  D_2.push_back(v * v);
311  }
312 
313  const double d_min = 0.05;
314  const double d_max = 0.50;
315  double beta = 0; // col.6, Algo.4
316  for (size_t i = 0; i < D_1.size(); ++i)
317  {
318  const double d = std::min(std::max(std::min(D_1.at(i), D_2.at(i)), d_min), d_max);
319  beta += 1.0 / d;
320  }
321  return beta;
322 }
