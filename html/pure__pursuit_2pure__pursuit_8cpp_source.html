64 bool PurePursuit::interpolateNextTarget(int next_waypoint, geometry_msgs::Point *next_target) const
65 {
66  constexpr double ERROR = pow(10, -5); // 0.00001
67 
68  int path_size = static_cast<int>(current_waypoints_.size());
69  if (next_waypoint == path_size - 1)
70  {
71  *next_target = current_waypoints_.at(next_waypoint).pose.pose.position;
72  return true;
73  }
74  double search_radius = lookahead_distance_;
75  geometry_msgs::Point zero_p;
76  geometry_msgs::Point end = current_waypoints_.at(next_waypoint).pose.pose.position;
77  geometry_msgs::Point start = current_waypoints_.at(next_waypoint - 1).pose.pose.position;
78 
79  // let the linear equation be "ax + by + c = 0"
80  // if there are two points (x1,y1) , (x2,y2), a = "y2-y1, b = "(-1) * x2 - x1" ,c = "(-1) * (y2-y1)x1 + (x2-x1)y1"
81  double a = 0;
82  double b = 0;
83  double c = 0;
84  double get_linear_flag = getLinearEquation(start, end, &a, &b, &c);
85  if (!get_linear_flag)
86  return false;
87 
88  // let the center of circle be "(x0,y0)", in my code , the center of circle is vehicle position
89  // the distance "d" between the foot of a perpendicular line and the center of circle is ...
90  // | a * x0 + b * y0 + c |
91  // d = -------------------------------
92  // √( a~2 + b~2)
93  double d = getDistanceBetweenLineAndPoint(current_pose_.position, a, b, c);
94 
95  // ROS_INFO("a : %lf ", a);
96  // ROS_INFO("b : %lf ", b);
97  // ROS_INFO("c : %lf ", c);
98  // ROS_INFO("distance : %lf ", d);
99 
100  if (d > search_radius)
101  return false;
102 
103  // unit vector of point 'start' to point 'end'
104  tf::Vector3 v((end.x - start.x), (end.y - start.y), 0);
105  tf::Vector3 unit_v = v.normalize();
106 
107  // normal unit vectors of v
108  tf::Vector3 unit_w1 = rotateUnitVector(unit_v, 90); // rotate to counter clockwise 90 degree
109  tf::Vector3 unit_w2 = rotateUnitVector(unit_v, -90); // rotate to counter clockwise 90 degree
110 
111  // the foot of a perpendicular line
112  geometry_msgs::Point h1;
113  h1.x = current_pose_.position.x + d * unit_w1.getX();
114  h1.y = current_pose_.position.y + d * unit_w1.getY();
115  h1.z = current_pose_.position.z;
116 
117  geometry_msgs::Point h2;
118  h2.x = current_pose_.position.x + d * unit_w2.getX();
119  h2.y = current_pose_.position.y + d * unit_w2.getY();
120  h2.z = current_pose_.position.z;
121 
122  // ROS_INFO("error : %lf", error);
123  // ROS_INFO("whether h1 on line : %lf", h1.y - (slope * h1.x + intercept));
124  // ROS_INFO("whether h2 on line : %lf", h2.y - (slope * h2.x + intercept));
125 
126  // check which of two foot of a perpendicular line is on the line equation
127  geometry_msgs::Point h;
128  if (fabs(a * h1.x + b * h1.y + c) < ERROR)
129  {
130  h = h1;
131  // ROS_INFO("use h1");
132  }
133  else if (fabs(a * h2.x + b * h2.y + c) < ERROR)
134  {
135  // ROS_INFO("use h2");
136  h = h2;
137  }
138  else
139  {
140  return false;
141  }
142 
143  // get intersection[s]
144  // if there is a intersection
145  if (d == search_radius)
146  {
147  *next_target = h;
148  return true;
149  }
150  else
151  {
152  // if there are two intersection
153  // get intersection in front of vehicle
154  double s = sqrt(pow(search_radius, 2) - pow(d, 2));
155  geometry_msgs::Point target1;
156  target1.x = h.x + s * unit_v.getX();
157  target1.y = h.y + s * unit_v.getY();
158  target1.z = current_pose_.position.z;
159 
160  geometry_msgs::Point target2;
161  target2.x = h.x - s * unit_v.getX();
162  target2.y = h.y - s * unit_v.getY();
163  target2.z = current_pose_.position.z;
164 
165  // ROS_INFO("target1 : ( %lf , %lf , %lf)", target1.x, target1.y, target1.z);
166  // ROS_INFO("target2 : ( %lf , %lf , %lf)", target2.x, target2.y, target2.z);
167  // displayLinePoint(a, b, c, target1, target2, h); // debug tool
168 
169  // check intersection is between end and start
170  double interval = getPlaneDistance(end, start);
171  if (getPlaneDistance(target1, end) < interval)
172  {
173  // ROS_INFO("result : target1");
174  *next_target = target1;
175  return true;
176  }
177  else if (getPlaneDistance(target2, end) < interval)
178  {
179  // ROS_INFO("result : target2");
180  *next_target = target2;
181  return true;
182  }
183  else
184  {
185  // ROS_INFO("result : false ");
186  return false;
187  }
188  }

196 void PurePursuit::getNextWaypoint()
197 {
198  int path_size = static_cast<int>(current_waypoints_.size());
199 
200  // if waypoints are not given, do nothing.
201  if (path_size == 0)
202  {
203  next_waypoint_number_ = -1;
204  return;
205  }
206 
207  // look for the next waypoint.
208  for (int i = 0; i < path_size; i++)
209  {
210  // if search waypoint is the last
211  if (i == (path_size - 1))
212  {
213  ROS_INFO("search waypoint is the last");
214  next_waypoint_number_ = i;
215  return;
216  }
217 
218  // if there exists an effective waypoint
219  if (getPlaneDistance(current_waypoints_.at(i).pose.pose.position, current_pose_.position) > lookahead_distance_)
220  {
221  next_waypoint_number_ = i;
222  return;
223  }
224  }
225 
226  // if this program reaches here , it means we lost the waypoint!
227  next_waypoint_number_ = -1;
228  return;
229 }

238 bool PurePursuit::canGetCurvature(double *output_kappa)
239 {
240  // search next waypoint
241  getNextWaypoint();
242  if (next_waypoint_number_ == -1)
243  {
244  ROS_INFO("lost next waypoint");
245  return false;
246  }
247  // check whether curvature is valid or not
248  bool is_valid_curve = false;
249  for (const auto &el : current_waypoints_)
250  {
251  if (getPlaneDistance(el.pose.pose.position, current_pose_.position) > minimum_lookahead_distance_)
252  {
253  is_valid_curve = true;
254  break;
255  }
256  }
257  if (!is_valid_curve)
258  {
259  return false;
260  }
261  // if is_linear_interpolation_ is false or next waypoint is first or last
262  if (!is_linear_interpolation_ || next_waypoint_number_ == 0 ||
263  next_waypoint_number_ == (static_cast<int>(current_waypoints_.size() - 1)))
264  {
265  next_target_position_ = current_waypoints_.at(next_waypoint_number_).pose.pose.position;
266  *output_kappa = calcCurvature(next_target_position_);
267  return true;
268  }
269 
270  // linear interpolation and calculate angular velocity
271  bool interpolation = interpolateNextTarget(next_waypoint_number_, &next_target_position_);
272 
273  if (!interpolation)
274  {
275  ROS_INFO_STREAM("lost target! ");
276  return false;
277  }
278 
279  // ROS_INFO("next_target : ( %lf , %lf , %lf)", next_target.x, next_target.y,next_target.z);
280 
281  *output_kappa = calcCurvature(next_target_position_);
282  return true;
283 }
