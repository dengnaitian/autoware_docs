188 static int getNextWaypoint(int closest_waypoint)
189 {
190  // if waypoints are not given, do nothing.
191  if (g_current_waypoints.getSize() == 0)
192  return -1;
193 
194  double lookahead_threshold = getLookAheadThreshold(closest_waypoint);
195 
196  //ROS_INFO_STREAM("threshold = " << lookahead_threshold);
197  // look for the next waypoint.
198  for (int i = closest_waypoint; i < g_current_waypoints.getSize(); i++)
199  {
200  //skip waypoint behind vehicle
201  if (calcRelativeCoordinate(g_current_waypoints.getCurrentWaypoints().waypoints[i].pose.pose.position,
202  g_current_pose.pose).x < 0)
203  continue;
204 
205  // if there exists an effective waypoint
206  if (getPlaneDistance(g_current_waypoints.getCurrentWaypoints().waypoints[i].pose.pose.position,
207  g_current_pose.pose.position) > lookahead_threshold)
208  return i;
209  }
210 
211  // if the program reaches here, it means we lost the waypoint.
212  return -1;
213 }

245 // Compute the goal state of the vehicle
247 static union State computeWaypointGoal(int next_waypoint)
248 {
249  union State l_goal;
250 
251  // Get the next waypoint position with respect to the vehicles frame
252  //l_goal.sx = _path_pp.transformWaypoint(next_waypoint).getX();
253  //l_goal.sy = _path_pp.transformWaypoint(next_waypoint).getY();
254 
255  l_goal.sx = calcRelativeCoordinate(g_current_waypoints.getWaypointPosition(next_waypoint),g_current_pose.pose).x;
256  l_goal.sy = calcRelativeCoordinate(g_current_waypoints.getWaypointPosition(next_waypoint),g_current_pose.pose).y;
257 
258  // Get the next next waypoint position
259  //double waypoint_lookahead_1_x = _path_pp.transformWaypoint(next_waypoint+1).getX();
260  // double waypoint_lookahead_1_y = _path_pp.transformWaypoint(next_waypoint+1).getY();
261  double waypoint_lookahead_1_x = calcRelativeCoordinate(g_current_waypoints.getWaypointPosition(next_waypoint+1),g_current_pose.pose).x;
262  double waypoint_lookahead_1_y = calcRelativeCoordinate(g_current_waypoints.getWaypointPosition(next_waypoint+1),g_current_pose.pose).y;
263 
264  // Get the next next next waypoint
265  // double waypoint_lookahead_2_x = _path_pp.transformWaypoint(next_waypoint+2).getX();
266  //double waypoint_lookahead_2_y = _path_pp.transformWaypoint(next_waypoint+2).getY();
267  double waypoint_lookahead_2_x = calcRelativeCoordinate(g_current_waypoints.getWaypointPosition(next_waypoint+2),g_current_pose.pose).x;
268  double waypoint_lookahead_2_y = calcRelativeCoordinate(g_current_waypoints.getWaypointPosition(next_waypoint+2),g_current_pose.pose).y;
269 
270  // Compute dX and dY relative to lookahead
271  double dX_1 = waypoint_lookahead_1_x - l_goal.sx;
272  double dY_1 = waypoint_lookahead_1_y - l_goal.sy;
273 
274  // Compute dX and dY relative to lookahead and next lookahead
275  double dX_2 = waypoint_lookahead_2_x - waypoint_lookahead_1_x;
276  double dY_2 = waypoint_lookahead_2_y - waypoint_lookahead_1_y;
277 
278  // Estimate desired orientation of the vehicle at next waypoint
279  l_goal.theta = atan(dY_1/dX_1);
280 
281  // Estimate the desired orientation at the next next waypoint
282  double theta_lookahead = atan(dY_2/dX_2);
283 
284  // Estimate the arc length
285  double ds = sqrt(pow(waypoint_lookahead_1_x - waypoint_lookahead_2_x, 2) + pow(waypoint_lookahead_1_x - waypoint_lookahead_2_y, 2));
286 
287  // Angle
288  double angle = (theta_lookahead-l_goal.theta)/2.00;
289 
290  // Estimate Kappa
291  double estimate = 2.00*sin(angle)/ds;
292 
293  l_goal.kappa = estimate;
294 
295  // Note we limit kappa from being too extreme
296  // 10.0 was arbitrary, we really need a better curvature estimate
297  l_goal.kappa = min((double)kmax/10.0, l_goal.kappa);
298 
299  l_goal.kappa = max ((double)kmin/10.0, l_goal.kappa);
300 
301  // Get the desired velocity at the closest waypoint
302  l_goal.v = g_current_waypoints.getWaypointVelocityMPS(next_waypoint);
303 
304  return l_goal;
305 }

310 static union State computeVeh(int old_time, double old_theta, int next_waypoint)
311 {
312  union State l_veh;
313 
314  // Goal is computed relative to vehicle coordinate frame
315  l_veh.sx=0.0;
316  l_veh.sy=0.0;
317 
318  // Compute yaw (orientation) relative to the world coordinate frame
319  // Necessary to compute curvature, but note that in the local coordinate frame yaw is still 0.0
320  tf::Quaternion q(g_current_pose.pose.orientation.x,g_current_pose.pose.orientation.y,g_current_pose.pose.orientation.z,g_current_pose.pose.orientation.w);
321  tf::Matrix3x3 m(q);
322  double roll,pitch,yaw;
323  m.getRPY(roll,pitch,yaw);
324 
325  // Because we are on the coordinate system of the base frame
326  l_veh.theta = 0.0;
327 
328  // Get the current velocity of the vehicle
329  l_veh.v = g_current_velocity;
330 
331  // Get the desired velocity
332  l_veh.vdes = g_current_waypoints.getWaypointVelocityMPS(next_waypoint);
333  ROS_INFO_STREAM("Desired Velocity: "<< l_veh.vdes);
334 
335  // Not using timing related stuff for now...
336  double w = g_current_angular_velocity;
337  ROS_INFO_STREAM("Current omega: " <<w);
338  l_veh.kappa = w/l_veh.vdes;
339 
340 
341  if (!g_sim_mode && g_prius_mode)
342  {
343  l_veh.kappa = g_can_info_curvature;
344  ROS_INFO_STREAM("Current kappa (prius): " <<l_veh.kappa);

349  l_veh.kappa = g_mkz_info_curvature;

359 
360  return l_veh;

364 // Compute trajectory
366 static union Spline waypointTrajectory(union State veh, union State goal, union Spline curvature, int next_waypoint)
367 {
368  curvature.success=TRUE;
369  bool convergence=FALSE;
370  int iteration = 0;
371  union State veh_next;
372  double dt = step_size;
373  veh.v=goal.v;
374 
375  // While loop for computing trajectory parameters
376  while(convergence == FALSE && iteration<4)
377  {
378  // Set time horizon
379  double horizon = curvature.s/veh.vdes;
380  ROS_INFO_STREAM("vdes: " << veh.vdes);
381  ROS_INFO_STREAM("horizon: " << horizon);
382 
383  // Run motion model
384  veh_next = motionModel(veh, goal, curvature, dt, horizon, 0);
385 
386  // Determine convergence criteria
387  convergence = checkConvergence(veh_next, goal);
388 
389  // If the motion model doesn't get us to the goal compute new parameters
390  if(convergence==FALSE)
391  {
392  // Update parameters
393  curvature = generateCorrection(veh, veh_next, goal, curvature, dt, horizon);
394  iteration++;
395 
396  // Escape route for poorly conditioned Jacobian
397  if(curvature.success==FALSE)
398  {
399  ROS_INFO_STREAM("Init State: sx "<<veh.sx<<" sy " <<veh.sy<<" theta "<<veh.theta<<" kappa "<<veh.kappa<<" v "<<veh.v);
400  ROS_INFO_STREAM("Goal State: sx "<<goal.sx<<" sy " <<goal.sy<<" theta "<<goal.theta<<" kappa "<<goal.kappa<<" v"<<goal.v);
401  break;
402  }
403  }
404  }
405 
406  if(convergence==FALSE)
407  {
408  ROS_INFO_STREAM("Next State: sx "<<veh_next.sx<<" sy " <<veh_next.sy<<" theta "<<veh_next.theta<<" kappa "<<veh_next.kappa<<" v "<<veh_next.v);
409  ROS_INFO_STREAM("Init State: sx "<<veh.sx<<" sy " <<veh.sy<<" theta "<<veh.theta<<" kappa "<<veh.kappa);
410  ROS_INFO_STREAM("Goal State: sx "<<goal.sx<<" sy " <<goal.sy<<" theta "<<goal.theta<<" kappa "<<goal.kappa);
411  curvature.success= FALSE;
412  }
413 
414  else
415  {
416  ROS_INFO_STREAM("Converged in "<<iteration<<" iterations");
417 
418  #ifdef LOG_OUTPUT
419  // Set time horizon
420  double horizon = curvature.s/v_0;
421  // Run motion model and log data for plotting
422  veh_next = motionModel(veh, goal, curvature, 0.1, horizon, 1);
423  fmm_sx<<"0.0 \n";
424  fmm_sy<<"0.0 \n";
425  #endif
426  }
427 
428  return curvature;
429 }
