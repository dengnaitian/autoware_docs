56 std::vector<int> createCostMap(const pcl::PointCloud<pcl::PointXYZ> &scan)
57 {
58  std::vector<int> cost_map(g_cell_width * g_cell_height, 0);
59  double map_center_x = (g_cell_width / 2.0) * g_resolution - g_offset_x;
60  double map_center_y = (g_cell_height / 2.0) * g_resolution - g_offset_y;
61 
62  // scan points are in sensor frame
63  for (const auto &p : scan.points)
64  {
65  if (p.z > HEIGHT_LIMIT)
66  continue;
67  if (std::fabs(p.x) < CAR_LENGTH && std::fabs(p.y) < CAR_WIDTH)
68  continue;
69 
70  // Calculate grid index
71  int grid_x = (p.x + map_center_x) / g_resolution;
72  int grid_y = (p.y + map_center_y) / g_resolution;
73  if (grid_x < 0 || grid_x >= g_cell_width || grid_y < 0 || grid_y >= g_cell_height)
74  continue;
75 
76  int index = g_cell_width * grid_y + grid_x;
77  cost_map[index] += 15;
78 
79  // Max cost value is 100
80  if (cost_map[index] > 100)
81  cost_map[index] = 100;
82  }
83 
84  return cost_map;
85 }

101 std::vector<int> filterCostMap(std::vector<int>& cost_map)
102 {
103  std::vector<int> filtered_cost_map(cost_map.size(), 0);
104 
105  // cells around reference (x, y)
106  std::vector<std::pair<int, int>> neighborhood
107  {
108  std::make_pair(-1, -1), std::make_pair( 0, -1), std::make_pair( 1, -1),
109  std::make_pair(-1, 0), std::make_pair( 1, 0),
110  std::make_pair(-1, 1), std::make_pair( 0, 1), std::make_pair( 1, 1),
111  };
112 
113  for (size_t size = cost_map.size(), i = 0; i < size; i++) {
114  int ref_cost = cost_map[i];
115 
116  int ref_x = i % g_cell_width;
117  int ref_y = (i - ref_x) / g_cell_width;
118 
119  // we don't have to filter if the cost is 0
120  if (ref_cost <= 0)
121  continue;
122 
123  filtered_cost_map[i] += ref_cost;
124 
125  // increase the cost for each neighborhood cell
126  for (const auto& n : neighborhood) {
127  int neighbor_x = ref_x + n.first;
128  int neighbor_y = ref_y + n.second;
129 
130  if (neighbor_x < 0 || neighbor_x >= g_cell_width || neighbor_y < 0 || neighbor_y >= g_cell_height)
131  continue;
132 
133  int neighbor_index = neighbor_x + neighbor_y * g_cell_width;
134  filtered_cost_map[neighbor_index] += ref_cost;
135  }
136 
137  }
138 
139  // handle the cost over 100
140  for (auto &cost : filtered_cost_map) {
141  if (cost > 100)
142  cost = 100;
143  }
144 
145  return filtered_cost_map;
146 }
