239 void SpinnakerCamera::publish_image(int index)
240 {
241  while (ros::ok())
242  {
243  try
244  {
245  ImagePtr pResultImage = pCamList_[index]->GetNextImage();
246  if (pResultImage->IsIncomplete())
247  {
248  ROS_WARN("[%s] [camera %d] Image incomplete with image status %d", __APP_NAME__, index,
249  pResultImage->GetImageStatus());
250  }
251  else
252  {
253  ros::Time receive_time = ros::Time::now();
254 
255  // create opencv mat
256  int pixel_format = pResultImage->GetPixelFormat();
257  std::string encoding_pattern;
258  switch (pixel_format)
259  {
260  case PixelFormatEnums::PixelFormat_BayerRG8:
261  encoding_pattern = "bayer_rggb8";
262  break;
263  case PixelFormatEnums::PixelFormat_BayerGR8:
264  encoding_pattern = "bayer_grbg8";
265  break;
266  case PixelFormatEnums::PixelFormat_BayerGB8:
267  encoding_pattern = "bayer_gbrg8";
268  break;
269  case PixelFormatEnums::PixelFormat_BayerBG8:
270  encoding_pattern = "bayer_bggr8";
271  break;
272  case PixelFormatEnums::PixelFormat_RGB8:
273  encoding_pattern = "rgb8";
274  break;
275  case PixelFormatEnums::PixelFormat_BGR8:
276  encoding_pattern = "bgr8";
277  break;
278  default:
279  encoding_pattern = "mono8";
280  }
281  unsigned int XPadding = pResultImage->GetXPadding();
282  unsigned int YPadding = pResultImage->GetYPadding();
283  unsigned int rowsize = pResultImage->GetWidth();
284  unsigned int colsize = pResultImage->GetHeight();
285 
286  // create and publish ros message
287  std_msgs::Header header;
288  sensor_msgs::Image msg;
289  header.stamp = receive_time;
290  header.frame_id = (camList_.GetSize() > 1) ? "camera" + std::to_string(index) : "camera";
291 
292  msg.header = header;
293  msg.height = pResultImage->GetHeight();
294  msg.width = pResultImage->GetWidth();
295  msg.encoding = encoding_pattern;
296  msg.step = pResultImage->GetStride();
297 
298  size_t image_size = pResultImage->GetImageSize();
299  msg.data.resize(image_size);
300  memcpy(msg.data.data(), pResultImage->GetData(), image_size);
301 
302  image_pubs_[index].publish(msg);
303  }
304  pResultImage->Release();
305  }
306  catch (Spinnaker::Exception& e)
307  {
308  ROS_ERROR("[%s] Error: %s", __APP_NAME__, e.what());
309  }
310  }

323  }
