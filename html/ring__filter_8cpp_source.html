61 static void scan_callback(const sensor_msgs::PointCloud2::ConstPtr& input)
62 {
63  pcl::PointCloud<pcl::PointXYZI> scan;
64  pcl::PointCloud<velodyne_pointcloud::PointXYZIR> tmp;
65  sensor_msgs::PointCloud2 filtered_msg;
66 
67  pcl::fromROSMsg(*input, scan);
68  pcl::fromROSMsg(*input, tmp);
69 
70  filter_start = std::chrono::system_clock::now();
71 
72  scan.points.clear();
73 
74  double square_measurement_range = measurement_range*measurement_range;
75 
76  for (pcl::PointCloud<velodyne_pointcloud::PointXYZIR>::const_iterator item = tmp.begin(); item != tmp.end(); item++)
77  {
78  pcl::PointXYZI p;
79  p.x = item->x;
80  p.y = item->y;
81  p.z = item->z;
82  p.intensity = item->intensity;
83 
84  double square_distance = p.x * p.x + p.y * p.y;
85 
86  if (item->ring % ring_div == 0 && square_distance <= square_measurement_range)
87  {
88  scan.points.push_back(p);
89  }
90  if (item->ring > ring_max)
91  {
92  ring_max = item->ring;
93  }
94  }
95 
96  pcl::PointCloud<pcl::PointXYZI>::Ptr scan_ptr(new pcl::PointCloud<pcl::PointXYZI>(scan));
97  pcl::PointCloud<pcl::PointXYZI>::Ptr filtered_scan_ptr(new pcl::PointCloud<pcl::PointXYZI>());
98 
99  // if voxel_leaf_size < 0.1 voxel_grid_filter cannot down sample (It is specification in PCL)
100  if (voxel_leaf_size >= 0.1)
101  {
102  // Downsampling the velodyne scan using VoxelGrid filter
103  pcl::VoxelGrid<pcl::PointXYZI> voxel_grid_filter;
104  voxel_grid_filter.setLeafSize(voxel_leaf_size, voxel_leaf_size, voxel_leaf_size);
105  voxel_grid_filter.setInputCloud(scan_ptr);
106  voxel_grid_filter.filter(*filtered_scan_ptr);
107 
108  pcl::toROSMsg(*filtered_scan_ptr, filtered_msg);
109  }
110  else
111  {
112  pcl::toROSMsg(*scan_ptr, filtered_msg);
113  }
114 
115  filter_end = std::chrono::system_clock::now();
116 
117  filtered_msg.header = input->header;
118  filtered_points_pub.publish(filtered_msg);
119 
120  points_downsampler_info_msg.header = input->header;
121  points_downsampler_info_msg.filter_name = "ring_filter";
122  points_downsampler_info_msg.measurement_range = measurement_range;
123  points_downsampler_info_msg.original_points_size = scan.size();
124  if (voxel_leaf_size >= 0.1)
125  {
126  points_downsampler_info_msg.filtered_points_size = filtered_scan_ptr->size();
127  }
128  else
129  {
130  points_downsampler_info_msg.filtered_points_size = scan_ptr->size();
131  }
132  points_downsampler_info_msg.original_ring_size = ring_max;
133  points_downsampler_info_msg.filtered_ring_size = ring_max / ring_div;
134  points_downsampler_info_msg.exe_time = std::chrono::duration_cast<std::chrono::microseconds>(filter_end - filter_start).count() / 1000.0;
135  points_downsampler_info_pub.publish(points_downsampler_info_msg);
136 
137  if(_output_log == true){
138  if(!ofs){
139  std::cerr << "Could not open " << filename << "." << std::endl;
140  exit(1);
141  }
142  ofs << points_downsampler_info_msg.header.seq << ","
143  << points_downsampler_info_msg.header.stamp << ","
144  << points_downsampler_info_msg.header.frame_id << ","
145  << points_downsampler_info_msg.filter_name << ","
146  << points_downsampler_info_msg.original_points_size << ","
147  << points_downsampler_info_msg.filtered_points_size << ","
148  << points_downsampler_info_msg.original_ring_size << ","
149  << points_downsampler_info_msg.filtered_ring_size << ","
150  << points_downsampler_info_msg.exe_time << ","
151  << std::endl;
152  }
