53 // has these responsibilities:
54 //
55 // - Act as a container for GUI elements DriveWidget and QLineEdit.

57 // - Saving and restoring internal state from a config file.
58 //
59 // We start with the constructor, doing the standard Qt thing of
60 // passing the optional *parent* argument on to the superclass
61 // constructor, and also zero-ing the velocities we will be

290 void DecisionMakerPanel::sendTopic(const QString &text)
291 {
292  if (text == "emergency")
293  {
294  sendEmergency();
295  }
296  else if (statecmd_publisher_)
297  {
298  std_msgs::String msg;
299  msg.data = text.toStdString();
300  statecmd_publisher_.publish(msg);
301  }
302 }

304 void DecisionMakerPanel::sendEmergency()
305 {
306  if (emergency_publisher_)
307  {
308  autoware_msgs::RemoteCmd remote_emergency;
309  remote_emergency.header.stamp = ros::Time::now();
310  remote_emergency.vehicle_cmd.emergency = 1;
311  emergency_publisher_.publish(remote_emergency);
312  }
313 }
