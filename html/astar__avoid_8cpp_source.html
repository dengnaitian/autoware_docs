105 void AstarAvoid::run()
106 {
107  // check topics
108  state_ = AstarAvoid::STATE::INITIALIZING;
109 
110  while (ros::ok())
111  {
112  ros::spinOnce();
113  if (checkInitialized())
114  {
115  break;
116  }
117  ROS_WARN("Waiting for subscribing topics...");
118  ros::Duration(1.0).sleep();
119  }
120 
121  // main loop
122  int end_of_avoid_index = -1;
123  ros::WallTime start_plan_time = ros::WallTime::now();
124  ros::WallTime start_avoid_time = ros::WallTime::now();
125 
126  // reset obstacle index
127  obstacle_waypoint_index_ = -1;
128 
129  // relaying mode at startup
130  state_ = AstarAvoid::STATE::RELAYING;
131 
132  // start publish thread
133  publish_thread_ = std::thread(&AstarAvoid::publishWaypoints, this);
134 
135  while (ros::ok())
136  {
137  ros::spinOnce();
138 
139  // relay mode
140  if (!enable_avoidance_)
141  {
142  rate_->sleep();
143  continue;
144  }
145 
146  // avoidance mode
147  bool found_obstacle = (obstacle_waypoint_index_ >= 0);
148  bool avoid_velocity = (current_velocity_.twist.linear.x < avoid_start_velocity_ / 3.6);
149 
150  // update state
151  if (state_ == AstarAvoid::STATE::RELAYING)
152  {
153  avoid_waypoints_ = base_waypoints_;
154 
155  if (found_obstacle)
156  {
157  ROS_INFO("RELAYING -> STOPPING, Decelerate for stopping");
158  state_ = AstarAvoid::STATE::STOPPING;
159  }
160  }
161  else if (state_ == AstarAvoid::STATE::STOPPING)
162  {
163  bool replan = ((ros::WallTime::now() - start_plan_time).toSec() > replan_interval_);
164 
165  if (!found_obstacle)
166  {
167  ROS_INFO("STOPPING -> RELAYING, Obstacle disappers");
168  state_ = AstarAvoid::STATE::RELAYING;
169  }
170  else if (replan && avoid_velocity)
171  {
172  ROS_INFO("STOPPING -> PLANNING, Start A* planning");
173  state_ = AstarAvoid::STATE::PLANNING;
174  }
175  }
176  else if (state_ == AstarAvoid::STATE::PLANNING)
177  {
178  start_plan_time = ros::WallTime::now();
179 
180  if (planAvoidWaypoints(end_of_avoid_index))
181  {
182  ROS_INFO("PLANNING -> AVOIDING, Found path");
183  state_ = AstarAvoid::STATE::AVOIDING;
184  start_avoid_time = ros::WallTime::now();
185  }
186  else
187  {
188  ROS_INFO("PLANNING -> STOPPING, Cannot find path");
189  state_ = AstarAvoid::STATE::STOPPING;
190  }
191  }
192  else if (state_ == AstarAvoid::STATE::AVOIDING)
193  {
194  bool reached = (getClosestWaypoint(avoid_waypoints_, current_pose_global_.pose) > end_of_avoid_index);
195  if (reached)
196  {
197  ROS_INFO("AVOIDING -> RELAYING, Reached goal");
198  state_ = AstarAvoid::STATE::RELAYING;
199  }
200  else if (found_obstacle && avoid_velocity)
201  {
202  bool replan = ((ros::WallTime::now() - start_avoid_time).toSec() > replan_interval_);
203  if (replan)
204  {
205  ROS_INFO("AVOIDING -> STOPPING, Abort avoiding");
206  state_ = AstarAvoid::STATE::STOPPING;
207  }
208  }
209  }
210 
211  rate_->sleep();
212  }
213 
214  terminate_thread_ = true;
215 }

234 bool AstarAvoid::planAvoidWaypoints(int& end_of_avoid_index)
235 {
236  bool found_path = false;
237  int closest_waypoint_index = getClosestWaypoint(avoid_waypoints_, current_pose_global_.pose);
238 
239  // update goal pose incrementally and execute A* search
240  for (int i = search_waypoints_delta_; i < static_cast<int>(search_waypoints_size_); i += search_waypoints_delta_)
241  {
242  // update goal index
243  int goal_waypoint_index = closest_waypoint_index + obstacle_waypoint_index_ + i;
244  if (goal_waypoint_index >= static_cast<int>(avoid_waypoints_.waypoints.size()))
245  {
246  break;
247  }
248 
249  // update goal pose
250  goal_pose_global_ = avoid_waypoints_.waypoints[goal_waypoint_index].pose;
251  goal_pose_local_.header = costmap_.header;
252  goal_pose_local_.pose = transformPose(goal_pose_global_.pose,
253  getTransform(costmap_.header.frame_id, goal_pose_global_.header.frame_id));
254 
255  // initialize costmap for A* search
256  astar_.initialize(costmap_);
257 
258  // execute astar search
259  // ros::WallTime start = ros::WallTime::now();
260  found_path = astar_.makePlan(current_pose_local_.pose, goal_pose_local_.pose);
261  // ros::WallTime end = ros::WallTime::now();
262 
263  static ros::Publisher pub = nh_.advertise<nav_msgs::Path>("debug", 1, true);
264 
265  // ROS_INFO("Astar planning: %f [s], at index = %d", (end - start).toSec(), goal_waypoint_index);
266 
267  if (found_path)
268  {
269  pub.publish(astar_.getPath());
270  end_of_avoid_index = goal_waypoint_index;
271  mergeAvoidWaypoints(astar_.getPath(), end_of_avoid_index);
272  if (avoid_waypoints_.waypoints.size() > 0)
273  {
274  ROS_INFO("Found GOAL at index = %d", goal_waypoint_index);
275  astar_.reset();
276  return true;
277  }
278  else
279  {
280  found_path = false;
281  }
282  }
283  astar_.reset();
284  }
285 
286  ROS_ERROR("Can't find goal...");
287  return false;
288 }

290 void AstarAvoid::mergeAvoidWaypoints(const nav_msgs::Path& path, const int& end_of_avoid_index)
291 {
292  autoware_msgs::Lane current_waypoints = avoid_waypoints_;
293 
294  // reset
295  std::lock_guard<std::mutex> lock(mutex_);
296  avoid_waypoints_.waypoints.clear();
297 
298  // add waypoints before start index
299  int closest_waypoint_index = getClosestWaypoint(current_waypoints, current_pose_global_.pose);
300  for (int i = 0; i < closest_waypoint_index; ++i)
301  {
302  avoid_waypoints_.waypoints.push_back(current_waypoints.waypoints.at(i));
303  }
304 
305  // set waypoints for avoiding
306  for (const auto& pose : path.poses)
307  {
308  autoware_msgs::Waypoint wp;
309  wp.pose.header = avoid_waypoints_.header;
310  wp.pose.pose = transformPose(pose.pose, getTransform(avoid_waypoints_.header.frame_id, pose.header.frame_id));
311  wp.pose.pose.position.z = current_pose_global_.pose.position.z; // height = const
312  wp.twist.twist.linear.x = avoid_waypoints_velocity_ / 3.6; // velocity = const
313  avoid_waypoints_.waypoints.push_back(wp);
314  }
315 
316  // add waypoints after goal index
317  for (int i = end_of_avoid_index; i < static_cast<int>(current_waypoints.waypoints.size()); ++i)
318  {
319  avoid_waypoints_.waypoints.push_back(current_waypoints.waypoints.at(i));
320  }
321 }

323 void AstarAvoid::publishWaypoints()
324 {
325  autoware_msgs::Lane current_waypoints;
326 
327  while (!terminate_thread_)
328  {
329  // select waypoints
330  switch (state_)
331  {
332  case AstarAvoid::STATE::RELAYING:
333  current_waypoints = base_waypoints_;
334  break;
335  case AstarAvoid::STATE::STOPPING:
336  // do nothing, keep current waypoints
337  break;
338  case AstarAvoid::STATE::PLANNING:
339  // do nothing, keep current waypoints
340  break;
341  case AstarAvoid::STATE::AVOIDING:
342  current_waypoints = avoid_waypoints_;
343  break;
344  default:
345  current_waypoints = base_waypoints_;
346  break;
347  }
348 
349  autoware_msgs::Lane safety_waypoints;
350  safety_waypoints.header = current_waypoints.header;
351  safety_waypoints.increment = current_waypoints.increment;
352 
353  // push waypoints from closest index
354  for (int i = 0; i < safety_waypoints_size_; ++i)
355  {
356  int index = getClosestWaypoint(current_waypoints, current_pose_global_.pose) + i;
357  if (index < 0 || static_cast<int>(current_waypoints.waypoints.size()) <= index)
358  {
359  break;
360  }
361  const autoware_msgs::Waypoint& wp = current_waypoints.waypoints[index];
362  safety_waypoints.waypoints.push_back(wp);
363  }
364 
365  if (safety_waypoints.waypoints.size() > 0)
366  {
367  safety_waypoints_pub_.publish(safety_waypoints);
368  }
369 
370  rate_->sleep();
371  }
372 }
