435 u16 __mip_interface_parse_input_buffer(mip_interface *device_interface)
436 {
437  u16 i, ret;
438  mip_header *header_ptr = (mip_header*)device_interface->mip_packet;
439  parser_callback_ptr callback_function = NULL;
440  void *callback_user_ptr = NULL;
441 
442  //Check that the parser is initialized
443  if(device_interface->state != MIP_INTERFACE_INITIALIZED)
444  return MIP_INTERFACE_ERROR;
445 
446 
448  //Are we searching for a start condition? Requires: SYNC byte 1 & 2, and a valid header
450 
451  if(device_interface->mip_packet_byte_count < sizeof(mip_header))
452  {
453 
455  //Get a start byte
457 
458  while((device_interface->mip_packet_byte_count == 0) && ring_buffer_count(&device_interface->input_buffer))
459  {
460  ret = ring_buffer_read(&device_interface->input_buffer, &device_interface->mip_packet[0], 1);
461 
462  //Found a potential start byte
463  if((ret == RING_BUFFER_OK) && (device_interface->mip_packet[0] == MIP_SYNC_BYTE1))
464  {
465  device_interface->mip_packet_byte_count = 1;
466  device_interface->parser_start_time = mip_sdk_get_time_ms();
467  }
468  }
469 
471  //Get the rest of the header
473 
474  if(device_interface->mip_packet_byte_count > 0)
475  {
476  //Need at least the size of (header - start_byte) in the buffer to start processing
477  if(ring_buffer_count(&device_interface->input_buffer) >= sizeof(mip_header) - 1)
478  {
479  //Try to get a header
480  for(i=0; i < sizeof(mip_header) - 1; i++)
481  {
482  ring_buffer_lookahead_read(&device_interface->input_buffer, i, &device_interface->mip_packet[i+1], 1);
483  }
484 
485  //If the header is valid, then continue to the next step
486  if((header_ptr->sync2 == MIP_SYNC_BYTE2) &&
487  (header_ptr->payload_size + MIP_HEADER_SIZE + MIP_CHECKSUM_SIZE <= MIP_MAX_PACKET_SIZE))
488  {
489  device_interface->mip_packet_byte_count = sizeof(mip_header);
490  }
491  else
492  {
493  //Header not found
494  device_interface->parser_in_sync = 0;
495  device_interface->mip_packet_byte_count = 0;
496  device_interface->parser_headers_skipped++;
497  }
498  }
499  //Check for timeout on incoming packet
500  else
501  {
502  //If a timeout happens, we probably had a badly-formed packet, start over
503  if(__mip_interface_time_timeout(device_interface->parser_start_time, device_interface->packet_timeout) == MIP_INTERFACE_TIMEOUT)
504  {
505  device_interface->parser_timeouts++;
506 
507  //Reset the parser
508  device_interface->mip_packet_byte_count = 0;
509  device_interface->parser_in_sync = 0;
510  }
511  }
512  }
513  }
514 
515 
517  //Header located, get the rest of the potential MIP packet
519 
520  if(device_interface->mip_packet_byte_count >= sizeof(mip_header))
521  {
522  //Wait for the rest of the packet to be available in the buffer
523  if(ring_buffer_count(&device_interface->input_buffer) >= (sizeof(mip_header) - 1 + header_ptr->payload_size + MIP_CHECKSUM_SIZE))
524  {
525  //Get the remaining packet bytes
526  for(i=0; i < header_ptr->payload_size + MIP_CHECKSUM_SIZE; i++)
527  {
528  ring_buffer_lookahead_read(&device_interface->input_buffer, sizeof(mip_header) - 1 + i, &device_interface->mip_packet[MIP_HEADER_SIZE + i], 1);
529  }
530 
532  //Valid MIP Packet Found
534 
535  if(mip_is_checksum_valid(device_interface->mip_packet) == MIP_OK)
536  {
537  device_interface->mip_packet_byte_count += header_ptr->payload_size + MIP_CHECKSUM_SIZE;
538 
539  //Trigger the callback with the valid packet
540  if(__mip_interface_find_callback(device_interface, header_ptr->descriptor_set, &callback_user_ptr, &callback_function) == MIP_INTERFACE_OK)
541  {
542  (*callback_function)(callback_user_ptr, device_interface->mip_packet, device_interface->mip_packet_byte_count, MIP_INTERFACE_CALLBACK_VALID_PACKET);
543  }
544 
545  ring_buffer_consume_entries(&device_interface->input_buffer, device_interface->mip_packet_byte_count - 1);
546 
547  device_interface->parser_in_sync = 1;
548 
549  //Reset the parser
550  device_interface->mip_packet_byte_count = 0;
551  }
552 
554  //Inalid MIP Packet: Bad checksum
556 
557  else
558  {
559  //Trigger the callback with a bad checksum
560  if(__mip_interface_find_callback(device_interface, header_ptr->descriptor_set, &callback_user_ptr, &callback_function) == MIP_INTERFACE_OK)
561  {
562  (*callback_function)(callback_user_ptr, device_interface->mip_packet, device_interface->mip_packet_byte_count, MIP_INTERFACE_CALLBACK_CHECKSUM_ERROR);
563  }
564 
565  if(device_interface->parser_in_sync)
566  {
567  device_interface->parser_num_bad_checksums++;
568  }
569 
570  device_interface->parser_in_sync = 0;
571  }
572 
573  //Reset the parser
574  device_interface->mip_packet_byte_count = 0;
575  }
576  //Check for timeout on incoming packet
577  else
578  {
579  //If a timeout happens, we probably had a bad-form packet, start over
580  if(__mip_interface_time_timeout(device_interface->parser_start_time, device_interface->packet_timeout) == MIP_INTERFACE_TIMEOUT)
581  {
582  //Trigger the callback with a timeout
583  if(__mip_interface_find_callback(device_interface, header_ptr->descriptor_set, &callback_user_ptr, &callback_function) == MIP_INTERFACE_OK)
584  {
585  (*callback_function)(callback_user_ptr, device_interface->mip_packet, device_interface->mip_packet_byte_count, MIP_INTERFACE_CALLBACK_TIMEOUT);
586  }
587 
588  device_interface->parser_timeouts++;
589  device_interface->parser_in_sync = 0;
590 
591  //Reset the parser
592  device_interface->mip_packet_byte_count = 0;
593  }
594  }
595  }
596 
597  return MIP_INTERFACE_OK;
598 }

679 u16 __mip_interface_time_timeout(u32 initial_time, u32 timeout_ms)
680 {

683  //Handle wrap-around case

685  {
686  if(((0xFFFFFFFF - initial_time) + current_time) >= timeout_ms)
687  return MIP_INTERFACE_TIMEOUT;
