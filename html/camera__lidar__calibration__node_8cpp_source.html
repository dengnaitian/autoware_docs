75  cv::Point3f get_rpy_from_matrix(const cv::Mat& in_rotation)
76  {
77 
78  assert(is_rotation_matrix(in_rotation));
79 
80  float sy = sqrt(in_rotation.at<double>(0,0) * in_rotation.at<double>(0,0) +
81  in_rotation.at<double>(1,0) * in_rotation.at<double>(1,0) );
82 
83  bool is_singular = sy < 1e-6;
84 
85  float x, y, z;
86  if (!is_singular)
87  {
88  x = atan2(in_rotation.at<double>(2,1), in_rotation.at<double>(2,2));
89  y = atan2(-in_rotation.at<double>(2,0), sy);
90  z = atan2(in_rotation.at<double>(1,0), in_rotation.at<double>(0,0));
91  }
92  else
93  {
94  x = atan2(-in_rotation.at<double>(1,2), in_rotation.at<double>(1,1));
95  y = atan2(-in_rotation.at<double>(2,0), sy);
96  z = 0;
97  }
98  return cv::Point3f(RAD2DEG(x), RAD2DEG(y), RAD2DEG(z));
99  }
