46 static FLOAT *padarray(FLOAT *feature,int *size,int padx,int pady)
47 {
48  const int NEW_Y=size[0]+pady*2;
49  const int NEW_X=size[1]+padx*2;
50  const int L=NEW_Y*padx;
51  const int SPL=size[0]+pady;
52  const int M_S = sizeof(FLOAT)*size[0];
53  FLOAT *new_feature = (FLOAT*)calloc(NEW_Y*NEW_X*size[2],sizeof(FLOAT)); // feature (pad-added)
54  FLOAT *P=new_feature;
55  FLOAT *S=feature;
56 
57  for(int ii=0;ii<size[2];ii++) //31
58  {
59  P+=L; //row-all
60  for(int jj=0;jj<size[1];jj++)
61  {
62  P+=pady;
63  memcpy(P, S,M_S);
64  S+=size[0];
65  P+=SPL;
66  }
67  P+=L; //row-all
68  }
69  size[0]=NEW_Y;
70  size[1]=NEW_X;
71  return(new_feature);

165 static void calc_a_score(FLOAT *ac_score,FLOAT *score,int *ssize,int *rsize,Model_info *MI,FLOAT scale)
166 {
167  const int IHEI = MI->IM_HEIGHT;
168  const int IWID = MI->IM_WIDTH;

170  int padx_n = MI->padx;

172 
173  int RY = (int)((FLOAT)rsize[0]*scale/2.0-1.0+block_pad);
174  int RX = (int)((FLOAT)rsize[1]*scale/2.0-1.0+block_pad);
175 
176  for(int ii=0;ii<IWID;ii++)
177  {
178  int Xn=(int)((FLOAT)ii/scale+padx_n);
179 
180  for(int jj=0;jj<IHEI;jj++)
181  {
182  int Yn =(int)((FLOAT)jj/scale+pady_n);
183  if(Yn<ssize[0] && Xn<ssize[1])
184  {
185  FLOAT sc = score[Yn+Xn*ssize[0]]; //get score of pixel
186 
187  int Im_Y = jj+RY;
188  int Im_X = ii+RX;
189  if(Im_Y<IHEI && Im_X<IWID)
190  {
191  FLOAT *PP=ac_score+Im_Y+Im_X*IHEI; //consider root rectangle size
192  if(sc>*PP) *PP=sc; //save max score
193  }
194  }
195  }
196  }
