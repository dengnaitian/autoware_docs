99 void DriveWorksApi::initSensors(std::vector<Camera> *cameras, uint32_t *numCameras,
100  dwSALHandle_t sal, DeviceArguments &arguments)
101 {
102  std::cout << "Init Sensors .. " << std::endl;
103  std::string selector = arguments.get("selector_mask");
104  dwStatus result;
105  // identify active ports
106  int idx = 0;
107  int cnt[3] = {0, 0, 0};
108  std::string port[3] = {"ab", "cd", "ef"};
109  for (size_t i = 0; i < selector.length() && i < 12; i++, idx++)
110  {
111  const char s = selector[i];
112  if (s == '1')
113  {
114  cnt[idx / 4]++;
115  }
116  }
117 
118  // parsing arguments
119  (*numCameras) = 0;
120  for (size_t p = 0; p < 3; p++)
121  {
122  if (cnt[p] > 0)
123  {
124  std::string params;
125  params += std::string("csi-port=") + port[p];
126  params += ",camera-type=" + arguments.get((std::string("type_") + port[p]).c_str());
127  params += ",camera-count=4"; // when using the mask, just ask for all cameras, mask will select properly
128 
129  if (selector.size() >= p*4)
130  {
131  params += ",camera-mask="+ selector.substr(p*4, std::min(selector.size() - p*4, size_t{4}));
132  }
133 
134  params += ",slave=" + arguments.get("slave");
135  params += ",cross-csi-sync=" + arguments.get("cross_csi_sync");
136  params += ",fifo-size=" + arguments.get("fifo_size");
137 
138 
139  //Debug arguments
140 
141  std::cout << "DEBUG ARGS PORT: " << p << std::endl;
142  std::cout << params.c_str() << std::endl;
143 
144  //End Debug
145 
146  dwSensorHandle_t salSensor = DW_NULL_HANDLE;
147  dwSensorParams salParams;
148  salParams.parameters = params.c_str();
149  salParams.protocol = "camera.gmsl";
150  result = dwSAL_createSensor(&salSensor, salParams, sal);
151  if (result == DW_SUCCESS)
152  {
153  Camera cam;
154  cam.sensor = salSensor;
155 
156  dwImageProperties cameraImageProperties;
157  dwSensorCamera_getImageProperties(&cameraImageProperties,
158  DW_CAMERA_PROCESSED_IMAGE,
159  salSensor);
160 
161  dwCameraProperties cameraProperties;
162  dwSensorCamera_getSensorProperties(&cameraProperties, salSensor);
163 
164  cam.width = cameraImageProperties.width;
165  cam.height = cameraImageProperties.height;
166  cam.numSiblings = cameraProperties.siblings;
167  cameras->push_back(cam);
168 
169  (*numCameras) += cam.numSiblings;
170  }
171  else
172  {
173  std::cerr << "Cannot create driver: " << salParams.protocol
174  << " with params: " << salParams.parameters << std::endl
175  << "Error: " << dwGetStatusName(result) << std::endl;
176 
177  if (result == DW_INVALID_ARGUMENT)
178  {
179  std::cerr << "It is possible the given camera is not supported. "
180  << "Please refer to the documentation for this sample."
181  << std::endl;
182  }
183  }
184  }
185  }

231 /*
232  * Init camera frames convertion to RGBA and image pools per port
233  */
234 void DriveWorksApi::initFrameImage(Camera* camera)
235 {
236  std::cout << "Init Camera Frame Pools .. " << std::endl;
237  // RGBA image pool for conversion from YUV camera output
238  // two RGBA frames per camera per sibling for a pool
239  // since image streamer might hold up-to one frame when using egl stream
240  dwStatus result;
241  int32_t pool_size = 2;
242  uint32_t numFramesRGBA = pool_size*camera->numSiblings;
243 
244  // temp variable for easy access and de-reference back to camera->frameRGBA in releasing nvidia image frame read
245  std::vector<dwImageNvMedia>& g_frameRGBA = camera->frameRGBA;
246 
247  g_frameRGBA.reserve(numFramesRGBA);
248  {
249  dwImageProperties cameraImageProperties;
250  dwSensorCamera_getImageProperties(&cameraImageProperties, DW_CAMERA_PROCESSED_IMAGE,camera->sensor);
251  dwImageProperties displayImageProperties = cameraImageProperties;
252  displayImageProperties.pxlFormat = DW_IMAGE_RGBA;
253  displayImageProperties.planeCount = 1;
254 
255  // format converter
256  result = dwImageFormatConverter_initialize(&camera->yuv2rgba, cameraImageProperties.type, sdk);
257  if (result != DW_SUCCESS)
258  {
259  std::cerr << "Cannot create pixel format converter : yuv->rgba" << dwGetStatusName(result) << std::endl;
260  g_run = false;
261  }
262 
263  // allocate image pool
264  for(uint32_t cameraIdx = 0; cameraIdx < camera->numSiblings; cameraIdx++)
265  {
266  for(int32_t k = 0; k < pool_size; k++)
267  {
268  dwImageNvMedia rgba{};
269  result = dwImageNvMedia_create(&rgba, &displayImageProperties, sdk);
270  if (result != DW_SUCCESS)
271  {
272  std::cerr << "Cannot create nvmedia image for pool:" << dwGetStatusName(result) << std::endl;
273  g_run = false;
274  break;
275  }
276  // don't forget to delete dwImage via g_frameRGBA when exit
277  g_frameRGBA.push_back(rgba);
278  camera->rgbaPool.push(&g_frameRGBA.back());
279  }
280  }
281 
282 
283  // NVMedia image compression definition.
284  for(uint32_t cameraIdx = 0; cameraIdx < camera->numSiblings; cameraIdx++)
285  {
286  NvMediaDevice *device;
287  device = NvMediaDeviceCreate();
288  if(!device)
289  {
290  std::cerr << "main: NvMediaDeviceCreate failed\n" << std::endl;
291  g_run = false;
292  }
293  NvMediaIJPE* jpegEncoder = NULL;
294  jpegEncoder = NvMediaIJPECreate(device, NvMediaSurfaceType_Image_YUV_420,(uint8_t) 1, max_jpeg_bytes );
295  if(!jpegEncoder)
296  {
297  std::cerr << "main: NvMediaIJPECreate failed\n" << std::endl;
298  g_run = false;
299  }
300  else
301  {
302  camera->jpegEncoders.push_back(jpegEncoder);
303  }
304  }
305  // allocate compressed image pool
306  for(uint32_t cameraIdx = 0; cameraIdx < camera->numSiblings; cameraIdx++)
307  {
308  for (int32_t k = 0; k < pool_size; k++)
309  {
310  uint8_t* jpeg_img = (uint8_t*) malloc( max_jpeg_bytes );
311  camera->jpegPool.push(jpeg_img);
312  }
313  }
314  // start camera capturing
315  g_run = g_run && dwSensor_start(camera->sensor) == DW_SUCCESS;
316  eof = false;
317  }
318 }

352 void DriveWorksApi::threadCameraPipeline(Camera* cameraSensor, uint32_t port, dwContextHandle_t sdk)
353 {
354  std::cout << "Start camera thread for port: " << port << std::endl;
355  // cv publishers
356  std::vector<std::unique_ptr<OpenCVConnector>> cv_connectors;
357  // init multiple cv cameras connection and topic name
358  for (uint32_t cameraIdx = 0; cameraIdx < cameraSensor->numSiblings; cameraIdx++)
359  {
360  // Topic mapping e.g. gmsl_image_raw_<nvidia cam port A=0, B=1, C=2>_<sibling id 0,1,2,3> : port_0/camera_1/(image_raw,image_raw/compressed)
361  const std::string topic = std::string("port_") + std::to_string(port) + std::string("/camera_") + std::to_string(cameraIdx);
362  const std::string camera_frame_id = std::string("gmsl_camera_") + std::to_string(port) + std::string("_") + std::to_string(cameraIdx);
363  const std::string cam_info_file = std::string("file://") + std::string(g_calibFolder) + std::to_string(port) + std::to_string(cameraIdx) + std::string("_calibration.yml");
364  std::unique_ptr<OpenCVConnector> cvPtr(new OpenCVConnector(topic, camera_frame_id, cam_info_file, 10));
365  cv_connectors.push_back(std::move(cvPtr));
366  }
367 
368  while (g_run && ros::ok())
369  {
370  bool eofAny = false;
371  // capture from all csi-ports
372  // NOTE if cross-csi-synch is active, all cameras will capture at the same time
373  {
374  if (eof)
375  {
376  eofAny = true;
377  continue;
378  }
379 
380  if (cameraSensor->rgbaPool.empty())
381  {
382  std::cerr << "Ran out of RGBA buffers, continuing" << std::endl;
383  continue;
384  }
385 
386  // capture from all cameras within a csi port
387  for (uint32_t cameraIdx = 0;cameraIdx < cameraSensor->numSiblings && !cameraSensor->rgbaPool.empty();cameraIdx++)
388  {
389  // capture, convert to rgba and return it
390  eof = captureCamera(cameraSensor->rgbaPool.front(),
391  cameraSensor->sensor,
392  port, cameraIdx,
393  cameraSensor->yuv2rgba,
394  cameraSensor->jpegPool.front(),
395  cameraSensor->jpegEncoders[cameraIdx]);
396  g_frameRGBAPtr[port][cameraIdx] = cameraSensor->rgbaPool.front();
397  cameraSensor->rgbaPool.pop();
398 
399  g_frameJPGPtr[port][cameraIdx] = cameraSensor->jpegPool.front();
400  cameraSensor->jpegPool.pop();
401 
402  if (!eof)
403  {
404  cameraSensor->rgbaPool.push(g_frameRGBAPtr[port][cameraIdx]);
405  cameraSensor->jpegPool.push(g_frameJPGPtr[port][cameraIdx]);
406  }
407 
408  eofAny |= eof;
409  }
410  }
411 
412  // stop to take screenshot (will cause a delay)
413  if (gTakeScreenshot)
414  {
415  {
416 
417  for (uint32_t cameraIdx = 0; cameraIdx < cameraSensor->numSiblings && !cameraSensor->rgbaPool.empty(); cameraIdx++)
418  {
419  //copy to memory replacing by //takeScreenshot(g_frameRGBAPtr[port][cameraIdx], port, cameraIdx);
420  dwImageNvMedia *frameNVMrgba = g_frameRGBAPtr[port][cameraIdx];
421  NvMediaImageSurfaceMap surfaceMap;
422 
423  if (NvMediaImageLock(frameNVMrgba->img, NVMEDIA_IMAGE_ACCESS_READ, &surfaceMap) == NVMEDIA_STATUS_OK)
424  {
425  // publish an image
426  if(gImageCompressed)
427  {
428  // compressed
429  cv_connectors[cameraIdx]->WriteToJpeg( g_frameJPGPtr[port][cameraIdx], g_compressedSize[port][cameraIdx]);
430  }
431  else
432  {
433  //raw (resize if set)
434  cv_connectors[cameraIdx]->WriteToOpenCV((unsigned char*)surfaceMap.surface[0].mapping,
435  frameNVMrgba->prop.width, frameNVMrgba->prop.height,
436  g_imageWidth,
437  g_imageHeight);
438  }
439 
440  NvMediaImageUnlock(frameNVMrgba->img);
441  }
442  else
443  {
444  std::cout << "CANNOT LOCK NVMEDIA IMAGE - NO SCREENSHOT\n";
445  }
446 
447  }
448  }
449  gScreenshotCount++;
450  }
451  std::this_thread::sleep_for(std::chrono::milliseconds(5));
452  g_run = g_run && !eofAny;
453  }//end while
454 }

457 /*
458  * Function to capture an image frame and convert to RGBA
459  */
460 dwStatus DriveWorksApi::captureCamera(dwImageNvMedia *frameNVMrgba,
461  dwSensorHandle_t cameraSensor, uint32_t port,
462  uint32_t sibling, dwImageFormatConverterHandle_t yuv2rgba,
463  uint8_t* jpeg_image, NvMediaIJPE *jpegEncoder)
464 {
465  dwCameraFrameHandle_t frameHandle;
466  dwImageNvMedia *frameNVMyuv = nullptr;
467  dwStatus result = DW_FAILURE;
468  result = dwSensorCamera_readFrame(&frameHandle, sibling, 300000, cameraSensor);
469  if (result != DW_SUCCESS)
470  {
471  std::cerr << "readFrameNvMedia: " << dwGetStatusName(result) << std::endl;
472  return result;
473  }
474 
475  result = dwSensorCamera_getImageNvMedia(&frameNVMyuv, DW_CAMERA_PROCESSED_IMAGE, frameHandle);
476  if( result != DW_SUCCESS )
477  {
478  std::cerr << "readImageNvMedia: " << dwGetStatusName(result) << std::endl;
479  }
480 
481  result = dwImageFormatConverter_copyConvertNvMedia(frameNVMrgba, frameNVMyuv, yuv2rgba);
482  if( result != DW_SUCCESS )
483  {
484  std::cerr << "copyConvertNvMedia: " << dwGetStatusName(result) << std::endl;
485  }
486 
487 
488  if(gImageCompressed)
489  {
490  NvMediaStatus nvStatus = NvMediaIJPEFeedFrame(jpegEncoder,frameNVMyuv->img,JPEG_quality);
491  if(nvStatus != NVMEDIA_STATUS_OK)
492  {
493  std::cerr <<"NvMediaIJPEFeedFrameQuality failed: %x\n" << nvStatus << std::endl;
494  }
495  nvStatus = NvMediaIJPEBitsAvailable(jpegEncoder, &g_compressedSize[port][sibling],NVMEDIA_ENCODE_BLOCKING_TYPE_IF_PENDING , 10000);
496  nvStatus = NvMediaIJPEGetBits(jpegEncoder, &g_compressedSize[port][sibling], jpeg_image, 0);
497  if(nvStatus != NVMEDIA_STATUS_OK && nvStatus != NVMEDIA_STATUS_NONE_PENDING)
498  {
499  std::cerr <<"main: Error getting encoded bits\n"<< std::endl;
500  }
501  }
502  result = dwSensorCamera_returnFrame(&frameHandle);
503  if( result != DW_SUCCESS )
504  {
505  std::cout << "returnFrameNvMedia: " << dwGetStatusName(result) << std::endl;
506  }
507 
508  return DW_SUCCESS;
509 }
