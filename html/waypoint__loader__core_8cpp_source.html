88 void WaypointLoaderNode::loadWaypointsForVer1(const char* filename, std::vector<autoware_msgs::Waypoint>* wps)
89 {
90  std::ifstream ifs(filename);
91 
92  if (!ifs)
93  {
94  return;
95  }
96 
97  std::string line;
98  std::getline(ifs, line); // Remove first line
99 
100  while (std::getline(ifs, line))
101  {
102  autoware_msgs::Waypoint wp;
103  parseWaypointForVer1(line, &wp);
104  wps->emplace_back(wp);
105  }
106 
107  size_t last = wps->size() - 1;
108  for (size_t i = 0; i < wps->size(); ++i)
109  {
110  if (i != last)
111  {
112  double yaw = atan2(wps->at(i + 1).pose.pose.position.y - wps->at(i).pose.pose.position.y,
113  wps->at(i + 1).pose.pose.position.x - wps->at(i).pose.pose.position.x);
114  wps->at(i).pose.pose.orientation = tf::createQuaternionMsgFromYaw(yaw);
115  }
116  else
117  {
118  wps->at(i).pose.pose.orientation = wps->at(i - 1).pose.pose.orientation;
119  }
120  }
121 }

189 void WaypointLoaderNode::parseWaypointForVer3(const std::string& line, const std::vector<std::string>& contents,
190  autoware_msgs::Waypoint* wp)
191 {
192  std::vector<std::string> columns;
193  parseColumns(line, &columns);
194  std::unordered_map<std::string, std::string> map;
195  for (size_t i = 0; i < contents.size(); i++)
196  {
197  map[contents.at(i)] = columns.at(i);
198  }
199 
200  wp->pose.pose.position.x = std::stod(map["x"]);
201  wp->pose.pose.position.y = std::stod(map["y"]);
202  wp->pose.pose.position.z = std::stod(map["z"]);
203  wp->pose.pose.orientation = tf::createQuaternionMsgFromYaw(std::stod(map["yaw"]));
204  wp->twist.twist.linear.x = kmph2mps(std::stod(map["velocity"]));
205  wp->change_flag = std::stoi(map["change_flag"]);
206  wp->wpstate.steering_state = (map.find("steering_flag") != map.end()) ? std::stoi(map["steering_flag"]) : 0;
207  wp->wpstate.accel_state = (map.find("accel_flag") != map.end()) ? std::stoi(map["accel_flag"]) : 0;
208  wp->wpstate.stop_state = (map.find("stop_flag") != map.end()) ? std::stoi(map["stop_flag"]) : 0;
209  wp->wpstate.event_state = (map.find("event_flag") != map.end()) ? std::stoi(map["event_flag"]) : 0;
210 }

212 FileFormat WaypointLoaderNode::checkFileFormat(const char* filename)
213 {
214  std::ifstream ifs(filename);
215 
216  if (!ifs)
217  {
218  return FileFormat::unknown;
219  }
220 
221  // get first line
222  std::string line;
223  std::getline(ifs, line);
224 
225  // parse first line
226  std::vector<std::string> parsed_columns;
227  parseColumns(line, &parsed_columns);
228 
229  // check if first element in the first column does not include digit
230  if (!std::any_of(parsed_columns.at(0).cbegin(), parsed_columns.at(0).cend(), isdigit))
231  {
232  return FileFormat::ver3;
233  }
234 
235  // if element consists only digit
236  int num_of_columns = countColumns(line);
237  ROS_INFO("columns size: %d", num_of_columns);
238 
239  return (num_of_columns == 3 ? FileFormat::ver1 // if data consists "x y z (velocity)"
240  :
241  num_of_columns == 4 ? FileFormat::ver2 // if data consists "x y z yaw (velocity)
242  :
243  FileFormat::unknown);
244 }

246 bool WaypointLoaderNode::verifyFileConsistency(const char* filename)
247 {
248  ROS_INFO("verify...");
249  std::ifstream ifs(filename);
250 
251  if (!ifs)
252  {
253  return false;
254  }
255 
256  FileFormat format = checkFileFormat(filename);
257  ROS_INFO("format: %d", static_cast<int>(format));
258  if (format == FileFormat::unknown)
259  {
260  ROS_ERROR("unknown file format");
261  return false;
262  }
263 
264  std::string line;
265  std::getline(ifs, line); // remove first line
266 
267  size_t ncol = format == FileFormat::ver1 ? 4 // x,y,z,velocity
268  :
269  format == FileFormat::ver2 ? 5 // x,y,z,yaw,velocity
270  :
271  countColumns(line);
272 
273  while (std::getline(ifs, line)) // search from second line
274  {
275  if (countColumns(line) != ncol)
276  {
277  return false;
278  }
279  }
280  return true;
281 }

283 void parseColumns(const std::string& line, std::vector<std::string>* columns)
284 {
285  std::istringstream ss(line);
286  std::string column;
287  while (std::getline(ss, column, ','))
288  {
289  while (1)
290  {
291  auto res = std::find(column.begin(), column.end(), ' ');
292  if (res == column.end())
293  {
294  break;
295  }
296  column.erase(res);
297  }
298  if (!column.empty())
299  {
300  columns->emplace_back(column);
301  }
302  }
303 }
