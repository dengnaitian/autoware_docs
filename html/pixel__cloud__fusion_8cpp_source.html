55 void ROSPixelCloudFusionApp::CloudCallback(const sensor_msgs::PointCloud2::ConstPtr &in_cloud_msg)
56 {
57  if (current_frame_.empty() || image_frame_id_ == "")
58  {
59  ROS_INFO("[%s] Waiting for Image frame to be available.", __APP_NAME__);
60  return;
61  }
62  if (!camera_lidar_tf_ok_)
63  {
64  camera_lidar_tf_ = FindTransform(image_frame_id_,
65  in_cloud_msg->header.frame_id);
66  }
67  if (!camera_info_ok_ || !camera_lidar_tf_ok_)
68  {
69  ROS_INFO("[%s] Waiting for Camera-Lidar TF and Intrinsics to be available.", __APP_NAME__);
70  return;
71  }
72 
73  pcl::PointCloud<pcl::PointXYZ>::Ptr in_cloud(new pcl::PointCloud<pcl::PointXYZ>);
74  pcl::PointCloud<pcl::PointXYZRGB>::Ptr out_cloud(new pcl::PointCloud<pcl::PointXYZRGB>);
75  pcl::fromROSMsg(*in_cloud_msg, *in_cloud);
76  std::unordered_map<cv::Point, pcl::PointXYZ> projection_map;
77 
78  std::vector<pcl::PointXYZ> cam_cloud(in_cloud->points.size());
79  for (size_t i = 0; i < in_cloud->points.size(); i++)
80  {
81  cam_cloud[i] = TransformPoint(in_cloud->points[i], camera_lidar_tf_);
82  int u = int(cam_cloud[i].x * fx_ / cam_cloud[i].z + cx_);
83  int v = int(cam_cloud[i].y * fy_ / cam_cloud[i].z + cy_);
84  if ((u >= 0) && (u < image_size_.width)
85  && (v >= 0) && (v < image_size_.height)
86  && cam_cloud[i].z > 0
87  )
88  {
89  projection_map.insert(std::pair<cv::Point, pcl::PointXYZ>(cv::Point(u, v), in_cloud->points[i]));
90  }
91  }
92 
93  out_cloud->points.clear();
94 
95 #pragma omp for
96  for (int row = 0; row < image_size_.height; row++)
97  {
98  for (int col = 0; col < image_size_.width; col++)
99  {
100  std::unordered_map<cv::Point, pcl::PointXYZ>::const_iterator iterator_3d_2d;
101  pcl::PointXYZ corresponding_3d_point;
102  pcl::PointXYZRGB colored_3d_point;
103  iterator_3d_2d = projection_map.find(cv::Point(col, row));
104  if (iterator_3d_2d != projection_map.end())
105  {
106  corresponding_3d_point = iterator_3d_2d->second;
107  cv::Vec3b rgb_pixel = current_frame_.at<cv::Vec3b>(row, col);
108  colored_3d_point.x = corresponding_3d_point.x;
109  colored_3d_point.y = corresponding_3d_point.y;
110  colored_3d_point.z = corresponding_3d_point.z;
111  colored_3d_point.r = rgb_pixel[2];
112  colored_3d_point.g = rgb_pixel[1];
113  colored_3d_point.b = rgb_pixel[0];
114  out_cloud->points.push_back(colored_3d_point);
115  }
116  }
117  }
118  // Publish PC
119  sensor_msgs::PointCloud2 cloud_msg;
120  pcl::toROSMsg(*out_cloud, cloud_msg);
121  cloud_msg.header = in_cloud_msg->header;
122  publisher_fused_cloud_.publish(cloud_msg);
123 }

125 void ROSPixelCloudFusionApp::IntrinsicsCallback(const sensor_msgs::CameraInfo &in_message)
126 {
127  image_size_.height = in_message.height;
128  image_size_.width = in_message.width;
129 
130  camera_instrinsics_ = cv::Mat(3, 3, CV_64F);
131  for (int row = 0; row < 3; row++)
132  {
133  for (int col = 0; col < 3; col++)
134  {
135  camera_instrinsics_.at<double>(row, col) = in_message.K[row * 3 + col];
136  }
137  }
138 
139  distortion_coefficients_ = cv::Mat(1, 5, CV_64F);
140  for (int col = 0; col < 5; col++)
141  {
142  distortion_coefficients_.at<double>(col) = in_message.D[col];
143  }
144 
145  fx_ = static_cast<float>(in_message.P[0]);
146  fy_ = static_cast<float>(in_message.P[5]);
147  cx_ = static_cast<float>(in_message.P[2]);
148  cy_ = static_cast<float>(in_message.P[6]);
149 
150  intrinsics_subscriber_.shutdown();
151  camera_info_ok_ = true;
152  ROS_INFO("[%s] CameraIntrinsics obtained.", __APP_NAME__);
153 }
