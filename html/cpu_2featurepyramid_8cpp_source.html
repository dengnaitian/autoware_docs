159  const int vis_R[2] ={visible[0]-1,visible[1]-1};
160  const int vp0=dims[0]-2;
161  const int vp1=dims[1]-2;
162  const int SQUARE =dims[0]*dims[1];
163  const FLOAT SBIN = FLOAT(sbin);

165

168  FLOAT *Norm = (FLOAT*)calloc(BLOCK_SQ,sizeof(FLOAT)); // Norm
169 
170  //feature(Output)
171  FLOAT *feat=(FLOAT*)calloc(OUT_SIZE[0]*OUT_SIZE[1]*OUT_SIZE[2],sizeof(FLOAT));
172 
173  //calculate HOG histgram
174  for(int x=1;x<vis_R[1];x++)
175  {
176  //parameters for interpolation
177  FLOAT xp=((FLOAT)x+0.5)/SBIN-0.5;
178  int ixp=(int)floor(xp);
179  int ixpp=ixp+1;
180  int ixp_b = ixp * blocks[0];
181  int ixpp_b = ixp_b + blocks[0];

183  FLOAT vx1=1.0-vx0;
184  bool flag1=true,flag2=true,flagX=true;
185  if(ixp<0) {flag1=false;flagX=false;}
186  if(ixpp>=blocks[1]) {flag2=false;flagX=false;}
187  int YC=min_i(x,vp1)*dims[0];
188  FLOAT *SRC_YC = SRC+YC;
189 
190  for(int y=1;y<vis_R[0];y++)
191  {
192  //first color channel
193  FLOAT *s=SRC_YC+min_i(y,vp0);
194  FLOAT dy=*(s+1)-*(s-1);
195  FLOAT dx=*(s+dims[0])-*(s-dims[0]);
196  FLOAT v=dx*dx+dy*dy;
197 
198  //second color channel
199  s+=SQUARE;
200  FLOAT dy2=*(s+1)-*(s-1);
201  FLOAT dx2=*(s+dims[0])-*(s-dims[0]);
202  FLOAT v2=dx2*dx2+dy2*dy2;
203 
204  //third color channel
205  s+=SQUARE;
206  FLOAT dy3=*(s+1)-*(s-1);
207  FLOAT dx3=*(s+dims[0])-*(s-dims[0]);
208  FLOAT v3=dx3*dx3+dy3*dy3;
209 
210  //pick channel with strongest gradient
211  if(v2>v){v=v2;dx=dx2;dy=dy2;}
212  if(v3>v){v=v3;dx=dx3;dy=dy3;}
213 
214  FLOAT best_dot=0.0;
215  int best_o=0;
216 
217  //snap to one of 18 orientations
218  for(int o=0;o<9;o++)
219  {
220  FLOAT dot=Hcos[o]*dx+Hsin[o]*dy;
221  if(dot>best_dot) {best_dot=dot;best_o=o;}
222  else if (-dot>best_dot) {best_dot=-dot;best_o=o+9;}
223  }
224 
225  //Add to 4 histgrams around pixel using linear interpolation
226  FLOAT yp=((FLOAT)y+0.5)/SBIN-0.5;
227  int iyp=(int)floor(yp);
228  int iypp=iyp+1;
229  FLOAT vy0=yp-(FLOAT)iyp;
230  FLOAT vy1=1.0-vy0;
231  v=sqrt(v);
232  int ODim=best_o*BLOCK_SQ;
233  FLOAT *Htemp = HHist+ODim;
234  FLOAT vx1Xv =vx1*v;
235  FLOAT vx0Xv = vx0*v;
236 
237  if(flagX)
238  {
239  if(iyp>=0)
240  {
241  *(Htemp+ ixp_b+iyp)+=vy1*vx1Xv; //1-少数をxyでかけたものにエッジ強度の2乗をかけたもの
242  *(Htemp+ ixpp_b+iyp)+=vy1*vx0Xv;
243  }
244  if (iypp<blocks[0])
245  {
246  *(Htemp+ ixp_b+iypp)+=vy0*vx1Xv;
247  *(Htemp+ ixpp_b+iypp)+=vy0*vx0Xv;
248  }
249  }
250  else if(flag1)
251  {
252  if (iyp>=0) *(Htemp+ixp_b+iyp)+=vy1*vx1Xv;
253  if (iypp<blocks[0]) *(Htemp+ixp_b+iypp)+=vy0*vx1Xv;
254  }
255  else if(flag2)
256  {
257  if(iyp>=0) *(Htemp+ixpp_b+iyp)+=vy1*vx0Xv;
258  if(iypp<blocks[0]) *(Htemp+ixpp_b+iypp)+=vy0*vx0Xv;
259  }
260  }
261  }
262 
263  //compute energy in each block by summing over orientations
264  for(int kk=0;kk<9;kk++)
265  {
266  FLOAT *src1=HHist+kk*BLOCK_SQ;
267  FLOAT *src2=src1+DIM_N;
268  FLOAT *dst=Norm;
269  FLOAT *end=Norm+BLOCK_SQ;
270  while(dst<end)
271  {
272  FLOAT sss=*src1+*src2;
273  *(dst++)+=sss*sss;
274  src1++;src2++;
275  }
276  }
277 
278  //compute features
279  for(int x=0;x<OUT_SIZE[1];x++)
280  {
281  FLOAT *dst_X = feat+x*OUT_SIZE[0];
282  int BB = x*blocks[0];
283  int BA = BB+blocks[0];
284 
285  FLOAT *pt = Norm+BA;
286  FLOAT nc1 = 1.0/sqrt(*pt+*(pt+1)+*(pt+blocks[0])+*(pt+BX)+eps);
287  pt = Norm+BB;
288  FLOAT nc3 = 1.0/sqrt(*pt+*(pt+1)+*(pt+blocks[0])+*(pt+BX)+eps);
289 
290  for(int y=0;y<OUT_SIZE[0];y++)
291  {
292  FLOAT *dst=dst_X+y;
293  FLOAT *src,*p,n1,n2,n3,n4;
294  //calculate normarize factor
295  int yp = y+1;
296 
297  p=Norm+BA+yp;
298  n1 = 1.0/sqrt(*p+*(p+1)+*(p+blocks[0])+*(p+BX)+eps);
299  n2 = nc1;
300  nc1 = n1;
301 
302  p=Norm+BB+yp;
303  n3 = 1.0/sqrt(*p+*(p+1)+*(p+blocks[0])+*(p+BX)+eps);
304  n4 = nc3;
305  nc3 = n3;
306 
307  //features for each orientations
308  FLOAT t1=0,t2=0,t3=0,t4=0;
309 
310  //contrast-sensitive features(18)
311  src=HHist+BA+yp;
312  for(int kk=0;kk<18;kk++)
313  {
314  FLOAT h1=min_2(*src*n1);
315  FLOAT h2=min_2(*src*n2);
316  FLOAT h3=min_2(*src*n3);
317  FLOAT h4=min_2(*src*n4);
318  *dst=0.5*(h1+h2+h3+h4);
319  t1+=h1;
320  t2+=h2;
321  t3+=h3;
322  t4+=h4;
323  dst+=O_DIM;
324  src+=BLOCK_SQ;
325  }
326 
327  //contrast-insensitive features(9)
328  src=HHist+BA+yp;
329  for(int kk=0;kk<9;kk++)
330  {
331  FLOAT sum = *src+*(src+DIM_N);
332  FLOAT h1=min_2(sum*n1);
333  FLOAT h2=min_2(sum*n2);
334  FLOAT h3=min_2(sum*n3);
335  FLOAT h4=min_2(sum*n4);
336  *dst=0.5*(h1+h2+h3+h4);
337  dst+=O_DIM;
338  src+=BLOCK_SQ;
339  }
340 
341  //texture gradient
342  *dst=0.2357*t1;
343  dst+=O_DIM;
344  *dst=0.2357*t2;
345  dst+=O_DIM;
346  *dst=0.2357*t3;
347  dst+=O_DIM;
348  *dst=0.2357*t4;
349  }
350  }
351 
352  //Release
353  s_free(HHist);
354  s_free(Norm);
355 
356  //size of feature(output)
357  *FTSIZE=OUT_SIZE[0];
358  *(FTSIZE+1)=OUT_SIZE[1];
359  // printf("feat%f\n",*(feat));
360  return(feat);
