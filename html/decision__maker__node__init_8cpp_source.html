287 void DecisionMakerNode::initVectorMap(void)
288 {
289  int _index = 0;
290 
291  const std::vector<CrossRoad> crossroads = g_vmap.findByFilter([](const CrossRoad& crossroad) { return true; });
292  if (crossroads.empty())
293  {
294  ROS_INFO("crossroads have not found\n");
295  return;
296  }
297 
298  for (const auto& cross_road : crossroads)
299  {
300  geometry_msgs::Point _prev_point;
301  Area area = g_vmap.findByKey(Key<Area>(cross_road.aid));
302  CrossRoadArea carea;
303  carea.id = _index++;
304  carea.area_id = area.aid;
305 
306  double x_avg = 0.0, x_min = 0.0, x_max = 0.0;
307  double y_avg = 0.0, y_min = 0.0, y_max = 0.0;
308  double z = 0.0;
309  int points_count = 0;
310 
311  const std::vector<Line> lines =
312  g_vmap.findByFilter([&area](const Line& line) { return area.slid <= line.lid && line.lid <= area.elid; });
313  for (const auto& line : lines)
314  {
315  const std::vector<Point> points =
316  g_vmap.findByFilter([&line](const Point& point) { return line.bpid == point.pid; });
317  for (const auto& point : points)
318  {
319  geometry_msgs::Point _point;
320  _point.x = point.ly;
321  _point.y = point.bx;
322  _point.z = point.h;
323 
324  if (_prev_point.x == _point.x && _prev_point.y == _point.y)
325  continue;
326 
327  _prev_point = _point;
328  points_count++;
329  carea.points.push_back(_point);
330 
331  // calc a centroid point and about intersects size
332  x_avg += _point.x;
333  y_avg += _point.y;
334  x_min = (x_min == 0.0) ? _point.x : std::min(_point.x, x_min);
335  x_max = (x_max == 0.0) ? _point.x : std::max(_point.x, x_max);
336  y_min = (y_min == 0.0) ? _point.y : std::min(_point.y, y_min);
337  y_max = (y_max == 0.0) ? _point.y : std::max(_point.y, y_max);
338  z = _point.z;
339  } // points iter
340  } // line iter
341  carea.bbox.pose.position.x = x_avg / (double)points_count * 1.5 /* expanding rate */;
342  carea.bbox.pose.position.y = y_avg / (double)points_count * 1.5;
343  carea.bbox.pose.position.z = z;
344  carea.bbox.dimensions.x = x_max - x_min;
345  carea.bbox.dimensions.y = y_max - y_min;
346  carea.bbox.dimensions.z = 2;
347  carea.bbox.label = 1;
348  intersects.push_back(carea);
349  }
350 }
