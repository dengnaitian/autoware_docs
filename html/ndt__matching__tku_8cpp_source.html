85 void pose_mod(Posture *pose)
86 {
87  while (pose->theta < -M_PI)
88  pose->theta += 2 * M_PI;
89  while (pose->theta > M_PI)

92  pose->theta2 += 2 * M_PI;
93  while (pose->theta2 > M_PI)
94  pose->theta2 -= 2 * M_PI;
95  while (pose->theta3 < -M_PI)
96  pose->theta3 += 2 * M_PI;
97  while (pose->theta3 > M_PI)
98  pose->theta3 -= 2 * M_PI;
