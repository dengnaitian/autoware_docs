79 void create_waypoint(const tablet_socket_msgs::route_cmd& msg)
80 {
81  std_msgs::Header header;
82  header.stamp = ros::Time::now();
83  header.frame_id = frame_id;
84 
85  if (all_vmap.points.empty() || all_vmap.lanes.empty() || all_vmap.nodes.empty()) {
86  cached_route.header = header;
87  cached_route.point = msg.point;
88  return;
89  }
90 
91  lane_planner::vmap::VectorMap coarse_vmap = lane_planner::vmap::create_coarse_vmap_from_route(msg);
92  if (coarse_vmap.points.size() < 2)
93  return;
94 
95  std::vector<lane_planner::vmap::VectorMap> fine_vmaps;
96  lane_planner::vmap::VectorMap fine_mostleft_vmap =
97  lane_planner::vmap::create_fine_vmap(lane_vmap, lane_planner::vmap::LNO_MOSTLEFT, coarse_vmap,
98  search_radius, waypoint_max);
99  if (fine_mostleft_vmap.points.size() < 2)
100  return;
101  fine_vmaps.push_back(fine_mostleft_vmap);
102 
103  int lcnt = count_lane(fine_mostleft_vmap);
104  for (int i = lane_planner::vmap::LNO_MOSTLEFT + 1; i <= lcnt; ++i) {
105  lane_planner::vmap::VectorMap v =
106  lane_planner::vmap::create_fine_vmap(lane_vmap, i, coarse_vmap, search_radius, waypoint_max);
107  if (v.points.size() < 2)
108  continue;
109  fine_vmaps.push_back(v);
110  }
111 
112  autoware_msgs::LaneArray lane_waypoint;
113  for (const lane_planner::vmap::VectorMap& v : fine_vmaps) {
114  autoware_msgs::Lane l;
115  l.header = header;
116  l.increment = 1;
117 
118  size_t last_index = v.points.size() - 1;
119  for (size_t i = 0; i < v.points.size(); ++i) {
120  double yaw;
121  if (i == last_index) {
122  geometry_msgs::Point p1 =
123  lane_planner::vmap::create_geometry_msgs_point(v.points[i]);
124  geometry_msgs::Point p2 =
125  lane_planner::vmap::create_geometry_msgs_point(v.points[i - 1]);
126  yaw = atan2(p2.y - p1.y, p2.x - p1.x);
127  yaw -= M_PI;
128  } else {
129  geometry_msgs::Point p1 =
130  lane_planner::vmap::create_geometry_msgs_point(v.points[i]);
131  geometry_msgs::Point p2 =
132  lane_planner::vmap::create_geometry_msgs_point(v.points[i + 1]);
133  yaw = atan2(p2.y - p1.y, p2.x - p1.x);
134  }
135 
136  autoware_msgs::Waypoint w;
137  w.pose.header = header;
138  w.pose.pose.position = lane_planner::vmap::create_geometry_msgs_point(v.points[i]);
139  w.pose.pose.orientation = tf::createQuaternionMsgFromYaw(yaw);
140  w.twist.header = header;
141  w.twist.twist.linear.x = velocity / 3.6; // to m/s
142  l.waypoints.push_back(w);
143  }
144  lane_waypoint.lanes.push_back(l);
145  }
146  waypoint_pub.publish(lane_waypoint);
147 
148  for (size_t i = 0; i < fine_vmaps.size(); ++i) {
149  std::stringstream ss;
150  ss << "_" << i;
151 
152  std::vector<std::string> v1 = split(output_file, '/');
153  std::vector<std::string> v2 = split(v1.back(), '.');
154  v2[0] = v2.front() + ss.str();
155  v1[v1.size() - 1] = join(v2, '.');
156  std::string path = join(v1, '/');
157 
158  lane_planner::vmap::write_waypoints(fine_vmaps[i].points, velocity, path);
159  }
160 }
