265 void MotionPrediction::GenerateCurbsObstacles(std::vector<PlannerHNS::DetectedObject>& curb_obstacles)
266 {
267  if(!bNewCurrentPos) return;
268 
269  for(unsigned int ic = 0; ic < m_Map.curbs.size(); ic++)
270  {
271 
272  if(m_Map.curbs.at(ic).points.size() > 0)
273  {
274  PlannerHNS::DetectedObject obj;
275  obj.center.pos = m_Map.curbs.at(ic).points.at(0);
276 
277  if(curb_obstacles.size()>0)
278  {
279  double distance_to_prev = hypot(curb_obstacles.at(curb_obstacles.size()-1).center.pos.y-obj.center.pos.y, curb_obstacles.at(curb_obstacles.size()-1).center.pos.x-obj.center.pos.x);
280  if(distance_to_prev < m_DistanceBetweenCurbs)
281  continue;
282  }
283 
284  double longitudinalDist = hypot(m_CurrentPos.pos.y -obj.center.pos.y, m_CurrentPos.pos.x-obj.center.pos.x);
285 
286  if(longitudinalDist > m_PlanningParams.horizonDistance)
287  continue;
288 
289  obj.bDirection = false;
290  obj.bVelocity = false;
291  obj.id = -1;
292  obj.t = PlannerHNS::SIDEWALK;
293  obj.label = "curb";
294  for(unsigned int icp=0; icp< m_Map.curbs.at(ic).points.size(); icp++)
295  {
296  obj.contour.push_back(m_Map.curbs.at(ic).points.at(icp));
297  }
298 
299  curb_obstacles.push_back(obj);
300  }
301  }
