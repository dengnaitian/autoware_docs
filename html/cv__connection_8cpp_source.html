48 OpenCVConnector::OpenCVConnector(std::string topic_name, std::string camera_frame_id, std::string cam_info_file, int buffer):
49 it(nh), counter(0), camera_id(camera_frame_id),camera_info_manager(ros::NodeHandle(topic_name), camera_frame_id)
50 {
51  //init image pub topic
52  std::string topic_raw = topic_name + std::string("/image_raw");
53  std::string topic_jpg = topic_name + std::string("/image_raw/compressed");
54  pub = it.advertise(topic_raw, buffer);
55  pub_jpg = nh.advertise<sensor_msgs::CompressedImage>(topic_jpg, buffer);
56  //init pub camera info topic
57  pub_caminfo = nh.advertise<sensor_msgs::CameraInfo>(camera_frame_id + std::string("/camera_info"), 1);
58 
59  //init camera info
60  if(camera_info_manager.validateURL(cam_info_file))
61  {
62  camera_info_manager.loadCameraInfo(cam_info_file);
63  camera_info = camera_info_manager.getCameraInfo();
64  }
65  else
66  {
67  ROS_ERROR("ERROR READING CALIBRATION FILE: %s", cam_info_file.c_str());
68  }
69 }

77 void OpenCVConnector::WriteToOpenCV(unsigned char* buffer, int width_in, int height_in, int width_pub, int height_pub)
78 {
79  cv::Mat mat_img(cv::Size(width_in, height_in), CV_8UC4, buffer); // create a cv::Mat from rgbaImage
80  cv::Mat dst;
81 
82  // if we need to resize
83  if((width_in != width_pub) || (height_in != height_pub))
84  {
85  cv::resize(mat_img, dst, cv::Size(width_pub, height_pub)); // resize to the publishing size
86  }
87  else
88  {
89  dst = mat_img;
90  }
91  cv::Mat converted; // new cv::Mat();
92  cv::cvtColor(dst,converted,cv::COLOR_RGBA2RGB); // COLOR_BGRA2BGR
93  cv_bridge::CvImage img_bridge;
94  sensor_msgs::Image img_msg; // message to be sent
95 
96  std_msgs::Header header; // empty header
97  header.seq = counter; // user defined counter
98  header.stamp = ros::Time::now(); // time
99  header.frame_id = camera_id; // camera id
100  img_bridge = cv_bridge::CvImage(header, sensor_msgs::image_encodings::RGB8, converted);
101  img_bridge.toImageMsg(img_msg); // from cv_bridge to sensor_msgs::Image
102  pub.publish(img_msg); // pub image
103 
104  //publish camera info
105  camera_info.header = header;
106  pub_caminfo.publish(camera_info);
107 }

109 void OpenCVConnector::WriteToJpeg(uint8_t* data, uint32_t compressed_size)
110 {
111  // publishing original size only
112  sensor_msgs::CompressedImage img_msg_compressed;
113  img_msg_compressed.data.resize(compressed_size);
114  memcpy(&img_msg_compressed.data[0], data, compressed_size);
115  std_msgs::Header header; // empty header
116  header.seq = counter; // user defined counter
117  header.stamp = ros::Time::now(); // time
118  header.frame_id = camera_id; // camera id
119  img_msg_compressed.header = header;
120  img_msg_compressed.format = "jpeg";
121  pub_jpg.publish(img_msg_compressed);
122 
123  //publish camera info
124  camera_info.header = header;
125  pub_caminfo.publish(camera_info);
126 }
