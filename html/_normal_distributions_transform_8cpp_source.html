268 template <typename PointSourceType, typename PointTargetType>
269 double NormalDistributionsTransform<PointSourceType, PointTargetType>::updateDerivatives(Eigen::Matrix<double, 6, 1> &score_gradient, Eigen::Matrix<double, 6, 6> &hessian,
270  Eigen::Matrix<double, 3, 6> point_gradient, Eigen::Matrix<double, 18, 6> point_hessian,
271  Eigen::Vector3d &x_trans, Eigen::Matrix3d &c_inv, bool compute_hessian)
272 {
273  Eigen::Vector3d cov_dxd_pi;
274  double e_x_cov_x = exp(-gauss_d2_ * x_trans.dot(c_inv * x_trans) / 2);
275  double score_inc = -gauss_d1_ * e_x_cov_x;
276 
277  e_x_cov_x = gauss_d2_ * e_x_cov_x;
278 
279  if (e_x_cov_x > 1 || e_x_cov_x < 0 || e_x_cov_x != e_x_cov_x) {
280  return 0.0;
281  }
282 
283  e_x_cov_x *= gauss_d1_;
284 
285  for (int i = 0; i < 6; i++) {
286  cov_dxd_pi = c_inv * point_gradient.col(i);
287 
288  score_gradient(i) += x_trans.dot(cov_dxd_pi) * e_x_cov_x;
289 
290  if (compute_hessian) {
291  for (int j = 0; j < hessian.cols(); j++) {
292  hessian(i, j) += e_x_cov_x * (-gauss_d2_ * x_trans.dot(cov_dxd_pi) * x_trans.dot(c_inv * point_gradient.col(j)) +
293  x_trans.dot(c_inv * point_hessian.block<3, 1>(3 * i, j)) +
294  point_gradient.col(j).dot(cov_dxd_pi));
295  }
296  }
297  }
298 
299  return score_inc;
300 }

310  sx = 0.0;
311  } else {
312  cx = cos(pose(3));
313  sx = sin(pose(3));
314  }
315 
316  if (fabs(pose(4)) < 10e-5) {

331

431  double step_init, double step_max, double step_min, double &score,

433  typename pcl::PointCloud<PointSourceType> &trans_cloud)
434 {

436  double d_phi_0 = -(score_gradient.dot(step_dir));
437 
438  Eigen::Matrix<double, 6, 1> x_t;
439 
440  if (d_phi_0 >= 0) {
441  if (d_phi_0 == 0) {
442  return 0;
443  } else {
444  d_phi_0 *= -1;
445  step_dir *= -1;
446  }
447  }
448 
449  int max_step_iterations = 10;
450  int step_iterations = 0;
451

453  double nu = 0.9;
454  double a_l = 0, a_u = 0;
455

457  double g_l = auxilaryFunction_dPsiMT(d_phi_0, d_phi_0, mu);
458 
459  double f_u = auxilaryFunction_PsiMT(a_u, phi_0, phi_0, d_phi_0, mu);
460  double g_u = auxilaryFunction_dPsiMT(d_phi_0, d_phi_0, mu);
461 
462  bool interval_converged = (step_max - step_min) > 0, open_interval = true;
463 
464  double a_t = step_init;
465  a_t = std::min(a_t, step_max);
466  a_t = std::max(a_t, step_min);
467 
468  x_t = x + step_dir * a_t;
469 
470  final_transformation_ = (Eigen::Translation<float, 3>(static_cast<float>(x_t(0)), static_cast<float>(x_t(1)), static_cast<float>(x_t(2))) *
471  Eigen::AngleAxis<float>(static_cast<float>(x_t(3)), Eigen::Vector3f::UnitX()) *
472  Eigen::AngleAxis<float>(static_cast<float>(x_t(4)), Eigen::Vector3f::UnitY()) *
473  Eigen::AngleAxis<float>(static_cast<float>(x_t(5)), Eigen::Vector3f::UnitZ())).matrix();
474 
475  transformPointCloud(*source_cloud_, trans_cloud, final_transformation_);
476 
477  score = computeDerivatives(score_gradient, hessian, trans_cloud, x_t, true);
478 
479  double phi_t = -score;
480  double d_phi_t = -(score_gradient.dot(step_dir));
481  double psi_t = auxilaryFunction_PsiMT(a_t, phi_t, phi_0, d_phi_0, mu);
482  double d_psi_t = auxilaryFunction_dPsiMT(d_phi_t, d_phi_0, mu);
483 
484  while (!interval_converged && step_iterations < max_step_iterations && !(psi_t <= 0 && d_phi_t <= -nu * d_phi_0)) {
485  if (open_interval) {
486  a_t = trialValueSelectionMT(a_l, f_l, g_l, a_u, f_u, g_u, a_t, psi_t, d_psi_t);
487  } else {
488  a_t = trialValueSelectionMT(a_l, f_l, g_l, a_u, f_u, g_u, a_t, phi_t, d_phi_t);
489  }
490 
491  a_t = (a_t < step_max) ? a_t : step_max;
492  a_t = (a_t > step_min) ? a_t : step_min;
493 
494  x_t = x + step_dir * a_t;
495 
496  final_transformation_ = (Eigen::Translation<float, 3>(static_cast<float>(x_t(0)), static_cast<float>(x_t(1)), static_cast<float>(x_t(2))) *
497  Eigen::AngleAxis<float>(static_cast<float>(x_t(3)), Eigen::Vector3f::UnitX()) *
498  Eigen::AngleAxis<float>(static_cast<float>(x_t(4)), Eigen::Vector3f::UnitY()) *
499  Eigen::AngleAxis<float>(static_cast<float>(x_t(5)), Eigen::Vector3f::UnitZ())).matrix();
500 
501  transformPointCloud(*source_cloud_, trans_cloud, final_transformation_);
502 
503  score = computeDerivatives(score_gradient, hessian, trans_cloud, x_t, false);
504 
505  phi_t -= score;
506  d_phi_t -= (score_gradient.dot(step_dir));
507  psi_t = auxilaryFunction_PsiMT(a_t, phi_t, phi_0, d_phi_0, mu);
508  d_psi_t = auxilaryFunction_dPsiMT(d_phi_t, d_phi_0, mu);
509 
510  if (open_interval && (psi_t <= 0 && d_psi_t >= 0)) {
511  open_interval = false;
512 
513  f_l += phi_0 - mu * d_phi_0 * a_l;
514  g_l += mu * d_phi_0;
515 
516  f_u += phi_0 - mu * d_phi_0 * a_u;
517  g_u += mu * d_phi_0;
518  }
519 
520  if (open_interval) {
521  interval_converged = updateIntervalMT(a_l, f_l, g_l, a_u, f_u, g_u, a_t, psi_t, d_psi_t);
522  } else {
523  interval_converged = updateIntervalMT(a_l, f_l, g_l, a_u, f_u, g_u, a_t, phi_t, d_phi_t);
524  }

539 template <typename PointSourceType, typename PointTargetType>
540 double NormalDistributionsTransform<PointSourceType, PointTargetType>::trialValueSelectionMT (double a_l, double f_l, double g_l,
541  double a_u, double f_u, double g_u,
542  double a_t, double f_t, double g_t)
543 {
544  // Case 1 in Trial Value Selection [More, Thuente 1994]
545  if (f_t > f_l) {
546  // Calculate the minimizer of the cubic that interpolates f_l, f_t, g_l and g_t
547  // Equation 2.4.52 [Sun, Yuan 2006]
548  double z = 3 * (f_t - f_l) / (a_t - a_l) - g_t - g_l;
549  double w = std::sqrt (z * z - g_t * g_l);
550  // Equation 2.4.56 [Sun, Yuan 2006]
551  double a_c = a_l + (a_t - a_l) * (w - g_l - z) / (g_t - g_l + 2 * w);
552 
553  // Calculate the minimizer of the quadratic that interpolates f_l, f_t and g_l
554  // Equation 2.4.2 [Sun, Yuan 2006]
555  double a_q = a_l - 0.5 * (a_l - a_t) * g_l / (g_l - (f_l - f_t) / (a_l - a_t));
556 
557  if (std::fabs (a_c - a_l) < std::fabs (a_q - a_l)) {
558  return (a_c);
559  } else {
560  return (0.5 * (a_q + a_c));
561  }
562  }
563  // Case 2 in Trial Value Selection [More, Thuente 1994]
564  else if (g_t * g_l < 0) {
565  // Calculate the minimizer of the cubic that interpolates f_l, f_t, g_l and g_t
566  // Equation 2.4.52 [Sun, Yuan 2006]
567  double z = 3 * (f_t - f_l) / (a_t - a_l) - g_t - g_l;
568  double w = std::sqrt (z * z - g_t * g_l);
569  // Equation 2.4.56 [Sun, Yuan 2006]
570  double a_c = a_l + (a_t - a_l) * (w - g_l - z) / (g_t - g_l + 2 * w);
571 
572  // Calculate the minimizer of the quadratic that interpolates f_l, g_l and g_t
573  // Equation 2.4.5 [Sun, Yuan 2006]
574  double a_s = a_l - (a_l - a_t) / (g_l - g_t) * g_l;
575 
576  if (std::fabs (a_c - a_t) >= std::fabs (a_s - a_t)) {
577  return (a_c);
578  } else {
579  return (a_s);
580  }
581  }
582  // Case 3 in Trial Value Selection [More, Thuente 1994]
583  else if (std::fabs (g_t) <= std::fabs (g_l)) {
584  // Calculate the minimizer of the cubic that interpolates f_l, f_t, g_l and g_t
585  // Equation 2.4.52 [Sun, Yuan 2006]
586  double z = 3 * (f_t - f_l) / (a_t - a_l) - g_t - g_l;
587  double w = std::sqrt (z * z - g_t * g_l);
588  double a_c = a_l + (a_t - a_l) * (w - g_l - z) / (g_t - g_l + 2 * w);
589 
590  // Calculate the minimizer of the quadratic that interpolates g_l and g_t
591  // Equation 2.4.5 [Sun, Yuan 2006]
592  double a_s = a_l - (a_l - a_t) / (g_l - g_t) * g_l;
593 
594  double a_t_next;
595 
596  if (std::fabs (a_c - a_t) < std::fabs (a_s - a_t)) {
597  a_t_next = a_c;
598  } else {
599  a_t_next = a_s;
600  }
601 
602  if (a_t > a_l) {
603  return (std::min (a_t + 0.66 * (a_u - a_t), a_t_next));
604  } else {
605  return (std::max (a_t + 0.66 * (a_u - a_t), a_t_next));
606  }
607  }
608  // Case 4 in Trial Value Selection [More, Thuente 1994]
609  else {
610  // Calculate the minimizer of the cubic that interpolates f_u, f_t, g_u and g_t
611  // Equation 2.4.52 [Sun, Yuan 2006]
612  double z = 3 * (f_t - f_u) / (a_t - a_u) - g_t - g_u;
613  double w = std::sqrt (z * z - g_t * g_u);
614  // Equation 2.4.56 [Sun, Yuan 2006]
615  return (a_u + (a_t - a_u) * (w - g_u - z) / (g_t - g_u + 2 * w));
616  }
617 }
