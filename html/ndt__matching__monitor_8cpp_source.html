43 geometry_msgs::PoseWithCovarianceStamped
44  ROSNDTMatchingMonitor::predict_next_pose(geometry_msgs::PoseWithCovarianceStamped prev_pose,
45  geometry_msgs::PoseWithCovarianceStamped current_pose)
46 {
47  geometry_msgs::PoseWithCovarianceStamped predicted_pose;
48 
49  double time_delta = current_pose.header.stamp.toSec() - prev_pose.header.stamp.toSec();
50 
51  if (time_delta == 0)
52  {
53  return current_pose;
54  }
55 
56  predicted_pose.header = current_pose.header;
57  predicted_pose.header.stamp.sec += time_delta;
58  predicted_pose.pose.pose.orientation = current_pose.pose.pose.orientation;
59 
60  predicted_pose.pose.pose.position.x = current_pose.pose.pose.position.x +
61  (current_pose.pose.pose.position.x - current_pose.pose.pose.position.x) / time_delta;
62  predicted_pose.pose.pose.position.y = current_pose.pose.pose.position.y +
63  (current_pose.pose.pose.position.y - current_pose.pose.pose.position.y) / time_delta;
64  predicted_pose.pose.pose.position.z = current_pose.pose.pose.position.z +
65  (current_pose.pose.pose.position.z - current_pose.pose.pose.position.z) / time_delta;
66 
67  return predicted_pose;
68 
69 }

71 void ROSNDTMatchingMonitor::initialpose_callback(const geometry_msgs::PoseWithCovarianceStamped::ConstPtr& input)
72 {
73  //if currently blocking reset and receiving a different blocking pose, then try to reset
74  if (ndt_status::NDT_FATAL == ndt_status_
75  && input->pose.pose.position.x != initialpose_.pose.pose.position.x
76  && input->pose.pose.position.y != initialpose_.pose.pose.position.y
77  && input->pose.pose.position.z != initialpose_.pose.pose.position.z
78  && input->pose.pose.orientation.x != initialpose_.pose.pose.orientation.x
79  && input->pose.pose.orientation.y != initialpose_.pose.pose.orientation.y
80  && input->pose.pose.orientation.z != initialpose_.pose.pose.orientation.z
81  && input->pose.pose.orientation.w != initialpose_.pose.pose.orientation.w)
82  {
83  ndt_status_ = ndt_status::NDT_NOT_INITIALIZED;
84  initialpose_ = *input;
85  initialized_ = false;
86  last_score_ = 0.;
87  ROS_INFO_STREAM(__APP_NAME__ << " RECEIVED NEW INITIAL POSE, RESETTING.");
88  }

91 void ROSNDTMatchingMonitor::ndt_pose_callback(const geometry_msgs::PoseStamped::ConstPtr& input)
92 {
93  geometry_msgs::PoseWithCovarianceStamped initialpose_msg;
94  initialpose_msg.header = input->header;
95  jsk_rviz_plugins::OverlayText rviz_info_text;
96  std_msgs::String ndt_status_msg;
97 
98  if (ndt_status::NDT_FATAL == ndt_status_)
99  {
100  rviz_info_text = ndt_fatal_text_;
101  ROS_ERROR_STREAM(__APP_NAME__ << " FATAL CANNOT RECOVER - STOPPING.");
102  initialpose_msg.pose = initialpose_.pose;
103  initialpose_pub_.publish(initialpose_msg);
104  overlay_info_text_pub_.publish(rviz_info_text);
105  stable_samples_ = 0;
106  return;
107  }
108 
109  if (iteration_count_ < iteration_threshold_warning_
110  && score_delta_ < score_delta_threshold_)
111  {
112  //update the last good pose
113  initialpose_.pose.pose = input->pose;
114  prev_initialpose_ = initialpose_;
115  if (stable_samples_ >= (unsigned int) min_stable_samples_)
116  {
117  ndt_status_ = ndt_status::NDT_OK;
118  initialized_ = true;
119  }
120  stable_samples_++;
121  }
122  else if ( initialized_ &&
123  ((iteration_count_ >= iteration_threshold_stop_)
124  || (iteration_count_ >= iteration_threshold_warning_ && score_delta_ >= score_delta_threshold_)))
125  {
126  ndt_status_ = ndt_status::NDT_ERROR;
127  if (gnss_pose_available_)
128  {
129  initialpose_msg.header.frame_id = gnss_pose_.header.frame_id;
130  initialpose_msg.pose = predict_next_pose(prev_gnss_pose_, gnss_pose_).pose;
131  }
132  else
133  {
134  initialpose_msg.pose = predict_next_pose(prev_initialpose_, initialpose_).pose;
135  //if there are too many predictions in a short time, set fatal status
136  if (prediction_samples_ > fatal_time_threshold_
137  && (input->header.stamp.toSec() - last_prediction_time_.toSec()) <= fatal_time_threshold_ )
138  {
139  ndt_status_ = ndt_status::NDT_FATAL;
140  }
141  last_prediction_time_ = input->header.stamp;
142  prediction_samples_++;
143  }
144 
145  if (gnss_pose_available_
146  || stable_samples_ >= (unsigned int) min_stable_samples_ )
147  {
148  ROS_INFO_STREAM(__APP_NAME__ << " Resetting position.");
149  initialpose_pub_.publish(initialpose_msg);
150  stable_samples_ = 0;
151  }
152  }
153  else if (initialized_ &&
154  ((iteration_count_ > iteration_threshold_warning_
155  && iteration_count_< iteration_threshold_stop_)
156  || ndt_status_ == ndt_status::NDT_ERROR))
157  {
158  ndt_status_ = ndt_status::NDT_WARNING;
159  prev_initialpose_ = initialpose_;
160  }
161 
162  prev_gnss_pose_ = gnss_pose_;
163 
164  if (!initialized_)
165  {
166  rviz_info_text = ndt_not_ready_text_;
167  } else
168  {
169  switch (ndt_status_)
170  {
171  case ndt_status::NDT_ERROR:
172  rviz_info_text = ndt_error_text_;
173  break;
174  case ndt_status::NDT_WARNING:
175  rviz_info_text = ndt_warn_text_;
176  break;
177  case ndt_status::NDT_OK:
178  rviz_info_text = ndt_normal_text_;
179  break;
180  case ndt_status::NDT_NOT_INITIALIZED:
181  default:
182  rviz_info_text = ndt_not_ready_text_;
183  break;
184  }
185  }
186 
187  rviz_info_text.text+=gnss_text_;
188 
189  ndt_status_msg.data = ndt_status_names[ndt_status_];
190 
191  overlay_info_text_pub_.publish(rviz_info_text);
192  ndt_status_pub_.publish(ndt_status_msg);
193  last_score_ = current_score_;
194 }

196 void ROSNDTMatchingMonitor::Run()
197 {
198  ros::NodeHandle nh;
199  ros::NodeHandle private_nh("~");
200 
201  // Geting parameters
202  private_nh.param("/monitor/iteration_threshold_warn", iteration_threshold_warning_, NDT_THRESHOLD_ITERATION_WARN);
203  private_nh.param("/monitor/iteration_threshold_stop", iteration_threshold_stop_, NDT_THRESHOLD_ITERATION_STOP);
204  private_nh.param("/monitor/score_delta_threshold", score_delta_threshold_, NDT_THRESHOLD_SCORE_MAX_DELTA);
205  private_nh.param("/monitor/min_stable_samples", min_stable_samples_, NDT_MIN_STABLE_SAMPLES);
206  private_nh.param("/monitor/fatal_time_threshold", fatal_time_threshold_, NDT_TIME_TO_FATAL_PREDICTIONS);
207 
208 
209  //swap values in case of error
210  if (iteration_threshold_warning_ > iteration_threshold_stop_) {
211  auto aux = iteration_threshold_stop_;
212  iteration_threshold_stop_ = iteration_threshold_warning_;
213  iteration_threshold_warning_ = aux;
214  }
215 
216  // Subscribers
217  ros::Subscriber ndt_stat_sub = nh.subscribe("/ndt_stat", 10, &ROSNDTMatchingMonitor::ndt_stat_callback, this);
218  ros::Subscriber ndt_pose_sub = nh.subscribe("/ndt_pose", 10, &ROSNDTMatchingMonitor::ndt_pose_callback, this);
219  ros::Subscriber initial_pose_sub = nh.subscribe("/initialpose", 10, &ROSNDTMatchingMonitor::initialpose_callback, this);
220  ros::Subscriber gnss_sub = nh.subscribe("gnss_pose", 10, &ROSNDTMatchingMonitor::gnss_callback, this);
221 
222  // Publishers
223  initialpose_pub_ = nh.advertise<geometry_msgs::PoseWithCovarianceStamped>("/initialpose", 1);
224  overlay_info_text_pub_ = nh.advertise<jsk_rviz_plugins::OverlayText>("/ndt_monitor/ndt_info_text", 1);
225  ndt_status_pub_ = nh.advertise<std_msgs::String>("/ndt_monitor/ndt_status", 1);
226 
227  ros::spin();
228 
229 }
