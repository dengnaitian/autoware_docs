47 void publish_scan(ros::Publisher *pub, uint32_t *range_values,
48  uint32_t n_range_values, uint32_t *intensity_values,
49  uint32_t n_intensity_values, double scale, ros::Time start,
50  double scan_time, bool inverted, float angle_min,
51  float angle_max, std::string frame_id)
52 {
53  static int scan_count = 0;
54  sensor_msgs::LaserScan scan_msg;
55  scan_msg.header.frame_id = frame_id;
56  scan_count++;
57  if(inverted) {
58  scan_msg.angle_min = angle_max;
59  scan_msg.angle_max = angle_min;
60  } else {
61  scan_msg.angle_min = angle_min;
62  scan_msg.angle_max = angle_max;
63  }
64  scan_msg.angle_increment = (scan_msg.angle_max - scan_msg.angle_min)
65  / (double)(n_range_values - 1);
66  scan_msg.scan_time = scan_time;
67  scan_msg.time_increment = scan_time / (2 * M_PI) * scan_msg.angle_increment;
68  scan_msg.range_min = 0;
69  scan_msg.range_max = 81.;
70  scan_msg.ranges.resize(n_range_values);
71  scan_msg.header.stamp = start;
72  for(size_t i = 0; i < n_range_values; i++) {
73  scan_msg.ranges[i] = (float)range_values[i] * (float)scale;
74  }
75  scan_msg.intensities.resize(n_intensity_values);
76  for(size_t i = 0; i < n_intensity_values; i++) {
77  scan_msg.intensities[i] = (float)intensity_values[i];
78  }
79 
80  pub->publish(scan_msg);
81 }
