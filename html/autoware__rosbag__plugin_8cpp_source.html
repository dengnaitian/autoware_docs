136 void Autoware_Rosbag_Plugin::on_button_record_start_clicked()
137 {
138  RecordParam recoParam;
139 
140  record_topics_.clear();
141 
142  /* Set topics parameter from checkboxes */
143  QList<QCheckBox *> list_checkboxes = ui->scrollAreaWidgetContents->findChildren<QCheckBox *>();
144  string topic_name;
145  for (int i = 0; i < list_checkboxes.size(); ++i) {
146  if (list_checkboxes.at(i)->isChecked())
147  {
148  topic_name = list_checkboxes.at(i)->text().toStdString();
149  if (topic_name.find("!") == string::npos)
150  record_topics_.push_back(topic_name);
151  else
152  record_topics_.push_back(topic_name.substr(0, topic_name.length() - strlen("(Not published!!)")));
153  }
154  }
155 
156  std::vector<std::string>::iterator ite = record_topics_.begin();
157  while( ite != record_topics_.end() )
158  {
159  recoParam.topics.push_back(*ite);
160  ite++;
161  }
162 
163  /* Set filename parameter */
164  recoParam.filename = record_filename_;
165 
166  /* Caculate max split size */
167  if (split_size != 0 && ui->checkBox_split_size->isChecked())
168  recoParam.max_size = ceil(split_size * 1000 * 1000 * 1000);
169  else
170  recoParam.max_size = 0;
171 
172  /* Caculate max split size */
173  if (split_duration != 0 && ui->checkBox_split_duration->isChecked())
174  recoParam.max_duration = ceil(split_duration * 60);
175  else
176  recoParam.max_duration = -1.0;
177 
178  /* Start record */
179  recordReq(recoParam);
180 
181  /*Start Timer*/
182  ui_timer_->start();
183 
184  ui->button_record_save->setDisabled(true);
185  ui->button_record_stop->setEnabled(true);
186  ui->button_record_start->setDisabled(true);
187  ui->button_record_configure->setDisabled(true);

190 int Autoware_Rosbag_Plugin::recordReq( RecordParam &recoParam )
191 {
192  recorder_opts_.reset( new rosbag_controller::RecorderOptions() );
193  recorder_opts_->prefix = recoParam.filename;
194  recorder_opts_->append_date = false;
195 
196  /* Default options */
197  recorder_opts_->buffer_size = 1048576 * 256;
198  recorder_opts_->chunk_size = 1024 * 768;
199  recorder_opts_->min_space = 1 * 1073741824ull;
200  recorder_opts_->min_space_str = "1G";
201 
202  /* Set max split size MB */
203  if (recoParam.max_size != 0)
204  {
205  recorder_opts_->split = true;
206  recorder_opts_->max_size = recoParam.max_size;
207  }
208 
209  /* Set max duration Sec */
210  if (recoParam.max_duration != -1.0)
211  {
212  recorder_opts_->split = true;
213  recorder_opts_->max_duration = ros::Duration(recoParam.max_duration);
214  }
215 
216  /* No specified topics then record all */
217  if (recoParam.topics.empty())
218  {
219  recorder_opts_->record_all = true;
220  }
221  else
222  {
223  /* Give specified topics to request */
224  std::vector<std::string>::iterator ite = recoParam.topics.begin();
225  while (ite < recoParam.topics.end())
226  {
227  recorder_opts_->topics.push_back( *ite );
228  ite++;
229  }
230  }
231 
232  ROS_INFO("%s L.%d - Set Recorder Option", __FUNCTION__, __LINE__);
233  ROS_INFO("%s L.%d - record_all [%d]", __FUNCTION__, __LINE__, recorder_opts_->record_all);
234  ROS_INFO("%s L.%d - regex [%d]", __FUNCTION__, __LINE__, recorder_opts_->regex);
235  ROS_INFO("%s L.%d - do_exclude [%d]", __FUNCTION__, __LINE__, recorder_opts_->do_exclude);
236  ROS_INFO("%s L.%d - quiet [%d]", __FUNCTION__, __LINE__, recorder_opts_->quiet);
237  ROS_INFO("%s L.%d - append_date [%d]", __FUNCTION__, __LINE__, recorder_opts_->append_date);
238  ROS_INFO("%s L.%d - verbose [%d]", __FUNCTION__, __LINE__, recorder_opts_->verbose);
239  ROS_INFO("%s L.%d - compression [%d]", __FUNCTION__, __LINE__, recorder_opts_->compression);
240  ROS_INFO("%s L.%d - prefix [%s]", __FUNCTION__, __LINE__, recorder_opts_->prefix.c_str());
241  ROS_INFO("%s L.%d - name [%s]", __FUNCTION__, __LINE__, recorder_opts_->name.c_str());
242  ROS_INFO("%s L.%d - buffer_size [%d]", __FUNCTION__, __LINE__, recorder_opts_->buffer_size);
243  ROS_INFO("%s L.%d - chunk_size [%d]", __FUNCTION__, __LINE__, recorder_opts_->chunk_size);
244  ROS_INFO("%s L.%d - limit [%d]", __FUNCTION__, __LINE__, recorder_opts_->limit);
245  ROS_INFO("%s L.%d - split [%d]", __FUNCTION__, __LINE__, recorder_opts_->split);
246  ROS_INFO("%s L.%d - max_size [%.2fGB]", __FUNCTION__, __LINE__, recorder_opts_->max_size / 1000.0 / 1000.0 / 1000.0);
247  ROS_INFO("%s L.%d - max_duration [%.1fmin]", __FUNCTION__, __LINE__, recorder_opts_->max_duration.toSec() / 60.0 );
248  ROS_INFO("%s L.%d - node [%s]", __FUNCTION__, __LINE__, recorder_opts_->node.c_str() );
249  ROS_INFO("%s L.%d - min_space [%d]", __FUNCTION__, __LINE__, recorder_opts_->min_space );
250  ROS_INFO("%s L.%d - min_space_str [%s]", __FUNCTION__, __LINE__, recorder_opts_->min_space_str.c_str() );
251  ROS_INFO("%s L.%d - topic num [%d]", __FUNCTION__, __LINE__, recorder_opts_->topics.size() );
252 
253  for( size_t i=0; i<recorder_opts_->topics.size(); i++ ) {
254  ROS_INFO("%s L.%d - [%d] %s", __FUNCTION__, __LINE__, i, recorder_opts_->topics.at(i).c_str() );
255  }
256 
257  doRecord( *recorder_opts_ );
258 
259  return 0;
260 }

320 void Autoware_Rosbag_Plugin::on_botton_topic_refresh_clicked()
321 {
322  ros::master::V_TopicInfo master_topics;
323  ros::master::getTopics(master_topics);
324 
325  QLayout* layout = ui->scrollAreaWidgetContents->layout();
326 
327  /* Delete previous topic scan result */
328  if (layout != 0)
329  {
330  QLayoutItem *item;
331  while ((item = layout->takeAt(0)) != 0)
332  {
333  delete item->widget();
334  delete item;
335  }
336  delete layout;
337  }
338 
339  QVBoxLayout *lay = new QVBoxLayout(ui->scrollAreaWidgetContents);
340  int topic_num = 0;
341 
342  /* Scan current topics */
343  std::vector<std::string> current_topic;
344  for (ros::master::V_TopicInfo::iterator it = master_topics.begin() ; it != master_topics.end(); it++) {
345  const ros::master::TopicInfo& info = *it;
346  current_topic.push_back(info.name);
347  }
348 
349  /* Soft topics */
350  std::sort(current_topic.begin(),current_topic.end());
351 
352  /* Build checkboxs for configure file */
353  if (!conf_topics_.empty())
354  {
355  std::vector<std::string>::iterator ite = conf_topics_.begin();
356  while (ite != conf_topics_.end())
357  {
358  if (std::find(current_topic.begin(), current_topic.end(), *ite) == current_topic.end())
359  {
360  QCheckBox *dynamic = new QCheckBox(QString::fromStdString(*ite + "(Not published!!)"));
361  dynamic->setChecked (true);
362  dynamic->setStyleSheet("color: red;"
363  "background-color: yellow;");
364  lay->addWidget(dynamic);
365  topic_num++;
366  ite++;
367  }
368  else
369  {
370  QCheckBox *dynamic = new QCheckBox(QString::fromStdString(*ite));
371  dynamic->setChecked (true);
372  lay->addWidget(dynamic);
373  topic_num++;
374  ite++;
375  }
376  }
377  }
378 
379  /* Build checkboxs for current topics */
380  std::vector<std::string>::iterator ite = current_topic.begin();
381  while (ite != current_topic.end())
382  {
383  if (std::find(conf_topics_.begin(), conf_topics_.end(), *ite) == conf_topics_.end())
384  {
385  QCheckBox *dynamic = new QCheckBox(QString::fromStdString(*ite));
386  dynamic->setChecked (false);
387  lay->addWidget(dynamic);
388  topic_num++;
389  }
390  ite++;
391  }
392  ui->scrollAreaWidgetContents->setLayout(lay);
