154 vector_map::Point find_departure_point(const VectorMap& lane_vmap, int lno,
155  const std::vector<vector_map::Point>& coarse_points,
156  double search_radius)
157 {
158  vector_map::Point coarse_p1 = coarse_points[0];
159  vector_map::Point coarse_p2 = coarse_points[1];
160 
161  vector_map::Point nearest_point = find_nearest_point(lane_vmap, coarse_p1);
162  if (nearest_point.pid < 0)
163  return nearest_point;
164 
165  std::vector<vector_map::Point> near_points = find_near_points(lane_vmap, coarse_p1, search_radius);
166  double coarse_angle = compute_direction_angle(coarse_p1, coarse_p2);
167  double score = 180 + search_radius; // XXX better way?
168  for (const vector_map::Point& p1 : near_points) {
169  vector_map::Lane l = find_lane(lane_vmap, lno, p1);
170  if (l.lnid < 0)
171  continue;
172 
173  vector_map::Point p2 = find_end_point(lane_vmap, l);
174  if (p2.pid < 0)
175  continue;
176 
177  double a = compute_direction_angle(p1, p2);
178  a = fabs(a - coarse_angle);
179  if (a > 180)
180  a = fabs(a - 360);
181  double d = hypot(p1.bx - coarse_p1.bx, p1.ly - coarse_p1.ly);
182  double s = a + d;
183  if (s <= score) {
184  nearest_point = p1;
185  score = s;
186  }
187  }
188 
189  return nearest_point;
190 }

334 vector_map::Lane find_next_branching_lane(const VectorMap& vmap, int lno, const vector_map::Lane& lane,
335  double coarse_angle, double search_radius)
336 {
337  vector_map::Lane error;
338  error.lnid = -1;
339 
340  vector_map::Point p1 = find_end_point(vmap, lane);
341  if (p1.pid < 0)
342  return error;
343 
344  std::vector<std::tuple<vector_map::Point, vector_map::Lane>> candidates;
345  for (const vector_map::Lane& l1 : vmap.lanes) {
346  if (lno != LNO_ALL && l1.lno != lno)
347  continue;
348  if (l1.lnid == lane.flid || l1.lnid == lane.flid2 || l1.lnid == lane.flid3 || l1.lnid == lane.flid4) {
349  vector_map::Lane l2 = l1;
350  vector_map::Point p = find_end_point(vmap, l2);
351  if (p.pid < 0)
352  continue;
353  vector_map::Point p2 = p;
354  double d = hypot(p2.bx - p1.bx, p2.ly - p1.ly);
355  while (d <= search_radius && l2.flid != 0 && !is_branching_lane(l2)) {
356  l2 = find_next_lane(vmap, LNO_ALL, l2);
357  if (l2.lnid < 0)
358  break;
359  p = find_end_point(vmap, l2);
360  if (p.pid < 0)
361  break;
362  p2 = p;
363  d = hypot(p2.bx - p1.bx, p2.ly - p1.ly);
364  }
365  candidates.push_back(std::make_tuple(p2, l1));
366  }
367  }
368 
369  if (candidates.empty())
370  return error;
371 
372  vector_map::Lane branching_lane;
373  double angle = 180;
374  for (const std::tuple<vector_map::Point, vector_map::Lane>& c : candidates) {
375  vector_map::Point p2 = std::get<0>(c);
376  double a = compute_direction_angle(p1, p2);
377  a = fabs(a - coarse_angle);
378  if (a > 180)
379  a = fabs(a - 360);
380  if (a <= angle) {
381  branching_lane = std::get<1>(c);
382  angle = a;
383  }
384  }
385 
386  return branching_lane;
387 }

391 void write_waypoints(const std::vector<vector_map::Point>& points, double velocity, const std::string& path)
392 {
393  if (points.size() < 2)
394  return;
395 
396  size_t last_index = points.size() - 1;
397  for (size_t i = 0; i < points.size(); ++i) {
398  double yaw;
399  if (i == last_index) {
400  geometry_msgs::Point p1 = create_geometry_msgs_point(points[i]);
401  geometry_msgs::Point p2 = create_geometry_msgs_point(points[i - 1]);
402  yaw = atan2(p2.y - p1.y, p2.x - p1.x);
403  yaw -= M_PI;
404  } else {
405  geometry_msgs::Point p1 = create_geometry_msgs_point(points[i]);
406  geometry_msgs::Point p2 = create_geometry_msgs_point(points[i + 1]);
407  yaw = atan2(p2.y - p1.y, p2.x - p1.x);
408  }
409 
410  write_waypoint(points[i], yaw, velocity, path, (i == 0));
411  }
412 }

588 VectorMap create_fine_vmap(const VectorMap& lane_vmap, int lno, const VectorMap& coarse_vmap, double search_radius,
589  int waypoint_max)
590 {
591  VectorMap fine_vmap;
592  VectorMap null_vmap;
593 
594  vector_map::Point departure_point;
595  departure_point.pid = -1;
596  if (lno == LNO_ALL)
597  departure_point = find_nearest_point(lane_vmap, coarse_vmap.points.front());
598  else {
599  for (int i = lno; i >= LNO_CROSSING; --i) {
600  departure_point = find_departure_point(lane_vmap, i, coarse_vmap.points, search_radius);
601  if (departure_point.pid >= 0)
602  break;
603  }
604  }
605  if (departure_point.pid < 0)
606  return null_vmap;
607 
608  vector_map::Point arrival_point;
609  arrival_point.pid = -1;
610  if (lno == LNO_ALL)
611  arrival_point = find_nearest_point(lane_vmap, coarse_vmap.points.back());
612  else {
613  for (int i = lno; i >= LNO_CROSSING; --i) {
614  arrival_point = find_arrival_point(lane_vmap, i, coarse_vmap.points, search_radius);
615  if (arrival_point.pid >= 0)
616  break;
617  }
618  }
619  if (arrival_point.pid < 0)
620  return null_vmap;
621 
622  vector_map::Point point = departure_point;
623  vector_map::Lane lane = find_lane(lane_vmap, LNO_ALL, point);
624  if (lane.lnid < 0)
625  return null_vmap;
626 
627  bool finish = false;
628  for (int i = 0; i < waypoint_max; ++i) {
629  fine_vmap.points.push_back(point);
630 
631  // last is equal to previous dtlane
632  vector_map::DTLane dtlane;
633  dtlane.did = -1;

672  if (coarse_p1.pid < 0)
673  return null_vmap;
674 
675  vector_map::Point coarse_p2;
676  double distance = -1;
677  for (const vector_map::Point& p : coarse_vmap.points) {
678  if (distance == -1) {
679  if (p.bx == coarse_p1.bx && p.ly == coarse_p1.ly)
680  distance = 0;
681  continue;
682  }
683  coarse_p2 = p;
684  distance = hypot(coarse_p2.bx - coarse_p1.bx, coarse_p2.ly - coarse_p1.ly);
685  if (distance > search_radius)
686  break;
687  }
688  if (distance <= 0)
689  return null_vmap;
690 
691  double coarse_angle = compute_direction_angle(coarse_p1, coarse_p2);
692  if (lno == LNO_ALL) {
693  lane = find_next_branching_lane(lane_vmap, LNO_ALL, lane, coarse_angle, search_radius);
694  } else {
695  vector_map::Lane l;
696  l.lnid = -1;
697  for (int j = lno; j >= LNO_CROSSING; --j) {
698  l = find_next_branching_lane(lane_vmap, j, lane, coarse_angle, search_radius);
699  if (l.lnid >= 0)
700  break;
701  }
702  lane = l;
703  }
704  } else {
705  lane = find_next_lane(lane_vmap, LNO_ALL, lane);
706  }
707  if (lane.lnid < 0)
708  return null_vmap;
709  }
710 
711  if (!finish) {
712  ROS_ERROR_STREAM("lane is too long");
713  return null_vmap;
714  }
715 
716  return fine_vmap;
