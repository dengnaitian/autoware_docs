146 void preCasting(const sensor_msgs::LaserScan& scan, std::vector<std::vector<Grid>>* precasted_grids)
147 {
148  int iangle_size = 2 * M_PI / scan.angle_increment;
149  // We decide if a grid is passed by laser by this search_step
150  double search_step = g_resolution / 10.0;
151 
152  // Grid indexes for each laser
153  std::vector<std::vector<int>> grid_indexes(iangle_size);
154 
155  for (int iangle = 0; iangle < iangle_size; iangle++)
156  {
157  // Angle of each laser from Lidar
158  double angle = scan.angle_increment * iangle; // + scan.angle_min;
159  double step = 0;
160 
161  while (1)
162  {
163  step += search_step;
164 
165  // Calculate a grid index passed by laser scan
166  int grid_x = step * cos(angle) / g_resolution + g_scan_size_x / 2.0;
167  int grid_y = step * sin(angle) / g_resolution + g_scan_size_y / 2.0;
168 
169  if (grid_x >= g_scan_size_x || grid_x < 0 || grid_y >= g_scan_size_y || grid_y < 0)
170  break;
171 
172  grid_indexes[iangle].push_back(grid_x + grid_y * g_scan_size_x);
173  }
174  }
175 
176  // Erase overlapping values
177  auto unique_grid_indexes = grid_indexes;
178  for (auto& vec : unique_grid_indexes)
179  vec.erase(std::unique(vec.begin(), vec.end()), vec.end());
180 
181  // Max size of step counts
182  // Precated laser steps for each grid are less than this value
183  // int max_count = g_resolution / search_step * sqrt(2) + 1;
184 
185  for (int i = 0; i < iangle_size; i++)
186  {
187  // Push grid indexes for each laser
188  // Each index has weight which is decided by passing laser rate
189  for (const auto& unique_index : unique_grid_indexes[i])
190  {
191  Grid g;
192  g.index = unique_index;
193  // g.weight = 1.0 * std::count(grid_indexes[i].begin(), grid_indexes[i].end(), unique_index) / max_count;
194  g.calcCoordinate();
195  g.calcRange();
196  precasted_grids->at(i).push_back(g);
197  }
198  }
199 }

202 // Delete old cost values
203 void deleteOldData(std::vector<Cost>* cost_map, double current_x, double current_y, double prev_x, double prev_y)
204 {
205  double begin_x = current_x;
206  double begin_y = current_y;
207  double end_x = prev_x;
208  double end_y = prev_y;
209 
210  if (begin_x > end_x)
211  std::swap(begin_x, end_x);
212  if (begin_y > end_y)
213  std::swap(begin_y, end_y);
214 
215  int ibegin_x = begin_x / g_resolution + g_scan_size_x / 2;
216  int ibegin_y = begin_y / g_resolution + g_scan_size_y / 2;
217  int iend_x = end_x / g_resolution + g_scan_size_x / 2;
218  int iend_y = end_y / g_resolution + g_scan_size_y / 2;
219 
220  for (int i = ibegin_x; i < iend_x; i++)
221  {
222  for (int j = 0; j < g_scan_size_y; j++)
223  {
224  int global_grid_x = (i + 100000 * g_scan_size_x) % g_scan_size_x;
225  cost_map->at(global_grid_x + j * g_scan_size_x).occupied = 0;
226  cost_map->at(global_grid_x + j * g_scan_size_x).free = 0;
227  cost_map->at(global_grid_x + j * g_scan_size_x).unknown = true;
228  }
229  }
230 
231  for (int i = 0; i < g_scan_size_x; i++)
232  {
233  for (int j = ibegin_y; j < iend_y; j++)
234  {
235  int global_grid_y = (j + 100000 * g_scan_size_y) % g_scan_size_y;
236  cost_map->at(i + global_grid_y * g_scan_size_x).occupied = 0;
237  cost_map->at(i + global_grid_y * g_scan_size_x).free = 0;
238  cost_map->at(i + global_grid_y * g_scan_size_x).unknown = true;
239  }

261 void createCostMap(const sensor_msgs::LaserScan& scan, const std::vector<std::vector<Grid>>& precasted_grids)
262 {
263  tf::StampedTransform transform;
264 
265  try
266  {
267  // What time should we use?
268  g_tf_listenerp->lookupTransform(OGM_FRAME, g_sensor_frame, ros::Time(0), transform);
269  }
270  catch (tf::TransformException ex)
271  {
272  ROS_ERROR("%s", ex.what());
273  return;
274  }
275 
276  // Save costs in this variable
277  static std::vector<Cost> cost_map(g_scan_size_x * g_scan_size_y);
278 
279  static bool initialized_map = false;
280  static double prev_x = transform.getOrigin().x();
281  static double prev_y = transform.getOrigin().y();
282  static nav_msgs::OccupancyGrid map;
283  setOccupancyGridMap(&map, scan.header, transform);
284  if (!initialized_map)
285  {
286  map.data.resize(g_map_size_x * g_map_size_y, -1);
287  initialized_map = true;
288  }
289 
290  // Since we implement as ring buffer, we have to delete old data regularly
291  deleteOldData(&cost_map, transform.getOrigin().x(), transform.getOrigin().y(), prev_x, prev_y);
292  prev_x = transform.getOrigin().x();
293  prev_y = transform.getOrigin().y();
294 
295  // Vehicle's orientation
296  double yaw = calcYawFromQuaternion(transform.getRotation());
297 
298  // Original yaw is -PI ~ PI, so make its range 0 ~ 2PI
299  if (yaw < 0)
300  yaw += 2 * M_PI;
301  static double laser_offset = fabs(scan.angle_min);
302 
303  // For tough_urg
304  // static double manual_offset = -10.0 / 180 * M_PI + M_PI;
305  // int index_offset = (yaw + laser_offset + manual_offset) / scan.angle_increment;
306 
307  int index_offset = (yaw + laser_offset) / scan.angle_increment;
308  static int iangle_size = 2 * M_PI / scan.angle_increment;
309 
310  //----------------- RING OCCUPANCY GRID MAPPING --------------
311  // Accumulate grid costs for each laser scan
312  for (size_t i = 0; i < scan.ranges.size(); i++)
313  {
314  double range = scan.ranges[i];
315 
316  // Ignore 0 ranges
317  if (range == 0)
318  continue;
319 
320  // If laserscan does not reach objects, make a range max
321  if (scan.ranges[i] > scan.range_max)
322  range = scan.range_max;
323 
324  int precasted_index = (i + index_offset) % iangle_size;
325  int obstacle_index = -1;
326  for (const auto& g : precasted_grids[precasted_index])
327  {
328  obstacle_index++;
329 
330  if (g.range > range)
331  break;
332 
333  // Free range
334  int global_index = g.calcGlobalIndex(transform);
335  cost_map[global_index].accumulateCost(0, FREE_INCREMENT);
336  }
337 
338  // Obstacle
339  int global_index = precasted_grids[precasted_index][obstacle_index].calcGlobalIndex(transform);
340  cost_map[global_index].accumulateCost(OCCUPIED_INCREMENT, 0);
341 
342  }
343 
344 
345  // Begining of grid index of publishing OGM
346  static int origin_index_x = (g_scan_size_x - g_map_size_x) / 2.0;
347  static int origin_index_y = (g_scan_size_y - g_map_size_y) / 2.0;
348  // static int origin_index = origin_index_x + origin_index_y * g_scan_size_x;
349 
350  // Set cost values for publishing OccuppancyGridMap
351  for (int i = 0; i < g_map_size_y; i++)
352  {
353  for (int j = 0; j < g_map_size_x; j++)
354  {
355  int scanmap_index_x = origin_index_x + j;
356  int scanmap_index_y = origin_index_y + i;
357 
358  int global_index = calcGlobalIndex(scanmap_index_x, scanmap_index_y, transform);
359  if (cost_map[global_index].unknown)
360  map.data[j + i * g_map_size_x] = -1;
361  else
362  map.data[j + i * g_map_size_x] = (cost_map[global_index].occupied + 8) * 6;
363  }
364  }
365 
366  g_map_pub.publish(map);
367 }
