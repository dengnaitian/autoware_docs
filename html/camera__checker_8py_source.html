166  def handle_monocular(self, msg):
167 
168  (image, camera) = msg
169  gray = self.mkgray(image)
170  C = self.image_corners(gray)
171  if C is not None:
172  linearity_rms = self.mc.linear_error(C, self.board)
173 
174  # Add in reprojection check
175  image_points = C
176  object_points = self.mc.mk_object_points([self.board], use_board_size=True)[0]
177  dist_coeffs = numpy.zeros((4, 1))
178  camera_matrix = numpy.array( [ [ camera.P[0], camera.P[1], camera.P[2] ],
179  [ camera.P[4], camera.P[5], camera.P[6] ],
180  [ camera.P[8], camera.P[9], camera.P[10] ] ] )
181  ok, rot, trans = cv2.solvePnP(object_points, image_points, camera_matrix, dist_coeffs)
182  # Convert rotation into a 3x3 Rotation Matrix
183  rot3x3, _ = cv2.Rodrigues(rot)
184  # Reproject model points into image
185  object_points_world = numpy.asmatrix(rot3x3) * numpy.asmatrix(object_points.squeeze().T) + numpy.asmatrix(trans)
186  reprojected_h = camera_matrix * object_points_world
187  reprojected = (reprojected_h[0:2, :] / reprojected_h[2, :])
188  reprojection_errors = image_points.squeeze().T - reprojected
189 
190  reprojection_rms = numpy.sqrt(numpy.sum(numpy.array(reprojection_errors) ** 2) / numpy.product(reprojection_errors.shape))
191 
192  # Print the results
193  print("Linearity RMS Error: %.3f Pixels Reprojection RMS Error: %.3f Pixels" % (linearity_rms, reprojection_rms))
194  else:
195  print('no chessboard')
