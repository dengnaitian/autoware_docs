214 void geo_pos_conv::set_llh_nmea_degrees(double latd, double lond, double h)
215 {
216  double lat, lad, lod, lon;
217  // 1234.56 -> 12'34.56 -> 12+ 34.56/60
218 
219  lad = floor(latd / 100.);
220  lat = latd - lad * 100.;
221  lod = floor(lond / 100.);
222  lon = lond - lod * 100.;
223 
224  // Changing Longitude and Latitude to Radians
225  m_lat = (lad + lat / 60.0) * M_PI / 180;
226  m_lon = (lod + lon / 60.0) * M_PI / 180;
