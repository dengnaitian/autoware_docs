120 void VelocitySetPath::avoidSuddenDeceleration(double velocity_change_limit, double deceleration, int closest_waypoint)
121 {
122  if (closest_waypoint < 0)
123  return;
124 
125  // not avoid braking
126  if (current_vel_ - new_waypoints_.waypoints[closest_waypoint].twist.twist.linear.x < velocity_change_limit)
127  return;
128 
129  //std::cout << "avoid sudden braking!" << std::endl;
130 
131  double square_vel = (current_vel_ - velocity_change_limit) * (current_vel_ - velocity_change_limit);
132  for (int i = 0;; i++)
133  {
134  if (!checkWaypoint(closest_waypoint + i, __FUNCTION__))
135  return;
136 
137  // sqrt(v^2 - 2ax)
138  double changed_vel = square_vel - 2 * deceleration * calcInterval(closest_waypoint, closest_waypoint + i);
139 
140  if (changed_vel < 0)
141  break;
142 
143  new_waypoints_.waypoints[closest_waypoint + i].twist.twist.linear.x = std::sqrt(changed_vel);

149 {

186 double VelocitySetPath::calcInterval(const int begin, const int end) const
187 {
188  // check index
189  if (begin < 0 || begin >= getPrevWaypointsSize() || end < 0 || end >= getPrevWaypointsSize())
190  {
191  ROS_WARN("Invalid index");
192  return 0;
193  }
194 
195  // Calculate the inteval of waypoints
196  double dist_sum = 0;
197  for (int i = begin; i < end; i++)
198  {
199  tf::Vector3 v1(prev_waypoints_.waypoints[i].pose.pose.position.x,
200  prev_waypoints_.waypoints[i].pose.pose.position.y, 0);
201 
202  tf::Vector3 v2(prev_waypoints_.waypoints[i + 1].pose.pose.position.x,
203  prev_waypoints_.waypoints[i + 1].pose.pose.position.y, 0);
204 
205  dist_sum += tf::tfDistance(v1, v2);
206  }
207 
208  return dist_sum;
209 }
