48 void parse_camera_info(const cv::Mat& camMat,
49  const cv::Mat& disCoeff,
50  const cv::Size& imgSize,
51  sensor_msgs::CameraInfo& msg)
52 {
53  msg.header.frame_id = "camera";
54  // msg.header.stamp = ros::Time::now();
55 
56  msg.height = imgSize.height;
57  msg.width = imgSize.width;
58 
59  for (int row=0; row<3; row++)
60  {
61  for (int col=0; col<3; col++)
62  {
63  msg.K[row * 3 + col] = camMat.at<double>(row, col);
64  }
65  }
66 
67  for (int row=0; row<3; row++)
68  {
69  for (int col=0; col<4; col++)
70  {
71  if (col == 3)
72  {
73  msg.P[row * 4 + col] = 0.0f;
74  } else
75  {
76  msg.P[row * 4 + col] = camMat.at<double>(row, col);
77  }
78  }
79  }
80 
81  for (int row=0; row<disCoeff.rows; row++)
82  {
83  for (int col=0; col<disCoeff.cols; col++)
84  {
85  msg.D.push_back(disCoeff.at<double>(row, col));
86  }
87  }
88 }

131 void initialize_cameras(std::vector<FlyCapture2::Camera *> &cameras,
132  FlyCapture2::BusManager *bus_manger,
133  int camera_num,
134  FlyCapture2::Mode desired_mode,
135  FlyCapture2::PixelFormat desired_pixel_format,
136  int timeout_ms)
137 {
138  // Connect to all detected cameras and attempt to set them to
139  // a common video mode and frame rate
140  for (int i = 0; i < camera_num; i++)
141  {
142  FlyCapture2::Camera *camera = new FlyCapture2::Camera();
143 
144  FlyCapture2::PGRGuid guid;
145  FlyCapture2::Error error = bus_manger->GetCameraFromIndex(i, &guid);
146  if (error != FlyCapture2::PGRERROR_OK)
147  {
148  error.PrintErrorTrace();
149  std::exit(-1);
150  }
151 
152  error = camera->Connect( &guid );
153  if (error != FlyCapture2::PGRERROR_OK)
154  {
155  error.PrintErrorTrace();
156  std::exit(-1);
157  }
158 
159  FlyCapture2::EmbeddedImageInfo image_info;
160  error = camera->GetEmbeddedImageInfo(&image_info);
161  if (error != FlyCapture2::PGRERROR_OK)
162  {
163  error.PrintErrorTrace();
164  std::exit(-1);
165  }
166 
167  image_info.timestamp.onOff = false;
168  error = camera->SetEmbeddedImageInfo(&image_info);
169  if (error != FlyCapture2::PGRERROR_OK)
170  {
171  error.PrintErrorTrace();
172  std::exit(-1);
173  }
174 
175  // Get the camera information
176  FlyCapture2::CameraInfo camera_info;
177  error = camera->GetCameraInfo(&camera_info);
178  if (error != FlyCapture2::PGRERROR_OK)
179  {
180  error.PrintErrorTrace();
181  std::exit(-1);
182  }
183 
184  //obtain working settings
185  FlyCapture2::VideoMode default_video_mode;
186  FlyCapture2::FrameRate default_frame_rate;
187 
188  error = camera->GetVideoModeAndFrameRate(&default_video_mode, &default_frame_rate);
189  if (error != FlyCapture2::PGRERROR_OK)
190  {
191  error.PrintErrorTrace();
192  std::exit(-1);
193  }
194 
195  //try to set Format7, according to the desired mode and pixel format.
196  FlyCapture2::Format7ImageSettings image_settings;
197  bool supported = false;
198  unsigned int packet_size;
199  float percentage;
200 
201  FlyCapture2::Format7Info format7_info;
202  format7_info.mode = desired_mode;
203 
204  error = camera->GetFormat7Info(&format7_info, &supported);
205  if (error != FlyCapture2::PGRERROR_OK)
206  {
207  error.PrintErrorTrace();
208  std::exit(-1);
209  }
210 
211  print_format7_info(format7_info, supported);
212 
213  if (supported)
214  {
215  error = camera->GetFormat7Configuration(&image_settings, &packet_size, &percentage);
216  if (error != FlyCapture2::PGRERROR_OK)
217  {
218  error.PrintErrorTrace();
219  std::exit(-1);
220  }
221 
222  image_settings.mode = desired_mode;
223  image_settings.pixelFormat = desired_pixel_format;
224  image_settings.offsetX = 0;
225  image_settings.offsetY = 0;
226  image_settings.width = format7_info.maxWidth;
227  image_settings.height = format7_info.maxHeight;
228 
229  FlyCapture2::Format7PacketInfo packet_info;
230  bool valid_settings = false;
231  error = camera->ValidateFormat7Settings(&image_settings, &valid_settings, &packet_info);
232  if (error != FlyCapture2::PGRERROR_OK)
233  {
234  error.PrintErrorTrace();
235  std::exit(-1);
236  }
237  packet_size = packet_info.recommendedBytesPerPacket;
238  error = camera->SetFormat7Configuration(&image_settings, packet_size);
239  if (error != FlyCapture2::PGRERROR_OK)
240  {
241  error.PrintErrorTrace();
242  std::exit(-1);
243  }
244 
245  error = camera->GetFormat7Configuration(&image_settings, &packet_size, &percentage);
246  if (error != FlyCapture2::PGRERROR_OK)
247  {
248  error.PrintErrorTrace();
249  std::exit(-1);
250  }
251 
252  print_image_settings(image_settings, packet_size, percentage);
253  }
254  else
255  {
256  ROS_ERROR("Selected Mode not supported, using last working mode.");
257  }
258 
259  FlyCapture2::FC2Config camera_config;
260  error = camera->GetConfiguration(&camera_config);
261  if (error != FlyCapture2::PGRERROR_OK)
262  {
263  error.PrintErrorTrace();
264  ROS_INFO("Could not read configuration from Camera");
265  }
266  else
267  {
268  if (timeout_ms > 0)
269  camera_config.grabTimeout = timeout_ms;
270 
271  error = camera->SetConfiguration(&camera_config);
272  if (error != FlyCapture2::PGRERROR_OK)
273  {
274  error.PrintErrorTrace();
275  ROS_INFO("Could not set configuration on Camera");
276  }
277  }
278 
279  print_camera_info(&camera_info);
280  cameras.push_back(camera);
281  }
282 }

457  std::string topic(std::string("image_raw"));

477  FlyCapture2::Image image;

483  }
484 
485  // check encoding pattern

496  break;

498  encoding_pattern = "bayer_bggr8";

502  }
503 
504  sensor_msgs::Image msg;
505  //publish*******************
506 
507  msg.header.seq = count;
508  std::string frame = "camera" + std::to_string(i);
509  msg.header.frame_id = frame;
510  msg.header.stamp.sec = ros::Time::now().sec;
511  msg.header.stamp.nsec = ros::Time::now().nsec;
512  msg.height = image.GetRows();
513  msg.width = image.GetCols();
514  msg.encoding = encoding_pattern;
515  msg.step = image.GetStride();
516 
517  size_t image_size = image.GetDataSize();
518  msg.data.resize(image_size);
519  memcpy(msg.data.data(), image.GetData(), image_size);
520

523  }
