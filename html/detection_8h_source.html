130  void propagate_prev_det(cv::Mat k, cv::Mat motion)
131  {
132  // std::cout << "### Propagate previous detection　###\n";
133  B1Q = origin_;
134  cv::Mat bvolume = bvolume_;
135  B1Q = B1Q + motion.colRange(0, 3).t();
136  bvolume += repeat(motion.colRange(0, 3), bvolume.size().height, 1);
137  bvolume -= repeat(B1Q.t(), bvolume.size().height, 1);
138  bvolume = get_boundingbox_volume(bvolume, motion.at<double>(0, 3)); //translated_cuboid);
139  origin_ = B1Q;
140  bvolume_ = bvolume;
141  bvolume_proj_ = bvolume_ * k.t();
142  bvolume_proj_.colRange(0, 3) /= repeat(bvolume_proj_.col(2), 1, 3);
143 
144  // Pre-calculate the convexhull -- 3d --
145  cv::Mat pd_3d_xz = cv::Mat(bvolume_.size().height, 2, CV_64FC1);
146  bvolume_.col(0).copyTo(pd_3d_xz.col(0));
147  bvolume_.col(2).copyTo(pd_3d_xz.col(1));
148  pd_3d_xz.convertTo(pd_3d_xz, CV_32FC1);
149 
150  pd_3d_convhull_ = get_convhull(pd_3d_xz * 100); // due to cast
151 
152  Paths pd_3d_convhull_clip(1);
153  for (const auto &e: pd_3d_convhull_)
154  {
155  pd_3d_convhull_clip[0] << IntPoint(e.x, e.y);
156  }
157  pd_3d_convhull_clip_ = pd_3d_convhull_clip;
158 
159  // Pre-calculate the convexhull -- 2d --
160  cv::Mat pd_2d = bvolume_proj_.colRange(0, 2);
161  pd_2d.convertTo(pd_2d, CV_32FC1);
162 
163  pd_2d_convhull_ = get_convhull(pd_2d);
164 
165  Paths pd_2d_convhull_clip(1);
166  for (const auto &e: pd_2d_convhull_)
167  {
168  pd_2d_convhull_clip[0] << IntPoint(e.x, e.y);
169  }
170  pd_2d_convhull_clip_ = pd_2d_convhull_clip;
171  }

173  void propagate_cur_det(cv::Mat cuboid, double h, cv::Mat k, cv::Mat inv_k, cv::Mat n)
174  {
175  // std::cout << "Propagate current detection\n";
176  cv::Mat b1Q = (cv::Mat_<double>(3, 1) << bbox_[0] + (bbox_[2] - bbox_[0]) / 2, bbox_[3], 1.0);
177  B1Q = (h * inv_k * b1Q) / (n * inv_k * b1Q);
178  // apply offset which is a function of yaw and get car's origin.
179  double offset_z = calc_offset_base_yaw();
180  B1Q.at<double>(0, 2) += offset_z;
181  cv::Mat bvolume = get_boundingbox_volume(cuboid, yaw_); //translated_cuboid);
182  origin_ = B1Q;
183  cv::Mat offset = (cv::Mat_<double>(1, 3) << 0, params_car_cuboid_.avg_w / 2., 0);
184  bvolume_ = bvolume - repeat(offset, bvolume.size().height, 1);
185  bvolume_proj_ = bvolume_ * k.t();
186  bvolume_proj_.colRange(0, 3) /= repeat(bvolume_proj_.col(2), 1, 3);
187 
188  // Pre-calculate the convexhull -- 3d --
189  cv::Mat bvolume_xy = cv::Mat(bvolume_.size().height, 2, CV_64FC1);
190  bvolume_.col(0).copyTo(bvolume_xy.col(0));
191  bvolume_.col(2).copyTo(bvolume_xy.col(1));
192  bvolume_xy.convertTo(bvolume_xy, CV_32FC1);
193 
194  cd_3d_convhull_ = get_convhull(bvolume_xy * 100); // due to cast
195 
196  Paths cd_3d_convhull_clip(1);
197  for (const auto &e: cd_3d_convhull_)
198  {
199  cd_3d_convhull_clip[0] << IntPoint(e.x, e.y);
200  }
201  cd_3d_convhull_clip_ = cd_3d_convhull_clip;
202 
203  // Pre-calculate the convexhull -- 2d --
204  cd_2d_ = (cv::Mat_<double>(4, 2) <<
205  bbox_[0], bbox_[1], bbox_[2], bbox_[1], bbox_[2], bbox_[3], bbox_[0], bbox_[3]);
206 
207  Paths cd_2d_convhull_clip(1);
208  for (int i = 0; i < 4; ++i)
209  {
210  cd_2d_convhull_clip[0] << IntPoint((int) cd_2d_.at<double>(i, 0), (int) cd_2d_.at<double>(i, 1));
211  }
212  cd_2d_convhull_clip_ = cd_2d_convhull_clip;
213 
214  cd_area_ = (bbox_[3] - bbox_[1]) * (bbox_[2] - bbox_[0]);
215  }

219  double calc_offset_base_yaw()
220  {
221  double offset;
222  double l = params_car_cuboid_.avg_l / 2.;
223  double w = params_car_cuboid_.avg_w / 2.;
224  cv::Mat car_bottom_plane = (cv::Mat_<double>(4, 2) << -l, -w, l, -w, l, w, -l, w);
225  car_bottom_plane = car_bottom_plane.t();
226 
227  cv::Mat rot = (cv::Mat_<double>(2, 2) << cos(yaw_), -sin(yaw_), sin(yaw_), cos(yaw_));
228  cv::Mat car_bottom_plane_rot = (rot * car_bottom_plane).t();
229  cv::Mat sum_mat;
230  cv::reduce(car_bottom_plane_rot, sum_mat, 1, CV_REDUCE_SUM);
231  double min_val, max_val;
232  CvPoint min_loc = cvPoint(0, 0);
233  CvPoint max_loc = cvPoint(0, 0);
234  CvMat cvmat = sum_mat;
235  cvMinMaxLoc(&cvmat, &min_val, &max_val, &min_loc, &max_loc);
236  offset =
237  abs(car_bottom_plane_rot.at<double>(max_loc.y, 1) - car_bottom_plane_rot.at<double>(min_loc.y, 1)) /
238  2.;
239  return offset;
240  }

242  cv::Mat get_boundingbox_volume(cv::Mat cuboid, double ry)
243  {
244  double ry_n = sigma_3D.at<double>(3, 3);
245  cv::Mat centered_pts = cuboid; //cuboid - repeat(B1Q.t(), 8, 1);
246  // ry_n = 0.5;
247  cv::Mat rot_pts_plus_yaw = centered_pts * rot_mat_y(ry_n).t();
248  cv::Mat rot_pts_minus_yaw = centered_pts * rot_mat_y(-ry_n).t();
249  cv::Mat pts;
250  vconcat(centered_pts, rot_pts_plus_yaw, pts);
251  vconcat(pts, rot_pts_minus_yaw, pts);
252 
253  cv::Mat scale_mat = sigma_3D(cv::Range(0, 3), cv::Range(0, 3));
254  cv::Mat bvolume = pts * scale_mat;
255  // cv::Mat m_bvolume = -bvolume;
256  // cv::cvThreshold(bvolume, bvolume);
257 
258  bvolume = bvolume * rot_mat_y(ry).t();
259  bvolume = bvolume + repeat(B1Q.t(), bvolume.size().height, 1);
260  return bvolume;
261  }

263  std::vector<cv::Point> get_convhull(cv::Mat mat_2d)
264  {
265  int size_pd = mat_2d.size().height;
266  std::vector<cv::Point> mat_2d_convhull(1);
267  std::vector<cv::Point> mat_2d_vec;
268  for (int i = 0; i < size_pd; ++i)
269  {
270  cv::Mat a = mat_2d.row(i);
271  cv::Point b = (cv::Point) a;
272  mat_2d_vec.push_back(b);
273  }
274 
275  cv::convexHull(mat_2d_vec, mat_2d_convhull, false);
276  return mat_2d_convhull;
277  }
