57 static void scan_callback(const sensor_msgs::PointCloud2::ConstPtr& input)
58 {
59  pcl::PointXYZI sampled_p;
60  pcl::PointCloud<pcl::PointXYZI> scan;
61 
62  pcl::fromROSMsg(*input, scan);
63 
64  if(measurement_range != MAX_MEASUREMENT_RANGE){
65  scan = removePointsByRange(scan, 0, measurement_range);
66  }
67 
68  pcl::PointCloud<pcl::PointXYZI>::Ptr filtered_scan_ptr(new pcl::PointCloud<pcl::PointXYZI>());
69  filtered_scan_ptr->header = scan.header;
70 
71  int points_num = scan.size();
72 
73  double w_total = 0.0;
74  double w_step = 0.0;
75  int m = 0;
76  double c = 0.0;
77 
78  filter_start = std::chrono::system_clock::now();
79 
80  for (pcl::PointCloud<pcl::PointXYZI>::const_iterator item = scan.begin(); item != scan.end(); item++)
81  {
82  w_total += item->x * item->x + item->y * item->y + item->z * item->z;
83  }
84  w_step = w_total / sample_num;
85 
86  pcl::PointCloud<pcl::PointXYZI>::const_iterator item = scan.begin();
87  for (m = 0; m < sample_num; m++)
88  {
89  while (m * w_step > c)
90  {
91  item++;
92  c += item->x * item->x + item->y * item->y + item->z * item->z;
93  }
94  sampled_p.x = item->x;
95  sampled_p.y = item->y;
96  sampled_p.z = item->z;
97  sampled_p.intensity = item->intensity;
98  filtered_scan_ptr->points.push_back(sampled_p);
99  }
100 
101  sensor_msgs::PointCloud2 filtered_msg;
102  pcl::toROSMsg(*filtered_scan_ptr, filtered_msg);
103 
104  filter_end = std::chrono::system_clock::now();
105 
106  filtered_msg.header = input->header;
107  filtered_points_pub.publish(filtered_msg);
108 
109  points_downsampler_info_msg.header = input->header;
110  points_downsampler_info_msg.filter_name = "distance_filter";
111  points_downsampler_info_msg.measurement_range = measurement_range;
112  points_downsampler_info_msg.original_points_size = points_num;
113  points_downsampler_info_msg.filtered_points_size = std::min((int)filtered_scan_ptr->size(), points_num);
114  points_downsampler_info_msg.original_ring_size = 0;
115  points_downsampler_info_msg.original_ring_size = 0;
116  points_downsampler_info_msg.exe_time = std::chrono::duration_cast<std::chrono::microseconds>(filter_end - filter_start).count() / 1000.0;
117  points_downsampler_info_pub.publish(points_downsampler_info_msg);
118 
119  if(_output_log == true){
120  if(!ofs){
121  std::cerr << "Could not open " << filename << "." << std::endl;
122  exit(1);
123  }
124  ofs << points_downsampler_info_msg.header.seq << ","
125  << points_downsampler_info_msg.header.stamp << ","
126  << points_downsampler_info_msg.header.frame_id << ","
127  << points_downsampler_info_msg.filter_name << ","
128  << points_downsampler_info_msg.original_points_size << ","
129  << points_downsampler_info_msg.filtered_points_size << ","
130  << points_downsampler_info_msg.original_ring_size << ","
131  << points_downsampler_info_msg.filtered_ring_size << ","
132  << points_downsampler_info_msg.exe_time << ","
133  << std::endl;
134  }
