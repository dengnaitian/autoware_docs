256 //
257 // Decode the input state data from the sensor into this
258 // objects structure.
259 //
260 void EvalCase::setLogicalInputState_from_UINT8(UINT8 value)
261 {
262  // Input 1
263  UINT8 inputValue = value & 0x03;
264  switch (inputValue)
265  {
266  case 0:
267  m_logicalInputState[0] = INPUT_INACTIVE;
268  break;
269  case 1:
270  m_logicalInputState[0] = INPUT_LOW_ACTIVE;
271  break;
272  case 2:
273  m_logicalInputState[0] = INPUT_HIGH_ACTIVE;
274  break;
275  default:
276  printError("EvalCase::setLogicalInputState_from_UINT8: Configuration of input 1 has an invalid value, aborting!");
277  return;
278  }
279 
280  // Input 2
281  inputValue = (value >> 2) & 0x03;
282  switch (inputValue)
283  {
284  case 0:
285  m_logicalInputState[1] = INPUT_INACTIVE;
286  break;
287  case 1:
288  m_logicalInputState[1] = INPUT_LOW_ACTIVE;
289  break;
290  case 2:
291  m_logicalInputState[1] = INPUT_HIGH_ACTIVE;
292  break;
293  default:
294  printError("EvalCase::setLogicalInputState_from_UINT8: Configuration of input 2 has an invalid value, aborting!");
295  return;
296  }
297 }
