188 void Adas::checkPath(const std::vector<PlannerHNS::DetectedObject>& obj_list, std::vector<PlannerHNS::WayPoint>& path, PlannerHNS::CAR_BASIC_INFO& carInfo, const PlannerHNS::WayPoint& currState, PlannerHNS::TrajectoryCost& trajectoryCost){
189 
190  double critical_long_front_distance = carInfo.wheel_base/2.0 + carInfo.length/2.0;
191  double critical_lateral_distance = carInfo.width/2.0 + 1;
192  trajectoryCost.closest_obj_distance = 1000;
193 
194  PlannerHNS::WayPoint p;
195  std::vector<PlannerHNS::WayPoint> contourPoints;
196  for(unsigned int io=0; io<obj_list.size(); io++)
197  {
198  for(unsigned int icon=0; icon < obj_list.at(io).contour.size(); icon++)
199  {
200  p.pos = obj_list.at(io).contour.at(icon);
201  p.v = obj_list.at(io).center.v;
202  p.id = io;
203  p.cost = sqrt(obj_list.at(io).w*obj_list.at(io).w + obj_list.at(io).l*obj_list.at(io).l);
204  contourPoints.push_back(p);
205  }
206  }
207  PlannerHNS::RelativeInfo car_info;
208  PlannerHNS::PlanningHelpers::GetRelativeInfo(path, currState, car_info);
209  int skip_id = -1;
210  for(unsigned int icon = 0; icon < contourPoints.size(); icon++){
211  if(skip_id == contourPoints.at(icon).id)
212  continue;
213  PlannerHNS::RelativeInfo obj_info;
214  PlannerHNS::PlanningHelpers::GetRelativeInfo(path,contourPoints.at(icon),obj_info);
215  double longitudinalDist = PlannerHNS::PlanningHelpers::GetExactDistanceOnTrajectory(path,car_info, obj_info);
216  if(obj_info.iFront == 0&& longitudinalDist > 0)
217  longitudinalDist = -longitudinalDist;
218 
219  double direct_distance = hypot(obj_info.perp_point.pos.y-contourPoints.at(icon).pos.y, obj_info.perp_point.pos.x-contourPoints.at(icon).pos.x);
220 
221  double lateralDist = fabs(obj_info.perp_distance);
222  if(longitudinalDist < -carInfo.length || longitudinalDist > 50 || lateralDist > 3){
223  continue;
224  }
225 
226  longitudinalDist = longitudinalDist - critical_long_front_distance;
227 
228  if(lateralDist <= critical_lateral_distance
229  && longitudinalDist >= -carInfo.length/1.5
230  && longitudinalDist < 50){
231  trajectoryCost.bBlocked = true;
232  //modified by hui
233  if(longitudinalDist >= -critical_long_front_distance && longitudinalDist < trajectoryCost.closest_obj_distance)
234  {
235  trajectoryCost.closest_obj_distance = longitudinalDist;
236  }
237  }
238  }
