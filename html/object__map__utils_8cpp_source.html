74  SearchAreaPoints(const vector_map::Area &in_area, const vector_map::VectorMap &in_vectormap)
75  {
76  std::vector<geometry_msgs::Point> area_points;
77  std::vector<geometry_msgs::Point> area_points_empty;
78 
79  if (in_area.aid == 0)
80  return area_points_empty;
81 
82  vector_map_msgs::Line line = in_vectormap.findByKey(vector_map::Key<vector_map_msgs::Line>(in_area.slid));
83  // must set beginning line
84  if (line.lid == 0 || line.blid != 0)
85  return area_points_empty;
86 
87  // Search all lines in in_area
88  while (line.flid != 0)
89  {
90  vector_map_msgs::Point bp = in_vectormap.findByKey(vector_map::Key<vector_map_msgs::Point>(line.bpid));
91  if (bp.pid == 0)
92  return area_points_empty;
93 
94  vector_map_msgs::Point fp = in_vectormap.findByKey(vector_map::Key<vector_map_msgs::Point>(line.fpid));
95  if (fp.pid == 0)
96  return area_points_empty;
97 
98  // 2 points of line
99  area_points.push_back(vector_map::convertPointToGeomPoint(bp));
100  area_points.push_back(vector_map::convertPointToGeomPoint(fp));
101 
102  line = in_vectormap.findByKey(vector_map::Key<vector_map_msgs::Line>(line.flid));
103  if (line.lid == 0)
104  return area_points_empty;
105  }
106 
107  vector_map_msgs::Point bp = in_vectormap.findByKey(vector_map::Key<vector_map_msgs::Point>(line.bpid));
108  vector_map_msgs::Point fp = in_vectormap.findByKey(vector_map::Key<vector_map_msgs::Point>(line.fpid));
109  if (bp.pid == 0 || fp.pid == 0)
110  return area_points_empty;
111 
112  area_points.push_back(vector_map::convertPointToGeomPoint(bp));
113  area_points.push_back(vector_map::convertPointToGeomPoint(fp));
114 
115  return area_points;
116  }
