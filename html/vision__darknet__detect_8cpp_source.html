62  image Yolo3Detector::convert_image(const sensor_msgs::ImageConstPtr& msg)
63  {
64  if (msg->encoding != sensor_msgs::image_encodings::BGR8)
65  {
66  ROS_ERROR("Unsupported encoding");
67  exit(-1);
68  }
69 
70  auto data = msg->data;
71  uint32_t height = msg->height, width = msg->width, offset = msg->step - 3 * width;
72  uint32_t i = 0, j = 0;
73  image im = make_image(width, height, 3);
74 
75  for (uint32_t line = height; line; line--)
76  {
77  for (uint32_t column = width; column; column--)
78  {
79  for (uint32_t channel = 0; channel < 3; channel++)
80  im.data[i + width * height * channel] = data[j++] / 255.;
81  i++;
82  }
83  j += offset;
84  }
85 
86  if (darknet_network_->w == (int) width && darknet_network_->h == (int) height)
87  {
88  return im;
89  }
90  image resized = resize_image(im, darknet_network_->w, darknet_network_->h);
91  free_image(im);
92  return resized;
93  }

95  std::vector< RectClassScore<float> > Yolo3Detector::forward(image& in_darknet_image)
96  {
97  float * in_data = in_darknet_image.data;
98  float *prediction = network_predict(darknet_network_, in_data);
99  layer output_layer = darknet_network_->layers[darknet_network_->n - 1];
100 
101  output_layer.output = prediction;
102  int nboxes = 0;
103  int num_classes = output_layer.classes;
104  detection *darknet_detections = get_network_boxes(darknet_network_, darknet_network_->w, darknet_network_->h, min_confidence_, .5, NULL, 0, &nboxes);
105 
106  do_nms_sort(darknet_detections, nboxes, num_classes, nms_threshold_);
107 
108  std::vector< RectClassScore<float> > detections;
109 
110  for (int i = 0; i < nboxes; i++)
111  {
112  int class_id = -1;
113  float score = 0.f;
114  //find the class
115  for(int j = 0; j < num_classes; ++j){
116  if (darknet_detections[i].prob[j] >= min_confidence_){
117  if (class_id < 0) {
118  class_id = j;
119  score = darknet_detections[i].prob[j];
120  }
121  }
122  }
123  //if class found
124  if (class_id >= 0)
125  {
126  RectClassScore<float> detection;
127 
128  detection.x = darknet_detections[i].bbox.x - darknet_detections[i].bbox.w/2;
129  detection.y = darknet_detections[i].bbox.y - darknet_detections[i].bbox.h/2;
130  detection.w = darknet_detections[i].bbox.w;
131  detection.h = darknet_detections[i].bbox.h;
132  detection.score = score;
133  detection.class_type = class_id;
134  //std::cout << detection.toString() << std::endl;
135 
136  detections.push_back(detection);
137  }
138  }
139  //std::cout << std::endl;
140  return detections;
141  }

146 void Yolo3DetectorNode::convert_rect_to_image_obj(std::vector< RectClassScore<float> >& in_objects, autoware_msgs::DetectedObjectArray& out_message)
147 {
148  for (unsigned int i = 0; i < in_objects.size(); ++i)
149  {
150  {
151  autoware_msgs::DetectedObject obj;
152 
153  obj.x = (in_objects[i].x /image_ratio_) - image_left_right_border_/image_ratio_;
154  obj.y = (in_objects[i].y /image_ratio_) - image_top_bottom_border_/image_ratio_;
155  obj.width = in_objects[i].w /image_ratio_;
156  obj.height = in_objects[i].h /image_ratio_;
157  if (in_objects[i].x < 0)
158  obj.x = 0;
159  if (in_objects[i].y < 0)
160  obj.y = 0;
161  if (in_objects[i].w < 0)
162  obj.width = 0;
163  if (in_objects[i].h < 0)
164  obj.height = 0;
165 
166  obj.score = in_objects[i].score;
167  if (use_coco_names_)
168  {
169  obj.label = in_objects[i].GetClassString();
170  }
171  else
172  {
173  if (in_objects[i].class_type < custom_names_.size())
174  obj.label = custom_names_[in_objects[i].class_type];
175  else
176  obj.label = "unknown";
177  }
178  obj.valid = true;
179 
180  out_message.objects.push_back(obj);
181 
182  }
183  }

197 image Yolo3DetectorNode::convert_ipl_to_image(const sensor_msgs::ImageConstPtr& msg)
198 {
199  cv_bridge::CvImagePtr cv_image = cv_bridge::toCvCopy(msg, "bgr8");//toCvCopy(image_source, sensor_msgs::image_encodings::BGR8);
200  cv::Mat mat_image = cv_image->image;
201 
202  int network_input_width = yolo_detector_.get_network_width();
203  int network_input_height = yolo_detector_.get_network_height();
204 
205  int image_height = msg->height,
206  image_width = msg->width;
207 
208  IplImage ipl_image;
209  cv::Mat final_mat;
210 
211  if (network_input_width!=image_width
212  || network_input_height != image_height)
213  {
214  //final_mat = cv::Mat(network_input_width, network_input_height, CV_8UC3, cv::Scalar(0,0,0));
215  image_ratio_ = (double ) network_input_width / (double)mat_image.cols;
216 
217  cv::resize(mat_image, final_mat, cv::Size(), image_ratio_, image_ratio_);
218  image_top_bottom_border_ = abs(final_mat.rows-network_input_height)/2;
219  image_left_right_border_ = abs(final_mat.cols-network_input_width)/2;
220  cv::copyMakeBorder(final_mat, final_mat,
221  image_top_bottom_border_, image_top_bottom_border_,
222  image_left_right_border_, image_left_right_border_,
223  cv::BORDER_CONSTANT, cv::Scalar(0,0,0));
224  }
225  else
226  final_mat = mat_image;
227 
228  ipl_image = final_mat;
229 
230  unsigned char *data = (unsigned char *)ipl_image.imageData;
231  int h = ipl_image.height;
232  int w = ipl_image.width;
233  int c = ipl_image.nChannels;
234  int step = ipl_image.widthStep;
235  int i, j, k;
236 
237  image darknet_image = make_image(w, h, c);
238 
239  for(i = 0; i < h; ++i){
240  for(k= 0; k < c; ++k){
241  for(j = 0; j < w; ++j){
242  darknet_image.data[k*w*h + i*w + j] = data[i*step + j*c + k]/255.;
243  }
244  }
245  }
246  rgbgr_image(darknet_image);
247  return darknet_image;
248 }

287 void Yolo3DetectorNode::Run()
288 {
289  //ROS STUFF
290  ros::NodeHandle private_node_handle("~");//to receive args
291 
292  //RECEIVE IMAGE TOPIC NAME
293  std::string image_raw_topic_str;
294  if (private_node_handle.getParam("image_raw_node", image_raw_topic_str))
295  {
296  ROS_INFO("Setting image node to %s", image_raw_topic_str.c_str());
297  }
298  else
299  {
300  ROS_INFO("No image node received, defaulting to /image_raw, you can use _image_raw_node:=YOUR_TOPIC");
301  image_raw_topic_str = "/image_raw";
302  }
303 
304  std::string network_definition_file;
305  std::string pretrained_model_file, names_file;
306  if (private_node_handle.getParam("network_definition_file", network_definition_file))
307  {
308  ROS_INFO("Network Definition File (Config): %s", network_definition_file.c_str());
309  }
310  else
311  {
312  ROS_INFO("No Network Definition File was received. Finishing execution.");
313  return;
314  }
315  if (private_node_handle.getParam("pretrained_model_file", pretrained_model_file))
316  {
317  ROS_INFO("Pretrained Model File (Weights): %s", pretrained_model_file.c_str());
318  }
319  else
320  {
321  ROS_INFO("No Pretrained Model File was received. Finishing execution.");
322  return;
323  }
324 
325  if (private_node_handle.getParam("names_file", names_file))
326  {
327  ROS_INFO("Names File: %s", names_file.c_str());
328  use_coco_names_ = false;
329  custom_names_ = read_custom_names_file(names_file);
330  }
331  else
332  {
333  ROS_INFO("No Names file was received. Using default COCO names.");
334  use_coco_names_ = true;
335  }
336 
337  private_node_handle.param<float>("score_threshold", score_threshold_, 0.5);
338  ROS_INFO("[%s] score_threshold: %f",__APP_NAME__, score_threshold_);
339 
340  private_node_handle.param<float>("nms_threshold", nms_threshold_, 0.45);
341  ROS_INFO("[%s] nms_threshold: %f",__APP_NAME__, nms_threshold_);
342 
343 
344  ROS_INFO("Initializing Yolo on Darknet...");
345  yolo_detector_.load(network_definition_file, pretrained_model_file, score_threshold_, nms_threshold_);
346  ROS_INFO("Initialization complete.");
347 
348  #if (CV_MAJOR_VERSION <= 2)
349  cv::generateColors(colors_, 80);
350  #else
351  generateColors(colors_, 80);
352  #endif
353 
354  publisher_objects_ = node_handle_.advertise<autoware_msgs::DetectedObjectArray>("/detection/image_detector/objects", 1);
355 
356  ROS_INFO("Subscribing to... %s", image_raw_topic_str.c_str());
357  subscriber_image_raw_ = node_handle_.subscribe(image_raw_topic_str, 1, &Yolo3DetectorNode::image_callback, this);
358 
359  std::string config_topic("/config");
360  config_topic += "/Yolo3";
361  subscriber_yolo_config_ = node_handle_.subscribe(config_topic, 1, &Yolo3DetectorNode::config_cb, this);
362 
363  ROS_INFO_STREAM( __APP_NAME__ << "" );
364 
365  ros::spin();
366  ROS_INFO("END Yolo");
367
