54 void LShapeFilter::getPointsInPointcloudFrame(cv::Point2f rect_points[],
55  std::vector<cv::Point2f>& pointcloud_frame_points,
56  const cv::Point& offset_point)
57 {
58  // loop 4 rect points
59  for (int point_i = 0; point_i < 4; point_i++)
60  {
61  cv::Point2f offset_point_float(offset_point.x, offset_point.y);
62 
63  cv::Point2f reverse_offset_point = rect_points[point_i] - offset_point_float;
64  // reverse from image coordinate to eucledian coordinate
65  float r_x = reverse_offset_point.x;
66  float r_y = pic_scale_ * roi_m_ - reverse_offset_point.y;
67  cv::Point2f eucledian_coordinate_pic_point(r_x, r_y);
68  // reverse to roi_m_*roi_m_ scale
69  cv::Point2f offset_pointcloud_point = eucledian_coordinate_pic_point * float(1/pic_scale_);
70  // reverse from (0 < x,y < roi_m_) to (roi_m_/2 < x,y < roi_m_/2)
71  cv::Point2f offset_vec_(roi_m_ / 2, roi_m_ / 2);
72  cv::Point2f pointcloud_point = offset_pointcloud_point - offset_vec_;
73  pointcloud_frame_points[point_i] = pointcloud_point;
74  }
75 }

77 void LShapeFilter::updateCpFromPoints(const std::vector<cv::Point2f>& pointcloud_frame_points,
78  autoware_msgs::DetectedObject& object)
79 {
80  cv::Point2f p1 = pointcloud_frame_points[0];
81  cv::Point2f p2 = pointcloud_frame_points[1];
82  cv::Point2f p3 = pointcloud_frame_points[2];
83  cv::Point2f p4 = pointcloud_frame_points[3];
84 
85  double s1 = ((p4.x - p2.x) * (p1.y - p2.y) - (p4.y - p2.y) * (p1.x - p2.x)) / 2;
86  double s2 = ((p4.x - p2.x) * (p2.y - p3.y) - (p4.y - p2.y) * (p2.x - p3.x)) / 2;
87  double cx = p1.x + (p3.x - p1.x) * s1 / (s1 + s2);
88  double cy = p1.y + (p3.y - p1.y) * s1 / (s1 + s2);
89 
90  object.pose.position.x = cx;
91  object.pose.position.y = cy;
92  object.pose.position.z = -sensor_height_ / 2;
93 }

95 void LShapeFilter::toRightAngleBBox(std::vector<cv::Point2f>& pointcloud_frame_points)
96 {
97  cv::Point2f p1 = pointcloud_frame_points[0];
98  cv::Point2f p2 = pointcloud_frame_points[1];
99  cv::Point2f p3 = pointcloud_frame_points[2];
100 
101  cv::Point2f vec1(p2.x - p1.x, p2.y - p1.y);
102  cv::Point2f vec2(p3.x - p2.x, p3.y - p2.y);
103 
104  // from the equation of inner product
105  double cos_theta = vec1.dot(vec2) / (norm(vec1) + norm(vec2));
106  double theta = acos(cos_theta);
107  double diff_theta = theta - M_PI / 2;
108 
109  if (std::abs(diff_theta) > 0.1)
110  {
111  double m1 = vec1.y / vec1.x;
112  double b1 = p3.y - m1 * p3.x;
113  double m2 = -1.0 / m1;
114  double b2 = p2.y - (m2 * p2.x);
115 
116  double x = (b2 - b1) / (m1 - m2);
117  double y = (b2 * m1 - b1 * m2) / (m1 - m2);
118 
119  double delta_x = x - p2.x;
120  double delta_y = y - p2.y;
121 
122  pointcloud_frame_points[2].x = x;
123  pointcloud_frame_points[2].y = y;
124  pointcloud_frame_points[3].x = pointcloud_frame_points[0].x + delta_x;
125  pointcloud_frame_points[3].y = pointcloud_frame_points[0].y + delta_y;
126  }
127 }

129 void LShapeFilter::updateDimentionAndEstimatedAngle(const std::vector<cv::Point2f>& pointcloud_frame_points,
130  autoware_msgs::DetectedObject& object)
131 {
132  // p1-p2 and p2-p3 is line segment, p1-p3 is diagonal
133  cv::Point2f p1 = pointcloud_frame_points[0];
134  cv::Point2f p2 = pointcloud_frame_points[1];
135  cv::Point2f p3 = pointcloud_frame_points[2];
136 
137  cv::Point2f vec1 = p1 - p2;
138  cv::Point2f vec2 = p3 - p2;
139  double dist1 = norm(vec1);
140  double dist2 = norm(vec2);
141  double bb_yaw;
142  // dist1 is length, dist2 is width
143  if (dist1 > dist2)
144  {
145  bb_yaw = atan2(p1.y - p2.y, p1.x - p2.x);
146  object.dimensions.x = dist1;
147  object.dimensions.y = dist2;
148  object.dimensions.z = 2;
149  }
150  // dist1 is width, dist2 is length
151  else
152  {
153  bb_yaw = atan2(p3.y - p2.y, p3.x - p2.x);
154  object.dimensions.x = dist2;
155  object.dimensions.y = dist1;
156  object.dimensions.z = 2;
157  }
158  // convert yaw to quartenion
159  tf::Matrix3x3 obs_mat;
160  obs_mat.setEulerYPR(bb_yaw, 0, 0);
161 
162  tf::Quaternion q_tf;
163  obs_mat.getRotation(q_tf);
164  object.pose.orientation.x = q_tf.getX();
165  object.pose.orientation.y = q_tf.getY();
166  object.pose.orientation.z = q_tf.getZ();
167  object.pose.orientation.w = q_tf.getW();
168 }

170 void LShapeFilter::getLShapeBB(const autoware_msgs::DetectedObjectArray& in_object_array,
171  autoware_msgs::DetectedObjectArray& out_object_array)
172 {
173  out_object_array.header = in_object_array.header;
174 
175  for (const auto& in_object : in_object_array.objects)
176  {
177  pcl::PointCloud<pcl::PointXYZ> cloud;
178 
179  // Convert from ros msg to PCL::pic_scalePointCloud data type
180  pcl::fromROSMsg(in_object.pointcloud, cloud);
181 
182  // calculating offset so that projecting pointcloud into cv::mat
183  cv::Mat m(pic_scale_ * roi_m_, pic_scale_ * roi_m_, CV_8UC1, cv::Scalar(0));
184  cv::Point2f tmp_pointcloud_point(cloud[0].x, cloud[0].y);
185  cv::Point2f tmp_pointcloud_offset(roi_m_ / 2, roi_m_ / 2);
186  cv::Point2f tmp_offset_pointcloud_point = tmp_pointcloud_point + tmp_pointcloud_offset;
187  cv::Point tmp_pic_point = tmp_offset_pointcloud_point * pic_scale_;
188 
189  int tmp_init_pic_x = tmp_pic_point.x;
190  int tmp_init_pic_y = pic_scale_ * roi_m_ - tmp_pic_point.y;
191 
192  cv::Point tmp_init_pic_point(tmp_init_pic_x, tmp_init_pic_y);
193  cv::Point tmp_init_offset_vec(roi_m_ * pic_scale_ / 2, roi_m_ * pic_scale_ / 2);
194  cv::Point offset_init_pic_point = tmp_init_offset_vec - tmp_init_pic_point;
195 
196  int num_points = cloud.size();
197  std::vector<cv::Point> point_vec(num_points);
198  std::vector<cv::Point2f> pointcloud_frame_points(4);
199 
200  // init variables
201  cv::Point2f min_m_p(0, 0);
202  cv::Point2f max_m_p(0, 0);
203  float min_m = std::numeric_limits<float>::max();
204  float max_m = std::numeric_limits<float>::lowest();
205 
206  for (int i_point = 0; i_point < num_points; i_point++)
207  {
208  const float p_x = cloud[i_point].x;
209  const float p_y = cloud[i_point].y;
210 
211  // cast (roi_m_/2 < x,y < roi_m_/2) into (0 < x,y < roi_m_)
212  cv::Point2f pointcloud_point(p_x, p_y);
213  cv::Point2f pointcloud_offset_vec(roi_m_ / 2, roi_m_ / 2);
214  cv::Point2f offset_pointcloud_point = pointcloud_point + pointcloud_offset_vec;
215  // cast (roi_m_)m*(roi_m_)m into pic_scale_
216  cv::Point scaled_point = offset_pointcloud_point * pic_scale_;
217  // cast into image coordinate
218  int pic_x = scaled_point.x;
219  int pic_y = pic_scale_ * roi_m_ - scaled_point.y;
220  // offset so that the object would be locate at the center
221  cv::Point pic_point(pic_x, pic_y);
222  cv::Point offset_point = pic_point + offset_init_pic_point;
223 
224  // Make sure points are inside the image size
225  if (offset_point.x > (pic_scale_ * roi_m_) || offset_point.x < 0 || offset_point.y < 0 ||
226  offset_point.y > (pic_scale_ * roi_m_))
227  {
228  continue;
229  }
230  // cast the pointcloud into cv::mat
231  m.at<uchar>(offset_point.y, offset_point.x) = 255;
232  point_vec[i_point] = offset_point;
233  // calculate min and max slope for x1, x3(edge points)
234  float delta_m = p_y / p_x;
235  if (delta_m < min_m)
236  {
237  min_m = delta_m;
238  min_m_p.x = p_x;
239  min_m_p.y = p_y;
240  }
241 
242  if (delta_m > max_m)
243  {
244  max_m = delta_m;
245  max_m_p.x = p_x;
246  max_m_p.y = p_y;
247  }
248  }
249  if (max_m == std::numeric_limits<float>::lowest() || min_m == std::numeric_limits<float>::max())
250  {
251  continue;
252  }
253  // L shape fitting parameters
254  cv::Point2f dist_vec = max_m_p - min_m_p;
255  float slope_dist = sqrt(dist_vec.x * dist_vec.x + dist_vec.y * dist_vec.y);
256  float slope = (max_m_p.y - min_m_p.y) / (max_m_p.x - min_m_p.x);
257 
258  // random variable
259  std::mt19937_64 mt;
260  mt.seed(in_object_array.header.stamp.toSec());
261  // mt.seed(0);
262  std::uniform_int_distribution<> rand_points(0, num_points - 1);
263 
264  // start l shape fitting for car like object
265  if (slope_dist > slope_dist_thres_ && num_points > num_points_thres_)
266  {
267  float max_dist = 0;
268  cv::Point2f max_p(0, 0);
269 
270  // get max distance from random sampling points
271  for (int i = 0; i < random_points_; i++)
272  {
273  int p_ind = rand_points(mt);
274  assert(p_ind >= 0 && p_ind < (cloud.size() - 1));
275  cv::Point2f p_i(cloud[p_ind].x, cloud[p_ind].y);
276 
277  // from equation of distance between line and point
278  float dist = std::abs(slope * p_i.x - 1 * p_i.y + max_m_p.y - slope * max_m_p.x) / std::sqrt(slope * slope + 1);
279  if (dist > max_dist)
280  {
281  max_dist = dist;
282  max_p = p_i;
283  }
284  }
285  // vector adding
286  cv::Point2f max_m_vec = max_m_p - max_p;
287  cv::Point2f min_m_vec = min_m_p - max_p;
288  cv::Point2f last_p = max_p + max_m_vec + min_m_vec;
289 
290  pointcloud_frame_points[0] = min_m_p;
291  pointcloud_frame_points[1] = max_p;
292  pointcloud_frame_points[2] = max_m_p;
293  pointcloud_frame_points[3] = last_p;
294  }
295  else
296  {
297  // MinAreaRect fitting
298  cv::RotatedRect rect_info = cv::minAreaRect(point_vec);
299  cv::Point2f rect_points[4];
300  rect_info.points(rect_points);
301  // covert points back to lidar coordinate
302  getPointsInPointcloudFrame(rect_points, pointcloud_frame_points, offset_init_pic_point);
303  }
304 
305  autoware_msgs::DetectedObject output_object;
306  output_object = in_object;
307 
308  // update output_object pose
309  updateCpFromPoints(pointcloud_frame_points, output_object);
310 
311  // update pointcloud_frame_points to make it right angle bbox
312  toRightAngleBBox(pointcloud_frame_points);
313 
314  // update output_object dimensions
315  updateDimentionAndEstimatedAngle(pointcloud_frame_points, output_object);
316 
317  out_object_array.objects.push_back(output_object);
318  }
319 }
