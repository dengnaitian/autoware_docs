85 cv::Rect ROSRangeVisionFusionApp::ProjectDetectionToRect(const autoware_msgs::DetectedObject &in_detection)
86 {
87  cv::Rect projected_box;
88 
89  Eigen::Vector3f pos;
90  pos << in_detection.pose.position.x,
91  in_detection.pose.position.y,
92  in_detection.pose.position.z;
93 
94  Eigen::Quaternionf rot(in_detection.pose.orientation.w,
95  in_detection.pose.orientation.x,
96  in_detection.pose.orientation.y,
97  in_detection.pose.orientation.z);
98 
99  std::vector<double> dims = {
100  in_detection.dimensions.x,
101  in_detection.dimensions.y,
102  in_detection.dimensions.z
103  };
104 
105  jsk_recognition_utils::Cube cube(pos, rot, dims);
106 
107  Eigen::Affine3f range_vision_tf;
108  tf::transformTFToEigen(camera_lidar_tf_, range_vision_tf);
109  jsk_recognition_utils::Vertices vertices = cube.transformVertices(range_vision_tf);
110 
111  std::vector<cv::Point> polygon;
112  for (auto &vertex : vertices)
113  {
114  cv::Point p = ProjectPoint(cv::Point3f(vertex.x(), vertex.y(), vertex.z()));
115  polygon.push_back(p);
116  }
117 
118  projected_box = cv::boundingRect(polygon);
119 
120  return projected_box;
121 }

144 void
145 ROSRangeVisionFusionApp::CalculateObjectFeatures(autoware_msgs::DetectedObject &in_out_object, bool in_estimate_pose)
146 {
147 
148  float min_x = std::numeric_limits<float>::max();
149  float max_x = -std::numeric_limits<float>::max();
150  float min_y = std::numeric_limits<float>::max();
151  float max_y = -std::numeric_limits<float>::max();
152  float min_z = std::numeric_limits<float>::max();
153  float max_z = -std::numeric_limits<float>::max();
154  float average_x = 0, average_y = 0, average_z = 0, length, width, height;
155  pcl::PointXYZ centroid, min_point, max_point, average_point;
156 
157  std::vector<cv::Point2f> object_2d_points;
158 
159  pcl::PointCloud<pcl::PointXYZ> in_cloud;
160  pcl::fromROSMsg(in_out_object.pointcloud, in_cloud);
161 
162  for (const auto &point : in_cloud.points)
163  {
164  average_x += point.x;
165  average_y += point.y;
166  average_z += point.z;
167  centroid.x += point.x;
168  centroid.y += point.y;
169  centroid.z += point.z;
170 
171  if (point.x < min_x)
172  min_x = point.x;
173  if (point.y < min_y)
174  min_y = point.y;
175  if (point.z < min_z)
176  min_z = point.z;
177  if (point.x > max_x)
178  max_x = point.x;
179  if (point.y > max_y)
180  max_y = point.y;
181  if (point.z > max_z)
182  max_z = point.z;
183 
184  cv::Point2f pt;
185  pt.x = point.x;
186  pt.y = point.y;
187  object_2d_points.push_back(pt);
188  }
189  min_point.x = min_x;
190  min_point.y = min_y;
191  min_point.z = min_z;
192  max_point.x = max_x;
193  max_point.y = max_y;
194  max_point.z = max_z;
195 
196  if (in_cloud.points.size() > 0)
197  {
198  centroid.x /= in_cloud.points.size();
199  centroid.y /= in_cloud.points.size();
200  centroid.z /= in_cloud.points.size();
201 
202  average_x /= in_cloud.points.size();
203  average_y /= in_cloud.points.size();
204  average_z /= in_cloud.points.size();
205  }
206 
207  average_point.x = average_x;
208  average_point.y = average_y;
209  average_point.z = average_z;
210 
211  length = max_point.x - min_point.x;
212  width = max_point.y - min_point.y;
213  height = max_point.z - min_point.z;
214 
215  geometry_msgs::PolygonStamped convex_hull;
216  std::vector<cv::Point2f> hull_points;
217  if (object_2d_points.size() > 0)
218  cv::convexHull(object_2d_points, hull_points);
219 
220  convex_hull.header = in_out_object.header;
221  for (size_t i = 0; i < hull_points.size() + 1; i++)
222  {
223  geometry_msgs::Point32 point;
224  point.x = hull_points[i % hull_points.size()].x;
225  point.y = hull_points[i % hull_points.size()].y;
226  point.z = min_point.z;
227  convex_hull.polygon.points.push_back(point);
228  }
229 
230  for (size_t i = 0; i < hull_points.size() + 1; i++)
231  {
232  geometry_msgs::Point32 point;
233  point.x = hull_points[i % hull_points.size()].x;
234  point.y = hull_points[i % hull_points.size()].y;
235  point.z = max_point.z;
236  convex_hull.polygon.points.push_back(point);
237  }
238 
239  double rz = 0;
240  if (in_estimate_pose)
241  {
242  cv::RotatedRect box = cv::minAreaRect(hull_points);
243  rz = box.angle * 3.14 / 180;
244  in_out_object.pose.position.x = box.center.x;
245  in_out_object.pose.position.y = box.center.y;
246  in_out_object.dimensions.x = box.size.width;
247  in_out_object.dimensions.y = box.size.height;
248  }
249 
250  in_out_object.convex_hull = convex_hull;
251 
252  in_out_object.pose.position.x = min_point.x + length / 2;
253  in_out_object.pose.position.y = min_point.y + width / 2;
254  in_out_object.pose.position.z = min_point.z + height / 2;
255 
256  in_out_object.dimensions.x = ((length < 0) ? -1 * length : length);
257  in_out_object.dimensions.y = ((width < 0) ? -1 * width : width);
258  in_out_object.dimensions.z = ((height < 0) ? -1 * height : height);
259 
260  tf::Quaternion quat = tf::createQuaternionFromRPY(0.0, 0.0, rz);
261  tf::quaternionTFToMsg(quat, in_out_object.pose.orientation);
262 }

328 autoware_msgs::DetectedObjectArray
329 ROSRangeVisionFusionApp::FuseRangeVisionDetections(
330  const autoware_msgs::DetectedObjectArray::ConstPtr &in_vision_detections,
331  const autoware_msgs::DetectedObjectArray::ConstPtr &in_range_detections)
332 {
333 
334  autoware_msgs::DetectedObjectArray range_in_cv;
335  autoware_msgs::DetectedObjectArray range_out_cv;
336  TransformRangeToVision(in_range_detections, range_in_cv, range_out_cv);
337 
338  autoware_msgs::DetectedObjectArray fused_objects;
339  fused_objects.header = in_range_detections->header;
340 
341  std::vector<std::vector<size_t> > vision_range_assignments(in_vision_detections->objects.size());
342  std::vector<bool> used_vision_detections(in_vision_detections->objects.size(), false);
343  std::vector<long> vision_range_closest(in_vision_detections->objects.size());
344 
345  for (size_t i = 0; i < in_vision_detections->objects.size(); i++)
346  {
347  auto vision_object = in_vision_detections->objects[i];
348 
349  cv::Rect vision_rect(vision_object.x, vision_object.y,
350  vision_object.width, vision_object.height);
351  int vision_rect_area = vision_rect.area();
352  long closest_index = -1;
353  double closest_distance = std::numeric_limits<double>::max();
354 
355  for (size_t j = 0; j < range_in_cv.objects.size(); j++)
356  {
357  double current_distance = GetDistanceToObject(range_in_cv.objects[j]);
358 
359  cv::Rect range_rect = ProjectDetectionToRect(range_in_cv.objects[j]);
360  int range_rect_area = range_rect.area();
361 
362  cv::Rect overlap = range_rect & vision_rect;
363  if ((overlap.area() > range_rect_area * overlap_threshold_)
364  || (overlap.area() > vision_rect_area * overlap_threshold_)
365  )
366  {
367  vision_range_assignments[i].push_back(j);
368  range_in_cv.objects[j].score = vision_object.score;
369  range_in_cv.objects[j].label = vision_object.label;
370  range_in_cv.objects[j].color = vision_object.color;
371  range_in_cv.objects[j].image_frame = vision_object.image_frame;
372  range_in_cv.objects[j].x = vision_object.x;
373  range_in_cv.objects[j].y = vision_object.y;
374  range_in_cv.objects[j].width = vision_object.width;
375  range_in_cv.objects[j].height = vision_object.height;
376  range_in_cv.objects[j].angle = vision_object.angle;
377  range_in_cv.objects[j].id = vision_object.id;
378  CheckMinimumDimensions(range_in_cv.objects[j]);
379  if (vision_object.pose.orientation.x > 0
380  || vision_object.pose.orientation.y > 0
381  || vision_object.pose.orientation.z > 0)
382  {
383  range_in_cv.objects[i].pose.orientation = vision_object.pose.orientation;
384  }
385  if (current_distance < closest_distance)
386  {
387  closest_index = j;
388  closest_distance = current_distance;
389  }
390  used_vision_detections[i] = true;
391  }//end if overlap
392  }//end for range_in_cv
393  vision_range_closest[i] = closest_index;
394  }
395 
396  std::vector<bool> used_range_detections(range_in_cv.objects.size(), false);
397  //only assign the closest
398  for (size_t i = 0; i < vision_range_assignments.size(); i++)
399  {
400  if (!range_in_cv.objects.empty() && vision_range_closest[i] >= 0)
401  {
402  used_range_detections[i] = true;
403  fused_objects.objects.push_back(range_in_cv.objects[vision_range_closest[i]]);
404  }
405  }
406  for (size_t i = 0; i < used_vision_detections.size(); i++)
407  {
408  if (!used_vision_detections[i])
409  {
410  fused_objects.objects.push_back(in_vision_detections->objects[i]);
411  }
412  }
413  //add also objects outside the image
414  for (auto &object: range_out_cv.objects)
415  {
416  fused_objects.objects.push_back(object);
417  }
418  //enable merged for visualization
419  for (auto &object : fused_objects.objects)
420  {
421  object.valid = true;
422  }
423 
424  return fused_objects;
425 }

427 void
428 ROSRangeVisionFusionApp::SyncedDetectionsCallback(
429  const autoware_msgs::DetectedObjectArray::ConstPtr &in_vision_detections,
430  const autoware_msgs::DetectedObjectArray::ConstPtr &in_range_detections)
431 {
432  autoware_msgs::DetectedObjectArray fusion_objects;
433  fusion_objects.objects.clear();
434 
435  if (empty_frames_ > 5)
436  {
437  ROS_INFO("[%s] Empty Detections. Make sure the vision and range detectors are running.", __APP_NAME__);
438  }
439 
440  if (nullptr == in_vision_detections
441  && nullptr == in_range_detections)
442  {
443  empty_frames_++;
444  return;
445  }
446 
447  if (nullptr == in_vision_detections
448  && nullptr != in_range_detections
449  && !in_range_detections->objects.empty())
450  {
451  publisher_fused_objects_.publish(in_range_detections);
452  empty_frames_++;
453  return;
454  }
455  if (nullptr == in_range_detections
456  && nullptr != in_vision_detections
457  && !in_vision_detections->objects.empty())
458  {
459  publisher_fused_objects_.publish(in_vision_detections);
460  empty_frames_++;
461  return;
462  }
463 
464  if (!camera_lidar_tf_ok_)
465  {
466  camera_lidar_tf_ = FindTransform(image_frame_id_,
467  in_range_detections->header.frame_id);
468  }
469  if (
470  !camera_lidar_tf_ok_ ||
471  !camera_info_ok_)
472  {
473  ROS_INFO("[%s] Missing Camera-LiDAR TF or CameraInfo", __APP_NAME__);
474  return;
475  }
476 
477  fusion_objects = FuseRangeVisionDetections(in_vision_detections, in_range_detections);
478 
479  publisher_fused_objects_.publish(fusion_objects);
480  empty_frames_ = 0;
481 
482  vision_detections_ = nullptr;
483  range_detections_ = nullptr;
