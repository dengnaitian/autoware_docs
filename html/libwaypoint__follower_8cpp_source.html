160 // get closest waypoint from current pose
161 int getClosestWaypoint(const autoware_msgs::Lane &current_path, geometry_msgs::Pose current_pose)
162 {
163  WayPoints wp;
164  wp.setPath(current_path);
165 
166  if (wp.isEmpty())
167  return -1;
168 
169  // search closest candidate within a certain meter
170  double search_distance = 5.0;
171  std::vector<int> waypoint_candidates;
172  for (int i = 1; i < wp.getSize(); i++)
173  {
174  if (getPlaneDistance(wp.getWaypointPosition(i), current_pose.position) > search_distance)
175  continue;
176 
177  if (!wp.isFront(i, current_pose))
178  continue;
179 
180  double angle_threshold = 90;
181  if (getRelativeAngle(wp.getWaypointPose(i), current_pose) > angle_threshold)
182  continue;
183 
184  waypoint_candidates.push_back(i);
185  }
186 
187  // get closest waypoint from candidates
188  if (!waypoint_candidates.empty())
189  {
190  int waypoint_min = -1;
191  double distance_min = DBL_MAX;
192  for (auto el : waypoint_candidates)
193  {
194  // ROS_INFO("closest_candidates : %d",el);
195  double d = getPlaneDistance(wp.getWaypointPosition(el), current_pose.position);
196  if (d < distance_min)
197  {
198  waypoint_min = el;
199  distance_min = d;
200  }
201  }
202  return waypoint_min;
203  }
204  else
205  {
206  ROS_INFO("no candidate. search closest waypoint from all waypoints...");
207  // if there is no candidate...
208  int waypoint_min = -1;
209  double distance_min = DBL_MAX;
210  for (int i = 1; i < wp.getSize(); i++)
211  {
212  if (!wp.isFront(i, current_pose))
213  continue;
214 
215  // if (!wp.isValid(i, current_pose))
216  // continue;
217 
218  double d = getPlaneDistance(wp.getWaypointPosition(i), current_pose.position);
219  if (d < distance_min)
220  {
221  waypoint_min = i;
222  distance_min = d;
223  }
224  }
225  return waypoint_min;
226  }
227 }

229 // let the linear equation be "ax + by + c = 0"
230 // if there are two points (x1,y1) , (x2,y2), a = "y2-y1, b = "(-1) * x2 - x1" ,c = "(-1) * (y2-y1)x1 + (x2-x1)y1"
231 bool getLinearEquation(geometry_msgs::Point start, geometry_msgs::Point end, double *a, double *b, double *c)
232 {
233  //(x1, y1) = (start.x, star.y), (x2, y2) = (end.x, end.y)
234  double sub_x = fabs(start.x - end.x);
235  double sub_y = fabs(start.y - end.y);
236  double error = pow(10, -5); // 0.00001
237 
238  if (sub_x < error && sub_y < error)
239  {
240  ROS_INFO("two points are the same point!!");
241  return false;
242  }
243 
244  *a = end.y - start.y;
245  *b = (-1) * (end.x - start.x);
246  *c = (-1) * (end.y - start.y) * start.x + (end.x - start.x) * start.y;
247 
248  return true;
249 }
