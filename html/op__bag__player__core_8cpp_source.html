156 void TestingUI::BagReaderModeMainLoop()
157 {
158  if(m_bBagOpen)
159  {
160  bool bFreshMsg = false;
161 
162  if(m_PlayMode == PLAY_FORWARD)
163  {
164  bFreshMsg = ReadNextFrame();
165  }
166  else if(m_PlayMode == PLAY_STEP_FORWARD)
167  {
168  if(!m_bStepDone)
169  {
170  bFreshMsg = ReadNextFrame();
171  m_bStepDone = true;
172  }
173  }
174  else if(m_PlayMode == PLAY_STEP_BACKWARD)
175  {
176  if(!m_bStepDone)
177  {
178  bFreshMsg = ReadPrevFrame();
179  m_bStepDone = true;
180  }
181  }
182 
183  if(bFreshMsg && m_pLatestCloud != NULL)
184  {
185  sensor_msgs::PointCloud2 cloud = *m_pLatestCloud;
186  cloud.header.stamp = ros::Time::now();
187  pub_Point_Raw.publish(cloud);
188  }
189 
190  if(bFreshMsg && m_pLatestImage != NULL)
191  {
192  sensor_msgs::Image img = *m_pLatestImage;
193  img.header.stamp = ros::Time().now();
194  pub_Image_Raw.publish(img);
195  }
196 
197  if(bFreshMsg && m_pLatestPose != NULL)
198  {
199  geometry_msgs::PoseStamped pos = *m_pLatestPose;
200  pos.header.stamp = ros::Time::now();
201  pub_NDT_pose.publish(pos);
202  }
203  }

222  }
223 
224  bImage = m_ImageReader.ReadNext(m_pLatestImage, &m_pLatestPose->header.stamp);
225  while(!m_ImageReader.ReadNext(m_pLatestImage, &m_pLatestPose->header.stamp))
226  {
227 
228  }
229  }
230 
231  return bPose || bCloud || bImage;

253 void TestingUI::InitNode(const BagReaderParams& params, const int& mode)
254 {

258  case 0:

266  break;
267  default:
268  break;
269  }
270 
271  ros::NodeHandle nh;
272  if(m_TestMode == ROSBAG_MODE)
273  {
274  m_bBagOpen = OpenROSBag();
275 
276  pub_Point_Raw = nh.advertise<sensor_msgs::PointCloud2>(m_BagParams.lidarTopic_pub, 10);
277  pub_Image_Raw = nh.advertise<sensor_msgs::Image>(m_BagParams.imageTopic_pub, 10);
278  pub_NDT_pose = nh.advertise<geometry_msgs::PoseStamped>(m_BagParams.poseTopic_pub, 10);
279  }
280  else if(m_TestMode == SIMULATION_MODE)
281  {
282  pub_SimuStepSignal = nh.advertise<geometry_msgs::TwistStamped>("simu_step_signal", 1);
283  pub_PredStepSignal = nh.advertise<geometry_msgs::TwistStamped>("pred_step_signal", 1);
284  pub_SimuGenSignal = nh.advertise<geometry_msgs::TwistStamped>("simu_generate_signal", 1);
