133 static void MqttReceiver::on_message(struct mosquitto *mosq, void *obj, const struct mosquitto_message *message)
134 {
135  if(message->payloadlen) {
136  string delim (",");
137  string msg_str((char *)message->payload, message->payloadlen);
138  vector<string> cmds;
139  boost::algorithm::split(cmds, msg_str, boost::is_any_of(","));
140 
141  if(cmds.size() == 8) {
142  autoware_msgs::RemoteCmd msg;
143  msg.vehicle_cmd.steer_cmd.steer = stof(cmds[0]) * steer_max_val;
144  msg.vehicle_cmd.accel_cmd.accel = stof(cmds[1]) * accel_max_val;
145  msg.vehicle_cmd.brake_cmd.brake = stof(cmds[2]) * brake_max_val;
146  msg.vehicle_cmd.gear = stoi(cmds[3]);
147  // lamp
148  switch(stoi(cmds[4])) {
149  msg.vehicle_cmd.lamp_cmd.l = 0;
150  msg.vehicle_cmd.lamp_cmd.r = 0;
151  case 0:
152  break;
153  case 1:
154  msg.vehicle_cmd.lamp_cmd.l = 1;
155  break;
156  case 2:
157  msg.vehicle_cmd.lamp_cmd.r = 1;
158  break;
159  case 3:
160  msg.vehicle_cmd.lamp_cmd.l = 1;
161  msg.vehicle_cmd.lamp_cmd.r = 1;
162  break;
163  default:
164  ROS_WARN("Invalid lamp_cmd");
165  break;
166  }
167  msg.vehicle_cmd.twist_cmd.twist.linear.x = stof(cmds[1]) * linear_x_max_val;
168  msg.vehicle_cmd.twist_cmd.twist.angular.z = stof(cmds[0]);
169  msg.vehicle_cmd.ctrl_cmd.linear_velocity = stof(cmds[1]) * linear_x_max_val;
170  msg.vehicle_cmd.ctrl_cmd.steering_angle = stof(cmds[0]) * steer_max_val;
171  msg.vehicle_cmd.mode = stoi(cmds[5]);
172  msg.vehicle_cmd.emergency = stoi(cmds[7]);
173  msg.control_mode = stoi(cmds[6]);
174  remote_cmd_pub.publish(msg);
175  }
176  else {
177  ROS_INFO("Failed to parse remote command.\n");
178  }
179  }
180 }
