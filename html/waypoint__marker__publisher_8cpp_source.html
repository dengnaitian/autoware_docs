93 void createGlobalLaneArrayVelocityMarker(const autoware_msgs::LaneArray& lane_waypoints_array)
94 {
95  visualization_msgs::MarkerArray tmp_marker_array;
96  // display by markers the velocity of each waypoint.
97  visualization_msgs::Marker velocity_marker;
98  velocity_marker.header.frame_id = "map";
99  velocity_marker.header.stamp = ros::Time::now();
100  velocity_marker.type = visualization_msgs::Marker::TEXT_VIEW_FACING;
101  velocity_marker.action = visualization_msgs::Marker::ADD;
102  velocity_marker.scale.z = 0.4;
103  velocity_marker.color.r = 1;
104  velocity_marker.color.g = 1;
105  velocity_marker.color.b = 1;
106  velocity_marker.color.a = 1.0;
107  velocity_marker.frame_locked = true;
108 
109  int count = 1;
110  for (auto lane : lane_waypoints_array.lanes)
111  {
112  velocity_marker.ns = "global_velocity_lane_" + std::to_string(count);
113  for (int i = 0; i < static_cast<int>(lane.waypoints.size()); i++)
114  {
115  velocity_marker.id = i;
116  geometry_msgs::Point relative_p;
117  relative_p.y = 0.5;
118  velocity_marker.pose.position = calcAbsoluteCoordinate(relative_p, lane.waypoints[i].pose.pose);
119  velocity_marker.pose.position.z += 0.2;
120 
121  // double to string
122  std::string vel = std::to_string(mps2kmph(lane.waypoints[i].twist.twist.linear.x));
123  velocity_marker.text = vel.erase(vel.find_first_of(".") + 2);
124 
125  tmp_marker_array.markers.push_back(velocity_marker);
126  }
127  count++;
128  }
129 
130  g_global_marker_array.markers.insert(g_global_marker_array.markers.end(), tmp_marker_array.markers.begin(),
131  tmp_marker_array.markers.end());
132 }

332 void lightCallback(const autoware_msgs::TrafficLightConstPtr& msg)
333 {
334  std_msgs::ColorRGBA global_color;
335  global_color.a = g_global_alpha;
336 
337  std_msgs::ColorRGBA local_color;
338  local_color.a = g_local_alpha;
339 
340  switch (msg->traffic_light)
341  {
342  case TRAFFIC_LIGHT_RED:
343  global_color.r = 1.0;
344  _global_color = global_color;
345  local_color.r = 1.0;
346  g_local_color = local_color;
347  break;
348  case TRAFFIC_LIGHT_GREEN:
349  global_color.g = 1.0;
350  _global_color = global_color;
351  local_color.g = 1.0;
352  g_local_color = local_color;
353  break;
354  case TRAFFIC_LIGHT_UNKNOWN:
355  global_color.b = 1.0;
356  global_color.g = 0.7;
357  _global_color = global_color;
358  local_color.b = 1.0;
359  local_color.g = 0.7;
360  g_local_color = local_color;
361  break;
362  default:
363  ROS_ERROR("unknown traffic_light");
364  return;
365  }
366 }

412 int main(int argc, char** argv)
413 {
414  ros::init(argc, argv, "waypoints_marker_publisher");
415  ros::NodeHandle nh;
416  ros::NodeHandle private_nh("~");
417 
418  // subscribe traffic light

420  ros::Subscriber light_managed_sub = nh.subscribe("light_color_managed", 10, receiveManualDetection);
421 
422  // subscribe global waypoints
423  ros::Subscriber lane_array_sub = nh.subscribe("lane_waypoints_array", 10, laneArrayCallback);
424  ros::Subscriber traffic_array_sub = nh.subscribe("traffic_waypoints_array", 10, laneArrayCallback);
425 
426  // subscribe local waypoints
427  ros::Subscriber final_sub = nh.subscribe("final_waypoints", 10, finalCallback);
428  ros::Subscriber closest_sub = nh.subscribe("closest_waypoint", 10, closestCallback);
429 
430  // subscribe config
431  ros::Subscriber config_sub = nh.subscribe("config/lane_stop", 10, configParameter);
432 
433  g_local_mark_pub = nh.advertise<visualization_msgs::MarkerArray>("local_waypoints_mark", 10, true);
434  g_global_mark_pub = nh.advertise<visualization_msgs::MarkerArray>("global_waypoints_mark", 10, true);
435 
436  // initialize path color
437  _initial_color.g = 0.7;
438  _initial_color.b = 1.0;
439  _global_color = _initial_color;
440  _global_color.a = g_global_alpha;
441  g_local_color = _initial_color;
442  g_local_color.a = g_local_alpha;
443 
444  ros::spin();
445 }
