126 void GroundFilter::InitRadiusTable(int in_model)
127 {
128  double a;
129  double b;
130  double theta;
131  switch (in_model)
132  {
133  case 64:
134  a = 1.0/3*M_PI/180;
135  b = max_slope_*M_PI/180;
136  for (int i = 0; i < 64; i++)
137  {
138  if (i <= 31)
139  {
140  if (i == 31) a = -a;
141  theta = (1.0/3*i - 2.0)*M_PI/180;
142  radius_table_[i] = fabs(sensor_height_*(1.0/(tan(theta)+tan(b)) - 1.0/(tan(a+theta)+tan(b))));
143  }
144  else
145  {
146  a = 0.5*M_PI/180;
147  theta = (8.83 + (0.5)*(i - 32.0))*M_PI/180;
148  radius_table_[i] = fabs(sensor_height_*(1.0/(tan(theta)+tan(b)) - 1.0/(tan(a+theta)+tan(b))));
149  }
150  }
151  break;
152  case 32:
153  a = 4.0/3*M_PI/180;
154  b = max_slope_*M_PI/180;
155  for (int i = 0; i < 32; i++)
156  {
157  theta = (-31.0/3 + (4.0/3)*i)*180/M_PI;
158  radius_table_[i] = fabs(sensor_height_*(1.0/(tan(theta)+tan(b)) - 1.0/(tan(a+theta)+tan(b))));
159  }
160  break;
161  case 16:
162  a = 2.0*M_PI/180;
163  b = max_slope_*M_PI/180;
164  for (int i = 0; i < 16; i++)
165  {
166  theta = (-30.0/2 + (2.0)*i)*180/M_PI;
167  radius_table_[i] = fabs(sensor_height_*(1.0/(tan(theta)+tan(b)) - 1.0/(tan(a+theta)+tan(b))));
168  }
169  break;
170  default:
171  a = 1.0/3*M_PI/180;
172  b = max_slope_*M_PI/180;
173  for (int i = 0; i < 64; i++)
174  {
175  if (i <= 31)
176  {
177  if (i == 31) a = -a;
178  theta = (1.0/3*i - 2.0)*M_PI/180;
179  radius_table_[i] = fabs(sensor_height_*(1.0/(tan(theta)+tan(b)) - 1.0/(tan(a+theta)+tan(b))));
180  }
181  else
182  {
183  a = 0.5*M_PI/180;
184  theta = (8.83 + (0.5)*(i - 32.0))*M_PI/180;
185  radius_table_[i] = fabs(sensor_height_*(1.0/(tan(theta)+tan(b)) - 1.0/(tan(a+theta)+tan(b))));
186  }
187  }

189  }

198 void GroundFilter::FilterGround(const pcl::PointCloud<velodyne_pointcloud::PointXYZIR>::ConstPtr &in_cloud_msg,
199  pcl::PointCloud<velodyne_pointcloud::PointXYZIR> &out_groundless_points,
200  pcl::PointCloud<velodyne_pointcloud::PointXYZIR> &out_ground_points)
201 {
202 
203  velodyne_pointcloud::PointXYZIR point;
204  InitDepthMap(horizontal_res_);
205 
206  for (size_t i = 0; i < in_cloud_msg->points.size(); i++)
207  {
208  double u = atan2(in_cloud_msg->points[i].y,in_cloud_msg->points[i].x) * 180/M_PI;
209  if (u < 0) { u = 360 + u; }
210  int column = horizontal_res_ - (int)((double)horizontal_res_ * u / 360.0) - 1;
211  int row = vertical_res_ - 1 - in_cloud_msg->points[i].ring;
212  index_map_.at<int>(row, column) = i;
213  }
214 
215  for (int i = 0; i < horizontal_res_; i++)
216  {
217  Label point_class[vertical_res_];
218  int point_index[vertical_res_];
219  int point_index_size = 0;
220  double z_max = 0;
221  double z_min = 0;
222  double r_ref = 0;
223  std::copy(class_label_, class_label_ + vertical_res_, point_class);
224  for (int j = 0; j < vertical_res_; j++)
225  {
226  if (index_map_.at<int>(j,i) > -1 && point_class[j] == UNKNOWN)
227  {
228  double x0 = in_cloud_msg->points[index_map_.at<int>(j, i)].x;
229  double y0 = in_cloud_msg->points[index_map_.at<int>(j, i)].y;
230  double z0 = in_cloud_msg->points[index_map_.at<int>(j, i)].z;
231  double r0 = sqrt(x0*x0 + y0*y0);
232  double r_diff = fabs(r0 - r_ref);
233  if (r_diff < radius_table_[j] || r_ref == 0)
234  {
235  r_ref = r0;
236  if (z0 > z_max || r_ref == 0) z_max = z0;
237  if (z0 < z_min || r_ref == 0) z_min = z0;
238  point_index[point_index_size] = j;
239  point_index_size++;
240  }
241  else
242  {
243  if (point_index_size > 1 && (z_max - z_min) > vertical_thres_)
244  {
245  for (int m = 0; m < point_index_size; m++)
246  {
247  int index = index_map_.at<int>(point_index[m],i);
248  point.x = in_cloud_msg->points[index].x;
249  point.y = in_cloud_msg->points[index].y;
250  point.z = in_cloud_msg->points[index].z;
251  point.intensity = in_cloud_msg->points[index].intensity;
252  point.ring = in_cloud_msg->points[index].ring;
253  out_groundless_points.push_back(point);
254  point_class[point_index[m]] = VERTICAL;
255  }
256  point_index_size = 0;
257  }
258  else
259  {
260  for (int m = 0; m < point_index_size; m++)
261  {
262  int index = index_map_.at<int>(point_index[m],i);
263  point.x = in_cloud_msg->points[index].x;
264  point.y = in_cloud_msg->points[index].y;
265  point.z = in_cloud_msg->points[index].z;
266  point.intensity = in_cloud_msg->points[index].intensity;
267  point.ring = in_cloud_msg->points[index].ring;
268  out_ground_points.push_back(point);
269  point_class[point_index[m]] = GROUND;
270  }
271  point_index_size = 0;
272  }
273  r_ref = r0;
274  z_max = z0;
275  z_min = z0;
276  point_index[point_index_size] = j;
277  point_index_size++;
278  }
279  }
280  if (j == vertical_res_ - 1 && point_index_size != 0)
281  {
282  if (point_index_size > 1 && (z_max - z_min) > vertical_thres_)
283  {
284  for (int m = 0; m < point_index_size; m++)
285  {
286  int index = index_map_.at<int>(point_index[m],i);
287  point.x = in_cloud_msg->points[index].x;
