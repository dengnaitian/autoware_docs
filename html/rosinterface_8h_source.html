172 {

187 template<class MSGTYPE>
188 MSGTYPE ROSSub<MSGTYPE>::getMessage()
189 {
190  MSGTYPE msg;
191  lock.lockForWrite();
192  if(receiveflag&&!msgs.isEmpty())
193  {
194  msg=msgs.front();
195  msgs.pop_front();
196  }
197  lock.unlock();
198  return msg;
199 }
