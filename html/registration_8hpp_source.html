119 template <typename PointSource, typename PointTarget, typename Scalar> inline double
120 pcl_omp::Registration<PointSource, PointTarget, Scalar>::getFitnessScore (
121  const std::vector<float> &distances_a,
122  const std::vector<float> &distances_b)
123 {
124  unsigned int nr_elem = static_cast<unsigned int> (std::min (distances_a.size (), distances_b.size ()));
125  Eigen::VectorXf map_a = Eigen::VectorXf::Map (&distances_a[0], nr_elem);
126  Eigen::VectorXf map_b = Eigen::VectorXf::Map (&distances_b[0], nr_elem);
127  return (static_cast<double> ((map_a - map_b).sum ()) / static_cast<double> (nr_elem));
128 }

131 template <typename PointSource, typename PointTarget, typename Scalar> inline double
132 pcl_omp::Registration<PointSource, PointTarget, Scalar>::getFitnessScore (double max_range)
133 {
134 
135  double fitness_score = 0.0;
136 
137  // Transform the input dataset using the final transformation
138  PointCloudSource input_transformed;
139  // transformPointCloud (*input_, input_transformed, final_transformation_);
140  input_transformed.resize (input_->size ());
141 
142 #ifdef _OPENMP
143  int nr = 0;
144 #pragma omp parallel
145  {
146 #pragma omp for
147 #endif
148  for (size_t i = 0; i < input_->size (); ++i)
149  {
150  const PointSource &src = input_->points[i];
151  PointTarget &tgt = input_transformed.points[i];
152  tgt.x = static_cast<float> (final_transformation_ (0, 0) * src.x + final_transformation_ (0, 1) * src.y + final_transformation_ (0, 2) * src.z + final_transformation_ (0, 3));
153  tgt.y = static_cast<float> (final_transformation_ (1, 0) * src.x + final_transformation_ (1, 1) * src.y + final_transformation_ (1, 2) * src.z + final_transformation_ (1, 3));
154  tgt.z = static_cast<float> (final_transformation_ (2, 0) * src.x + final_transformation_ (2, 1) * src.y + final_transformation_ (2, 2) * src.z + final_transformation_ (2, 3));
155  }
156 
157  std::vector<int> nn_indices (1);
158  std::vector<float> nn_dists (1);
159 
160  // For each point in the source dataset
161 #ifndef _OPENMP
162  int nr = 0;
163 #endif
164 #ifdef _OPENMP
165 #pragma omp for private(nn_dists, nn_indices) reduction(+:fitness_score)
166 #endif
167  for (size_t i = 0; i < input_transformed.points.size (); ++i)
168  {
169  // Find its nearest neighbor in the target
170  tree_->nearestKSearch (input_transformed.points[i], 1, nn_indices, nn_dists);
171 
172  // Deal with occlusions (incomplete targets)
173  if (nn_dists[0] <= max_range)
174  {
175  // Add to the fitness score
176  fitness_score += nn_dists[0];
177  nr++;
178  }
179  }
180 #ifdef _OPENMP
181  }
182 #endif
183 
184  if (nr > 0)
185  return (fitness_score / nr);
186  else
187  return (std::numeric_limits<double>::max ());
188 
189 }

200 pcl_omp::Registration<PointSource, PointTarget, Scalar>::align (PointCloudSource &output, const Matrix4& guess)
201 {
202  if (!initCompute ())
203  return;
204 
205  // Resize the output dataset
206  if (output.points.size () != indices_->size ())
207  output.points.resize (indices_->size ());
208  // Copy the header
209  output.header = input_->header;
210  // Check if the output will be computed for all points or only a subset
211  if (indices_->size () != input_->points.size ())
212  {
213  output.width = static_cast<uint32_t> (indices_->size ());
214  output.height = 1;
215  }
216  else
217  {
218  output.width = static_cast<uint32_t> (input_->width);
219  output.height = input_->height;
220  }
221  output.is_dense = input_->is_dense;
222 
223  // Copy the point data to output
224  for (size_t i = 0; i < indices_->size (); ++i)
225  output.points[i] = input_->points[(*indices_)[i]];
226 
227  // Set the internal point representation of choice unless otherwise noted
228  if (point_representation_ && !force_no_recompute_)
229  tree_->setPointRepresentation (point_representation_);
230 
231  // Perform the actual transformation computation
232  converged_ = false;
233  final_transformation_ = transformation_ = previous_transformation_ = Matrix4::Identity ();
234 
235  // Right before we estimate the transformation, we set all the point.data[3] values to 1 to aid the rigid
236  // transformation
237  for (size_t i = 0; i < indices_->size (); ++i)
238  output.points[i].data[3] = 1.0;
239 
240  computeTransformation (output, guess);
241 
242  deinitCompute ();
