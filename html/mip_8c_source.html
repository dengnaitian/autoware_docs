280 u16 mip_finalize(u8 *mip_buffer)
281 {
282  u16 checksum, checksum_offset;
283  mip_header *header_ptr = (mip_header*)mip_buffer;
284 
285  //Null check info
286  if(mip_buffer == 0)
287  return 0;
288 
289  //Calculate the checksum offset
290  checksum_offset = header_ptr->payload_size + MIP_HEADER_SIZE;
291 
292  //Check that the offset is valid
293  if(checksum_offset > MIP_MAX_PACKET_SIZE - MIP_CHECKSUM_SIZE)
294  return 0;
295 
296  checksum = mip_calculate_checksum(mip_buffer);
297 
298  //Put the checksum in the packet
299  mip_buffer[checksum_offset] = (checksum >> 8) & 0xFF;
300  mip_buffer[checksum_offset + 1] = checksum & 0xFF;
301 
302  return header_ptr->payload_size + MIP_HEADER_SIZE + MIP_CHECKSUM_SIZE;
303 }

669 u16 mip_calculate_checksum(u8 *mip_buffer)
670 {
671  mip_header *header_ptr = (mip_header*)mip_buffer;
672  u16 i, checksum, checksum_range_size;
673  u8 checksum_byte1 = 0, checksum_byte2 = 0;
674 
675 
676  //Null check info
677  if(mip_buffer == 0)
678  return 0;
679 
680  //Calculate the size of the checksum data range
681  checksum_range_size = header_ptr->payload_size + MIP_HEADER_SIZE;
682 
683  //Check that the size is valid
684  if(checksum_range_size > MIP_MAX_PACKET_SIZE - MIP_CHECKSUM_SIZE)
685  return 0;
686 
687  //Calculate the fletcher checksum
688  for(i=0; i<checksum_range_size; i++)
689  {
690  checksum_byte1 += mip_buffer[i];
691  checksum_byte2 += checksum_byte1;
692  }
693 
694  //Form the packet checksum
695  checksum = ((u16)checksum_byte1<<8) + (u16)checksum_byte2;
696 
697  return checksum;
698 }
