81  bool open(void)
82  {
83  // Open device file
84  if (imu.openPort(device_) < 0)
85  {
86  ROS_ERROR("Failed to open device %s", device_.c_str());
87  }
88  // Wait 10ms for SPI ready

92  ROS_INFO("Product ID: %x\n", pid);

94  int publish_imu_data()
95  {
96  sensor_msgs::Imu data;
97  data.header.frame_id = frame_id_;
98  data.header.stamp = ros::Time::now();
99 
100  // Linear acceleration
101  data.linear_acceleration.x = imu.accl[0];
102  data.linear_acceleration.y = imu.accl[1];
103  data.linear_acceleration.z = imu.accl[2];
104 
105  // Angular velocity
106  data.angular_velocity.x = imu.gyro[0];
107  data.angular_velocity.y = imu.gyro[1];
108  data.angular_velocity.z = imu.gyro[2];
109 
110  // Orientation (not provided)
111  data.orientation.x = 0;
112  data.orientation.y = 0;
113  data.orientation.z = 0;
114  data.orientation.w = 1;
115 
116  imu_data_pub_.publish(data);
117  }
118  bool spin()
119  {
120  ros::Rate loop_rate(rate_);
121 
122  while (ros::ok())
123  {
124  if (burst_mode_)
125  {
126  if (imu.update_burst() == 0)
127  {
128  publish_imu_data();
129  }
130  else
131  {
132  ROS_ERROR("Cannot update burst");
133  }
134  }
135  else
136  {
137  if (imu.update() == 0)
138  {
139  publish_imu_data();
140  }
141  else
142  {
143  ROS_ERROR("Cannot update");
144  }
145  }
146  ros::spinOnce();
147  loop_rate.sleep();
148  }
149  return true;
150  }
151 };
