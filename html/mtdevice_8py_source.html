126  def read_msg(self):
127  """Low-level message receiving function."""
128  start = time.time()
129  while (time.time()-start)<self.timeout:
130  new_start = time.time()

132  # Makes sure the buffer has 'size' bytes.
133  def waitfor(size=1):

135  if time.time()-new_start >= self.timeout:
