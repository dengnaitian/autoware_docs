177  int counter = 0;
178  int i;
179  double xinters;
180  GPSPoint p1,p2;
181  int N = polygon.points.size();
182  if(N <=0 ) return -1;
183 
184  p1 = polygon.points.at(0);
185  for (i=1;i<=N;i++)
186  {
187  p2 = polygon.points.at(i % N);
188 
189  if (p.y > MIN(p1.y,p2.y))
190  {
191  if (p.y <= MAX(p1.y,p2.y))
192  {
193  if (p.x <= MAX(p1.x,p2.x))
194  {
195  if (p1.y != p2.y)
196  {
197  xinters = (p.y-p1.y)*(p2.x-p1.x)/(p2.y-p1.y)+p1.x;
198  if (p1.x == p2.x || p.x <= xinters)
199  counter++;
200  }
201  }
202  }
203  }
