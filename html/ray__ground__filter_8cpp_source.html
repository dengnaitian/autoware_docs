71 void RayGroundFilter::ConvertXYZIToRTZColor(const pcl::PointCloud<pcl::PointXYZI>::Ptr in_cloud,
72  PointCloudXYZIRTColor& out_organized_points,
73  std::vector<pcl::PointIndices>& out_radial_divided_indices,
74  std::vector<PointCloudXYZIRTColor>& out_radial_ordered_clouds)
75 {
76  out_organized_points.resize(in_cloud->points.size());
77  out_radial_divided_indices.clear();
78  out_radial_divided_indices.resize(radial_dividers_num_);
79  out_radial_ordered_clouds.resize(radial_dividers_num_);
80 
81  for(size_t i=0; i< in_cloud->points.size(); i++)
82  {
83  PointXYZIRTColor new_point;
84  auto radius = (float) sqrt(
85  in_cloud->points[i].x*in_cloud->points[i].x
86  + in_cloud->points[i].y*in_cloud->points[i].y
87  );
88  auto theta = (float) atan2(in_cloud->points[i].y, in_cloud->points[i].x) * 180 / M_PI;
89  if (theta < 0){ theta+=360; }
90 
91  auto radial_div = (size_t) floor(theta/radial_divider_angle_);
92  auto concentric_div = (size_t) floor(fabs(radius/concentric_divider_distance_));
93 
94  new_point.point = in_cloud->points[i];
95  new_point.radius = radius;
96  new_point.theta = theta;
97  new_point.radial_div = radial_div;
98  new_point.concentric_div = concentric_div;
99  new_point.red = (size_t) colors_[new_point.radial_div % color_num_].val[0];
100  new_point.green = (size_t) colors_[new_point.radial_div % color_num_].val[1];
101  new_point.blue = (size_t) colors_[new_point.radial_div % color_num_].val[2];
102  new_point.original_index = i;
103 
104  out_organized_points[i] = new_point;
105 
106  //radial divisions
107  out_radial_divided_indices[radial_div].indices.push_back(i);
108 
109  out_radial_ordered_clouds[radial_div].push_back(new_point);
110 
111  }//end for
112 
113  //order radial points on each division
114 #pragma omp for
115  for(size_t i=0; i< radial_dividers_num_; i++)
116  {
117  std::sort(out_radial_ordered_clouds[i].begin(), out_radial_ordered_clouds[i].end(),
118  [](const PointXYZIRTColor& a, const PointXYZIRTColor& b){ return a.radius < b.radius; });
119  }
120 }

128 void RayGroundFilter::ClassifyPointCloud(std::vector<PointCloudXYZIRTColor>& in_radial_ordered_clouds,
129  pcl::PointIndices& out_ground_indices,
130  pcl::PointIndices& out_no_ground_indices)
131 {
132  out_ground_indices.indices.clear();
133  out_no_ground_indices.indices.clear();
134 #pragma omp for
135  for (size_t i=0; i < in_radial_ordered_clouds.size(); i++)//sweep through each radial division
136  {
137  float prev_radius = 0.f;
138  float prev_height = - sensor_height_;
139  bool prev_ground = false;
140  bool current_ground = false;
141  for (size_t j=0; j < in_radial_ordered_clouds[i].size(); j++)//loop through each point in the radial div
142  {
143  float points_distance = in_radial_ordered_clouds[i][j].radius - prev_radius;
144  float height_threshold = tan(DEG2RAD(local_max_slope_)) * points_distance;
145  float current_height = in_radial_ordered_clouds[i][j].point.z;
146  float general_height_threshold = tan(DEG2RAD(general_max_slope_)) * in_radial_ordered_clouds[i][j].radius;
147 
148  //for points which are very close causing the height threshold to be tiny, set a minimum value
149  if (points_distance > concentric_divider_distance_ && height_threshold < min_height_threshold_)
150  { height_threshold = min_height_threshold_; }
151 
152  //check current point height against the LOCAL threshold (previous point)
153  if (current_height <= (prev_height + height_threshold)
154  && current_height >= (prev_height - height_threshold)
155  )
156  {
157  //Check again using general geometry (radius from origin) if previous points wasn't ground
158  if (!prev_ground)
159  {
160  if(current_height <= (-sensor_height_ + general_height_threshold)
161  && current_height >= (-sensor_height_ - general_height_threshold))
162  {
163  current_ground = true;
164  }
165  else
166  {current_ground = false;}
167  }
168  else
169  {
170  current_ground = true;
171  }
172  }
173  else
174  {
175  //check if previous point is too far from previous one, if so classify again
176  if (points_distance > reclass_distance_threshold_ &&
177  (current_height <= (-sensor_height_ + height_threshold)
178  && current_height >= (-sensor_height_ - height_threshold))
179  )
180  {
181  current_ground = true;
182  }
183  else
184  {current_ground = false;}
185  }
186 
187  if (current_ground)
188  {
189  out_ground_indices.indices.push_back(in_radial_ordered_clouds[i][j].original_index);
190  prev_ground=true;
191  }
192  else
193  {
194  out_no_ground_indices.indices.push_back(in_radial_ordered_clouds[i][j].original_index);
195  prev_ground = false;
196  }
197 
198  prev_radius = in_radial_ordered_clouds[i][j].radius;
199  prev_height = in_radial_ordered_clouds[i][j].point.z;
200  }
201  }
202 }
