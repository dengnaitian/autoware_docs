75  void SimuDecisionMaker::LocalizeMe(const double& dt)
76 {
77  //calculate the new x, y ,
78  WayPoint currPose = state;
79 
80  if(m_CurrentShift == SHIFT_POS_DD)
81  {
82  m_OdometryState.pos.x += m_CurrentVelocity * dt * cos(currPose.pos.a);
83  m_OdometryState.pos.y += m_CurrentVelocity * dt * sin(currPose.pos.a);
84  m_OdometryState.pos.a += m_CurrentVelocity * dt * tan(m_CurrentSteering) / m_CarInfo.wheel_base;
85 
86  }
87  else if(m_CurrentShift == SHIFT_POS_RR )
88  {
89  m_OdometryState.pos.x += -m_CurrentVelocity * dt * cos(currPose.pos.a);
90  m_OdometryState.pos.y += -m_CurrentVelocity * dt * sin(currPose.pos.a);
91  m_OdometryState.pos.a += -m_CurrentVelocity * dt * tan(m_CurrentSteering);
92  }
93 
94  m_OdometryState.pos.a = atan2(sin(m_OdometryState.pos.a), cos(m_OdometryState.pos.a));
95  m_OdometryState.pos.a = UtilityHNS::UtilityH::FixNegativeAngle(m_OdometryState.pos.a);
96 
97  state.pos.a = m_OdometryState.pos.a;
98  state.pos.x = m_OdometryState.pos.x - (m_CurrentVelocity*dt* (m_CarInfo.wheel_base) * cos (m_OdometryState.pos.a));
99  state.pos.y = m_OdometryState.pos.y - (m_CurrentVelocity*dt* (m_CarInfo.wheel_base/2.0) * sin (m_OdometryState.pos.a));
