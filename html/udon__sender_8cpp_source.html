179 void accept_sock(int backlog, std::size_t bufsize, int period, std::uint16_t port)
180 {
181  int listen_fd = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);
182  if (listen_fd < 0) {
183  ROS_ERROR_STREAM("socket: " << std::strerror(errno));
184  return;
185  }
186 
187  const int on = 1;
188  if (setsockopt(listen_fd, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on)) < 0) {
189  ROS_ERROR_STREAM("setsockopt: " << std::strerror(errno));
190  goto close_listen_fd;
191  }
192 
193  sockaddr_in server_addr;
194  std::memset(&server_addr, 0, sizeof(server_addr));
195  server_addr.sin_family = AF_INET;
196  server_addr.sin_port = htons(port);
197  server_addr.sin_addr.s_addr = htonl(INADDR_ANY);
198  if (bind(listen_fd, reinterpret_cast<const sockaddr *>(&server_addr), sizeof(server_addr)) < 0) {
199  ROS_ERROR_STREAM("bind: " << std::strerror(errno));
200  goto close_listen_fd;
201  }
202 
203  if (listen(listen_fd, backlog) < 0) {
204  ROS_ERROR_STREAM("listen: " << std::strerror(errno));
205  goto close_listen_fd;
206  }
207 
208  char astr[INET_ADDRSTRLEN];
209  if (inet_ntop(AF_INET, &server_addr.sin_addr, astr, sizeof(astr)) == nullptr) {
210  ROS_ERROR_STREAM("inet_ntop: " << std::strerror(errno));
211  goto close_listen_fd;
212  }
213  ROS_INFO_STREAM("listen " << astr << ":" << ntohs(server_addr.sin_port));
214 
215  int connect_fd;
216  while (true) {
217  sockaddr_in client_addr;
218  socklen_t len = sizeof(client_addr);
219  connect_fd = accept(listen_fd, reinterpret_cast<sockaddr *>(&client_addr), &len);
220  if (connect_fd < 0) {
221  ROS_ERROR_STREAM("accept: " << std::strerror(errno));
222  goto close_listen_fd;
223  }
224 
225  if (inet_ntop(AF_INET, &client_addr.sin_addr, astr, sizeof(astr)) == nullptr) {
226  ROS_ERROR_STREAM("inet_ntop: " << std::strerror(errno));
227  goto close_connect_fd;
228  }
229  ROS_INFO_STREAM("connect " << astr << ":" << ntohs(client_addr.sin_port));
230 
231  try {
232  std::thread sender(send_info, client_addr, connect_fd, bufsize, period);
233  sender.detach();
234  } catch (std::exception &ex) {
235  ROS_ERROR_STREAM("std::thread::thread: " << ex.what());
236  goto close_connect_fd;
237  }
238  }
239 
240 close_connect_fd:
241  close(connect_fd);
242 close_listen_fd:
243  close(listen_fd);
244 }
