156  if (recorder_status_ != Stopped) {
157  ROS_ERROR("cannot start a new record since recording now.");
158  return 1;
159  }
160 
161  if (options_.topics.size() == 0) {
162  if (options_.limit > 0) {
163  ROS_ERROR("Specifing a count is not valid with automatic topic subscription.");
164  return 1;
165  }
166 
167  if (!options_.record_all && (options_.node == std::string(""))) {
168  ROS_ERROR("No topics specified.");
169  return 1;
170  }
171  }
172

188  ros::Subscriber trigger_sub;
189 
190  recorder_status_ = Recording;

192 
193  record_thread_.reset(new boost::thread(boost::bind(&Recorder::doRecord, this)));
194 
195  ros::Timer check_master_timer;
196  if (options_.record_all || options_.regex || (options_.node != std::string("")))
197  {

261 bool Recorder::shouldSubscribeToTopic(std::string const& topic, bool from_node) {
262  // ignore already known topics
263  if (isSubscribed(topic)) {
264  return false;
265  }
266 
267  // subtract exclusion regex, if any
268  if(options_.do_exclude && boost::regex_match(topic, options_.exclude_regex)) {
269  return false;
270  }
271 
272  if(options_.record_all || from_node) {
273  return true;
274  }
275 
276  if (options_.regex) {
277  // Treat the topics as regular expressions
278  foreach(string const& regex_str, options_.topics) {
279  boost::regex e(regex_str);
280  boost::smatch what;
281  if (boost::regex_match(topic, what, e, boost::match_extra))
282  return true;
283  }
284  }
285  else {
286  foreach(string const& t, options_.topics)
287  if (t == topic)
288  return true;
289  }
290 
291  return false;
292 }

310  Time rectime = Time::now();
311 
312  if (options_.verbose)
313  cout << "Received message on topic " << subscriber->getTopic() << endl;
314 
315  OutgoingMessage out(topic, msg_event.getMessage(), msg_event.getConnectionHeaderPtr(), rectime);
316 
317  {
318  boost::mutex::scoped_lock lock(queue_mutex_);
319 
320  queue_->push(out);
321  queue_size_ += out.msg->size();
322 
323  // Check to see if buffer has been exceeded
324  while (options_.buffer_size > 0 && queue_size_ > options_.buffer_size) {
325  OutgoingMessage drop = queue_->front();
326  queue_->pop();
327  queue_size_ -= drop.msg->size();
328 
329  Time now = Time::now();
330  if (now > last_buffer_warn_ + ros::Duration(5.0)) {
331  ROS_WARN("rosbag record buffer exceeded. Dropping oldest queued message.");
332  last_buffer_warn_ = now;
333  }
334  }
335  }
336 
337 // queue_condition_.notify_all();
338  if (!options_.snapshot)
339  queue_condition_.notify_all();
340 
341  // If we are book-keeping count, decrement and possibly shutdown
342  if ((*count) > 0) {
343  (*count)--;
344  if ((*count) == 0) {
345  subscriber->shutdown();
346 
347  num_subscribers_--;
348 
349  if (num_subscribers_ == 0) {
350  recorder_error_ = 1;
351  }
352  }

467 void Recorder::doRecord() {
468  // Open bag file for writing
469  startWriting();
470 
471  // Schedule the disk space check
472  warn_next_ = ros::WallTime();
473  checkDisk();
474  check_disk_next_ = ros::WallTime::now() + ros::WallDuration().fromSec(20.0);
475 
476  // Technically the queue_mutex_ should be locked while checking empty.
477  // Except it should only get checked if the node is not ok, and thus
478  // it shouldn't be in contention.
479  ros::NodeHandle nh;
480  while (recorder_status_ == Recording && (nh.ok() || !queue_->empty())) {
481  boost::unique_lock<boost::mutex> lock(queue_mutex_);
482 
483  bool finished = false;
484  while (queue_->empty()) {
485  if (!nh.ok()) {
486  lock.release()->unlock();
487  finished = true;
488  break;
489  }
490  boost::xtime xt;
491 #if BOOST_VERSION >= 105000
492  boost::xtime_get(&xt, boost::TIME_UTC_);
493 #else
494  boost::xtime_get(&xt, boost::TIME_UTC);
495 #endif
496  xt.nsec += 250000000;
497  queue_condition_.timed_wait(lock, xt);
498  if (checkDuration(ros::Time::now()))
499  {
500  finished = true;
501  break;
502  }
503  if (recorder_status_ != Recording)
504  {
505  finished = true;
506  break;
507  }
508  }
509  if (finished)
510  break;
511 
512  OutgoingMessage out = queue_->front();
513  queue_->pop();
514  queue_size_ -= out.msg->size();
515 
516  lock.release()->unlock();
517 
518  if (checkSize())
519  break;
520 
521 // if (checkDuration(out.time))
522 // break;
523 
524  if (scheduledCheckDisk() && checkLogging()) {
525  try {
526  bag_.write(out.topic, out.time, *out.msg, out.connection_header);
527  } catch (rosbag::BagIOException e) {
528  ROS_ERROR("write failure");
529  recorder_error_ = 1;
530  break;
531  }
532  }
533 
534  if (recorder_error_ != 0)
535  break;
536  }
537 
538  stopWriting();
539 }

580 void Recorder::doCheckMaster(ros::TimerEvent const& e, ros::NodeHandle& node_handle) {
581  ros::master::V_TopicInfo topics;
582  if (ros::master::getTopics(topics)) {
583  foreach(ros::master::TopicInfo const& t, topics) {
584  if (shouldSubscribeToTopic(t.name))
585  subscribe(t.name);
586  }
587  }
588 
589  if (options_.node != std::string(""))
590  {
591 
592  XmlRpc::XmlRpcValue req;
593  req[0] = ros::this_node::getName();
594  req[1] = options_.node;
595  XmlRpc::XmlRpcValue resp;
596  XmlRpc::XmlRpcValue payload;
597 
598  if (ros::master::execute("lookupNode", req, resp, payload, true))
599  {
600  std::string peer_host;
601  uint32_t peer_port;
602 
603  if (!ros::network::splitURI(static_cast<std::string>(resp[2]), peer_host, peer_port))
604  {
605  ROS_ERROR("Bad xml-rpc URI trying to inspect node at: [%s]", static_cast<std::string>(resp[2]).c_str());
606  } else {
607 
608  XmlRpc::XmlRpcClient c(peer_host.c_str(), peer_port, "/");
609  XmlRpc::XmlRpcValue req2;
610  XmlRpc::XmlRpcValue resp2;
611  req2[0] = ros::this_node::getName();
612  c.execute("getSubscriptions", req2, resp2);
613 
614  if (!c.isFault() && resp2.valid() && resp2.size() > 0 && static_cast<int>(resp2[0]) == 1)
615  {
616  for(int i = 0; i < resp2[2].size(); i++)
617  {
618  if (shouldSubscribeToTopic(resp2[2][i][0], true))
619  subscribe(resp2[2][i][0]);
620  }
621  } else {
622  ROS_ERROR("Node at: [%s] failed to return subscriptions.", static_cast<std::string>(resp[2]).c_str());

624  }
625  }
