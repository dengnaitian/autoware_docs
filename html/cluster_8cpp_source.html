122 void Cluster::SetCloud(const pcl::PointCloud<pcl::PointXYZ>::Ptr in_origin_cloud_ptr,
123  const std::vector<int>& in_cluster_indices, std_msgs::Header in_ros_header, int in_id, int in_r,
124  int in_g, int in_b, std::string in_label, bool in_estimate_pose)
125 {
126  label_ = in_label;
127  id_ = in_id;
128  r_ = in_r;
129  g_ = in_g;
130  b_ = in_b;
131  // extract pointcloud using the indices
132  // calculate min and max points
133  pcl::PointCloud<pcl::PointXYZRGB>::Ptr current_cluster(new pcl::PointCloud<pcl::PointXYZRGB>);
134  float min_x = std::numeric_limits<float>::max();
135  float max_x = -std::numeric_limits<float>::max();
136  float min_y = std::numeric_limits<float>::max();
137  float max_y = -std::numeric_limits<float>::max();
138  float min_z = std::numeric_limits<float>::max();
139  float max_z = -std::numeric_limits<float>::max();
140  float average_x = 0, average_y = 0, average_z = 0;
141 
142  for (auto pit = in_cluster_indices.begin(); pit != in_cluster_indices.end(); ++pit)
143  {
144  // fill new colored cluster point by point
145  pcl::PointXYZRGB p;
146  p.x = in_origin_cloud_ptr->points[*pit].x;
147  p.y = in_origin_cloud_ptr->points[*pit].y;
148  p.z = in_origin_cloud_ptr->points[*pit].z;
149  p.r = in_r;
150  p.g = in_g;
151  p.b = in_b;
152 
153  average_x += p.x;
154  average_y += p.y;
155  average_z += p.z;
156  centroid_.x += p.x;
157  centroid_.y += p.y;
158  centroid_.z += p.z;
159  current_cluster->points.push_back(p);
160 
161  if (p.x < min_x)
162  min_x = p.x;
163  if (p.y < min_y)
164  min_y = p.y;
165  if (p.z < min_z)
166  min_z = p.z;
167  if (p.x > max_x)
168  max_x = p.x;
169  if (p.y > max_y)
170  max_y = p.y;
171  if (p.z > max_z)
172  max_z = p.z;
173  }
174  // min, max points
175  min_point_.x = min_x;
176  min_point_.y = min_y;
177  min_point_.z = min_z;
178  max_point_.x = max_x;
179  max_point_.y = max_y;
180  max_point_.z = max_z;
181 
182  // calculate centroid, average
183  if (in_cluster_indices.size() > 0)
184  {
185  centroid_.x /= in_cluster_indices.size();
186  centroid_.y /= in_cluster_indices.size();
187  centroid_.z /= in_cluster_indices.size();
188 
189  average_x /= in_cluster_indices.size();
190  average_y /= in_cluster_indices.size();
191  average_z /= in_cluster_indices.size();
192  }
193 
194  average_point_.x = average_x;
195  average_point_.y = average_y;
196  average_point_.z = average_z;
197 
198  // calculate bounding box
199  length_ = max_point_.x - min_point_.x;
200  width_ = max_point_.y - min_point_.y;
201  height_ = max_point_.z - min_point_.z;
202 
203  bounding_box_.header = in_ros_header;
204 
205  bounding_box_.pose.position.x = min_point_.x + length_ / 2;
206  bounding_box_.pose.position.y = min_point_.y + width_ / 2;
207  bounding_box_.pose.position.z = min_point_.z + height_ / 2;
208 
209  bounding_box_.dimensions.x = ((length_ < 0) ? -1 * length_ : length_);
210  bounding_box_.dimensions.y = ((width_ < 0) ? -1 * width_ : width_);
211  bounding_box_.dimensions.z = ((height_ < 0) ? -1 * height_ : height_);
212 
213  // pose estimation
214  double rz = 0;
215 
216  {
217  std::vector<cv::Point2f> points;
218  for (unsigned int i = 0; i < current_cluster->points.size(); i++)
219  {
220  cv::Point2f pt;
221  pt.x = current_cluster->points[i].x;
222  pt.y = current_cluster->points[i].y;
223  points.push_back(pt);
224  }
225 
226  std::vector<cv::Point2f> hull;
227  cv::convexHull(points, hull);
228 
229  polygon_.header = in_ros_header;
230  for (size_t i = 0; i < hull.size() + 1; i++)
231  {
232  geometry_msgs::Point32 point;
233  point.x = hull[i % hull.size()].x;
234  point.y = hull[i % hull.size()].y;
235  point.z = min_point_.z;
236  polygon_.polygon.points.push_back(point);
237  }
238 
239  for (size_t i = 0; i < hull.size() + 1; i++)
240  {
241  geometry_msgs::Point32 point;
242  point.x = hull[i % hull.size()].x;
243  point.y = hull[i % hull.size()].y;
244  point.z = max_point_.z;
245  polygon_.polygon.points.push_back(point);
246  }
247  if (in_estimate_pose)
248  {
249  cv::RotatedRect box = minAreaRect(hull);
250  rz = box.angle * 3.14 / 180;
251  bounding_box_.pose.position.x = box.center.x;
252  bounding_box_.pose.position.y = box.center.y;
253  bounding_box_.dimensions.x = box.size.width;
254  bounding_box_.dimensions.y = box.size.height;
255  }
256  }
257 
258  // set bounding box direction
259  tf::Quaternion quat = tf::createQuaternionFromRPY(0.0, 0.0, rz);
260  tf::quaternionTFToMsg(quat, bounding_box_.pose.orientation);
261 
262  current_cluster->width = current_cluster->points.size();
263  current_cluster->height = 1;
264  current_cluster->is_dense = true;
265 
266  // Get EigenValues, eigenvectors
267  if (current_cluster->points.size() > 3)
268  {
269  pcl::PCA<pcl::PointXYZ> current_cluster_pca;
270  pcl::PointCloud<pcl::PointXYZ>::Ptr current_cluster_mono(new pcl::PointCloud<pcl::PointXYZ>);
271 
272  pcl::copyPointCloud<pcl::PointXYZRGB, pcl::PointXYZ>(*current_cluster, *current_cluster_mono);
273 
274  current_cluster_pca.setInputCloud(current_cluster_mono);
275  eigen_vectors_ = current_cluster_pca.getEigenVectors();
276  eigen_values_ = current_cluster_pca.getEigenValues();
277  }
278 
279  valid_cluster_ = true;
280  pointcloud_ = current_cluster;
281 }

283 std::vector<float> Cluster::GetFpfhDescriptor(const unsigned int& in_ompnum_threads,
284  const double& in_normal_search_radius,
285  const double& in_fpfh_search_radius)
286 {
287  std::vector<float> cluster_fpfh_histogram(33, 0.0);
288 
289  pcl::search::KdTree<pcl::PointXYZRGB>::Ptr norm_tree(new pcl::search::KdTree<pcl::PointXYZRGB>);
290  if (pointcloud_->points.size() > 0)
291  norm_tree->setInputCloud(pointcloud_);
292 
293  pcl::PointCloud<pcl::Normal>::Ptr normals(new pcl::PointCloud<pcl::Normal>);
294  pcl::NormalEstimationOMP<pcl::PointXYZRGB, pcl::Normal> normal_estimation;
295  normal_estimation.setNumberOfThreads(in_ompnum_threads);
296  normal_estimation.setInputCloud(pointcloud_);
297  normal_estimation.setSearchMethod(norm_tree);
298  normal_estimation.setViewPoint(std::numeric_limits<float>::max(), std::numeric_limits<float>::max(),
299  std::numeric_limits<float>::max());
300  normal_estimation.setRadiusSearch(in_normal_search_radius);
301  normal_estimation.compute(*normals);
302 
303  pcl::PointCloud<pcl::FPFHSignature33>::Ptr fpfh_histograms(new pcl::PointCloud<pcl::FPFHSignature33>());
304 
305  pcl::FPFHEstimationOMP<pcl::PointXYZRGB, pcl::Normal, pcl::FPFHSignature33> fpfh;
306  fpfh.setNumberOfThreads(in_ompnum_threads);
307  fpfh.setInputCloud(pointcloud_);
308  fpfh.setInputNormals(normals);
309  fpfh.setSearchMethod(norm_tree);
310  fpfh.setRadiusSearch(in_fpfh_search_radius);
311  fpfh.compute(*fpfh_histograms);
312 
313  float fpfh_max = std::numeric_limits<float>::min();
314  float fpfh_min = std::numeric_limits<float>::max();
315 
316  for (unsigned int i = 0; i < fpfh_histograms->size(); i++) // for each point fpfh
317  {
318  for (unsigned int j = 0; j < cluster_fpfh_histogram.size();
319  j++) // sum each histogram's bin for all points, get min/max
320  {
321  cluster_fpfh_histogram[j] = cluster_fpfh_histogram[j] + fpfh_histograms->points[i].histogram[j];
322  if (cluster_fpfh_histogram[j] < fpfh_min)
323  fpfh_min = cluster_fpfh_histogram[j];
324  if (cluster_fpfh_histogram[j] > fpfh_max)
325  fpfh_max = cluster_fpfh_histogram[j];
326  }
327 
328  float fpfh_dif = fpfh_max - fpfh_min;
329  for (unsigned int j = 0; fpfh_dif > 0 && j < cluster_fpfh_histogram.size();
330  j++) // substract the min from each and normalize
331  {
332  cluster_fpfh_histogram[j] = (cluster_fpfh_histogram[j] - fpfh_min) / fpfh_dif;
333  }
334  }
335 
336  return cluster_fpfh_histogram;
337 }
