92 MqttSender::MqttSender() :
93  node_handle_("~")
94 {
95  // ROS Subscriber
96  Subs["can_info"] = node_handle_.subscribe("/can_info", 100, &MqttSender::canInfoCallback, this);
97  Subs["target_velocity_array"] = node_handle_.subscribe("/target_velocity_array", 1, &MqttSender::targetVelocityArrayCallback, this);
98  Subs["twist_cmd"] = node_handle_.subscribe("/twist_cmd", 1, &MqttSender::twistCmdCallback, this);
99  Subs["state"] = node_handle_.subscribe("/state", 1, &MqttSender::stateCallback, this);
100  Subs["drive_mode"] = node_handle_.subscribe("/mode_info", 1, &MqttSender::modeInfoCallback, this);
101  Subs["current_pose"] = node_handle_.subscribe("/current_pose", 1, &MqttSender::currentPoseCallback, this);
102 
103  // MQTT PARAMS
104  mosquitto_lib_init();
105 
106  // Load Config
107  MqttSender::load_config();
108 
109  mqtt_topic_can_info_ = "vehicle/" + to_string(vehicle_id) + "/can_info";
110  mqtt_topic_state_ = "vehicle/" + to_string(vehicle_id) + "/state";
111  mqtt_topic_target_velocity_ = "vehicle/" + to_string(vehicle_id) + "/target_velocity";
112  mqtt_topic_current_target_velocity_ = "vehicle/" + to_string(vehicle_id) + "/current_velocity";
113  mqtt_topic_current_pose_ = "vehicle/" + to_string(vehicle_id) + "/current_pose";
114  mqtt_topic_drive_mode_ = "vehicle/" + to_string(vehicle_id) + "/drive_mode";
115 
116  mqtt_client = mosquitto_new(mqtt_client_id.c_str(), true, NULL);
117  mosquitto_connect_callback_set(mqtt_client, &MqttSender::on_connect);
118  mosquitto_disconnect_callback_set(mqtt_client, &MqttSender::on_disconnect);
119 
120  if(mosquitto_connect_bind(mqtt_client, mqtt_address.c_str(), mqtt_port, mqtt_timeout, NULL)){
121  ROS_INFO("Failed to connect broker.\n");
122  mosquitto_lib_cleanup();
123  exit(EXIT_FAILURE);
124  }
125 }

143 static void MqttSender::load_config()
144 {
145  string config_file_path = ros::package::getPath(MQTT_NODE_NAME) + MQTT_CONFIG_FILE_NAME;
146  ROS_INFO("Load Config File: %s %s(%d)\n", config_file_path.c_str(), __FUNCTION__, __LINE__);
147 
148  YAML::Node config = YAML::LoadFile(config_file_path);
149  vehicle_id = config["mqtt"]["VEHICLEID"].as<int>();
150  mqtt_client_id = "vehicle_" + to_string(vehicle_id) + "_snd";
151  mqtt_address = config["mqtt"]["ADDRESS"].as<string>();
152  mqtt_port = config["mqtt"]["PORT"].as<int>();
153  mqtt_qos = config["mqtt"]["QOS"].as<int>();
154  mqtt_timeout = config["mqtt"]["TIMEOUT"].as<int>();
155 
156  caninfo_downsample = config["mqtt"]["CANINFO_DOWNSAMPLE"].as<float>();
157  gear_d = config["mqtt"]["GEAR_D"].as<int>();
158  gear_n = config["mqtt"]["GEAR_N"].as<int>();
159  unordered_map<string, ros::Subscriber> Subs;
160  gear_r = config["mqtt"]["GEAR_R"].as<int>();
161  gear_p = config["mqtt"]["GEAR_P"].as<int>();
162 
163  ROS_INFO("MQTT Sender ADDR: %s:%d, ID: %s\n", mqtt_address.c_str(), mqtt_port, mqtt_client_id.c_str());
164 
165  ROS_INFO(
166  "[MQTT Receiver Settings] vehicle_id: %d, \
167  caninfo_downsample: %f, gear_d: %d, gear_n: %d, gear_r: %d, \
168  gear_p: %d", vehicle_id, caninfo_downsample, gear_d, gear_n, gear_r, gear_p);
169 }
