77 void PurePursuitNode::run()
78 {
79  ROS_INFO_STREAM("pure pursuit start");
80  ros::Rate loop_rate(LOOP_RATE_);
81  while (ros::ok())
82  {
83  ros::spinOnce();
84  if (!is_pose_set_ || !is_waypoint_set_ || !is_velocity_set_ || !is_config_set_)
85  {
86  ROS_WARN("Necessary topics are not subscribed yet ... ");
87  loop_rate.sleep();
88  continue;
89  }
90 
95  pp_.setLookaheadDistance(computeLookaheadDistance());
96  pp_.setMinimumLookaheadDistance(minimum_lookahead_distance_);
97 
101  double kappa = 0;
102  bool can_get_curvature = pp_.canGetCurvature(&kappa);
103 
104  publishTwistStamped(can_get_curvature, kappa);
105  publishControlCommandStamped(can_get_curvature, kappa);
106  node_status_publisher_ptr_->NODE_ACTIVATE();
107  node_status_publisher_ptr_->CHECK_RATE("/topic/rate/vehicle/slow",8,5,1,"topic vehicle_cmd publish rate low.");
108  // for visualization with Rviz
109  pub11_.publish(displayNextWaypoint(pp_.getPoseOfNextWaypoint()));
110  pub13_.publish(displaySearchRadius(pp_.getCurrentPose().position, pp_.getLookaheadDistance()));
111  pub12_.publish(displayNextTarget(pp_.getPoseOfNextTarget()));
112  pub15_.publish(displayTrajectoryCircle(
113  waypoint_follower::generateTrajectoryCircle(pp_.getPoseOfNextTarget(), pp_.getCurrentPose())));
114  std_msgs::Float32 angular_gravity_msg;
115  angular_gravity_msg.data = computeAngularGravity(computeCommandVelocity(), kappa);
116  pub16_.publish(angular_gravity_msg);
117 
118  publishDeviationCurrentPosition(pp_.getCurrentPose().position, pp_.getCurrentWaypoints());
119 
120  is_pose_set_ = false;
121  is_velocity_set_ = false;
122  is_waypoint_set_ = false;
123 
124  loop_rate.sleep();
125  }
126 }

160 double PurePursuitNode::computeLookaheadDistance() const
161 {
162  if (param_flag_ == enumToInteger(Mode::dialog))
163  return const_lookahead_distance_;
164 
165  double maximum_lookahead_distance = current_linear_velocity_ * 10;
166  double ld = current_linear_velocity_ * lookahead_distance_ratio_;
167 
168  return ld < minimum_lookahead_distance_ ? minimum_lookahead_distance_ :
169  ld > maximum_lookahead_distance ? maximum_lookahead_distance : ld;
170 }

180 double PurePursuitNode::computeCommandAccel() const
181 {
182  const geometry_msgs::Pose current_pose = pp_.getCurrentPose();
183  const geometry_msgs::Pose target_pose = pp_.getCurrentWaypoints().at(1).pose.pose;
184 
185  // v^2 - v0^2 = 2ax
186  const double x =
187  std::hypot(current_pose.position.x - target_pose.position.x, current_pose.position.y - target_pose.position.y);
188  const double v0 = current_linear_velocity_;
189  const double v = computeCommandVelocity();
190  const double a = (v * v - v0 * v0) / (2 * x);
191  return a;
192 }

194 double PurePursuitNode::computeAngularGravity(double velocity, double kappa) const
195 {
196  const double gravity = 9.80665;
197  return (velocity * velocity) / (1.0 / kappa * gravity);
