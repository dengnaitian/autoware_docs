169 static int getConnect(int port, int *sock, int *asock)
170 {
171  struct sockaddr_in addr;
172  struct sockaddr_in client;
173  socklen_t len;
174  int yes = 1;
175 
176  assert(sock != NULL);
177  assert(asock != NULL);
178 
179  if (*sock < 0) {
180  *sock = socket(AF_INET, SOCK_STREAM, 0);
181  if (*sock == -1) {
182  perror("socket");
183  return -1;
184  }
185 
186  std::memset(&addr, 0, sizeof(sockaddr_in));
187  addr.sin_family = AF_INET;
188  addr.sin_port = htons(port);
189  addr.sin_addr.s_addr = htonl(INADDR_ANY);
190  //make it available immediately to connect
191  setsockopt(*sock, SOL_SOCKET, SO_REUSEADDR,
192  (const char *)&yes, sizeof(yes));
193  if (bind(*sock, (struct sockaddr *)&addr, sizeof(addr)) == -1) {
194  perror("bind");
195  return -1;
196  }
197  if (listen(*sock, 5) == -1) {
198  perror("listen");
199  return -1;
200  }
201  }
202 
203  len = sizeof(client);
204  *asock = accept(*sock, (struct sockaddr *)&client, &len);
205  if(*asock == -1) {
206  perror("accept");
207  return -1;
208  }
209 
210  return 0;
211 }
