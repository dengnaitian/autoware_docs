276 void PointPillars::generateAnchors(float* anchors_px_, float* anchors_py_, float* anchors_pz_, float* anchors_dx_,
277  float* anchors_dy_, float* anchors_dz_, float* anchors_ro_)
278 {
279  // zero clear
280  for (size_t i = 0; i < NUM_ANCHOR_; i++)
281  {
282  anchors_px_[i] = 0;
283  anchors_py_[i] = 0;
284  anchors_pz_[i] = 0;
285  anchors_dx_[i] = 0;
286  anchors_dy_[i] = 0;
287  anchors_dz_[i] = 0;
288  anchors_ro_[i] = 0;
289  box_anchors_min_x_[i] = 0;
290  box_anchors_min_y_[i] = 0;
291  box_anchors_max_x_[i] = 0;
292  box_anchors_max_y_[i] = 0;
293  }
294 
295  float x_stride = PILLAR_X_SIZE_ * 2.0f;
296  float y_stride = PILLAR_Y_SIZE_ * 2.0f;
297  float x_offset = MIN_X_RANGE_ + PILLAR_X_SIZE_;
298  float y_offset = MIN_Y_RANGE_ + PILLAR_Y_SIZE_;
299 
300  float anchor_x_count[NUM_ANCHOR_X_INDS_] = { 0 };
301  for (size_t i = 0; i < NUM_ANCHOR_X_INDS_; i++)
302  {
303  anchor_x_count[i] = float(i) * x_stride + x_offset;
304  }
305  float anchor_y_count[NUM_ANCHOR_Y_INDS_] = { 0 };
306  for (size_t i = 0; i < NUM_ANCHOR_Y_INDS_; i++)
307  {
308  anchor_y_count[i] = float(i) * y_stride + y_offset;
309  }
310 
311  float anchor_r_count[NUM_ANCHOR_R_INDS_] = { 0, M_PI / 2 };
312 
313  // np.meshgrid
314  for (size_t y = 0; y < NUM_ANCHOR_Y_INDS_; y++)
315  {
316  for (size_t x = 0; x < NUM_ANCHOR_X_INDS_; x++)
317  {
318  for (size_t r = 0; r < NUM_ANCHOR_R_INDS_; r++)
319  {
320  int ind = y * NUM_ANCHOR_X_INDS_ * NUM_ANCHOR_R_INDS_ + x * NUM_ANCHOR_R_INDS_ + r;
321  anchors_px_[ind] = anchor_x_count[x];
322  anchors_py_[ind] = anchor_y_count[y];
323  anchors_ro_[ind] = anchor_r_count[r];
324  anchors_pz_[ind] = -1 * SENSOR_HEIGHT_;
325  anchors_dx_[ind] = ANCHOR_DX_SIZE_;
326  anchors_dy_[ind] = ANCHOR_DY_SIZE_;
327  anchors_dz_[ind] = ANCHOR_DZ_SIZE_;
328  }
329  }
330  }

351 void PointPillars::convertAnchors2BoxAnchors(float* anchors_px, float* anchors_py, float* anchors_dx, float* anchors_dy,
352  float* box_anchors_min_x_, float* box_anchors_min_y_,
353  float* box_anchors_max_x_, float* box_anchors_max_y_)
354 {
355  // flipping box's dimension
356  float flipped_anchors_dx[NUM_ANCHOR_] = { 0 };
357  float flipped_anchors_dy[NUM_ANCHOR_] = { 0 };
358  for (size_t x = 0; x < NUM_ANCHOR_X_INDS_; x++)
359  {
360  for (size_t y = 0; y < NUM_ANCHOR_Y_INDS_; y++)
361  {
362  int base_ind = x * NUM_ANCHOR_Y_INDS_ * NUM_ANCHOR_R_INDS_ + y * NUM_ANCHOR_R_INDS_;
363  flipped_anchors_dx[base_ind + 0] = ANCHOR_DX_SIZE_;
364  flipped_anchors_dy[base_ind + 0] = ANCHOR_DY_SIZE_;
365  flipped_anchors_dx[base_ind + 1] = ANCHOR_DY_SIZE_;
366  flipped_anchors_dy[base_ind + 1] = ANCHOR_DX_SIZE_;
367  }
368  }
369  for (size_t x = 0; x < NUM_ANCHOR_X_INDS_; x++)
370  {
371  for (size_t y = 0; y < NUM_ANCHOR_Y_INDS_; y++)
372  {
373  for (size_t r = 0; r < NUM_ANCHOR_R_INDS_; r++)
374  {
375  int ind = x * NUM_ANCHOR_Y_INDS_ * NUM_ANCHOR_R_INDS_ + y * NUM_ANCHOR_R_INDS_ + r;
376  box_anchors_min_x_[ind] = anchors_px[ind] - flipped_anchors_dx[ind] / 2.0f;
377  box_anchors_min_y_[ind] = anchors_py[ind] - flipped_anchors_dy[ind] / 2.0f;
378  box_anchors_max_x_[ind] = anchors_px[ind] + flipped_anchors_dx[ind] / 2.0f;
379  box_anchors_max_y_[ind] = anchors_py[ind] + flipped_anchors_dy[ind] / 2.0f;
380  }
381  }
382  }

386 {
387  // create a TensorRT model from the onnx model and serialize it to a stream
388  nvinfer1::IHostMemory* pfe_trt_model_stream{ nullptr };
