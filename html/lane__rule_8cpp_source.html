77 autoware_msgs::Lane apply_acceleration(const autoware_msgs::Lane& lane, double acceleration, size_t start_index,
78  size_t fixed_cnt, double fixed_vel)
79 {
80  autoware_msgs::Lane l = lane;
81 
82  if (fixed_cnt == 0)
83  return l;
84 
85  double square_vel = fixed_vel * fixed_vel;
86  double distance = 0;
87  for (size_t i = start_index; i < l.waypoints.size(); ++i)
88  {
89  if (i - start_index < fixed_cnt)
90  {
91  l.waypoints[i].twist.twist.linear.x = fixed_vel;
92  continue;
93  }
94 
95  geometry_msgs::Point a = l.waypoints[i - 1].pose.pose.position;
96  geometry_msgs::Point b = l.waypoints[i].pose.pose.position;
97  distance += hypot(b.x - a.x, b.y - a.y);
98 
99  double v = sqrt(square_vel + 2 * acceleration * distance);
100  if (v < l.waypoints[i].twist.twist.linear.x)
101  l.waypoints[i].twist.twist.linear.x = v;
102  else
103  break;
104  }
105 
106  return l;
107 }

109 autoware_msgs::Lane apply_crossroad_acceleration(const autoware_msgs::Lane& lane, double acceleration)
110 {
111  autoware_msgs::Lane l = lane;
112 
113  bool crossroad = false;
114  std::vector<size_t> start_indexes;
115  std::vector<size_t> end_indexes;
116  for (size_t i = 0; i < l.waypoints.size(); ++i)
117  {
118  vector_map::DTLane dtlane = lane_planner::vmap::create_vector_map_dtlane(l.waypoints[i].dtlane);
119  if (i == 0)
120  {
121  crossroad = lane_planner::vmap::is_crossroad_dtlane(dtlane);
122  continue;
123  }
124  if (crossroad)
125  {
126  if (!lane_planner::vmap::is_crossroad_dtlane(dtlane))
127  {
128  end_indexes.push_back(i - 1);
129  crossroad = false;
130  }
131  }
132  else
133  {
134  if (lane_planner::vmap::is_crossroad_dtlane(dtlane))
135  {
136  start_indexes.push_back(i);
137  crossroad = true;
138  }
139  }
140  }
141  if (start_indexes.empty() && end_indexes.empty())
142  return l;
143 
144  for (const size_t i : end_indexes)
145  l = apply_acceleration(l, acceleration, i, 1, l.waypoints[i].twist.twist.linear.x);
146 
147  std::reverse(l.waypoints.begin(), l.waypoints.end());
148 
149  std::vector<size_t> reverse_start_indexes;
150  for (const size_t i : start_indexes)
151  reverse_start_indexes.push_back(l.waypoints.size() - i - 1);
152  std::reverse(reverse_start_indexes.begin(), reverse_start_indexes.end());
153 
154  for (const size_t i : reverse_start_indexes)
155  l = apply_acceleration(l, acceleration, i, 1, l.waypoints[i].twist.twist.linear.x);
156 
157  std::reverse(l.waypoints.begin(), l.waypoints.end());
158 
159  return l;
160 }

195 std::vector<vector_map::Point> create_stop_points(const lane_planner::vmap::VectorMap& vmap)
196 {
197  std::vector<vector_map::Point> stop_points;
198  for (const vector_map::StopLine& s : vmap.stoplines)
199  {
200  for (const vector_map::Lane& l : vmap.lanes)
201  {
202  if (l.lnid != s.linkid)
203  continue;
204  for (const vector_map::Node& n : vmap.nodes)
205  {
206  if (n.nid != l.bnid)
207  continue;
208  for (const vector_map::Point& p : vmap.points)
209  {
210  if (p.pid != n.pid)
211  continue;
212  bool hit = false;
213  for (const vector_map::Point& sp : stop_points)
214  {
215  if (sp.pid == p.pid)
216  {
217  hit = true;
218  break;
219  }
220  }
221  if (!hit)
222  stop_points.push_back(p);
223  }
224  }
225  }
226  }
227 
228  return stop_points;
229 }

231 std::vector<size_t> create_stop_indexes(const lane_planner::vmap::VectorMap& vmap, const autoware_msgs::Lane& lane,
232  double stopline_search_radius)
233 {
234  std::vector<size_t> stop_indexes;
235  for (const vector_map::Point& p : create_stop_points(vmap))
236  {
237  size_t index = SIZE_MAX;
238  double distance = DBL_MAX;
239  for (size_t i = 0; i < lane.waypoints.size(); ++i)
240  {
241  vector_map::Point point = lane_planner::vmap::create_vector_map_point(lane.waypoints[i].pose.pose.position);
242  double d = hypot(p.bx - point.bx, p.ly - point.ly);
243  if (d <= distance)
244  {
245  index = i;
246  distance = d;
247  }
248  }
249  if (index != SIZE_MAX && distance <= stopline_search_radius)
250  {
251  stop_indexes.push_back(index);
252  }
253  }
254  std::sort(stop_indexes.begin(), stop_indexes.end());
255 
256  return stop_indexes;
257 }

286 bool is_fine_vmap(const lane_planner::vmap::VectorMap& fine_vmap, const autoware_msgs::Lane& lane)
287 {
288  if (fine_vmap.points.size() != lane.waypoints.size())
289  return false;

292  {
293  vector_map::Point point = lane_planner::vmap::create_vector_map_point(lane.waypoints[i].pose.pose.position);
294  double distance = hypot(fine_vmap.points[i].bx - point.bx, fine_vmap.points[i].ly - point.ly);

296  return false;
297  }
298

302 double create_reduction(const lane_planner::vmap::VectorMap& fine_vmap, int index)
303 {
304  const vector_map::DTLane& dtlane = fine_vmap.dtlanes[index];
305 
306  if (lane_planner::vmap::is_straight_dtlane(dtlane))
307  return 1;
308 
309  if (lane_planner::vmap::is_curve_dtlane(dtlane))
310  {
311  if (lane_planner::vmap::is_crossroad_dtlane(dtlane))
312  return lane_planner::vmap::compute_reduction(dtlane, crossroad_radius_min * crossroad_weight);
313 
314  if (lane_planner::vmap::is_connection_dtlane(fine_vmap, index))
315  return 1;
316 
317  return lane_planner::vmap::compute_reduction(dtlane, curve_radius_min * curve_weight);
318  }
319 
320  if (lane_planner::vmap::is_clothoid_dtlane(dtlane))
321  return lane_planner::vmap::compute_reduction(dtlane, clothoid_radius_min * clothoid_weight);
322 
323  return 1;
324 }

378 void create_waypoint(const autoware_msgs::LaneArray& msg)
379 {
380  std_msgs::Header header;
381  header.stamp = ros::Time::now();
382  header.frame_id = frame_id;
383 
384  cached_waypoint.lanes.clear();
385  cached_waypoint.lanes.shrink_to_fit();
386  cached_waypoint.id = msg.id;
387  for (const autoware_msgs::Lane& l : msg.lanes)
388  cached_waypoint.lanes.push_back(create_new_lane(l, header));
389  if (all_vmap.points.empty() || all_vmap.lanes.empty() || all_vmap.nodes.empty() || all_vmap.stoplines.empty() ||
390  all_vmap.dtlanes.empty())
391  {
392  traffic_pub.publish(cached_waypoint);
393  return;
394  }
395 
396 #ifdef DEBUG
397  marker_cnt = msg.lanes.size();
398 #endif // DEBUG
399 
400  autoware_msgs::LaneArray traffic_waypoint;
401  autoware_msgs::LaneArray red_waypoint;
402  autoware_msgs::LaneArray green_waypoint;
403  traffic_waypoint.id = red_waypoint.id = green_waypoint.id = msg.id;
404  for (size_t i = 0; i < msg.lanes.size(); ++i)
405  {
406  autoware_msgs::Lane lane = create_new_lane(msg.lanes[i], header);
407 
408  lane_planner::vmap::VectorMap coarse_vmap = lane_planner::vmap::create_coarse_vmap_from_lane(lane);
409  if (coarse_vmap.points.size() < 2)
410  {
411  traffic_waypoint.lanes.push_back(lane);
412  continue;
413  }
414 
415  lane_planner::vmap::VectorMap fine_vmap = lane_planner::vmap::create_fine_vmap(
416  lane_vmap, lane_planner::vmap::LNO_ALL, coarse_vmap, search_radius, waypoint_max);
417  if (fine_vmap.points.size() < 2 || !is_fine_vmap(fine_vmap, lane))
418  {
419  traffic_waypoint.lanes.push_back(lane);
420  green_waypoint.lanes.push_back(lane);
421  lane = apply_stopline_acceleration(lane, config_acceleration, config_stopline_search_radius,
422  config_number_of_zeros_ahead, config_number_of_zeros_behind);
423  red_waypoint.lanes.push_back(lane);
424  continue;
425  }
426 
427  for (size_t j = 0; j < lane.waypoints.size(); ++j)
428  {
429  lane.waypoints[j].twist.twist.linear.x *= create_reduction(fine_vmap, j);
430  if (fine_vmap.dtlanes[j].did >= 0)
431  {
432  lane.waypoints[j].dtlane = lane_planner::vmap::create_waypoint_follower_dtlane(fine_vmap.dtlanes[j]);
433  }
434  }
435 
436  /* velocity smoothing */
437  for (int k = 0; k < config_number_of_smoothing_count; ++k)
438  {
439  autoware_msgs::Lane temp_lane = lane;
440  if (lane.waypoints.size() >= 3)
441  {
442  for (size_t j = 1; j < lane.waypoints.size() - 1; ++j)
443  {
444  if (lane.waypoints.at(j).twist.twist.linear.x != 0)
445  {
446  lane.waypoints[j].twist.twist.linear.x =
447  (temp_lane.waypoints.at(j - 1).twist.twist.linear.x + temp_lane.waypoints.at(j).twist.twist.linear.x +
448  temp_lane.waypoints.at(j + 1).twist.twist.linear.x) /
449  3;
450  }
451  }
452  }
453  }
454 
455  lane = apply_crossroad_acceleration(lane, config_acceleration);
456 
457  traffic_waypoint.lanes.push_back(lane);
458  green_waypoint.lanes.push_back(lane);
459 
460  lane = apply_stopline_acceleration(lane, config_acceleration, fine_vmap, config_number_of_zeros_ahead,
461  config_number_of_zeros_behind);
462 
463  red_waypoint.lanes.push_back(lane);
464 
465 #ifdef DEBUG
466  std::stringstream ss;
467  ss << "_" << i;
468 
469  visualization_msgs::Marker m = debug_marker;
470  m.ns = "lane" + ss.str();
471  m.color = create_color(i);
472 
473  lane_planner::vmap::publish_add_marker(marker_pub, m, fine_vmap.points);
474 #endif // DEBUG
475  }
476 
477  traffic_pub.publish(traffic_waypoint);
478  red_pub.publish(red_waypoint);
479  green_pub.publish(green_waypoint);
480 }

482 void update_values()
483 {
484  if (all_vmap.points.empty() || all_vmap.lanes.empty() || all_vmap.nodes.empty() || all_vmap.stoplines.empty() ||
485  all_vmap.dtlanes.empty())
486  return;
487 
488  lane_vmap = lane_planner::vmap::create_lane_vmap(all_vmap, lane_planner::vmap::LNO_ALL);
489 
490  curve_radius_min = lane_planner::vmap::RADIUS_MAX;
491  crossroad_radius_min = lane_planner::vmap::RADIUS_MAX;
492  clothoid_radius_min = lane_planner::vmap::RADIUS_MAX;
493  for (const vector_map::DTLane& d : lane_vmap.dtlanes)
494  {
495  double radius_min = fabs(d.r);
496  if (lane_planner::vmap::is_curve_dtlane(d))
497  {
498  if (lane_planner::vmap::is_crossroad_dtlane(d))
499  {
500  if (radius_min < crossroad_radius_min)
501  crossroad_radius_min = radius_min;
502  }
503  else
504  {
505  if (radius_min < curve_radius_min)
506  curve_radius_min = radius_min;
507  }
508  }
509  else if (lane_planner::vmap::is_clothoid_dtlane(d))
510  {
511  if (radius_min < clothoid_radius_min)
512  clothoid_radius_min = radius_min;
513  }
514  }
515 
516 #ifdef DEBUG
517  for (int i = 0; i < marker_cnt; ++i)
518  {
519  std::stringstream ss;
520  ss << "_" << i;
521 
522  visualization_msgs::Marker m = debug_marker;
523  m.ns = "lane" + ss.str();
524 
525  lane_planner::vmap::publish_delete_marker(marker_pub, m);
526  }
527  marker_cnt = 0;
528 #endif // DEBUG
529 
530  if (!cached_waypoint.lanes.empty())
531  {
532  autoware_msgs::LaneArray update_waypoint = cached_waypoint;
533  create_waypoint(update_waypoint);
534  }
535 }
