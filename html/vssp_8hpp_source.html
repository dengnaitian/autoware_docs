227  template<class DATA_TYPE>
228  void rangeToXYZ(const vssp::range_header &range_header,
229  const vssp::range_index &range_index,
230  boost::shared_array<uint16_t> &index,
231  boost::shared_array<vssp::xyzi> &points)
232  {
233  int i = 0;
234 
235  double h_head = range_header.line_head_h_angle_ratio * 2.0 * M_PI / 65535.0;
236  double h_tail = range_header.line_tail_h_angle_ratio * 2.0 * M_PI / 65535.0;
237  const DATA_TYPE *data =
238  boost::asio::buffer_cast
239  <const DATA_TYPE*>(buf.data());
240  for(int s = 0; s < range_index.nspots; s ++)
241  {
242  double spot = s + range_header.spot;
243  double h_rad = h_head + (h_tail - h_head) * tblH[spot];
244  double h_cos = cos(h_rad);
245  double h_sin = sin(h_rad);
246  vssp::xyzi dir(tblV[spot].s, tblV[spot].c, h_sin, h_cos);
247  for(int e = index[s]; e < index[s+1]; e++)
248  points[i++] = dir * data[e];
249  }
250  };
251  void on_read(const boost::system::error_code& error)
252  {
253  auto timeRead = std::chrono::system_clock::now();
254  auto time = timeReadLast;
255  auto duration = timeRead - timeReadLast;
256  auto lengthTotal = buf.size();
257  if(error == boost::asio::error::eof)
258  {
259  // Connection closed
260  closed = true;
261  return;
262  }
263  else if(error)
264  {
265  // Connection error
266  closed = true;
267  return;
268  }
269  while(true)
270  {
271  if(buf.size() < sizeof(vssp::header))
272  {
273  break;
274  }
275  // Read packet header
276  const vssp::header header =
277  *boost::asio::buffer_cast<const vssp::header*>(buf.data());
278  if(header.mark != vssp::VSSP_MARK)
279  {
280  // Invalid packet
281  // find VSSP mark
282  const uint8_t *data =
283  boost::asio::buffer_cast<const uint8_t*>(buf.data());
284  for(size_t i = 1; i < buf.size() - sizeof(uint32_t); i ++)
285  {
286  const uint32_t *mark =
287  reinterpret_cast<const uint32_t*>(data + i);
288  if(*mark == vssp::VSSP_MARK)
289  {
290  buf.consume(i);
291  break;
292  }
293  }
294  break;
295  }
296  if(buf.size() < header.length) break;
297  auto delay = std::chrono::duration_cast
298  <std::chrono::microseconds>
299  (std::chrono::system_clock::now() - time);
300  time += duration * header.length / lengthTotal;
301 
302  size_t length = header.length - header.header_length;
303  buf.consume(header.header_length);
304 
305  do
306  {
307  if(header.status != vssp::STATUS_OK) break;
308 
309  switch(header.type)
310  {
311  case TYPE_GET:
312  // Response to get command
313  {
314  const std::string data(boost::asio::buffer_cast<const char*>(buf.data()));
315  std::vector<std::string> lines;
316  boost::algorithm::split(lines, data, boost::algorithm::is_any_of("\n\r"));
317  if(lines.size() == 0) break;
318 
319  if(lines[0].compare(0, 7, "GET:tbl") == 0)
320  {
321  if(lines.size() < 2) break;
322  std::vector<std::string> cells;
323  boost::algorithm::split(cells, lines[1], boost::algorithm::is_any_of(","));
324  int i = 0;
325 
326  if(lines[0].compare("GET:tblv") == 0)
327  {
328  tblV.reset(new table_sincos[cells.size()]);
329  for(auto &cell: cells)
330  {
331  double rad = (double)std::strtol(cell.c_str(), NULL, 16) * 2.0 * M_PI / 65535;
332  sincos(rad, &tblV[i].s, &tblV[i].c);
333  i ++;
334  }
335  tblV_loaded = true;
336  }
337  else if(lines[0].compare("GET:tblh") == 0)
338  {
339  tblH.reset(new double[cells.size()]);
340  for(auto &cell: cells)
341  {
342  tblH[i] = (double)std::strtol(cell.c_str(), NULL, 16) / 65535;
343  i ++;
344  }
345  tblH_loaded = true;
346  }
347  }
348  }
349  break;
350  case TYPE_SET:
351  // Response to set command
352  break;
353  case TYPE_DAT:
354  // Response to data request command
355  break;
356  case TYPE_VER:
357  // Response to version request command
358  break;
359  case TYPE_PNG:
360  // Response to ping command
361  if(!cbPing.empty())
362  cbPing(header, delay);
363  break;
364  case TYPE_RI:
365  case TYPE_RO:
366  // Range data
367  if(!tblH_loaded || !tblV_loaded || cbPoint.empty())
368  {
369  // Something wrong
370  break;
371  }
372  {
373  // Decode range data header
374  const vssp::range_header range_header =
375  *boost::asio::buffer_cast<const vssp::range_header*>(buf.data());
376  buf.consume(range_header.header_length);
377  length -= range_header.header_length;
378 
379  // Decode range index header
380  const vssp::range_index range_index =
381  *boost::asio::buffer_cast<const vssp::range_index*>(buf.data());
382  size_t index_length = range_index.index_length;
383  buf.consume(sizeof(vssp::range_index));
384  index_length -= sizeof(vssp::range_index);
385  length -= sizeof(vssp::range_index);
386 
387  // Decode range index
388  boost::shared_array<uint16_t> index(new uint16_t[range_index.nspots+1]);
389  std::memcpy(index.get(),
390  boost::asio::buffer_cast<const vssp::range_index*>(buf.data()),
391  sizeof(uint16_t) * (range_index.nspots+1));
392  buf.consume(index_length);
393  length -= index_length;
394 
395  // Decode echo size
396  const vssp::data_range_size data_range_size = {index[range_index.nspots]};
397 
398  // Decode range data
399  boost::shared_array<vssp::xyzi> points(new vssp::xyzi[index[range_index.nspots]]);
400  switch(header.type)
401  {
402  case TYPE_RI:
403  // Range and Intensity
404  rangeToXYZ<vssp::data_range_intensity>
405  (range_header, range_index, index, points);
406  break;
407  case TYPE_RO:
408  // Range
409  rangeToXYZ<vssp::data_range_only>
410  (range_header, range_index, index, points);
411  break;
412  }
413  cbPoint(header, range_header, range_index, points, delay, data_range_size);
414  }
415  break;
416  case TYPE_AX:
417  // Aux data
418  {
419  // Decode range data header
420  const vssp::aux_header aux_header =
421  *boost::asio::buffer_cast<const vssp::aux_header*>(buf.data());
422  buf.consume(aux_header.header_length);
423  length -= aux_header.header_length;
424 
425  // Decode aux data
426  boost::shared_array<vssp::aux> auxs(new vssp::aux[aux_header.data_count]);
427  for(int i = 0; i < aux_header.data_count; i ++)
428  {
429  const vssp::aux_data *data =
430  boost::asio::buffer_cast
431  <const vssp::aux_data*>(buf.data());
432  int offset = 0;
433  for(aux_id b = vssp::AX_MASK_LAST;
434  b >= vssp::AX_MASK_FIRST;
435  b = static_cast<aux_id>(b - 1))
436  {
437  if(aux_header.data_bitfield & (1 << static_cast<int>(b)))
438  auxs[i][b] = aux_factor[b] * data[offset ++].val;
439  }
440  buf.consume(sizeof(int32_t) * offset);
441  length -= sizeof(int32_t) * offset;
442  }
443  if(!cbAux.empty()) cbAux(header, aux_header, auxs, delay);
444  }
445  break;
446  }
447  }
448  while(false);
449  buf.consume(length);
450  }
451  receivePackets();
452  timeReadLast = timeRead;
453  return;
454  };
