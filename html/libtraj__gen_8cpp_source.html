79 // ------------INITIALIZE PARAMETERS----------//
80 // Heuristic to compute initial paramters given
81 // INPUT: Initial state, and a goal state
82 // OUTPUT: Estimated control parameters
83 // Use #define USE_HEURISTIC_INIT if using Nagy 2001 method
84 // This is necessary for cubic splines
85 // Use #define FIRST_ORDER if working with first order splines

87 union Spline initParams(union State veh, union State goal)
88 {
89 
90  // Local variables for init and goal:
91  double sx_f = goal.sx;
92  double sy_f = goal.sy;
93  double theta_f = goal.theta;
94  double kappa_0 = veh.kappa;
95  double kappa_f = goal.kappa;
96 
97  // Initialize output
98  union Spline curvature;
99 
100  // Convenience
101  double d_theta = abs(theta_f);
102 
103  // Heuristic for initial guess from Nagy and Kelly, 2001
104  // 'Trajectory Generation for Car-like Robots using
105  // Cubic Curvature Polynomials'
106  #ifdef USE_HEURISTIC_INIT
107  // Order of computation matters, need to compute d before s and so on...
108 
109  // Commented out, because unused for cubic splines
110  //double c = 0.0;
111 
112  // Note we must cast to double to avoid ambiguity
113  double d = sqrt((double)pow(sx_f,2) +(double) pow(sy_f,2));
114 
115  double s = d * ((pow(d_theta,2))/5.0 + 1.0) + (2.0/5.0)*d_theta;
116 
117  // Commented out because unused for cubic splines
118  //double a = (6* (theta_f/(pow(s,2)))) - (2*kappa_0/s) + (4*kappa_f/s);
119 
120  double b = (3/(pow(s,2))) * (kappa_0 + kappa_f) + (6*theta_f/(pow(s,3)));
121 
122  double si=0.00;
123  curvature.kappa_0 = veh.kappa;
124  curvature.kappa_1=(1.00/49.00)*(8.00*b*si - 8.00*b*curvature.s - 26.00*curvature.kappa_0 - curvature.kappa_3);
125  curvature.kappa_2=0.25*(curvature.kappa_3 -2.00*curvature.kappa_0 +5.00*curvature.kappa_1);
126  curvature.kappa_3 = goal.kappa;
127  curvature.s = s;
128 
129  #endif
130 
131  // Alternative heuristic initial guess for first order spline
132  // From Thomas Howard's dissertation page 38
133  #ifdef FIRST_ORDER
134  double d = sqrt((double)pow(sx_f,2) +(double) pow(sy_f,2));
135  double s = d * ((pow(d_theta,2))/5 + 1) + (2/5) *d_theta;
136  union Spline curvature;
137  double si=0.00;
138  curvature.kappa_0 = veh.kappa;
139  curvature.kappa_1 = 0.00;
140  curvature.kappa_2 = 0.00;
141  curvature.kappa_3 = goal.kappa;
142  curvature.s = s;
143  #endif
144 
145  curvature.success=TRUE;
146 
147  // Return the initialized curvature union
148  return curvature;
149

153 // ------------SPEED CONTROL----------//
154 // Function for computing safe (feasible) speed and curvature
155 // INPUT: Next vehicle state
156 // OUTPUT: Updated next vehicle state
157 
158 union State speedControlLogic(union State veh_next)
159 {
160 
161  // Calculate speed (look up from next state vector)
162  double vcmd = abs(veh_next.v);
163  double kappa_next = veh_next.kappa;
164 
165  // Compute safe speed
166  double compare_v=(kappa_next-ascl)/bscl;
167  double vcmd_max = max((double)vscl, compare_v);
168 
169  // Compute safe curvature
170  double compare_kappa = ascl + (bscl * vcmd);
171  double kmax_scl = min((double) kmax, compare_kappa);
172 
173  // Check if max curvature for speed is exceeded
174  if(kappa_next >= kmax_scl)
175  {
176  // Check for safe speed
177  vcmd = sf * vcmd_max;
178  }
179 
180  // Update velocity command, this is not quite equivalent to Ferguson, Howard, & Liukhachev
181  veh_next.v = vcmd;
182 
183  // Return vehicle state
184  return veh_next;
185 }

187 // ------------RESPONSE TO CONTROL----------//
188 // Function for computing the vehicles response to control inputs
189 // Mainly deals with delays associated with controls
190 // Also considers upper and lower safety bounds
191 // INPUT: Current vehicle state, next vehicle state, sampling time
192 // OUTPUT: Next vehicle state
193 
194 union State responseToControlInputs(union State veh, union State veh_next, double dt)
195 {
196 
197  // Local variables:
198  double kappa = veh.kappa;
199  double kappa_next = veh_next.kappa;
200  double v = veh.v;
201  double v_next = veh_next.v;
202 
203  // Compute curvature rate command
204  double kdot = (kappa_next - kappa)/dt;
205 
206  // Check against upper bound on curvature rate
207  kdot = min(kdot, (double) dkmax);
208 
209  // Check against lower bound on curvature rate
210  kdot = max(kdot, (double) dkmin);
211 
212  // Call the speedControlLogic function
213  veh_next = speedControlLogic(veh_next);
214 
215  // Compute curvature at the next vehicle state
216  kappa_next = kappa + kdot*dt;
217 
218  // Check upper bound on curvature
219  kappa_next = min(kappa_next, (double) kmax);
220 
221  // Check lower bound on curvature
222  kappa_next = max(kappa_next, (double) kmin);
223 
224  // Compute acceleration command
225  double vdot = (v_next -v)/dt;
226 
227  // Check for upper bound on acceleration
228  vdot = min(vdot, (double) dvmax);
229 
230  // Check for lower bound on acceleration
231  vdot = max(vdot, (double) dvmin);
232 
233  // Compute velocity at next state
234  veh_next.v = v + vdot*dt;
235 
236  // Return next vehicle state
237  return veh_next;
238 }

244 double getCurvatureCommand(union Spline curvature, double dt, double v, double t)
245 {
246  // Local variables for curvature constants
247  double kappa_0 = curvature.kappa_0;
248  double kappa_1 = curvature.kappa_1;
249  double kappa_2 = curvature.kappa_2;
250  double kappa_3 = curvature.kappa_3;
251  double s = curvature.s;
252 
253  // Init next command
254  double k_next_cmd =0.0;
255 
256  // Assume position at t=0, si=0
257  double si = 0.0;
258 
259  // Estimate distance traveled on spiral
260  double st = v*t;
261 
262  // These equations can be found in McNaughton, page 76
263  // They make the knot points equally spaced.
264  // They apply to stable cubic paths
265  #ifdef CUBIC_STABLE
266  double a =0;
267  double b =0;
268  double c= 0;
269  double d = 0;
270  a = kappa_0;
271  b = (-0.50)*(-2*kappa_3 + 11*kappa_0 - 18*kappa_1 + 9*kappa_2)/(s-si);
272  c = (4.50)*(-kappa_3 + 2*kappa_0 - 5*kappa_1 +4*kappa_2)/(pow((s-si), 2));
273  d = (-4.50)*(-kappa_3 + kappa_0 - 3*kappa_1 + 3*kappa_2)/(pow((s-si), 3));
274  k_next_cmd = a + b*st + c*pow(st,2) + d*pow(st,3);
275  #endif
276 
277  // These equations can be found in McNaughton, page 77
278  // They make the knot points equally spaced.
279  // They apply to stable quintic paths
280  #ifdef QUINTIC_STABLE
281  double a = kappa_0;
282  double b = kappa_1;
283  double c = kappa_2/2.0;
284  double d = -(575*kappa_0 -648*kappa_3 +81*kappa_4 -8*kappa_5 +170*kappa_1*st +22*kappa_2*pow(st,2)/ (8*pow(st,3));
285  double e = 9*(37*kappa_0 -45*kappa_3 +9*kappa_4 -kappa_5 +10*kappa_1*st +kappa_2*pow(st,2))/(2*pow(st,4));
286  double f = -9*(85*kappa_0 -108*kappa_3 +27*kappa_4 -4*kappa_5 +22*kappa_1*st +2*kappa_2*pow(st,2))/(8*pow(st,5));
287  k_next_cmd = a + b*st + c*pow(st,2) + d*pow(st,3) + e*pow(st,4) +f*pow(st,5);
288  #endif
289 
290  // Compute next command from Howard page 39
291  #ifdef FIRST_ORDER
292  k_next_cmd = kappa_0 + ((kappa_1 - kappa_0)/s) * (st);
293  #endif
294 
295  // Return value
296  return k_next_cmd;
297

321 union State motionModel(union State veh, union State goal, union Spline curvature, double dt, double horizon, int flag)
322 {
323  // Initialized the elapsed time to 0.0 s
324  double t =0.0;
325  // Setup local data structures for holding
326  // the previous and next vehicle states
327  union State veh_next;
328  union State veh_temp;
329  // Set veh_temp to the current state
330  veh_temp=veh;
331  // Compute the stop time for the simulation
332  horizon = curvature.s/goal.v;
333 
334  while(t < horizon)
335  {
336  // Read state vector
337  double sx = veh_temp.sx;
338  double sy = veh_temp.sy;
339  double v = veh_temp.v;
340  double theta = veh_temp.theta;
341  double kappa = veh_temp.kappa;
342  double v_goal= goal.v;
343 
344  // Compute change in 2D x-position (this is x_dot)
345  double sx_next = sx + (v * cos(theta) * dt);
346 
347  veh_next.sx = sx_next;
348 
349 
350  // Compute change in 2D y-position (this is y_dot)
351  double sy_next = sy + (v * sin(theta) * dt);
352 
353  veh_next.sy = sy_next;
354 
355  // Compute change in 2D orientation (this is theta_dot)
356  double theta_next = theta + (v * kappa * dt);
357 
358  veh_next.theta = theta_next;
359 
360  // Get curvature command
361  double kappa_next = getCurvatureCommand(curvature, dt , veh.v,t);
362 
363  veh_next.kappa = kappa_next;
364 
365  // Get velocity command
366  double v_next = getVelocityCommand(v_goal, v);
367 
368  veh_next.v = v_next;
369 
370  // Get acceleration command, not used yet...
371  //double a_next_cmd = 0;
372 
373 
374  // Update the next vehicle state
375  veh_next = responseToControlInputs(veh, veh_next, dt);
376 
377  // Increment the timestep
378  t=t+dt;
379 
380  // Update veh_temp
381  veh_temp=veh_next;
382 
383  // Write to log file if we are testing
384  if(flag==1)
385  {
386  fmm_sx <<veh_temp.sx<<", ";
387  fmm_sy <<veh_temp.sy<<", ";
388  fmm_v <<veh_temp.v<<", ";
389  fmm_theta << veh_temp.theta<<", ";
390  fmm_kappa<<veh_temp.kappa<<", ";
391  }
392 
393  }
394  // Print information to console
395  #ifdef DEBUG
396  cout << "Next state sx: " << veh_next.sx << endl;
397  cout << "Next state sy: " << veh_next.sy << endl;
398  cout << "Next state theta: " << veh_next.theta << endl;
399  cout << "Next state v: " << veh_next.v << endl;
400  cout << "Next state kappa: " << veh_next.kappa<< endl;
401  #endif
402 
403 
404  // Return the state at the end of the trajectory
405  return veh_next;
406 }

461 // ------------ESTIMATE JACOBIAN----------//
462 // Forward difference method via predictive motion model
463 // INPUT: Initial State, Target State, Parameterized Action,
464 // Predictive Motion Model, Action Parameters
465 // OUTPUT: Jacobian Estimate
466 
467 union State pDerivEstimate(union State veh, union State veh_next, union State goal, union Spline curvature, int p_id, double h, double dt, double horizon, int stateIndex)
468 {
469  #ifdef DEBUG
470  cout << "Function: pDerivEstimate()"<< endl;
471  #endif
472 
473  // Index for iterating through parameters
474  int i = 0;
475 
476  // Compute difference between desired sx and actual
477  // May need to update with intial state if not 0,0,0,0,0
478  union State delta_state;
479 
480  // Compute the difference between the end state of the current spline and the goal
481  for (i=0; i<stateIndex; i++)
482  {
483  delta_state.state_value[i] = (goal.state_value[i] - veh_next.state_value[i]);
484  }
485 
486  // Create parameter perturbation vector
487  union Spline perturb_curve = curvature;
488 
489  // Perturb parameter
490  perturb_curve.spline_value[p_id] = curvature.spline_value[p_id] +h;
491 
492  // Check forward motion model with perturbed parameter
493  union State perturb_next = motionModel(veh, goal, perturb_curve, dt, horizon, 0);
494 
495  // Now create a new structure to hold the difference between goal and perturbed forward motion model
496  union State delta_perturb_state;
497 
498  // Compute difference between goal sx and actual
499  for (i=0; i<stateIndex; i++)
500  {
501  delta_perturb_state.state_value[i] = goal.state_value[i] - perturb_next.state_value[i];
502  }
503 
504  // Structure to hold the difference between the perturbed error and unperturbed erro
505  union State partial_state_vector;
506 
507  // Estimate partial derivative
508  for (i=0; i<stateIndex; i++)
509  {
510  double num = (delta_perturb_state.state_value[i] - delta_state.state_value[i]);
511  partial_state_vector.state_value[i] = num/h;
512  }
513 
514  return partial_state_vector;
515 
516 }

518 // ------------UPDATE PARAMETERS----------//
519 // Inverts the Jacobian and computes the parameter update
520 
521 union Spline generateCorrection(union State veh, union State veh_next, union State goal, union Spline curvature, double dt, double horizon)
522 {
523  // Compute jacobian with forward gradient
524  int i;
525  int stateIndex=3;
526  int j;
527 
528  // Matrix J will contain the Jacobian
529  arma::mat J(stateIndex,stateIndex);
530 
531  // Initialize it to zero
532  J.fill(0.0);
533 
534  // Parameter perturbation vector
535  arma::vec h(stateIndex);
536 
537  // How much to perturb each parameter
538  // Note h(0) is larger than others because it perturbs length ~20m
539  // Rather than kappa < ~.19 rad/m
540  h(0)= (double) h_global*10;
541  h(1)= (double) h_global;
542  h(2)= (double) h_global;
543 
544  // For each parameter compute the vector associated with a small perturbation of its value
545  for (i=0; i<stateIndex; i++)
546  {
547  union State temp;
548  temp= pDerivEstimate(veh, veh_next, goal, curvature, i, h(i), dt, horizon, stateIndex);
549 
550  // For each element of the state vector place the value in the Jacobian matrix (column-wise)
551  for (j=0; j<stateIndex; j++)
552  {
553  J(j,i) = temp.state_value[j];
554  }
555  }
556 
557  // Put parameters in a vector
558  arma::vec dX(stateIndex);
559  dX.fill(0.0);
560 
561  for(i=0; i<stateIndex; i++)
562  {
563  dX(i) = goal.state_value[i]-veh_next.state_value[i];
564  }
565 
566  // Solve for delta P
567  // Note that we use a try catch routine as it is possible for J to be singular
568  arma::vec dP(stateIndex);
569  try
570  {
571  J=J.i();
572  }
573  catch(const std::exception& e)
574  {
575  curvature.success=FALSE;
576  return curvature;
577  cout<< "goal.sx: "<< goal.sx<<endl;
578  cout<< "goal.sy: "<< goal.sx<<endl;
579  cout<< "goal.theta: "<< goal.sx<<endl;
580  }
581 
582  dP=J*dX;
583 
584  // Update curvature parameters
585  for(i=0; i<stateIndex; i++)
586  {
587  curvature.spline_value[i]=curvature.spline_value[i]-dP(i);
588  }
589 
590  // Return new spline
591  return curvature;
592 }
