39  pnh_.advertise<jsk_rviz_plugins::OverlayText>("fatal_text", 1);
40  node_status_sub_ = nh_.subscribe("/node_status", 10,
41  &HealthAggregator::nodeStatusCallback, this);

70 void HealthAggregator::publishSystemStatus() {
71  ros::Rate rate = ros::Rate(autoware_health_checker::UPDATE_RATE);
72  while (ros::ok()) {
73  mtx_.lock();
74  system_status_.header.stamp = ros::Time::now();
75  updateConnectionStatus();
76  updateTopicStatus();
77  system_status_pub_.publish(system_status_);
78  text_pub_[autoware_health_checker::LEVEL_OK].publish(
79  generateOverlayText(system_status_, autoware_health_checker::LEVEL_OK));
80  text_pub_[autoware_health_checker::LEVEL_WARN].publish(generateOverlayText(
81  system_status_, autoware_health_checker::LEVEL_WARN));
82  text_pub_[autoware_health_checker::LEVEL_ERROR].publish(generateOverlayText(
83  system_status_, autoware_health_checker::LEVEL_ERROR));
84  text_pub_[autoware_health_checker::LEVEL_FATAL].publish(generateOverlayText(
85  system_status_, autoware_health_checker::LEVEL_FATAL));
86  system_status_.topic_statistics.clear();
87  system_status_.node_status.clear();
88  system_status_.hardware_status.clear();
89  mtx_.unlock();
90  rate.sleep();
91  }
92  return;
93 }

176 std::vector<autoware_system_msgs::DiagnosticStatus>
177 HealthAggregator::filterNodeStatus(autoware_system_msgs::SystemStatus status,
178  uint8_t level) {
179  std::vector<autoware_system_msgs::DiagnosticStatus> ret;
180  for (auto node_status_itr = status.node_status.begin();
181  node_status_itr != status.node_status.end(); node_status_itr++) {
182  if (node_status_itr->node_activated) {
183  for (auto array_itr = node_status_itr->status.begin();
184  array_itr != node_status_itr->status.end(); array_itr++) {
185  for (auto itr = array_itr->status.begin();
186  itr != array_itr->status.end(); itr++) {
187  if (itr->level == level) {
188  ret.push_back(*itr);
189  }
190  }
191  }
192  }
193  }
194  return ret;
195 }

197 boost::optional<autoware_system_msgs::HardwareStatus> HealthAggregator::convert(
198  const diagnostic_msgs::DiagnosticArray::ConstPtr msg) {
199  autoware_system_msgs::HardwareStatus status;
200  if (msg->status.size() == 0) {
201  return boost::none;
202  }
203  status.header = msg->header;
204  for (auto diag_itr = msg->status.begin(); diag_itr != msg->status.end();
205  diag_itr++) {
206  status.hardware_name = diag_itr->hardware_id;
207  autoware_system_msgs::DiagnosticStatus diag;
208  autoware_system_msgs::DiagnosticStatusArray diag_array;
209  diag.header = msg->header;
210  diag.key = diag_itr->hardware_id;
211  diag.description = diag_itr->message;
212  diag.type = autoware_system_msgs::DiagnosticStatus::HARDWARE;
213  if (diag_itr->level == diagnostic_msgs::DiagnosticStatus::OK) {
214  diag.level = autoware_health_checker::LEVEL_OK;
215  } else if (diag_itr->level == diagnostic_msgs::DiagnosticStatus::WARN) {
216  diag.level = autoware_health_checker::LEVEL_WARN;
217  } else if (diag_itr->level == diagnostic_msgs::DiagnosticStatus::ERROR) {
218  diag.level = autoware_health_checker::LEVEL_ERROR;
219  } else if (diag_itr->level == diagnostic_msgs::DiagnosticStatus::STALE) {
220  diag.level = autoware_health_checker::LEVEL_FATAL;
221  }
222  using namespace boost::property_tree;
223  std::stringstream ss;
224  ptree pt;
225  for (auto value_itr = diag_itr->values.begin();
226  value_itr != diag_itr->values.end(); value_itr++) {
227  pt.put(value_itr->key + ".string", value_itr->value);
228  }
229  write_json(ss, pt);
230  diag.value = ss.str();
231  diag_array.status.push_back(diag);
232  status.status.push_back(diag_array);
233  }
234  return status;
235 }
