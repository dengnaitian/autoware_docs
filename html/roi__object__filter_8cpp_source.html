25 void
26 RosRoiObjectFilterApp::SyncedDetectionsCallback(const autoware_msgs::DetectedObjectArray::ConstPtr &in_detections,
27  const grid_map_msgs::GridMap::ConstPtr &in_gridmap)
28 {
29  if (nullptr == in_detections
30  || nullptr == in_gridmap)
31  {
32  ROS_INFO("[%s] Empty input messages, for details check the topics : %s and %s.", __APP_NAME__,
33  wayarea_gridmap_topic_.c_str(), objects_src_topic_.c_str());
34  return;
35  }
36 
37  grid_map::GridMap current_grid;
38  grid_map::GridMapRosConverter::fromMessage(*in_gridmap, current_grid);
39  autoware_msgs::DetectedObjectArray in_roi_objects;
40  autoware_msgs::DetectedObjectArray out_roi_objects;
41 
42  in_roi_objects.header = in_detections->header;
43  out_roi_objects.header = in_detections->header;
44 
45  if (current_grid.exists(gridmap_layer_))
46  {
47  // check if centroids are inside the drivable area
48  cv::Mat grid_image;
49  grid_map::GridMapCvConverter::toImage<unsigned char, 1>(current_grid, gridmap_layer_, CV_8UC1,
50  GRID_MIN_VALUE, GRID_MAX_VALUE, grid_image);
51 
52  #pragma omp for
53  for (unsigned int i = 0; i < in_detections->objects.size(); i++)
54  {
55  geometry_msgs::Point original_centroid_point, final_centroid_point;
56  original_centroid_point = in_detections->objects[i].pose.position;
57 
58  if (current_grid.getFrameId() != in_detections->header.frame_id)
59  {
60  tf::StampedTransform grid_sensor_tf = FindTransform(current_grid.getFrameId(), in_detections->header.frame_id);
61  final_centroid_point = TransformPoint(original_centroid_point, grid_sensor_tf);
62  }
63  else
64  {
65  final_centroid_point = original_centroid_point;
66  }
67 
68  bool point_in_grid = CheckPointInGrid(current_grid, grid_image, final_centroid_point);
69 
70  std::vector<std::string>::const_iterator found = std::find(exception_list_.begin(), exception_list_.end(),
71  in_detections->objects[i].label);
72  bool is_exception = (found != exception_list_.end());
73 
74  if (is_exception || point_in_grid)
75  {
76  in_roi_objects.objects.push_back(in_detections->objects[i]);
77  }
78  else
79  {
80  out_roi_objects.objects.push_back(in_detections->objects[i]);
81  }
82  }
83  }
84  else
85  {
86  ROS_INFO("[%s] %s layer not contained in the OccupancyGrid", __APP_NAME__, gridmap_layer_.c_str());
87  }
88 
89  roi_objects_publisher_.publish(in_roi_objects);
90 
91  wayarea_gridmap_ = nullptr;
92  object_detections_ = nullptr;
93 }

150 bool
151 RosRoiObjectFilterApp::CheckPointInGrid(const grid_map::GridMap &in_grid_map,
152  const cv::Mat &in_grid_image,
153  const geometry_msgs::Point &in_point)
154 {
155  // calculate out_grid_map position
156  grid_map::Position map_pos = in_grid_map.getPosition();
157  double origin_x_offset = in_grid_map.getLength().x() / 2.0 - map_pos.x();
158  double origin_y_offset = in_grid_map.getLength().y() / 2.0 - map_pos.y();
159  // coordinate conversion for cv image
160  double cv_x = (in_grid_map.getLength().y() - origin_y_offset - in_point.y) / in_grid_map.getResolution();
161  double cv_y = (in_grid_map.getLength().x() - origin_x_offset - in_point.x) / in_grid_map.getResolution();
162 
163  // check coords are inside the gridmap
164  if (cv_x < 0 || cv_x > in_grid_image.cols || cv_y < 0 || cv_y > in_grid_image.rows)
165  {
166  return false;
167  }
168 
169  //_gridmap_no_road_value if road
170  if (in_grid_image.at<uchar>(cv_y, cv_x) != gridmap_no_road_value_)
171  {
172  return true;
173  }
174 
175  return false;
176 }

180 RosRoiObjectFilterApp::InitializeRosIo(ros::NodeHandle &in_private_handle)
181 {
182  std::string exception_list;
183 
184  ros_namespace_ = ros::this_node::getNamespace();
185 
186  if (ros_namespace_.substr(0, 2) == "//")
187  {
188  ros_namespace_.erase(ros_namespace_.begin());
189  }
190 
191  objects_src_topic_ = ros_namespace_ + "/objects";
192  wayarea_gridmap_topic_= "/grid_map_wayarea";
193  std::string roi_topic_str = ros_namespace_+ "/objects_filtered";
194 
195  ROS_INFO("[%s] objects_src_topic: %s", __APP_NAME__, objects_src_topic_.c_str());
196  ROS_INFO("[%s] wayarea_gridmap_topic: %s", __APP_NAME__, wayarea_gridmap_topic_.c_str());
197 
198  in_private_handle.param<bool>("sync_topics", sync_topics_, false);
199  ROS_INFO("[%s] sync_topics: %d", __APP_NAME__, sync_topics_);
200 
201  in_private_handle.param<std::string>("wayarea_gridmap_layer", gridmap_layer_, "wayarea");
202  ROS_INFO("[%s] wayarea_gridmap_layer: %s", __APP_NAME__, gridmap_layer_.c_str());
203 
204  in_private_handle.param<int>("wayarea_no_road_value", gridmap_no_road_value_, GRID_MAX_VALUE);
205  ROS_INFO("[%s] wayarea_no_road_value: %ds", __APP_NAME__, gridmap_no_road_value_);
206 
207  in_private_handle.param<std::string>("exception_list", exception_list, "[person, bicycle]");
208  ROS_INFO("[%s] exception_list: %s", __APP_NAME__, exception_list.c_str());
209 
210  YAML::Node exception_list_yaml;
211  try
212  {
213  exception_list_yaml = YAML::Load(exception_list);
214  for (const auto &exception : exception_list_yaml)
215  {
216  exception_list_.push_back(exception.as<std::string>());
217  }
218  }
219  catch (const std::exception &e)
220  {
221  ROS_INFO("[%s] Incorrect format for Exception List (exception_list) param, using default.", __APP_NAME__);
222  exception_list_ = {"person", "bycicle"};
223  }
224 
225  if (!sync_topics_)
226  {
227  detections_range_subscriber_ = in_private_handle.subscribe(objects_src_topic_,
228  1,
229  &RosRoiObjectFilterApp::DetectionsCallback, this);
230 
231  wayarea_gridmap_subscriber_ = in_private_handle.subscribe(wayarea_gridmap_topic_,
232  1,
233  &RosRoiObjectFilterApp::WayAreaGridMapCallback,
234  this);
235  }
236  else
237  {
238  detections_filter_subscriber_ = new message_filters::Subscriber<autoware_msgs::DetectedObjectArray>(node_handle_,
239  objects_src_topic_,
240  1);
241  gridmap_filter_subscriber_ = new message_filters::Subscriber<grid_map_msgs::GridMap>(node_handle_,
242  wayarea_gridmap_topic_, 1);
243  detections_synchronizer_ =
244  new message_filters::Synchronizer<SyncPolicyT>(SyncPolicyT(10),
245  *detections_filter_subscriber_,
246  *gridmap_filter_subscriber_);
247  detections_synchronizer_->registerCallback(boost::bind(&RosRoiObjectFilterApp::SyncedDetectionsCallback,
248  this,
249  _1, _2));
250  }
251 
252  roi_objects_publisher_ = node_handle_.advertise<autoware_msgs::DetectedObjectArray>(roi_topic_str, 1);
253  ROS_INFO("[%s] Publishing filtered objects in %s", __APP_NAME__, roi_topic_str.c_str());
254 }
