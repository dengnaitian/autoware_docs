198 void ImmUkfPda::measurementValidation(const autoware_msgs::DetectedObjectArray& input, UKF& target,
199  const bool second_init, const Eigen::VectorXd& max_det_z,
200  const Eigen::MatrixXd& max_det_s,
201  std::vector<autoware_msgs::DetectedObject>& object_vec,
202  std::vector<bool>& matching_vec)
203 {
204  // alert: different from original imm-pda filter, here picking up most likely measurement
205  // if making it allows to have more than one measurement, you will see non semipositive definite covariance
206  bool exists_smallest_nis_object = false;
207  double smallest_nis = std::numeric_limits<double>::max();
208  int smallest_nis_ind = 0;
209  for (size_t i = 0; i < input.objects.size(); i++)
210  {
211  double x = input.objects[i].pose.position.x;
212  double y = input.objects[i].pose.position.y;
213 
214  Eigen::VectorXd meas = Eigen::VectorXd(2);
215  meas << x, y;
216 
217  Eigen::VectorXd diff = meas - max_det_z;
218  double nis = diff.transpose() * max_det_s.inverse() * diff;
219 
220  if (nis < gating_thres_)
221  {
222  if (nis < smallest_nis)
223  {
224  smallest_nis = nis;
225  target.object_ = input.objects[i];
226  smallest_nis_ind = i;
227  exists_smallest_nis_object = true;
228  }
229  }
230  }
231  if (exists_smallest_nis_object)
232  {
233  matching_vec[smallest_nis_ind] = true;
234  if (use_vectormap_ && has_subscribed_vectormap_)
235  {
236  autoware_msgs::DetectedObject direction_updated_object;
237  bool use_direction_meas =
238  updateDirection(smallest_nis, target.object_, direction_updated_object, target);
239  if (use_direction_meas)
240  {
241  object_vec.push_back(direction_updated_object);
242  }
243  else
244  {
245  object_vec.push_back(target.object_);
246  }
247  }
248  else
249  {
250  object_vec.push_back(target.object_);
251  }
252  }
253 }

274 bool ImmUkfPda::storeObjectWithNearestLaneDirection(const autoware_msgs::DetectedObject& in_object,
275  autoware_msgs::DetectedObject& out_object)
276 {
277  geometry_msgs::Pose lane_frame_pose = getTransformedPose(in_object.pose, tracking_frame2lane_frame_);
278  double min_dist = std::numeric_limits<double>::max();
279 
280  double min_yaw = 0;
281  for (auto const& lane : lanes_)
282  {
283  vector_map_msgs::Node node = vmap_.findByKey(vector_map::Key<vector_map_msgs::Node>(lane.bnid));
284  vector_map_msgs::Point point = vmap_.findByKey(vector_map::Key<vector_map_msgs::Point>(node.pid));
285  double distance = std::sqrt(std::pow(point.bx - lane_frame_pose.position.y, 2) +
286  std::pow(point.ly - lane_frame_pose.position.x, 2));
287  if (distance < min_dist)
288  {
289  min_dist = distance;
290  vector_map_msgs::Node front_node = vmap_.findByKey(vector_map::Key<vector_map_msgs::Node>(lane.fnid));
291  vector_map_msgs::Point front_point = vmap_.findByKey(vector_map::Key<vector_map_msgs::Point>(front_node.pid));
292  min_yaw = std::atan2((front_point.bx - point.bx), (front_point.ly - point.ly));
293  }
294  }
295 
296  bool success = false;
297  if (min_dist < nearest_lane_distance_thres_)
298  {
299  success = true;
300  }
301  else
302  {
303  return success;
304  }
305 
306  // map yaw in rotation matrix representation
307  tf::Quaternion map_quat = tf::createQuaternionFromYaw(min_yaw);
308  tf::Matrix3x3 map_matrix(map_quat);
309 
310  // vectormap_frame to tracking_frame rotation matrix
311  tf::Quaternion rotation_quat = lane_frame2tracking_frame_.getRotation();
312  tf::Matrix3x3 rotation_matrix(rotation_quat);
313 
314  // rotated yaw in matrix representation
315  tf::Matrix3x3 rotated_matrix = rotation_matrix * map_matrix;
316  double roll, pitch, yaw;
317  rotated_matrix.getRPY(roll, pitch, yaw);
318 
319  out_object = in_object;
320  out_object.angle = yaw;
321  return success;
322 }

373 void ImmUkfPda::secondInit(UKF& target, const std::vector<autoware_msgs::DetectedObject>& object_vec, double dt)
374 {
375  if (object_vec.size() == 0)
376  {
377  target.tracking_num_ = TrackingState::Die;
378  return;
379  }
380  // record init measurement for env classification
381  target.init_meas_ << target.x_merge_(0), target.x_merge_(1);
382 
383  // state update
384  double target_x = object_vec[0].pose.position.x;
385  double target_y = object_vec[0].pose.position.y;
386  double target_diff_x = target_x - target.x_merge_(0);
387  double target_diff_y = target_y - target.x_merge_(1);
388  double target_yaw = atan2(target_diff_y, target_diff_x);
389  double dist = sqrt(target_diff_x * target_diff_x + target_diff_y * target_diff_y);
390  double target_v = dist / dt;
391 
392  while (target_yaw > M_PI)
393  target_yaw -= 2. * M_PI;
394  while (target_yaw < -M_PI)
395  target_yaw += 2. * M_PI;
396 
397  target.x_merge_(0) = target.x_cv_(0) = target.x_ctrv_(0) = target.x_rm_(0) = target_x;
398  target.x_merge_(1) = target.x_cv_(1) = target.x_ctrv_(1) = target.x_rm_(1) = target_y;
399  target.x_merge_(2) = target.x_cv_(2) = target.x_ctrv_(2) = target.x_rm_(2) = target_v;
400  target.x_merge_(3) = target.x_cv_(3) = target.x_ctrv_(3) = target.x_rm_(3) = target_yaw;
401 
402  target.tracking_num_++;
403  return;
404 }

446 bool ImmUkfPda::probabilisticDataAssociation(const autoware_msgs::DetectedObjectArray& input, const double dt,
447  std::vector<bool>& matching_vec,
448  std::vector<autoware_msgs::DetectedObject>& object_vec, UKF& target)
449 {
450  double det_s = 0;
451  Eigen::VectorXd max_det_z;
452  Eigen::MatrixXd max_det_s;
453  bool success = true;
454 
455  if (use_sukf_)
456  {
457  max_det_z = target.z_pred_ctrv_;
458  max_det_s = target.s_ctrv_;
459  det_s = max_det_s.determinant();
460  }
461  else
462  {
463  // find maxDetS associated with predZ
464  target.findMaxZandS(max_det_z, max_det_s);
465  det_s = max_det_s.determinant();
466  }
467 
468  // prevent ukf not to explode
469  if (std::isnan(det_s) || det_s > prevent_explosion_thres_)
470  {
471  target.tracking_num_ = TrackingState::Die;
472  success = false;
473  return success;
474  }
475 
476  bool is_second_init;
477  if (target.tracking_num_ == TrackingState::Init)
478  {
479  is_second_init = true;
480  }
481  else
482  {
483  is_second_init = false;
484  }
485 
486  // measurement gating
487  measurementValidation(input, target, is_second_init, max_det_z, max_det_s, object_vec, matching_vec);
488 
489  // second detection for a target: update v and yaw
490  if (is_second_init)
491  {
492  secondInit(target, object_vec, dt);
493  success = false;
494  return success;
495  }
496 
497  updateTargetWithAssociatedObject(object_vec, target);
498 
499  if (target.tracking_num_ == TrackingState::Die)
500  {
501  success = false;
502  return success;
503  }
504  return success;
505 }

528 void ImmUkfPda::staticClassification()
529 {
530  for (size_t i = 0; i < targets_.size(); i++)
531  {
532  // targets_[i].x_merge_(2) is referred for estimated velocity
533  double current_velocity = std::abs(targets_[i].x_merge_(2));
534  targets_[i].vel_history_.push_back(current_velocity);
535  if (targets_[i].tracking_num_ == TrackingState::Stable && targets_[i].lifetime_ > life_time_thres_)
536  {
537  int index = 0;
538  double sum_vel = 0;
539  double avg_vel = 0;
540  for (auto rit = targets_[i].vel_history_.rbegin(); index < static_num_history_thres_; ++rit)
541  {
542  index++;
543  sum_vel += *rit;
544  }
545  avg_vel = double(sum_vel / static_num_history_thres_);
546 
547  if(avg_vel < static_velocity_thres_ && current_velocity < static_velocity_thres_)
548  {
549  targets_[i].is_static_ = true;
550  }
551  }
552  }

584 autoware_msgs::DetectedObjectArray

589  return in_detected_objects;

592  resulting_objects.header = in_detected_objects.header;

601  }
602  }
603  //assign objects to the points
604  std::vector<std::vector<size_t>> matching_objects(centroids.size());
605  for(size_t k=0; k<in_detected_objects.objects.size(); k++)
606  {
607  const auto& object=in_detected_objects.objects[k];
608  for(size_t i=0; i< centroids.size(); i++)
609  {
610  if (arePointsClose(object.pose.position, centroids[i], merge_distance_threshold_))
611  {
612  matching_objects[i].push_back(k);//store index of matched object to this point
613  }
614  }
615  }
616  //get oldest object on each point
617  for(size_t i=0; i< matching_objects.size(); i++)
618  {
619  size_t oldest_object_index = 0;
620  int oldest_lifespan = -1;
621  std::string best_label;
622  for(size_t j=0; j<matching_objects[i].size(); j++)
623  {
624  size_t current_index = matching_objects[i][j];
625  int current_lifespan = targets_[in_tracker_indices[current_index]].lifetime_;
626  if (current_lifespan > oldest_lifespan)
627  {
628  oldest_lifespan = current_lifespan;
629  oldest_object_index = current_index;
630  }
631  if (!targets_[in_tracker_indices[current_index]].label_.empty() &&
632  targets_[in_tracker_indices[current_index]].label_ != "unknown")
633  {
634  best_label = targets_[in_tracker_indices[current_index]].label_;
635  }
636  }
637  // delete nearby targets except for the oldest target
638  for(size_t j=0; j<matching_objects[i].size(); j++)
639  {
640  size_t current_index = matching_objects[i][j];
641  if(current_index != oldest_object_index)
642  {
643  targets_[in_tracker_indices[current_index]].tracking_num_= TrackingState::Die;
644  }
645  }
646  autoware_msgs::DetectedObject best_object;
647  best_object = in_detected_objects.objects[oldest_object_index];
648  if (best_label != "unknown"
649  && !best_label.empty())
650  {
651  best_object.label = best_label;
652  }
653 
654  resulting_objects.objects.push_back(best_object);
655  }
656 
657  return resulting_objects;

661 void ImmUkfPda::makeOutput(const autoware_msgs::DetectedObjectArray& input,
662  const std::vector<bool> &matching_vec,
663  autoware_msgs::DetectedObjectArray& detected_objects_output)
664 {
665  autoware_msgs::DetectedObjectArray tmp_objects;
666  tmp_objects.header = input.header;
667  std::vector<size_t> used_targets_indices;
668  for (size_t i = 0; i < targets_.size(); i++)
669  {
670 
671  double tx = targets_[i].x_merge_(0);
672  double ty = targets_[i].x_merge_(1);
673 
674  double tv = targets_[i].x_merge_(2);
675  double tyaw = targets_[i].x_merge_(3);
676  double tyaw_rate = targets_[i].x_merge_(4);
677 
678  while (tyaw > M_PI)
679  tyaw -= 2. * M_PI;
680  while (tyaw < -M_PI)
681  tyaw += 2. * M_PI;
682 
683  tf::Quaternion q = tf::createQuaternionFromYaw(tyaw);
684 
685  autoware_msgs::DetectedObject dd;
686  dd = targets_[i].object_;
687  dd.id = targets_[i].ukf_id_;
688  dd.velocity.linear.x = tv;
689  dd.acceleration.linear.y = tyaw_rate;
690  dd.velocity_reliable = targets_[i].is_stable_;
691  dd.pose_reliable = targets_[i].is_stable_;
692 
693 
694  if (!targets_[i].is_static_ && targets_[i].is_stable_)
695  {
696  // Aligh the longest side of dimentions with the estimated orientation
697  if(targets_[i].object_.dimensions.x < targets_[i].object_.dimensions.y)
698  {
699  dd.dimensions.x = targets_[i].object_.dimensions.y;
700  dd.dimensions.y = targets_[i].object_.dimensions.x;
701  }
702 
703  dd.pose.position.x = tx;
704  dd.pose.position.y = ty;
705 
706  if (!std::isnan(q[0]))
707  dd.pose.orientation.x = q[0];
708  if (!std::isnan(q[1]))
709  dd.pose.orientation.y = q[1];
710  if (!std::isnan(q[2]))
711  dd.pose.orientation.z = q[2];
712  if (!std::isnan(q[3]))
713  dd.pose.orientation.w = q[3];
714  }
715  updateBehaviorState(targets_[i], dd);
716 
717  if (targets_[i].is_stable_ || (targets_[i].tracking_num_ >= TrackingState::Init &&
718  targets_[i].tracking_num_ < TrackingState::Stable))
719  {
720  tmp_objects.objects.push_back(dd);
721  used_targets_indices.push_back(i);
722  }
723  }
724  detected_objects_output = removeRedundantObjects(tmp_objects, used_targets_indices);
725 }

779  autoware_msgs::DetectedObjectArray& detected_objects_output)
780 {
781  double timestamp = input.header.stamp.toSec();
782  std::vector<bool> matching_vec(input.objects.size(), false);
783 
784  if (!init_)
785  {
786  initTracker(input, timestamp);
787  makeOutput(input, matching_vec, detected_objects_output);
788  return;
789  }
790 
791  double dt = (timestamp - timestamp_);
792  timestamp_ = timestamp;
793 
794 
795  // start UKF process
796  for (size_t i = 0; i < targets_.size(); i++)
797  {
798  targets_[i].is_stable_ = false;
799  targets_[i].is_static_ = false;
800 
801  if (targets_[i].tracking_num_ == TrackingState::Die)
802  {
803  continue;
804  }
805  // prevent ukf not to explode
806  if (targets_[i].p_merge_.determinant() > prevent_explosion_thres_ ||
807  targets_[i].p_merge_(4, 4) > prevent_explosion_thres_)
808  {
809  targets_[i].tracking_num_ = TrackingState::Die;
810  continue;
811  }
812 
813  targets_[i].prediction(use_sukf_, has_subscribed_vectormap_, dt);
814 
815  std::vector<autoware_msgs::DetectedObject> object_vec;
816  bool success = probabilisticDataAssociation(input, dt, matching_vec, object_vec, targets_[i]);
817  if (!success)
818  {
819  continue;
820  }
821 
822  targets_[i].update(use_sukf_, detection_probability_, gate_probability_, gating_thres_, object_vec);
823  }
824  // end UKF process
825 
826  // making new ukf target for no data association objects
827  makeNewTargets(timestamp, input, matching_vec);
828 
829  // static dynamic classification
830  staticClassification();
831 
832  // making output for visualization
833  makeOutput(input, matching_vec, detected_objects_output);
