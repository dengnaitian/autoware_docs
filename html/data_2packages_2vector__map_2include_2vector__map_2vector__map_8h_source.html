234 template <class T, class U>
235 class Handle
236 {
237 private:
238  ros::Subscriber sub_;
239  Updater<T, U> update_;
240  std::vector<Callback<U>> cbs_;
241  std::map<Key<T>, T> map_;
242 
243  void subscribe(const U& msg)
244  {
245  update_(map_, msg);
246  for (const auto& cb : cbs_)
247  cb(msg);
248  }
249 
250 public:
251  Handle()
252  {
253  }
254 
255  void registerSubscriber(ros::NodeHandle& nh, const std::string& topic_name)
256  {
257  sub_ = nh.subscribe(topic_name, 1, &Handle<T, U>::subscribe, this);
258  }
259 
260  void registerUpdater(const Updater<T, U>& update)
261  {
262  update_ = update;
263  }
264 
265  void registerCallback(const Callback<U>& cb)
266  {
267  cbs_.push_back(cb);
268  }
269 
270  T findByKey(const Key<T>& key) const
271  {
272  auto it = map_.find(key);
273  if (it == map_.end())
274  return T();
275  return it->second;
276  }
277 
278  std::vector<T> findByFilter(const Filter<T>& filter) const
279  {
280  std::vector<T> vector;
281  for (const auto& pair : map_)
282  {
283  if (filter(pair.second))
284  vector.push_back(pair.second);
285  }
286  return vector;
287  }
288 
289  bool empty() const
290  {
291  return map_.empty();
292  }

295 template <class T>
296 std::vector<T> parse(const std::string& csv_file)
297 {
298  std::ifstream ifs(csv_file.c_str());
299  std::string line;
300  std::getline(ifs, line); // remove first line
301  std::vector<T> objs;
302  while (std::getline(ifs, line))
303  {
304  T obj;
305  std::istringstream iss(line);
306  iss >> obj;
307  objs.push_back(obj);
308  }
309  return objs;
310 }
