178 vector<vector<float> > DrawingHelpers::PreparePathForDrawing(const std::vector<PlannerHNS::WayPoint>& path,
179  std::vector<std::vector<PlannerHNS::WayPoint> >& redyForDraw, double w, double resolution)
180 {
181  vector<vector<float> > colorProfiles;
182  if(path.size() < 2) return colorProfiles;
183  int size = path.size();
184  WayPoint p1 = path[0];
185  WayPoint p2 =p1;
186  WayPoint prev_point = p1;
187  WayPoint center, prev_center ,pa, pb, pc, pd;
188  double a = 0;
189  double prev_angle = 0;
190  vector<WayPoint> four_temp;
191  vector<float> color_vector;
192 
193  for(int i=0; i < size ; i++)
194  {
195 
196  color_vector.clear();
197  four_temp.clear();
198 
199  pa = p2 = path[i];
200 
201  color_vector.push_back(p1.v/12.0);
202  color_vector.push_back(p1.v/12.0);
203  color_vector.push_back(p1.v/12.0);
204  colorProfiles.push_back(color_vector);
205 
206  if(distance2points(p1.pos, p2.pos) < resolution)
207  continue;
208 
209  center.pos.x = p1.pos.x + (p2.pos.x-p1.pos.x)/2.0;
210  center.pos.y = p1.pos.y + (p2.pos.y-p1.pos.y)/2.0;
211 
212  a = atan2(p2.pos.y- p1.pos.y, p2.pos.x- p1.pos.x);
213 
214  pa.pos.x = p1.pos.x - w * cos(a - M_PI/2.0);
215  pa.pos.y = p1.pos.y - w * sin(a - M_PI/2.0);
216  pa.pos.z = p1.pos.z;
217 
218  pb.pos.x = p1.pos.x + w * cos(a - M_PI/2.0);
219  pb.pos.y = p1.pos.y + w * sin(a - M_PI/2.0);
220  pb.pos.z = p1.pos.z;
221 
222 
223  pc.pos.x = p2.pos.x + w * cos(a - M_PI/2.0);
224  pc.pos.y = p2.pos.y + w * sin(a - M_PI/2.0);
225  pc.pos.z = p2.pos.z;
226 
227  pd.pos.x = p2.pos.x - w * cos(a - M_PI/2.0);
228  pd.pos.y = p2.pos.y - w * sin(a - M_PI/2.0);
229  pd.pos.z = p2.pos.z;
230 
231  if(!(prev_point.pos.x == p1.pos.x && prev_point.pos.y == p1.pos.y))
232  {
233  prev_angle = atan2(p1.pos.y- prev_point.pos.y, p1.pos.x- prev_point.pos.x);
234 
235  pa.pos.x = p1.pos.x - w * cos(prev_angle - M_PI/2.0);
236  pa.pos.y = p1.pos.y - w * sin(prev_angle - M_PI/2.0);
237 
238  pb.pos.x = p1.pos.x + w * cos(prev_angle - M_PI/2.0);
239  pb.pos.y = p1.pos.y + w * sin(prev_angle - M_PI/2.0);
240  }
241 
242  four_temp.push_back(pa);
243  four_temp.push_back(pb);
244  four_temp.push_back(pc);
245  four_temp.push_back(pd);
246 
247  redyForDraw.push_back(four_temp);
248 
249  prev_point = p1;
250  p1 = p2;
251  }
252  return colorProfiles;
253 }

292 void DrawingHelpers::DrawCostPath(const std::vector<PlannerHNS::WayPoint*>& path_points, const double& z, const double& width)
293 {
294  if(path_points.size()==0) return;
295 
296  WayPoint p1 = *path_points[0];
297  float color[3] = {0,0,0};
298 
299  double max_cost = 0;
300  for(unsigned int i=0; i < path_points.size(); i++)
301  {
302  if(path_points.at(i)->cost > max_cost)
303  max_cost = path_points.at(i)->cost;
304  }
305 
306  int size = path_points.size();
307 
308  for(int i=0; i < size; i++)
309  {
310  p1 = *path_points[i];
311  double norm_cost = path_points.at(i)->cost / max_cost * 2.0;
312  if(norm_cost <= 1.0)
313  {
314  color[0] = norm_cost;
315  color[1] = 1.0;
316  }
317  else if(norm_cost > 1.0)
318  {
319  color[0] = 1.0;
320  color[1] = 2.0 - norm_cost;
321  }
322 
323  glColor3f(color[0], color[1], color[2]);
