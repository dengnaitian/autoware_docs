76 static void initialpose_callback(const geometry_msgs::PoseWithCovarianceStamped::ConstPtr& input)
77 {
78 
79  pose guess_pose;
80 
81  tf::TransformListener listener;
82  tf::StampedTransform transform;
83  try
84  {
85  ros::Time now = ros::Time(0);
86  listener.waitForTransform("/map", input->header.frame_id, now, ros::Duration(10.0));
87  listener.lookupTransform("/map", input->header.frame_id, now, transform);
88  }
89  catch (tf::TransformException& ex)
90  {
91  ROS_ERROR("%s", ex.what());
92  }
93 
94  tf::Quaternion q(input->pose.pose.orientation.x, input->pose.pose.orientation.y, input->pose.pose.orientation.z, input->pose.pose.orientation.w);
95  tf::Matrix3x3 m(q);
96 
97  guess_pose.x = input->pose.pose.position.x + transform.getOrigin().x();
98  guess_pose.y = input->pose.pose.position.y + transform.getOrigin().y();
99  guess_pose.z = input->pose.pose.position.z + transform.getOrigin().z();
100  m.getRPY(guess_pose.roll, guess_pose.pitch, guess_pose.yaw);
101 
102  if (_get_height == true)
103  {
104  double min_distance = DBL_MAX;
105  double nearest_z = guess_pose.z;
106  for (const auto& p : map)
107  {
108  double distance = hypot(guess_pose.x - p.x, guess_pose.y - p.y);
109  if (distance < min_distance)
110  {
111  min_distance = distance;
112  nearest_z = p.z;
113  }
114  }
115  guess_pose.z = nearest_z;
116  }
117 
118  Eigen::Translation3f init_translation(guess_pose.x, guess_pose.y, guess_pose.z);
119  Eigen::AngleAxisf init_rotation_x(guess_pose.roll, Eigen::Vector3f::UnitX());
120  Eigen::AngleAxisf init_rotation_y(guess_pose.pitch, Eigen::Vector3f::UnitY());
121  Eigen::AngleAxisf init_rotation_z(guess_pose.yaw, Eigen::Vector3f::UnitZ());
122 
123  Eigen::Matrix4f init_guess = (init_translation * init_rotation_z * init_rotation_y * init_rotation_x).matrix();
124 
125  pcl::PointCloud<pcl::PointXYZI>::Ptr output_cloud(new pcl::PointCloud<pcl::PointXYZI>);
126  ndt.align(*output_cloud, init_guess);
127 
128  Eigen::Matrix4f t = ndt.getFinalTransformation();
129 
130  pcl::PointCloud<pcl::PointXYZI>::Ptr transformed_additional_map_ptr (new pcl::PointCloud<pcl::PointXYZI>());
131  transformed_additional_map_ptr->header.frame_id = "/map";
132  pcl::transformPointCloud(*additional_map_ptr, *transformed_additional_map_ptr, t);
133  sensor_msgs::PointCloud2::Ptr msg_ptr(new sensor_msgs::PointCloud2);
134 
135  pcl::toROSMsg(*transformed_additional_map_ptr, *msg_ptr);
136  msg_ptr->header.frame_id = "/map";
137  ndt_map_pub.publish(*msg_ptr);
138 
139 
140  pcl::io::savePCDFileBinary(output_pcd, *transformed_additional_map_ptr);
141  std::cout << "Saved " << output_pcd << ": " << transformed_additional_map_ptr->points.size() << " points." << std::endl;
142 
143  pcl::PointCloud<pcl::PointXYZRGB> cloud_rgb;
144  cloud_rgb.width = transformed_additional_map_ptr->width;
145  cloud_rgb.height = transformed_additional_map_ptr->height;
146  cloud_rgb.points.resize(cloud_rgb.width * cloud_rgb.height);
147 
148  for(size_t i = 0; i < cloud_rgb.points.size(); i++){
149  cloud_rgb.points[i].x = transformed_additional_map_ptr->points[i].x;
