19 bool FeatureGenerator::init(caffe::Blob<float>* out_blob)
20 {
21  out_blob_ = out_blob;
22 
23  // raw feature parameters
24  range_ = 60;
25  width_ = 512;
26  height_ = 512;
27  min_height_ = -5.0;
28  max_height_ = 5.0;
29  CHECK_EQ(width_, height_)
30  << "Current implementation version requires input_width == input_height.";
31 
32  // set output blob and log lookup table
33  out_blob_->Reshape(1, 8, height_, width_);
34 
35  log_table_.resize(256);
36  for (size_t i = 0; i < log_table_.size(); ++i) {
37  log_table_[i] = std::log1p(static_cast<float>(i));
38  }
39 
40  float* out_blob_data = nullptr;
41  out_blob_data = out_blob_->mutable_cpu_data();
42 
43  int channel_index = 0;
44  max_height_data_ = out_blob_data + out_blob_->offset(0, channel_index++);
45  mean_height_data_ = out_blob_data + out_blob_->offset(0, channel_index++);
46  count_data_ = out_blob_data + out_blob_->offset(0, channel_index++);
47  direction_data_ = out_blob_data + out_blob_->offset(0, channel_index++);
48  top_intensity_data_ = out_blob_data + out_blob_->offset(0, channel_index++);
49  mean_intensity_data_ = out_blob_data + out_blob_->offset(0, channel_index++);
50  distance_data_ = out_blob_data + out_blob_->offset(0, channel_index++);
51  nonempty_data_ = out_blob_data + out_blob_->offset(0, channel_index++);
52  CHECK_EQ(out_blob_->offset(0, channel_index), out_blob_->count());
53 
54  // compute direction and distance features
55  int siz = height_ * width_;
56  std::vector<float> direction_data(siz);
57  std::vector<float> distance_data(siz);
58 
59  for (int row = 0; row < height_; ++row) {
60  for (int col = 0; col < width_; ++col) {
61  int idx = row * width_ + col;
62  // * row <-> x, column <-> y
63  float center_x = Pixel2Pc(row, height_, range_);
64  float center_y = Pixel2Pc(col, width_, range_);
65  constexpr double K_CV_PI = 3.1415926535897932384626433832795;
66  direction_data[idx] =
67  static_cast<float>(std::atan2(center_y, center_x) / (2.0 * K_CV_PI));
68  distance_data[idx] =
69  static_cast<float>(std::hypot(center_x, center_y) / 60.0 - 0.5);
70  }
71  }
72  caffe::caffe_copy(siz, direction_data.data(), direction_data_);
73  caffe::caffe_copy(siz, distance_data.data(), distance_data_);
74 
75  return true;
76 }

86 void FeatureGenerator::generate(
87  const pcl::PointCloud<pcl::PointXYZI>::Ptr& pc_ptr) {
88  const auto& points = pc_ptr->points;
89 
90  // DO NOT remove this line!!!
91  // Otherwise, the gpu_data will not be updated for the later frames.
92  // It marks the head at cpu for blob.
93  out_blob_->mutable_cpu_data();
94 
95  int siz = height_ * width_;
96  caffe::caffe_set(siz, float(-5), max_height_data_);
97  caffe::caffe_set(siz, float(0), mean_height_data_);
98  caffe::caffe_set(siz, float(0), count_data_);
99  caffe::caffe_set(siz, float(0), top_intensity_data_);
100  caffe::caffe_set(siz, float(0), mean_intensity_data_);
101  caffe::caffe_set(siz, float(0), nonempty_data_);
102 
103  map_idx_.resize(points.size());
104  float inv_res_x =
105  0.5 * static_cast<float>(width_) / static_cast<float>(range_);
106  float inv_res_y =
107  0.5 * static_cast<float>(height_) / static_cast<float>(range_);
108 
109  for (size_t i = 0; i < points.size(); ++i) {
110  if (points[i].z <= min_height_ || points[i].z >= max_height_) {
111  map_idx_[i] = -1;
112  continue;
113  }
114  // * the coordinates of x and y are exchanged here
115  // (row <-> x, column <-> y)
116  int pos_x = F2I(points[i].y, range_, inv_res_x); // col
117  int pos_y = F2I(points[i].x, range_, inv_res_y); // row
118  if (pos_x >= width_ || pos_x < 0 || pos_y >= height_ || pos_y < 0) {
119  map_idx_[i] = -1;
120  continue;
121  }
122  map_idx_[i] = pos_y * width_ + pos_x;
123 
124  int idx = map_idx_[i];
125  float pz = points[i].z;
126  float pi = points[i].intensity / 255.0;
127  if (max_height_data_[idx] < pz) {
128  max_height_data_[idx] = pz;
129  top_intensity_data_[idx] = pi;
130  }
131  mean_height_data_[idx] += static_cast<float>(pz);
132  mean_intensity_data_[idx] += static_cast<float>(pi);
133  count_data_[idx] += float(1);
134  }
135 
136  for (int i = 0; i < siz; ++i) {
137  constexpr double EPS = 1e-6;
138  if (count_data_[i] < EPS) {
139  max_height_data_[i] = float(0);
140  } else {
141  mean_height_data_[i] /= count_data_[i];
142  mean_intensity_data_[i] /= count_data_[i];
143  nonempty_data_[i] = float(1);
144  }
145  count_data_[i] = logCount(static_cast<int>(count_data_[i]));
146  }
147 }
