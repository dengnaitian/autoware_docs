179 void ContourTracker::ImportCloudClusters(const autoware_msgs::CloudClusterArrayConstPtr& msg, std::vector<PlannerHNS::DetectedObject>& originalClusters)
180 {
181  originalClusters.clear();
182  m_nOriginalPoints = 0;
183  m_nContourPoints = 0;
184  m_FilteringTime = 0;
185  m_PolyEstimationTime = 0;
186  struct timespec filter_time, poly_est_time;
187 
188  PlannerHNS::DetectedObject obj;
189  PlannerHNS::GPSPoint avg_center;
190  PolygonGenerator polyGen(m_Params.nQuarters);
191  pcl::PointCloud<pcl::PointXYZ> point_cloud;
192 
193  if(bMap)
194  m_ClosestLanesList = PlannerHNS::MappingHelpers::GetClosestLanesFast(m_CurrentPos, m_Map, m_Params.DetectionRadius);
195 
196  //Filter the detected Obstacles:
197  //std::cout << "Filter the detected Obstacles: " << std::endl;
198  for(unsigned int i=0; i < msg->clusters.size(); i++)
199  {
200  obj.center.pos.x = msg->clusters.at(i).centroid_point.point.x;
201  obj.center.pos.y = msg->clusters.at(i).centroid_point.point.y;
202  obj.center.pos.z = msg->clusters.at(i).centroid_point.point.z;
203  obj.center.pos.a = msg->clusters.at(i).estimated_angle;
204 
205  obj.distance_to_center = hypot(obj.center.pos.y-m_CurrentPos.pos.y, obj.center.pos.x-m_CurrentPos.pos.x);
206 
207  obj.actual_yaw = msg->clusters.at(i).estimated_angle;
208 
209  obj.w = msg->clusters.at(i).dimensions.x;
210  obj.l = msg->clusters.at(i).dimensions.y;
211  obj.h = msg->clusters.at(i).dimensions.z;
212 
213  UtilityHNS::UtilityH::GetTickCount(filter_time);
214  if(!IsCar(obj, m_CurrentPos, m_Map)) continue;
215  m_FilteringTime += UtilityHNS::UtilityH::GetTimeDiffNow(filter_time);
216 
217  obj.id = msg->clusters.at(i).id;
218  obj.originalID = msg->clusters.at(i).id;
219  obj.label = msg->clusters.at(i).label;
220 
221  if(msg->clusters.at(i).indicator_state == 0)
222  obj.indicator_state = PlannerHNS::INDICATOR_LEFT;
223  else if(msg->clusters.at(i).indicator_state == 1)
224  obj.indicator_state = PlannerHNS::INDICATOR_RIGHT;
225  else if(msg->clusters.at(i).indicator_state == 2)
226  obj.indicator_state = PlannerHNS::INDICATOR_BOTH;
227  else if(msg->clusters.at(i).indicator_state == 3)
228  obj.indicator_state = PlannerHNS::INDICATOR_NONE;
229 
230 
231  UtilityHNS::UtilityH::GetTickCount(poly_est_time);
232  point_cloud.clear();
233  pcl::fromROSMsg(msg->clusters.at(i).cloud, point_cloud);
234 
235  obj.contour = polyGen.EstimateClusterPolygon(point_cloud ,obj.center.pos,avg_center, m_Params.PolygonRes);
236 
237  m_PolyEstimationTime += UtilityHNS::UtilityH::GetTimeDiffNow(poly_est_time);
238  m_nOriginalPoints += point_cloud.points.size();
239  m_nContourPoints += obj.contour.size();
240  originalClusters.push_back(obj);
241 
242  }

245 bool ContourTracker::IsCar(const PlannerHNS::DetectedObject& obj, const PlannerHNS::WayPoint& currState, PlannerHNS::RoadNetwork& map)
246 {
247 
248  if(bMap)
249  {
250  bool bOnLane = false;
251  // std::cout << "Debug Obj: " << obj.id << ", Closest Lane: " << m_ClosestLanesList.size() << std::endl;
252 
253  for(unsigned int i =0 ; i < m_ClosestLanesList.size(); i++)
254  {
255 
256  PlannerHNS::RelativeInfo info;
257  PlannerHNS::PlanningHelpers::GetRelativeInfoLimited(m_ClosestLanesList.at(i)->points, obj.center, info);
258  PlannerHNS::WayPoint wp = m_ClosestLanesList.at(i)->points.at(info.iFront);
259 
260  double direct_d = hypot(wp.pos.y - obj.center.pos.y, wp.pos.x - obj.center.pos.x);
261 
262  // std::cout << "- Distance To Car: " << obj.distance_to_center << ", PerpD: " << info.perp_distance << ", DirectD: " << direct_d << ", bAfter: " << info.bAfter << ", bBefore: " << info.bBefore << std::endl;
263 
264  if((info.bAfter || info.bBefore) && direct_d > m_MapFilterDistance*2.0)
265  continue;
266 
267  if(fabs(info.perp_distance) <= m_MapFilterDistance)
268  {
269  bOnLane = true;
270  break;
271  }
272  }
273 
274  if(bOnLane == false)
275  return false;
276  }
277 
278  if(!m_Params.bEnableSimulation)
279  {
280  double object_size = hypot(obj.w, obj.l);
281 
282  //std::cout << "Filter the detected Obstacles: (" << obj.distance_to_center << ",>" << m_Params.DetectionRadius << " | "<< object_size << ",< " << m_Params.MinObjSize << "| " << object_size << ", >" << m_Params.MaxObjSize << ")"<< std::endl;
283 
284  if(obj.distance_to_center > m_Params.DetectionRadius || object_size < m_Params.MinObjSize || object_size > m_Params.MaxObjSize)
285  return false;
286  }
287 
288  if(m_Params.bEnableSimulation)
289  {
290  PlannerHNS::Mat3 rotationMat(-currState.pos.a);
291  PlannerHNS::Mat3 translationMat(-currState.pos.x, -currState.pos.y);
292 
293  PlannerHNS::GPSPoint relative_point = translationMat*obj.center.pos;
294  relative_point = rotationMat*relative_point;
295 
296  double distance_x = fabs(relative_point.x - m_Params.VehicleLength/3.0);
297  double distance_y = fabs(relative_point.y);
298 
299  if(distance_x <= m_Params.VehicleLength*0.5 && distance_y <= m_Params.VehicleWidth*0.5) // don't detect yourself
300  return false;
301  }
302 
303  return true;
304 }
