21 void ReadNMEASentence::Parse(char letter)
22 {
23  switch (Stat)
24  {
25  case waitFirstChar :
26  if (letter == '$')
27  {
28  Stat = command;
29  calculatedCheckSum = 0;
30  receivedCheckSum = 0;
31  nmeaCmd = "";
32  dataString = "";
33  }
34  break;
35 
36  case command :
37  if (letter != ',')
38  {
39  nmeaCmd += letter;
40  }
41  else
42  {
43  Stat = data;
44  }
45  calculatedCheckSum ^= letter;
46  break;
47 
48  case data :
49  if (letter != '*')
50  {
51  dataString += letter;
52  calculatedCheckSum ^= letter;
53  }
54  else
55  {
56  Stat = checkSum1;
57  }
58  break;
59 
60  case checkSum1 :
61  if (letter >= '0' && letter <= '9')
62  {
63  receivedCheckSum = ((letter - '0')&0x0f)<<4;
64  }
65  else if (letter >= 'A' && letter <= 'F')
66  {
67  receivedCheckSum = ((letter - 'A' + 10)&0x0f)<<4;
68  }
69  else receivedCheckSum = 0;
70  Stat = checkSum2;
71  break;
72 
73  case checkSum2 :
74  if (letter >= '0' && letter <= '9')
75  {
76  receivedCheckSum |= (letter - '0')&0x0f;
77  }
78  else if (letter >= 'A' && letter <= 'F')
79  {
80  receivedCheckSum |= ((letter - 'A' + 10)&0x0f);
81  }
82  else receivedCheckSum = 0;
83  Stat = waitFirstChar;
84 
85  if (receivedCheckSum == calculatedCheckSum)
86  {
87  commandCount++;
88  if (nmeaCmd == "GPRMC" || nmeaCmd == "GNRMC")
89  {
90  GxRMC();
91  }
92  else if (nmeaCmd == "HEHDT" || nmeaCmd == "GPHDT")
93  {
94  HEHDT();
95  }
96  else if (nmeaCmd == "GPGGA")
97  {
98  GPGGA();
99  }
100  }
101 
102  break;
103 
104  default :
105  break;
106  }
107 }
