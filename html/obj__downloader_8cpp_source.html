117 static void marker_publisher(const std_msgs::String& msg)
118 {
119  visualization_msgs::Marker sphere_list;
120  sphere_list.header.frame_id = "/mobility";
121  sphere_list.header.stamp = ros::Time::now();
122  sphere_list.ns = "mo_pictograms";
123  sphere_list.id = 0;
124  sphere_list.type = visualization_msgs::Marker::SPHERE_LIST;
125  sphere_list.action = visualization_msgs::Marker::ADD;
126  sphere_list.lifetime = ros::Duration();
127  sphere_list.color.r = 1.0;
128  sphere_list.color.g = 0.0;
129  sphere_list.color.b = 0.0;
130  sphere_list.color.a = 1.0;
131  sphere_list.scale.x = 10.0;
132  sphere_list.scale.y = 10.0;
133  sphere_list.scale.z = 10.0;
134 
135  geo_pos_conv geo;
136  geo.set_plane(7);
137  // Loading data.
138  std::vector<std::string> db_data = split(msg.data,'\n');
139 
140  std::vector<CarInformation> cars;
141  for(const std::string& row : db_data) {
142  if(row.empty())
143  continue;
144 
145  CarInformation car;
146  int ret = result_to_car_info(row, car);
147  if (ret != 0)
148  continue;
149 
150  // Convert from lat,lon to x,y
151  // geo.set_llh_nmea_degrees(car.lat, car.lon, car.ele);
152  if(car.lat >= -180 && car.lat <= 180 && car.lon >= -180 && car.lon <= 180){
153  geo.llh_to_xyz(car.lat, car.lon, car.ele);
154  car.x = geo.x();
155  car.y = geo.y();
156  car.z = geo.z();
157  }
158 
159  geometry_msgs::Point p;
160  // swap x and y
161  p.x = car.y;
162  p.y = car.x;
163  p.z = car.z;
164 
165  cars.push_back(car);
166  sphere_list.points.push_back(p);
167 
168  car.dump();
169  }
170 
171  pub.publish(sphere_list);
172 }

174 static std::string construct_select_statement(DataType type)
175 {
176  std::stringstream ss;
177 
178  switch (type) {
179  case RANGE:
180  ss << "SELECT 0,lat,lon,x,y,z,tm FROM pos"
181  << " WHERE ((lat >= "<< fixed << setprecision(7) << positionRange[0]
182  << " AND lat < " << fixed << setprecision(7) << positionRange[1]
183  << " AND lon >= " << fixed << setprecision(7) << positionRange[2]
184  << " AND lon < " << fixed << setprecision(7) << positionRange[3] << ")"
185  << " OR (x >= " << fixed << setprecision(7) << geoPosition[0]
186  << " AND x < " << fixed << setprecision(7) << geoPosition[1]
187  << " AND y >= " << fixed << setprecision(7) << geoPosition[2]
188  << " AND y < " << fixed << setprecision(7) << geoPosition[3] << "))"
189  << " AND tm > TO_TIMESTAMP(Second,SINCE_EPOCH(Second,current_timestamp)-1) and tm <= current_timestamp;";
190  break;
191  case TEST:
192  ss << "SELECT 0,lat,lon,0,x,y,z,tm FROM pos"
193  << " WHERE ((lat >= " << fixed << setprecision(7) << positionRange[0]
194  << " AND lat < " << fixed << setprecision(7) << positionRange[1]
195  << " AND lon >= " << fixed << setprecision(7) << positionRange[2]
196  << " AND lon < " << fixed << setprecision(7) << positionRange[3] << ")"
197  << " OR (x >= " << fixed << setprecision(7) << geoPosition[0]
198  << " AND x < " << fixed << setprecision(7) << geoPosition[1]
199  << " AND y >= " << fixed << setprecision(7) << geoPosition[2]
200  << " AND y < " << fixed << setprecision(7) << geoPosition[3] << "))"
201  << " AND id = '0' ORDER BY tm DESC LIMIT 1;";
202  break;
203  case DB1:
204  ss << "SELECT id,lat,lon,ele,timestamp FROM select_test"
205  << " WHERE timestamp = (select max(timestamp) from select_test)"
206  << " AND lat >= " << fixed << setprecision(7) << positionRange[0]
207  << " AND lat < " << fixed << setprecision(7) << positionRange[1]
208  << " AND lon >= " << fixed << setprecision(7) << positionRange[2]
209  << " AND lon < " << fixed << setprecision(7) << positionRange[3] << ";<E>";
210  break;
211  case NORMAL:
212  default:
213  ss << "SELECT 0,lat,lon,0,tm FROM pos"
214  << " WHERE lat >= 30 AND lat < 40"
215  << " AND lon >= 130 AND lon < 140"
216  << " AND tm > TO_TIMESTAMP(Second,SINCE_EPOCH(Second,current_timestamp)-1) AND tm <= current_timestamp;";
217  break;
218  }
219 
220  return ss.str();
221 }
