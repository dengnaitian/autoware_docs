198 void UKF::initialize(const Eigen::VectorXd& z, const double timestamp, const int target_id)
199 {
200  ukf_id_ = target_id;
201 
202  // first measurement
203  x_merge_ << 0, 0, 0, 0, 0.1;
204 
205  // init covariance matrix by hardcoding since no clue about initial state covrariance
206  p_merge_ << 0.5, 0, 0, 0, 0, 0, 0.5, 0, 0, 0, 0, 0, 3, 0, 0, 0, 0, 0, 10, 0, 0, 0, 0, 0, 1;
207 
208  // set weights
209  // reference from "The Unscented Kalman Filter for Nonlinear Estimation, Eric A. Wan and Rudolph van der Merwe, 2000"
210  // alpha = 0.0025, beta = 2, k = 0
211  double alpha = 0.0025;
212  double beta = 2;
213  double k = 0;
214  lambda_ = alpha * alpha * (num_state_ + k) - num_state_;
215  double weight_s_0 = lambda_ / (lambda_ + num_state_);
216  double weight_c_0 = lambda_ / (lambda_ + num_state_) + (1 - alpha * alpha + beta);
217  weights_s_(0) = weight_s_0;
218  weights_c_(0) = weight_c_0;
219  for (int i = 1; i < 2 * num_state_ + 1; i++)
220  { // 2n+1 weights
221  double weight = 0.5 / (num_state_ + lambda_);
222  weights_s_(i) = weight;
223  weights_c_(i) = weight;
224  }
225 
226  // init timestamp
227  time_ = timestamp;
228 
229  x_merge_(0) = z(0);
230  x_merge_(1) = z(1);
231 
232  z_pred_cv_(0) = z(0);
233  z_pred_cv_(1) = z(1);
234 
235  z_pred_ctrv_(0) = z(0);
236  z_pred_ctrv_(1) = z(1);
237 
238  z_pred_rm_(0) = z(0);
239  z_pred_rm_(1) = z(1);
240 
241  x_cv_ = x_ctrv_ = x_rm_ = x_merge_;
242  p_cv_ = p_ctrv_ = p_rm_ = p_merge_;
243 
244  s_cv_ << 1, 0, 0, 1;
245  s_ctrv_ << 1, 0, 0, 1;
246  s_rm_ << 1, 0, 0, 1;
247 
248  // initialize R covariance
249  r_cv_ << std_laspx_ * std_laspx_, 0, 0, std_laspy_ * std_laspy_;
250  r_ctrv_ << std_laspx_ * std_laspx_, 0, 0, std_laspy_ * std_laspy_;
251  r_rm_ << std_laspx_ * std_laspx_, 0, 0, std_laspy_ * std_laspy_;
252 
253  // initialize lidar-lane R covariance
254  // clang-format off
255  lidar_direction_r_cv_ << std_laspx_ * std_laspx_, 0, 0,
256  0, std_laspy_ * std_laspy_, 0,
257  0, 0, std_lane_direction_*std_lane_direction_;
258  lidar_direction_r_ctrv_ << std_laspx_ * std_laspx_, 0, 0,
259  0, std_laspy_ * std_laspy_, 0,
260  0, 0, std_lane_direction_*std_lane_direction_;
261  lidar_direction_r_rm_ << std_laspx_ * std_laspx_, 0, 0,
262  0, std_laspy_ * std_laspy_, 0,
263  0, 0, std_lane_direction_*std_lane_direction_;
264  // clang-format on
265 
266  // init tracking num
267  tracking_num_ = 1;
268 }

270 void UKF::updateModeProb(const std::vector<double>& lambda_vec)
271 {
272  double cvGauss = lambda_vec[0];
273  double ctrvGauss = lambda_vec[1];
274  double rmGauss = lambda_vec[2];
275  double sumGauss = cvGauss * mode_prob_cv_ + ctrvGauss * mode_prob_ctrv_ + rmGauss * mode_prob_rm_;
276  mode_prob_cv_ = (cvGauss * mode_prob_cv_) / sumGauss;
277  mode_prob_ctrv_ = (ctrvGauss * mode_prob_ctrv_) / sumGauss;
278  mode_prob_rm_ = (rmGauss * mode_prob_rm_) / sumGauss;
279  // prevent each prob from becoming 0
280  if (fabs(mode_prob_cv_) < 0.0001)
281  mode_prob_cv_ = 0.0001;
282  if (fabs(mode_prob_ctrv_) < 0.0001)
283  mode_prob_ctrv_ = 0.0001;
284  if (fabs(mode_prob_rm_) < 0.0001)
285  mode_prob_rm_ = 0.0001;
286 }

332 {

358  x_ctrv_ = mode_match_prob_cv2ctrv_ * x_pre_cv + mode_match_prob_ctrv2ctrv_ * x_pre_ctrv +
359  mode_match_prob_rm2ctrv_ * x_pre_rm;
360  x_rm_ = mode_match_prob_cv2rm_ * x_pre_cv + mode_match_prob_ctrv2rm_ * x_pre_ctrv + mode_match_prob_rm2rm_ * x_pre_rm;
361 
362  // not interacting yaw(-pi ~ pi)
363  x_cv_(3) = x_pre_cv(3);
364  x_ctrv_(3) = x_pre_ctrv(3);
365  x_rm_(3) = x_pre_rm(3);
366 
367  // normalizing angle
368  while (x_cv_(3) > M_PI)
369  x_cv_(3) -= 2. * M_PI;
370  while (x_cv_(3) < -M_PI)
371  x_cv_(3) += 2. * M_PI;

373  x_ctrv_(3) -= 2. * M_PI;

375  x_ctrv_(3) += 2. * M_PI;
376  while (x_rm_(3) > M_PI)
377  x_rm_(3) -= 2. * M_PI;
378  while (x_rm_(3) < -M_PI)
379  x_rm_(3) += 2. * M_PI;
380 
381  p_cv_ = mode_match_prob_cv2cv_ * (p_pre_cv + (x_pre_cv - x_cv_) * (x_pre_cv - x_cv_).transpose()) +

539  num_meas_state = num_lidar_direction_state_;
540  z_pred = Eigen::VectorXd(num_meas_state);
541  s_pred = Eigen::MatrixXd(num_meas_state, num_meas_state);
542  kalman_gain = Eigen::MatrixXd(num_state_, num_meas_state);
543  z_pred = z_pred_lidar_direction_ctrv_;
544  s_pred = s_lidar_direction_ctrv_;
545  kalman_gain = k_lidar_direction_ctrv_;
546  }
547  else
548  {
549  num_meas_state = num_lidar_state_;
550  z_pred = Eigen::VectorXd(num_meas_state);
551  s_pred = Eigen::MatrixXd(num_meas_state, num_meas_state);
552  kalman_gain = Eigen::MatrixXd(num_state_, num_meas_state);
553  z_pred = z_pred_ctrv_;
554  s_pred = s_ctrv_;
555  kalman_gain = k_ctrv_;
556  }
557  }
558  else
559  {
560  x = x_rm_;
561  p = p_rm_;
562  if (is_direction_rm_available_)
563  {
564  is_direction_available = true;
565  num_meas_state = num_lidar_direction_state_;
566  z_pred = Eigen::VectorXd(num_meas_state);
567  s_pred = Eigen::MatrixXd(num_meas_state, num_meas_state);
568  kalman_gain = Eigen::MatrixXd(num_state_, num_meas_state);
569  z_pred = z_pred_lidar_direction_rm_;
570  s_pred = s_lidar_direction_rm_;
571  kalman_gain = k_lidar_direction_rm_;
572  }
573  else
574  {
575  num_meas_state = num_lidar_state_;
576  z_pred = Eigen::VectorXd(num_meas_state);
577  s_pred = Eigen::MatrixXd(num_meas_state, num_meas_state);
578  kalman_gain = Eigen::MatrixXd(num_state_, num_meas_state);
579  z_pred = z_pred_rm_;
580  s_pred = s_rm_;
581  kalman_gain = k_rm_;
582  }
583  }
584 
585  for (size_t i = 0; i < num_meas; i++)
586  {
587  Eigen::VectorXd meas = Eigen::VectorXd(num_meas_state);
588  meas(0) = object_vec[i].pose.position.x;
589  meas(1) = object_vec[i].pose.position.y;
590  if (is_direction_available)
591  meas(2) = object_vec[i].angle;
592  meas_vec.push_back(meas);
593  Eigen::VectorXd diff = meas - z_pred;
594  diff_vec.push_back(diff);
595  double e = exp(-0.5 * diff.transpose() * s_pred.inverse() * diff);
596  e_vec.push_back(e);
597  e_sum += e;
598  }
599  double beta_zero = b / (b + e_sum);
600 
601  std::vector<double> beta_vec;
602 
603  if (num_meas != 0)
604  {
605  std::vector<double>::iterator max_iter = std::max_element(e_vec.begin(), e_vec.end());
606  int max_ind = std::distance(e_vec.begin(), max_iter);
607  likely_meas = meas_vec[max_ind];
608  }
609 
610  for (size_t i = 0; i < num_meas; i++)
611  {
612  double temp = e_vec[i] / (b + e_sum);
613  beta_vec.push_back(temp);
614  }
615  Eigen::VectorXd sigma_x;
616  sigma_x.setZero(num_meas_state);
617 
618  for (size_t i = 0; i < num_meas; i++)
619  {
620  sigma_x += beta_vec[i] * diff_vec[i];
621  }
622 
623  Eigen::MatrixXd sigma_p;
624  sigma_p.setZero(num_meas_state, num_meas_state);
625 
626  for (size_t i = 0; i < num_meas; i++)
627  {
628  sigma_p += (beta_vec[i] * diff_vec[i] * diff_vec[i].transpose() - sigma_x * sigma_x.transpose());
629  }
630 
631  // update x and P
632  Eigen::MatrixXd updated_x(x_cv_.rows(), x_cv_.cols());
633  updated_x = x + kalman_gain * sigma_x;
634 
635  updated_x(3) = normalizeAngle(updated_x(3));
636 
637  Eigen::MatrixXd updated_p(p_cv_.rows(), p_cv_.cols());
638  if (num_meas != 0)
639  {
640  updated_p = beta_zero * p + (1 - beta_zero) * (p - kalman_gain * s_pred * kalman_gain.transpose()) +
641  kalman_gain * sigma_p * kalman_gain.transpose();
642  }
643  else
644  {
645  updated_p = p - kalman_gain * s_pred * kalman_gain.transpose();
646  }
647 
648  double lambda;
649  if (num_meas != 0)
650  {
651  lambda =
652  (1 - gate_probability * detection_probability) / pow(Vk, num_meas) +
653  detection_probability * pow(Vk, 1 - num_meas) * e_sum / (num_meas * sqrt(2 * M_PI * s_pred.determinant()));
654  }
655  else
656  {
657  lambda = (1 - gate_probability * detection_probability);
658  }
659 
660  lambda_vec.push_back(lambda);
661 
662  if (motion_ind == MotionModel::CV)
663  {
664  x_cv_ = updated_x;
665  p_cv_ = updated_p;
666  }
667  else if (motion_ind == MotionModel::CTRV)
668  {
669  x_ctrv_ = updated_x;
670  p_ctrv_ = updated_p;
671  }
672  else
673  {
674  x_rm_ = updated_x;
675  p_rm_ = updated_p;
676  }
677  }

770 void UKF::ctrv(const double p_x, const double p_y, const double v, const double yaw, const double yawd,
771  const double delta_t, std::vector<double>& state)
772 {
773  // predicted state values
774  double px_p, py_p;
775 
776  // avoid division by zero
777  if (fabs(yawd) > 0.001)
778  {
779  px_p = p_x + v / yawd * (sin(yaw + yawd * delta_t) - sin(yaw));
780  py_p = p_y + v / yawd * (cos(yaw) - cos(yaw + yawd * delta_t));
781  }
782  else
783  {
784  px_p = p_x + v * delta_t * cos(yaw);
785  py_p = p_y + v * delta_t * sin(yaw);

789  double yawd_p = yawd;
790 
791  while (yaw_p > M_PI)
792  yaw_p -= 2. * M_PI;
793  while (yaw_p < -M_PI)
794  yaw_p += 2. * M_PI;
795 
796  state[0] = px_p;
797  state[1] = py_p;
798  state[2] = v_p;
799  state[3] = yaw_p;

840 {

842  {
843  return;
844  }

846  double dt_3 = dt_2 * dt;
847  double dt_4 = dt_3 * dt;

849  double sin_yaw = sin(yaw);
850  double cos_2_yaw = cos(yaw) * cos(yaw);
851  double sin_2_yaw = sin(yaw) * sin(yaw);
852  double cos_sin = cos_yaw * sin_yaw;
853

855  double cv_var_yawdd = std_cv_yawdd_ * std_cv_yawdd_;

861  double rm_var_yawdd = std_rm_yawdd_ * std_rm_yawdd_;
862 
863  q_cv_ << 0.5 * 0.5 * dt_4 * cos_2_yaw * cv_var_a, 0.5 * 0.5 * dt_4 * cos_sin * cv_var_a,
864  0.5 * dt_3 * cos_yaw * cv_var_a, 0, 0, 0.5 * 0.5 * dt_4 * cos_sin * cv_var_a,
865  0.5 * 0.5 * dt_4 * sin_2_yaw * cv_var_a, 0.5 * dt_3 * sin_yaw * cv_var_a, 0, 0, 0.5 * dt_3 * cos_yaw * cv_var_a,
866  0.5 * dt_3 * sin_yaw * cv_var_a, dt_2 * cv_var_a, 0, 0, 0, 0, 0, 0.5 * 0.5 * dt_4 * cv_var_yawdd,
867  0.5 * dt_3 * cv_var_yawdd, 0, 0, 0, 0.5 * dt_3 * cv_var_yawdd, dt_2 * cv_var_yawdd;
868  q_ctrv_ << 0.5 * 0.5 * dt_4 * cos_2_yaw * ctrv_var_a, 0.5 * 0.5 * dt_4 * cos_sin * ctrv_var_a,
869  0.5 * dt_3 * cos_yaw * ctrv_var_a, 0, 0, 0.5 * 0.5 * dt_4 * cos_sin * ctrv_var_a,
870  0.5 * 0.5 * dt_4 * sin_2_yaw * ctrv_var_a, 0.5 * dt_3 * sin_yaw * ctrv_var_a, 0, 0,
871  0.5 * dt_3 * cos_yaw * ctrv_var_a, 0.5 * dt_3 * sin_yaw * ctrv_var_a, dt_2 * ctrv_var_a, 0, 0, 0, 0, 0,
872  0.5 * 0.5 * dt_4 * ctrv_var_yawdd, 0.5 * dt_3 * ctrv_var_yawdd, 0, 0, 0, 0.5 * dt_3 * ctrv_var_yawdd,
873  dt_2 * ctrv_var_yawdd;

881 void UKF::predictionMotion(const double delta_t, const int model_ind)
882 {
883  /*****************************************************************************
884  * Initialize model parameters
885  ****************************************************************************/
886  Eigen::MatrixXd x(x_cv_.rows(), 1);
887  Eigen::MatrixXd p(p_cv_.rows(), p_cv_.cols());
888  Eigen::MatrixXd q(p_cv_.rows(), p_cv_.cols());
889  Eigen::MatrixXd x_sig_pred(x_sig_pred_cv_.rows(), x_sig_pred_cv_.cols());
890  if (model_ind == MotionModel::CV)
891  {
892  x = x_cv_.col(0);
893  p = p_cv_;
894  q = q_cv_;
895  x_sig_pred = x_sig_pred_cv_;
896  }
897  else if (model_ind == MotionModel::CTRV)
898  {
899  x = x_ctrv_.col(0);
900  p = p_ctrv_;
901  q = q_ctrv_;
902  x_sig_pred = x_sig_pred_ctrv_;
903  }
904  else
905  {
906  x = x_rm_.col(0);
907  p = p_rm_;
908  q = q_rm_;
909  x_sig_pred = x_sig_pred_rm_;
910  }
911 
912  /*****************************************************************************
913  * Create Sigma Points
914  ****************************************************************************/
915 
916  Eigen::MatrixXd x_sig = Eigen::MatrixXd(num_state_, 2 * num_state_ + 1);
917 
918  // create square root matrix
919  Eigen::MatrixXd L = p.llt().matrixL();
920 
921  // create augmented sigma points
922  x_sig.col(0) = x;
923  for (int i = 0; i < num_state_; i++)
924  {
925  Eigen::VectorXd pred1 = x + sqrt(lambda_ + num_state_) * L.col(i);
926  Eigen::VectorXd pred2 = x - sqrt(lambda_ + num_state_) * L.col(i);
927 
928  while (pred1(3) > M_PI)
929  pred1(3) -= 2. * M_PI;
930  while (pred1(3) < -M_PI)
931  pred1(3) += 2. * M_PI;
932 
933  while (pred2(3) > M_PI)
934  pred2(3) -= 2. * M_PI;
935  while (pred2(3) < -M_PI)
936  pred2(3) += 2. * M_PI;
937 
938  x_sig.col(i + 1) = pred1;
939  x_sig.col(i + 1 + num_state_) = pred2;
940  }
941 
942  /*****************************************************************************
943  * Predict Sigma Points
944  ****************************************************************************/
945  // predict sigma points
946  for (int i = 0; i < 2 * num_state_ + 1; i++)
947  {
948  // extract values for better readability
949  double p_x = x_sig(0, i);
950  double p_y = x_sig(1, i);
951  double v = x_sig(2, i);
952  double yaw = x_sig(3, i);
953  double yawd = x_sig(4, i);
954 
955  std::vector<double> state(5);
956  if (model_ind == MotionModel::CV)
957  cv(p_x, p_y, v, yaw, yawd, delta_t, state);
958  else if (model_ind == MotionModel::CTRV)
959  ctrv(p_x, p_y, v, yaw, yawd, delta_t, state);
960  else
961  randomMotion(p_x, p_y, v, yaw, yawd, delta_t, state);
962 
963  // write predicted sigma point into right column
964  x_sig_pred(0, i) = state[0];
965  x_sig_pred(1, i) = state[1];
966  x_sig_pred(2, i) = state[2];
967  x_sig_pred(3, i) = state[3];
968  x_sig_pred(4, i) = state[4];
969  }
970 
971  /*****************************************************************************
972  * Convert Predicted Sigma Points to Mean/Covariance
973  ****************************************************************************/
974  // predicted state mean
975  x.fill(0.0);
976  for (int i = 0; i < 2 * num_state_ + 1; i++)
977  { // iterate over sigma points
978  x = x + weights_s_(i) * x_sig_pred.col(i);
979  }
980 
981  while (x(3) > M_PI)
982  x(3) -= 2. * M_PI;
983  while (x(3) < -M_PI)
984  x(3) += 2. * M_PI;
985  // predicted state covariance matrix
986  p.fill(0.0);
987  for (int i = 0; i < 2 * num_state_ + 1; i++)
988  { // iterate over sigma points
989  // state difference
990  Eigen::VectorXd x_diff = x_sig_pred.col(i) - x;
991  // angle normalization
992  while (x_diff(3) > M_PI)
993  x_diff(3) -= 2. * M_PI;
994  while (x_diff(3) < -M_PI)
995  x_diff(3) += 2. * M_PI;
996  p = p + weights_c_(i) * x_diff * x_diff.transpose();
997  }
998 
999  p = p + q;
1000 
1001  /*****************************************************************************
1002  * Update model parameters
1003  ****************************************************************************/
1004  if (model_ind == MotionModel::CV)
1005  {
1006  x_cv_.col(0) = x;
1007  p_cv_ = p;
1008  x_sig_pred_cv_ = x_sig_pred;
1009  }
1010  else if (model_ind == MotionModel::CTRV)
1011  {
1012  x_ctrv_.col(0) = x;
1013  p_ctrv_ = p;
1014  x_sig_pred_ctrv_ = x_sig_pred;
1015  }
1016  else
1017  {
1018  x_rm_.col(0) = x;
1019  p_rm_ = p;
1020  x_sig_pred_rm_ = x_sig_pred;
1021  }
1022 }
