111  const double& velocity, PlannerHNS::WayPoint& pursuite_point, PlannerHNS::WayPoint& prep,
112  double& lateral_err, double& follow_distance)
113 {
114  if(path.size()==0) return false;
115 
116  follow_distance = fabs(velocity) * (m_Params.SteeringDelay+0.5);
117  if(follow_distance < m_Params.minPursuiteDistance)
118  follow_distance = m_Params.minPursuiteDistance;
119 
120  RelativeInfo info;
121  PlanningHelpers::GetRelativeInfo(path, state, info);
122  unsigned int point_index = 0;
123  pursuite_point = PlanningHelpers::GetFollowPointOnTrajectory(path, info, follow_distance, point_index);
124  prep = info.perp_point;
125  lateral_err = info.perp_distance;
126  m_iPrevWayPoint = info.iFront;
127 
128  double d_critical = (-velocity*velocity)/2.0*m_VehicleInfo.max_deceleration;
129  int nPointToEnd = path.size() - m_iPrevWayPoint;
130  double totalD = m_WayPointsDensity*nPointToEnd;
131 
132  if(totalD <= 1 || totalD <= d_critical)
133  {
134  m_bEndPath = true;
135  //cout << "Critical Distance: " << d_critical << endl;
136  }
137  else
138  m_bEndPath = false;

214 int TrajectoryFollower::VeclocityControllerUpdate(const double& dt, const PlannerHNS::VehicleState& CurrStatus,
215  const PlannerHNS::BehaviorState& CurrBehavior, double& desiredVelocity, PlannerHNS::SHIFT_POS& desiredShift)
216 {

224

226 
227  //desiredVelocity = m_PerpendicularPoint.v;
228  }
229  else if(CurrBehavior.state == FORWARD_STATE || CurrBehavior.state == OBSTACLE_AVOIDANCE_STATE )

241  else

243 
244  if(desiredVelocity > CurrBehavior.maxVelocity)
245  desiredVelocity = CurrBehavior.maxVelocity;

255  desiredVelocity = 0;
256  }
257  else if(CurrBehavior.state == FOLLOW_STATE)

269  desiredVelocity = (deceleration_critical * dt) + CurrStatus.speed;
270 
271  if(desiredVelocity > CurrBehavior.maxVelocity)
272  desiredVelocity = CurrBehavior.maxVelocity;
273 
274  if((desiredVelocity < 0.1 && desiredVelocity > -0.1) || CurrBehavior.followDistance <= 0) //use only effective velocities
275  desiredVelocity = 0;
276 
277  //cout << "Follow State: acceleration = " << deceleration_critical << ", speed = " << CurrStatus.speed << ", Distance = " << CurrBehavior.followDistance<< endl;
278  }
279  else
280  {
281  desiredVelocity = 0;
282  }
283 
284  if(desiredVelocity > m_VehicleInfo.max_speed_forward)
285  desiredVelocity = m_VehicleInfo.max_speed_forward;
286  else if (desiredVelocity < 0)
287  desiredVelocity = 0;
288 
289  desiredShift = PlannerHNS::SHIFT_POS_DD;
290  if(m_bEnableLog)
291  m_LogVelocityPIDData.push_back(m_pidVelocity.ToString());
292  return 1;
