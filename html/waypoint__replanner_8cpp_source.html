71 void WaypointReplanner::replanLaneWaypointVel(autoware_msgs::Lane& lane)
72 {
73  if (lane.waypoints.empty())
74  {
75  return;
76  }
77  const int dir = getDirection(lane);
78  const unsigned long last = lane.waypoints.size() - 1;
79  changeVelSign(lane, true);
80  limitVelocityByRange(0, last, 0, velocity_max_, lane);
81  if (resample_mode_)
82  {
83  resampleLaneWaypoint(resample_interval_, lane, dir);
84  }
85  if (replan_curve_mode_)
86  {
87  std::vector<double> curve_radius;
88  KeyVal curve_list;
89  createRadiusList(lane, curve_radius);
90  createCurveList(curve_radius, curve_list);
91  if (overwrite_vmax_mode_)
92  {// set velocity_max for all_point
93  setVelocityByRange(0, last, 0, velocity_max_, lane);
94  }
95  // set velocity by curve
96  for (const auto& el : curve_list)
97  {
98  const double& radius = el.second.second;
99  double vmin = velocity_max_ - vel_param_ * (r_th_ - radius);
100  vmin = (vmin < velocity_min_) ? velocity_min_ : vmin;
101  limitVelocityByRange(el.first, el.second.first, velocity_offset_, vmin, lane);
102  }
103  }
104  // set velocity on start & end of lane
105  if (replan_endpoint_mode_)
106  {
107  const unsigned long zerospeed_start = last - end_point_offset_;
108  const unsigned long lowspeed_start = zerospeed_start - braking_distance_;
109  raiseVelocityByRange(0, last, 0, velocity_min_, lane);
110  limitVelocityByRange(0, 0, 0, velocity_min_, lane);
111  limitVelocityByRange(lowspeed_start, last, 0, velocity_min_, lane);
112  setVelocityByRange(zerospeed_start, last, 0, 0.0, lane);
113  }
114  if (dir < 0)
115  {
116  changeVelSign(lane, false);
117  }
118 }

120 void WaypointReplanner::resampleLaneWaypoint(const double resample_interval, autoware_msgs::Lane& lane, int dir)
121 {
122  if (lane.waypoints.size() < 2)
123  {
124  return;
125  }
126  autoware_msgs::Lane original_lane(lane);
127  lane.waypoints.clear();
128  lane.waypoints.emplace_back(original_lane.waypoints[0]);
129  lane.waypoints.reserve(ceil(1.5 * calcPathLength(original_lane) / resample_interval_));
130 
131  for (unsigned long i = 1; i < original_lane.waypoints.size(); i++)
132  {
133  CbufGPoint curve_point = getCrvPointsOnResample(lane, original_lane, i);
134  const std::vector<double> curve_param = calcCurveParam(curve_point);
135  lane.waypoints.back().twist.twist = original_lane.waypoints[i - 1].twist.twist;
136  lane.waypoints.back().wpstate = original_lane.waypoints[i - 1].wpstate;
137  lane.waypoints.back().change_flag = original_lane.waypoints[i - 1].change_flag;
138  // if going straight
139  if (curve_param.empty())
140  {
141  resampleOnStraight(curve_point, lane);
142  }
143  // else if turnning curve
144  else
145  {
146  resampleOnCurve(curve_point[1], curve_param, lane, dir);
147  }
148  }
149  lane.waypoints[0].pose.pose.orientation = lane.waypoints[1].pose.pose.orientation;
150  lane.waypoints.back().twist.twist = original_lane.waypoints.back().twist.twist;
151  lane.waypoints.back().wpstate = original_lane.waypoints.back().wpstate;
152  lane.waypoints.back().change_flag = original_lane.waypoints.back().change_flag;
153 }

155 void WaypointReplanner::resampleOnStraight(const CbufGPoint& curve_point, autoware_msgs::Lane& lane)
156 {
157  if (curve_point.size() != 3)
158  {
159  return;
160  }
161  autoware_msgs::Waypoint wp(lane.waypoints.back());
162  const geometry_msgs::Point& pt = wp.pose.pose.position;
163  const double yaw = atan2(curve_point[2].y - curve_point[0].y, curve_point[2].x - curve_point[0].x);
164  wp.pose.pose.orientation = tf::createQuaternionMsgFromYaw(yaw);
165 
166  const std::vector<double> nvec = { curve_point[1].x - pt.x, curve_point[1].y - pt.y, curve_point[1].z - pt.z };
167  double dist = sqrt(nvec[0] * nvec[0] + nvec[1] * nvec[1]);
168  std::vector<double> resample_vec = nvec;
169  const double coeff = resample_interval_ / dist;
170  for (auto& el : resample_vec)
171  {
172  el *= coeff;
173  }
174  for (; dist > resample_interval_; dist -= resample_interval_)
175  {
176  wp.pose.pose.position.x += resample_vec[0];
177  wp.pose.pose.position.y += resample_vec[1];
178  wp.pose.pose.position.z += resample_vec[2];
179  lane.waypoints.emplace_back(wp);
180  }
181 }

183 void WaypointReplanner::resampleOnCurve(const geometry_msgs::Point& target_point,
184  const std::vector<double>& curve_param, autoware_msgs::Lane& lane, int dir)
185 {
186  if (curve_param.size() != 3)
187  {
188  return;
189  }
190  autoware_msgs::Waypoint wp(lane.waypoints.back());
191  const double& cx = curve_param[0];
192  const double& cy = curve_param[1];
193  const double& radius = curve_param[2];
194  const double reverse_angle = (dir < 0) ? M_PI : 0.0;
195 
196  const geometry_msgs::Point& p0 = wp.pose.pose.position;
197  const geometry_msgs::Point& p1 = target_point;
198  double theta = fmod(atan2(p1.y - cy, p1.x - cx) - atan2(p0.y - cy, p0.x - cx), 2 * M_PI);
199  int sgn = (theta > 0.0) ? (1) : (-1);
200  if (fabs(theta) > M_PI)
201  {
202  theta -= 2 * sgn * M_PI;
203  }
204  sgn = (theta > 0.0) ? (1) : (-1);
205  // interport
206  double t = atan2(p0.y - cy, p0.x - cx);
207  double dist = radius * fabs(theta);
208  const double resample_dz = resample_interval_ * (p1.z - p0.z) / dist;
209  for (; dist > resample_interval_; dist -= resample_interval_)
210  {
211  if (lane.waypoints.size() == lane.waypoints.capacity())
212  {
213  break;
214  }
215  t += sgn * resample_interval_ / radius;
216  const double yaw = fmod(t + sgn * M_PI / 2.0, 2 * M_PI) + reverse_angle;
217  wp.pose.pose.position.x = cx + radius * cos(t);
218  wp.pose.pose.position.y = cy + radius * sin(t);
219  wp.pose.pose.position.z += resample_dz;
220  wp.pose.pose.orientation = tf::createQuaternionMsgFromYaw(yaw);
221  lane.waypoints.emplace_back(wp);
222  }
223 }

295 void WaypointReplanner::createCurveList(const std::vector<double>& curve_radius, KeyVal& curve_list)
296 {
297  unsigned long index = 0;
298  bool on_curve = false;
299  double radius_localmin = DBL_MAX;
300  for (unsigned long i = 1; i < curve_radius.size(); i++)
301  {
302  if (!on_curve && curve_radius[i] <= r_th_ && curve_radius[i - 1] > r_th_)
303  {
304  index = i;
305  on_curve = true;
306  }
307  else if (on_curve && curve_radius[i - 1] <= r_th_ && curve_radius[i] > r_th_)
308  {
309  on_curve = false;
310  if (radius_localmin < r_min_)
311  {
312  radius_localmin = r_min_;
313  }
314  curve_list[index] = std::make_pair(i, radius_localmin);
315  radius_localmin = DBL_MAX;
316  }
317  if (!on_curve)
318  {
319  continue;
320  }
321  if (radius_localmin > curve_radius[i])
322  {
323  radius_localmin = curve_radius[i];
324  }
325  }

396 void WaypointReplanner::limitAccelDecel(const unsigned long idx, autoware_msgs::Lane& lane)
397 {
398  const double acc[2] = { accel_limit_, decel_limit_ };
399  const unsigned long end_idx[2] = { lane.waypoints.size() - idx, idx + 1 };
400  const int sgn[2] = { 1, -1 };
401  for (int j = 0; j < 2; j++) // [j=0]: accel_limit_process, [j=1]: decel_limit_process
402  {
403  double v = lane.waypoints[idx].twist.twist.linear.x;
404  unsigned long next = idx + sgn[j];
405  for (unsigned long i = 1; i < end_idx[j]; i++, next += sgn[j])
406  {
407  const geometry_msgs::Point& p0 = lane.waypoints[next - sgn[j]].pose.pose.position;
408  const geometry_msgs::Point& p1 = lane.waypoints[next].pose.pose.position;
409  const double dist = std::hypot(p0.x - p1.x, p0.y - p1.y);
410  v = sqrt(2 * acc[j] * dist + v * v);
411  if (v > velocity_max_ || v > lane.waypoints[next].twist.twist.linear.x)
412  {
413  break;
414  }
415  lane.waypoints[next].twist.twist.linear.x = v;
416  }
417  }
418 }

420 // get curve 3-Parameter [center_x, center_y, radius] with 3 point input. If error occured, return empty vector.
421 const std::vector<double> WaypointReplanner::calcCurveParam(CbufGPoint p) const
422 {
423  for (int i = 0; i < 3; i++, p.push_back(p.front())) // if exception occured, change points order
424  {
425  const double d = 2 * ((p[0].y - p[2].y) * (p[0].x - p[1].x) - (p[0].y - p[1].y) * (p[0].x - p[2].x));
426  if (fabs(d) < 1e-8)
427  {
428  continue;
429  }
430  const std::vector<double> x2 = { p[0].x * p[0].x, p[1].x * p[1].x, p[2].x * p[2].x };
431  const std::vector<double> y2 = { p[0].y * p[0].y, p[1].y * p[1].y, p[2].y * p[2].y };
432  const double a = y2[0] - y2[1] + x2[0] - x2[1];
433  const double b = y2[0] - y2[2] + x2[0] - x2[2];
434  std::vector<double> param(3);
435  const double cx = param[0] = ((p[0].y - p[2].y) * a - (p[0].y - p[1].y) * b) / d;
436  const double cy = param[1] = ((p[0].x - p[2].x) * a - (p[0].x - p[1].x) * b) / -d;
437  param[2] = sqrt((cx - p[0].x) * (cx - p[0].x) + (cy - p[0].y) * (cy - p[0].y));
438  return param;
439  }
440  return std::vector<double>(); // error
441 }
