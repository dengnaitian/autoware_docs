127 void LaneSelectNode::processing()
128 {
129  if (!isAllTopicsSubscribed())
130  return;
131 
132  // search closest waypoint number for each lanes
133  if (!getClosestWaypointNumberForEachLanes())
134  {
135  publishClosestWaypoint(-1);
136  publishVehicleLocation(-1, lane_array_id_);
137  resetLaneIdx();
138  return;
139  }
140 
141  // if closest waypoint on current lane is -1,
142  if (std::get<1>(tuple_vec_.at(current_lane_idx_)) == -1)
143  {
144  publishClosestWaypoint(-1);
145  publishVehicleLocation(-1, lane_array_id_);
146  resetLaneIdx();
147  return;
148  }
149 
150  findNeighborLanes();
151  ROS_INFO("current_lane_idx: %d", current_lane_idx_);
152  ROS_INFO("right_lane_idx: %d", right_lane_idx_);
153  ROS_INFO("left_lane_idx: %d", left_lane_idx_);
154 
155  if (current_state_ == "LANE_CHANGE")
156  {
157  try
158  {
159  changeLane();
160  std::get<1>(lane_for_change_) =
161  getClosestWaypointNumber(std::get<0>(lane_for_change_), current_pose_.pose, current_velocity_.twist,
162  std::get<1>(lane_for_change_), distance_threshold_);
163  std::get<2>(lane_for_change_) = static_cast<ChangeFlag>(
164  std::get<0>(lane_for_change_).waypoints.at(std::get<1>(lane_for_change_)).change_flag);
165  ROS_INFO("closest: %d", std::get<1>(lane_for_change_));
166  publishLane(std::get<0>(lane_for_change_));
167  publishLaneID(std::get<0>(lane_for_change_));
168  publishClosestWaypoint(std::get<1>(lane_for_change_));
169  publishChangeFlag(std::get<2>(lane_for_change_));
170  publishVehicleLocation(std::get<1>(lane_for_change_), lane_array_id_);
171  }
172  catch (std::out_of_range)
173  {
174  ROS_WARN("Failed to get closest waypoint num\n");
175  }
176  }
177  else
178  {
179  updateChangeFlag();
180  createLaneForChange();
181 
182  publishLane(std::get<0>(tuple_vec_.at(current_lane_idx_)));
183  publishClosestWaypoint(std::get<1>(tuple_vec_.at(current_lane_idx_)));
184  publishChangeFlag(std::get<2>(tuple_vec_.at(current_lane_idx_)));
185  publishVehicleLocation(std::get<1>(tuple_vec_.at(current_lane_idx_)), lane_array_id_);
186  }
187  publishVisualizer();
188  resetSubscriptionFlag();
189 }

205 void LaneSelectNode::createLaneForChange()
206 {
207  std::get<0>(lane_for_change_).waypoints.clear();
208  std::get<0>(lane_for_change_).waypoints.shrink_to_fit();
209  std::get<1>(lane_for_change_) = -1;
210 
211  const autoware_msgs::Lane &cur_lane = std::get<0>(tuple_vec_.at(current_lane_idx_));
212  const int32_t &clst_wp = std::get<1>(tuple_vec_.at(current_lane_idx_));
213 
214  int32_t num_lane_change = getClosestLaneChangeWaypointNumber(cur_lane.waypoints, clst_wp);
215  ROS_INFO("num_lane_change: %d", num_lane_change);
216  if (num_lane_change < 0 || num_lane_change >= static_cast<int32_t>(cur_lane.waypoints.size()))
217  {
218  ROS_WARN("current lane doesn't have change flag");
219  return;
220  }
221 
222  if ((static_cast<ChangeFlag>(cur_lane.waypoints.at(num_lane_change).change_flag) == ChangeFlag::right &&
223  right_lane_idx_ < 0) ||
224  (static_cast<ChangeFlag>(cur_lane.waypoints.at(num_lane_change).change_flag) == ChangeFlag::left &&
225  left_lane_idx_ < 0))
226  {
227  ROS_WARN("current lane doesn't have the lane for lane change");
228  return;
229  }
230 
231  double dt = getTwoDimensionalDistance(cur_lane.waypoints.at(num_lane_change).pose.pose.position,
232  cur_lane.waypoints.at(clst_wp).pose.pose.position);
233  double dt_by_vel = current_velocity_.twist.linear.x * lane_change_target_ratio_ > lane_change_target_minimum_ ?
234  current_velocity_.twist.linear.x * lane_change_target_ratio_ :
235  lane_change_target_minimum_;
236  ROS_INFO("dt : %lf, dt_by_vel : %lf", dt, dt_by_vel);
237  autoware_msgs::Lane &nghbr_lane =
238  static_cast<ChangeFlag>(cur_lane.waypoints.at(num_lane_change).change_flag) == ChangeFlag::right ?
239  std::get<0>(tuple_vec_.at(right_lane_idx_)) :
240  std::get<0>(tuple_vec_.at(left_lane_idx_));
241  const int32_t &nghbr_clst_wp =
242  static_cast<ChangeFlag>(cur_lane.waypoints.at(num_lane_change).change_flag) == ChangeFlag::right ?
243  std::get<1>(tuple_vec_.at(right_lane_idx_)) :
244  std::get<1>(tuple_vec_.at(left_lane_idx_));
245 
246  int32_t target_num = -1;
247  for (uint32_t i = nghbr_clst_wp; i < nghbr_lane.waypoints.size(); i++)
248  {
249  if (i == nghbr_lane.waypoints.size() - 1 ||
250  dt + dt_by_vel < getTwoDimensionalDistance(nghbr_lane.waypoints.at(nghbr_clst_wp).pose.pose.position,
251  nghbr_lane.waypoints.at(i).pose.pose.position))
252  {
253  target_num = i;
254  break;
255  }
256  }
257 
258  ROS_INFO("target_num : %d", target_num);
259  if (target_num < 0)
260  return;
261 
262  std::get<0>(lane_for_change_).header.stamp = nghbr_lane.header.stamp;
263  std::vector<autoware_msgs::Waypoint> hermite_wps = generateHermiteCurveForROS(
264  cur_lane.waypoints.at(num_lane_change).pose.pose, nghbr_lane.waypoints.at(target_num).pose.pose,
265  cur_lane.waypoints.at(num_lane_change).twist.twist.linear.x, vlength_hermite_curve_);
266 
267  for (auto &&el : hermite_wps)
268  el.change_flag = cur_lane.waypoints.at(num_lane_change).change_flag;
269 
270  std::get<0>(lane_for_change_).waypoints.reserve(nghbr_lane.waypoints.size() + hermite_wps.size());
271  std::move(hermite_wps.begin(), hermite_wps.end(), std::back_inserter(std::get<0>(lane_for_change_).waypoints));
272  auto itr = nghbr_lane.waypoints.begin();
273  std::advance(itr, target_num);
274  for (auto i = itr; i != nghbr_lane.waypoints.end(); i++)
275  {
276  if (getTwoDimensionalDistance(itr->pose.pose.position, i->pose.pose.position) < lane_change_interval_)
277  i->change_flag = enumToInteger(ChangeFlag::straight);
278  else
279  break;
280  }
281  std::copy(itr, nghbr_lane.waypoints.end(), std::back_inserter(std::get<0>(lane_for_change_).waypoints));
282 }

284 void LaneSelectNode::updateChangeFlag()
285 {
286  for (auto &el : tuple_vec_)
287  {
288  std::get<2>(el) = (std::get<1>(el) != -1) ?
289  static_cast<ChangeFlag>(std::get<0>(el).waypoints.at(std::get<1>(el)).change_flag) :
290  ChangeFlag::unknown;
291 
292  if (std::get<2>(el) == ChangeFlag::right && right_lane_idx_ == -1)
293  std::get<2>(el) = ChangeFlag::unknown;
294  else if (std::get<2>(el) == ChangeFlag::left && left_lane_idx_ == -1)
295  std::get<2>(el) = ChangeFlag::unknown;
296 
297  ROS_INFO("change_flag: %d", enumToInteger(std::get<2>(el)));
298  }
299 }

301 void LaneSelectNode::changeLane()
302 {
303  if (std::get<2>(tuple_vec_.at(current_lane_idx_)) == ChangeFlag::right && right_lane_idx_ != -1 &&
304  std::get<1>(tuple_vec_.at(right_lane_idx_)) != -1)
305  {
306  current_lane_idx_ = right_lane_idx_;
307  }
308  else if (std::get<2>(tuple_vec_.at(current_lane_idx_)) == ChangeFlag::left && left_lane_idx_ != -1 &&
309  std::get<1>(tuple_vec_.at(left_lane_idx_)) != -1)
310  {
311  current_lane_idx_ = left_lane_idx_;
312  }
313 
314  findNeighborLanes();
315  return;
316 }

318 bool LaneSelectNode::getClosestWaypointNumberForEachLanes()
319 {
320  for (auto &el : tuple_vec_)
321  {
322  std::get<1>(el) = getClosestWaypointNumber(std::get<0>(el), current_pose_.pose, current_velocity_.twist,
323  std::get<1>(el), distance_threshold_);
324  ROS_INFO("closest: %d", std::get<1>(el));
325  }
326 
327  // confirm if all closest waypoint numbers are -1. If so, output warning
328  int32_t accum = 0;
329  for (const auto &el : tuple_vec_)
330  {
331  accum += std::get<1>(el);
332  }
333  if (accum == (-1) * static_cast<int32_t>(tuple_vec_.size()))
334  {
335  ROS_WARN("Cannot get closest waypoints. All closest waypoints are changed to -1...");
336  return false;
337  }
338 
339  return true;

355 int32_t LaneSelectNode::findMostClosestLane(const std::vector<uint32_t> idx_vec, const geometry_msgs::Point p)
356 {
357  std::vector<double> dist_vec;
358  dist_vec.reserve(idx_vec.size());
359  for (const auto &el : idx_vec)
360  {
361  int32_t closest_number = std::get<1>(tuple_vec_.at(el));
362  dist_vec.push_back(
363  getTwoDimensionalDistance(p, std::get<0>(tuple_vec_.at(el)).waypoints.at(closest_number).pose.pose.position));
364  }
365  std::vector<double>::iterator itr = std::min_element(dist_vec.begin(), dist_vec.end());
366  return idx_vec.at(std::distance(dist_vec.begin(), itr));
367 }

369 void LaneSelectNode::findNeighborLanes()
370 {
371  int32_t current_closest_num = std::get<1>(tuple_vec_.at(current_lane_idx_));
372  const geometry_msgs::Pose &current_closest_pose =
373  std::get<0>(tuple_vec_.at(current_lane_idx_)).waypoints.at(current_closest_num).pose.pose;
374 
375  std::vector<uint32_t> left_lane_idx_vec;
376  left_lane_idx_vec.reserve(tuple_vec_.size());
377  std::vector<uint32_t> right_lane_idx_vec;
378  right_lane_idx_vec.reserve(tuple_vec_.size());
379  for (uint32_t i = 0; i < tuple_vec_.size(); i++)
380  {
381  if (i == static_cast<uint32_t>(current_lane_idx_) || std::get<1>(tuple_vec_.at(i)) == -1)
382  continue;
383 
384  int32_t target_num = std::get<1>(tuple_vec_.at(i));
385  const geometry_msgs::Point &target_p = std::get<0>(tuple_vec_.at(i)).waypoints.at(target_num).pose.pose.position;
386 
387  geometry_msgs::Point converted_p = convertPointIntoRelativeCoordinate(target_p, current_closest_pose);
388 
389  ROS_INFO("distance: %lf", converted_p.y);
390  if (fabs(converted_p.y) > distance_threshold_)
391  {
392  ROS_INFO("%d lane is far from current lane...", i);
393  continue;
394  }
395 
396  if (converted_p.y > 0)
397  left_lane_idx_vec.push_back(i);
398  else
399  right_lane_idx_vec.push_back(i);
400  }
401 
402  if (!left_lane_idx_vec.empty())
403  left_lane_idx_ = findMostClosestLane(left_lane_idx_vec, current_closest_pose.position);
404  else
405  left_lane_idx_ = -1;
406 
407  if (!right_lane_idx_vec.empty())
408  right_lane_idx_ = findMostClosestLane(right_lane_idx_vec, current_closest_pose.position);
409  else
410  right_lane_idx_ = -1;
411 }

772 // get closest waypoint from current pose
773 int32_t getClosestWaypointNumber(const autoware_msgs::Lane &current_lane, const geometry_msgs::Pose &current_pose,
774  const geometry_msgs::Twist &current_velocity, const int32_t previous_number,
775  const double distance_threshold)
776 {
777  if (current_lane.waypoints.empty())
778  return -1;
779 
780  std::vector<uint32_t> idx_vec;
781  // if previous number is -1, search closest waypoint from waypoints in front of current pose
782  if (previous_number == -1)
783  {
784  idx_vec.reserve(current_lane.waypoints.size());
785  for (uint32_t i = 0; i < current_lane.waypoints.size(); i++)
786  {
787  geometry_msgs::Point converted_p =
788  convertPointIntoRelativeCoordinate(current_lane.waypoints.at(i).pose.pose.position, current_pose);
789  double angle = getRelativeAngle(current_lane.waypoints.at(i).pose.pose, current_pose);
790  if (converted_p.x > 0 && angle < 90)
791  idx_vec.push_back(i);
792  }
793  }
794  else
795  {
796  if (distance_threshold <
797  getTwoDimensionalDistance(current_lane.waypoints.at(previous_number).pose.pose.position, current_pose.position))
798  {
799  ROS_WARN("Current_pose is far away from previous closest waypoint. Initilized...");
800  return -1;
801  }
802 
803  double ratio = 3;
804  double minimum_dt = 2.0;
805  double dt = current_velocity.linear.x * ratio > minimum_dt ? current_velocity.linear.x * ratio : minimum_dt;
806 
807  auto range_max = static_cast<uint32_t>(previous_number + dt) < current_lane.waypoints.size() ?
808  static_cast<uint32_t>(previous_number + dt) :
809  current_lane.waypoints.size();
810  for (uint32_t i = static_cast<uint32_t>(previous_number); i < range_max; i++)
811  {
812  geometry_msgs::Point converted_p =
813  convertPointIntoRelativeCoordinate(current_lane.waypoints.at(i).pose.pose.position, current_pose);
814  double angle = getRelativeAngle(current_lane.waypoints.at(i).pose.pose, current_pose);
815  if (converted_p.x > 0 && angle < 90)
816  idx_vec.push_back(i);
817  }
818  }
819 
820  if (idx_vec.empty())
821  return -1;
822 
823  std::vector<double> dist_vec;
824  dist_vec.reserve(idx_vec.size());
825  for (const auto &el : idx_vec)
826  {
827  double dt = getTwoDimensionalDistance(current_pose.position, current_lane.waypoints.at(el).pose.pose.position);
828  dist_vec.push_back(dt);
829  }
830  std::vector<double>::iterator itr = std::min_element(dist_vec.begin(), dist_vec.end());
831  int32_t found_number = idx_vec.at(static_cast<uint32_t>(std::distance(dist_vec.begin(), itr)));
832  return found_number;
833 }
