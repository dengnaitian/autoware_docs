143  feature_generator_->generate(pc_ptr);
144 
145 // network forward process
146  caffe_net_->Forward();
147 #ifndef USE_CAFFE_GPU
148 // caffe::Caffe::set_mode(caffe::Caffe::CPU);
149 #else
150 // int gpu_id = 0;

152 // caffe::Caffe::set_mode(caffe::Caffe::GPU);
153 // caffe::Caffe::DeviceQuery();
154 #endif
155 
156  // clutser points and construct segments/objects
157  float objectness_thresh = 0.5;
158  bool use_all_grids_for_clustering = true;
159  cluster2d_->cluster(*category_pt_blob_, *instance_pt_blob_, pc_ptr,
160  valid_idx, objectness_thresh,
161  use_all_grids_for_clustering);
162  cluster2d_->filter(*confidence_pt_blob_, *height_pt_blob_);
163  cluster2d_->classify(*class_pt_blob_);
