253  std::vector<std::vector<PlannerHNS::WayPoint> >& generatedTotalPaths)

260 
261  if (ret == 0)
262  {
263  std::cout << "Can't Generate Global Path for Start (" << startPoint.pos.ToString() << ") and Goal ("

266  }
267

276  }
277  }
278 
279  for (unsigned int i = 0; i < generatedTotalPaths.size(); i++)
280  {
281  PlannerHNS::PlanningHelpers::CalcAngleAndCost(generatedTotalPaths.at(i));
282  if (m_GlobalPathID > 10000)
283  m_GlobalPathID = 1;
284 
285  for (unsigned int j = 0; j < generatedTotalPaths.at(i).size(); j++)
286  generatedTotalPaths.at(i).at(j).gid = m_GlobalPathID;

289 
290  std::cout << "New DP Path -> " << generatedTotalPaths.at(i).size() << std::endl;
291  }

296  std::cout << "Can't Generate Global Path for Start (" << startPoint.pos.ToString() << ") and Goal ("

298  }
