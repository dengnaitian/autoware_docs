53 Eigen::MatrixXd ObjectsToCostmap::makeRectanglePoints(const autoware_msgs::DetectedObject& in_object,
54  const double expand_rectangle_size)
55 {
56  double length = in_object.dimensions.x + expand_rectangle_size;
57  double width = in_object.dimensions.y + expand_rectangle_size;
58  Eigen::MatrixXd origin_points(NUMBER_OF_DIMENSIONS, NUMBER_OF_POINTS);
59  origin_points << length / 2, length / 2, -length / 2, -length / 2, width / 2, -width / 2, -width / 2, width / 2;
60 
61  double yaw = tf::getYaw(in_object.pose.orientation);
62  Eigen::MatrixXd rotation_matrix(NUMBER_OF_DIMENSIONS, NUMBER_OF_DIMENSIONS);
63  rotation_matrix << std::cos(yaw), -std::sin(yaw), std::sin(yaw), std::cos(yaw);
64  Eigen::MatrixXd rotated_points = rotation_matrix * origin_points;
65 
66  double dx = in_object.pose.position.x;
67  double dy = in_object.pose.position.y;
68  Eigen::MatrixXd transformed_points(NUMBER_OF_DIMENSIONS, NUMBER_OF_POINTS);
69  Eigen::MatrixXd ones = Eigen::MatrixXd::Ones(1, NUMBER_OF_POINTS);
70  transformed_points.row(0) = rotated_points.row(0) + dx * ones;
71  transformed_points.row(1) = rotated_points.row(1) + dy * ones;
72  return transformed_points;
73 }

88 geometry_msgs::Point ObjectsToCostmap::makeExpandedPoint(const geometry_msgs::Point& in_centroid,
89  const geometry_msgs::Point32& in_corner_point,
90  const double expand_polygon_size)
91 {
92  geometry_msgs::Point expanded_point;
93  if(expand_polygon_size == 0)
94  {
95  expanded_point.x = in_corner_point.x;
96  expanded_point.y = in_corner_point.y;
97  return expanded_point;
98  }
99  double theta = std::atan2(in_corner_point.y - in_centroid.y, in_corner_point.x - in_centroid.x);
100  double delta_x = expand_polygon_size * std::cos(theta);
101  double delta_y = expand_polygon_size * std::sin(theta);
102  expanded_point.x = in_corner_point.x + delta_x;
103  expanded_point.y = in_corner_point.y + delta_y;
104  return expanded_point;
105 }

108 grid_map::Polygon ObjectsToCostmap::makePolygonFromObjectConvexHull(const autoware_msgs::DetectedObject& in_object,
109  const double expand_polygon_size)
110 {
111  grid_map::Polygon polygon;
112  polygon.setFrameId(in_object.header.frame_id);
113 
114  double initial_z = in_object.convex_hull.polygon.points[0].z;
115  for (size_t index = 0; index < in_object.convex_hull.polygon.points.size(); index++)
116  {
117  if(in_object.convex_hull.polygon.points[index].z == initial_z)
118  {
119  geometry_msgs::Point centroid = in_object.pose.position;
120  geometry_msgs::Point expanded_point = makeExpandedPoint(centroid,
121  in_object.convex_hull.polygon.points[index], expand_polygon_size);
122  polygon.addVertex(grid_map::Position(expanded_point.x, expanded_point.y));
123  }
124  }
125  return polygon;
126 }

142 grid_map::Matrix ObjectsToCostmap::makeCostmapFromObjects(const grid_map::GridMap& costmap,
143  const double expand_polygon_size,
144  const double size_of_expansion_kernel,
145  const autoware_msgs::DetectedObjectArray::ConstPtr& in_objects,
146  const bool use_objects_convex_hull)
147 {
148  grid_map::GridMap objects_costmap = costmap;
149  objects_costmap.add(OBJECTS_COSTMAP_LAYER_, 0);
150  objects_costmap.add(EXPANDED_OBJECTS_COSTMAP_LAYER_, 0);
151 
152  const double not_expand_polygon_size = 0;
153  for (const auto& object : in_objects->objects)
154  {
155  grid_map::Polygon polygon, expanded_polygon;
156  if(use_objects_convex_hull)
157  {
158  polygon = makePolygonFromObjectConvexHull(object, not_expand_polygon_size);
159  expanded_polygon = makePolygonFromObjectConvexHull(object, expand_polygon_size);
160  }
161  else
162  {
163  polygon = makePolygonFromObjectBox(object, not_expand_polygon_size);
164  expanded_polygon = makePolygonFromObjectBox(object, expand_polygon_size);
165  }
166  setCostInPolygon(polygon, OBJECTS_COSTMAP_LAYER_, object.score, objects_costmap);
167  setCostInPolygon(expanded_polygon, EXPANDED_OBJECTS_COSTMAP_LAYER_, object.score, objects_costmap);
168  }
169  // Applying mean filter to expanded gridmap
170  const grid_map::SlidingWindowIterator::EdgeHandling edge_handling =
171  grid_map::SlidingWindowIterator::EdgeHandling::CROP;
172  for (grid_map::SlidingWindowIterator iterator(objects_costmap, EXPANDED_OBJECTS_COSTMAP_LAYER_, edge_handling,
173  size_of_expansion_kernel);
174  !iterator.isPastEnd(); ++iterator)
175  {
176  objects_costmap.at(EXPANDED_OBJECTS_COSTMAP_LAYER_, *iterator) =
177  iterator.getData().meanOfFinites(); // Blurring.
178  }
179 
180  objects_costmap[OBJECTS_COSTMAP_LAYER_] =
181  objects_costmap[OBJECTS_COSTMAP_LAYER_].cwiseMax(objects_costmap[EXPANDED_OBJECTS_COSTMAP_LAYER_]);
182 
183  return objects_costmap[OBJECTS_COSTMAP_LAYER_];
184 }
