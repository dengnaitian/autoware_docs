346 visualization_msgs::MarkerArray
347 VisualizeDetectedObjects::ObjectsToArrows(const autoware_msgs::DetectedObjectArray &in_objects)
348 {
349  visualization_msgs::MarkerArray arrow_markers;
350  for (auto const &object: in_objects.objects)
351  {
352  if (IsObjectValid(object) && object.pose_reliable)
353  {
354  double velocity = object.velocity.linear.x;
355 
356  if (abs(velocity) >= arrow_speed_threshold_)
357  {
358  visualization_msgs::Marker arrow_marker;
359  arrow_marker.lifetime = ros::Duration(marker_display_duration_);
360 
361  tf::Quaternion q(object.pose.orientation.x,
362  object.pose.orientation.y,
363  object.pose.orientation.z,
364  object.pose.orientation.w);
365  double roll, pitch, yaw;
366 
367  tf::Matrix3x3(q).getRPY(roll, pitch, yaw);
368 
369  // in the case motion model fit opposite direction
370  if (velocity < -0.1)
371  {
372  yaw += M_PI;
373  // normalize angle
374  while (yaw > M_PI)
375  yaw -= 2. * M_PI;
376  while (yaw < -M_PI)
377  yaw += 2. * M_PI;
378  }
379 
380  tf::Matrix3x3 obs_mat;
381  tf::Quaternion q_tf;
382 
383  obs_mat.setEulerYPR(yaw, 0, 0); // yaw, pitch, roll
384  obs_mat.getRotation(q_tf);
385 
386  arrow_marker.header = in_objects.header;
387  arrow_marker.ns = ros_namespace_ + "/arrow_markers";
388  arrow_marker.action = visualization_msgs::Marker::ADD;
389  arrow_marker.type = visualization_msgs::Marker::ARROW;
390 
391  // green
392  if (object.color.a == 0)
393  {
394  arrow_marker.color = arrow_color_;
395  }
396  else
397  {
398  arrow_marker.color = object.color;
399  }
400  arrow_marker.id = marker_id_++;
401 
402  // Set the pose of the marker. This is a full 6DOF pose relative to the frame/time specified in the header
403  arrow_marker.pose.position.x = object.pose.position.x;
404  arrow_marker.pose.position.y = object.pose.position.y;
405  arrow_marker.pose.position.z = arrow_height_;
406 
407  arrow_marker.pose.orientation.x = q_tf.getX();
408  arrow_marker.pose.orientation.y = q_tf.getY();
409  arrow_marker.pose.orientation.z = q_tf.getZ();
410  arrow_marker.pose.orientation.w = q_tf.getW();
411 
412  // Set the scale of the arrow -- 1x1x1 here means 1m on a side
413  arrow_marker.scale.x = 3;
414  arrow_marker.scale.y = 0.1;
415  arrow_marker.scale.z = 0.1;
416 
417  arrow_markers.markers.push_back(arrow_marker);
418  }//velocity threshold
419  }//valid object
420  }//end for
421  return arrow_markers;
422 }//ObjectsToArrows

424 visualization_msgs::MarkerArray
425 VisualizeDetectedObjects::ObjectsToLabels(const autoware_msgs::DetectedObjectArray &in_objects)
426 {
427  visualization_msgs::MarkerArray label_markers;
428  for (auto const &object: in_objects.objects)
429  {
430  if (IsObjectValid(object))
431  {
432  visualization_msgs::Marker label_marker;
433 
434  label_marker.lifetime = ros::Duration(marker_display_duration_);
435  label_marker.header = in_objects.header;
436  label_marker.ns = ros_namespace_ + "/label_markers";
437  label_marker.action = visualization_msgs::Marker::ADD;
438  label_marker.type = visualization_msgs::Marker::TEXT_VIEW_FACING;
439  label_marker.scale.x = 1.5;
440  label_marker.scale.y = 1.5;
441  label_marker.scale.z = 1.5;
442 
443  label_marker.color = label_color_;
444 
445  label_marker.id = marker_id_++;
446 
447  if(!object.label.empty() && object.label != "unknown")
448  label_marker.text = object.label + " "; //Object Class if available
449 
450  std::stringstream distance_stream;
451  distance_stream << std::fixed << std::setprecision(1)
452  << sqrt((object.pose.position.x * object.pose.position.x) +
453  (object.pose.position.y * object.pose.position.y));
454  std::string distance_str = distance_stream.str() + " m";
455  label_marker.text += distance_str;
456 
457  if (object.velocity_reliable)
458  {
459  double velocity = object.velocity.linear.x;
460  if (velocity < -0.1)
461  {
462  velocity *= -1;
463  }
464 
465  if (abs(velocity) < object_speed_threshold_)
466  {
467  velocity = 0.0;
468  }
469 
470  tf::Quaternion q(object.pose.orientation.x, object.pose.orientation.y,
471  object.pose.orientation.z, object.pose.orientation.w);
472 
473  double roll, pitch, yaw;
474  tf::Matrix3x3(q).getRPY(roll, pitch, yaw);
475 
476  // convert m/s to km/h
477  std::stringstream kmh_velocity_stream;
478  kmh_velocity_stream << std::fixed << std::setprecision(1) << (velocity * 3.6);
479  std::string text = "\n<" + std::to_string(object.id) + "> " + kmh_velocity_stream.str() + " km/h";
480  label_marker.text += text;
481  }
482 
483  label_marker.pose.position.x = object.pose.position.x;
484  label_marker.pose.position.y = object.pose.position.y;
485  label_marker.pose.position.z = label_height_;
486  label_marker.scale.z = 1.0;
487  if (!label_marker.text.empty())
488  label_markers.markers.push_back(label_marker);
489  }
490  } // end in_objects.objects loop
491 
492  return label_markers;
493 }//ObjectsToLabels
