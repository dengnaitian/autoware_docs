115 template <class T, class U>
116 U createObjectArray(const std::string& file_path)
117 {
118  U obj_array;
119  // NOTE: Autoware want to use map messages with or without /use_sim_time.
120  // Therefore we don't set obj_array.header.stamp.
121  // obj_array.header.stamp = ros::Time::now();
122  obj_array.header.frame_id = "map";
123  obj_array.data = vector_map::parse<T>(file_path);
124  return obj_array;
125 }

484  visualization_msgs::MarkerArray marker_array;

499  }
500 
501  Pole pole;

504  pole = vmap.findByKey(Key<Pole>(signal.plid));

509  }
510  }
511 
512  visualization_msgs::Marker vector_marker;
