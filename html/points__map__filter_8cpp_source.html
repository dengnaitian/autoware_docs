58 void points_map_filter::run() {
59  std::lock_guard<std::mutex> lock(mtx_);
60  map_pub_ =
61  nh_.advertise<sensor_msgs::PointCloud2>("/points_map/filtered", 10);
62  map_sub_ =
63  nh_.subscribe("/points_map", 1, &points_map_filter::map_callback_, this);
64  pose_sub_ = nh_.subscribe("/current_pose", 1,
65  &points_map_filter::current_pose_callback_, this);

80 void points_map_filter::current_pose_callback_(
81  const geometry_msgs::PoseStamped::ConstPtr msg) {
82  std::lock_guard<std::mutex> lock(mtx_);
83  ROS_INFO_STREAM("pose received");
84  if (!last_load_pose_ && map_recieved_) {
85  last_load_pose_ = *msg;
86  pcl::PointCloud<pcl::PointXYZ>::Ptr cloud_filtered(
87  new pcl::PointCloud<pcl::PointXYZ>);
88  pass_.setInputCloud(map_cloud_);
89  pass_.setFilterFieldName("x");
90  pass_.setFilterLimits(
91  last_load_pose_.get().pose.position.x - (load_grid_size_ / 2),
92  last_load_pose_.get().pose.position.x + (load_grid_size_ / 2));
93  pass_.filter(*cloud_filtered);
94  pass_.setInputCloud(cloud_filtered);
95  pass_.setFilterFieldName("y");
96  pass_.setFilterLimits(
97  last_load_pose_.get().pose.position.y - (load_grid_size_ / 2),
98  last_load_pose_.get().pose.position.y + (load_grid_size_ / 2));
99  pass_.filter(*cloud_filtered);
100  sensor_msgs::PointCloud2 msg;
101  pcl::toROSMsg(*cloud_filtered, msg);
102  ROS_INFO_STREAM("update map");
103  map_pub_.publish(msg);
104  } else if (last_load_pose_ && map_recieved_) {
105  double dist = std::sqrt(
106  std::pow(last_load_pose_.get().pose.position.x - msg->pose.position.x,
107  2) +
108  std::pow(last_load_pose_.get().pose.position.y - msg->pose.position.y,
109  2));
110  if (dist > load_trigger_distance_) {
111  last_load_pose_ = *msg;
112  pcl::PointCloud<pcl::PointXYZ>::Ptr cloud_filtered(
113  new pcl::PointCloud<pcl::PointXYZ>);
114  pass_.setInputCloud(map_cloud_);
115  pass_.setFilterFieldName("x");
116  pass_.setFilterLimits(
117  last_load_pose_.get().pose.position.x - (load_grid_size_ / 2),
118  last_load_pose_.get().pose.position.x + (load_grid_size_ / 2));
119  pass_.filter(*cloud_filtered);
120  pass_.setInputCloud(cloud_filtered);
121  pass_.setFilterFieldName("y");
122  pass_.setFilterLimits(
123  last_load_pose_.get().pose.position.y - (load_grid_size_ / 2),
124  last_load_pose_.get().pose.position.y + (load_grid_size_ / 2));
125  pass_.filter(*cloud_filtered);
126  sensor_msgs::PointCloud2 msg;
127  pcl::toROSMsg(*cloud_filtered, msg);
128  map_pub_.publish(msg);
129  ROS_INFO_STREAM("update map");
130  }
131  }
132  return;
133 }
