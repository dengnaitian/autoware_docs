50 int adxl345::Imu::open_device(const std::string device)
51 {
52  struct termios config;
53 
54  fd_ = open(device.c_str(), O_RDWR | O_NOCTTY);
55  if (fd_ < 0)
56  {
57  return -1;
58  }
59  if (tcgetattr(fd_, &defaults_) < 0)
60  {
61  perror("tcgetattr");
62  return -1;
63  }
64  cfmakeraw(&config);
65  if (tcsetattr(fd_, TCSANOW, &config) < 0)
66  {
67  perror("tcsetattr config");
68  return -1;
69  }
70  // set SPI mode
71  unsigned char buff[20];
72  buff[0] = 0x5A;
73  buff[1] = 0x02; // Set mode command
74  buff[2] = 0x93; // Set SPI mode 3
75  // buff[3] = 5; // 1MHz clock speed
76  buff[3] = 11; // 500kHz clock speed
77 
78  if (write(fd_, buff, 4) < 0)
79  {
80  ROS_ERROR("write error");
81  }
82  if (tcdrain(fd_) < 0)
83  {
84  ROS_ERROR("set_spi_mode tcdrain");
85  }
86  if (read(fd_, buff, 4) < 0)
87  {
88  ROS_ERROR("set_spi_mode read");
89  }
90  // Read back error byte
91  if (buff[0] != 0xFF)
92  {
93  ROS_ERROR("**set_spi_mode: Error setting spi mode!**\n\n");
94  }
95  write_address(0x2d, 0x08);
96 
97  return 0;
98 }
