214 /*
215  * Project a point from world coordinate to image plane
216  */
217 bool project2(const Point3 &pt, int &u, int &v, bool useOpenGLCoord = false)
218 {
219  float nearPlane = 1.0;
220  float farPlane = 200.0;
221  Point3 _pt = transform(pt, trf);
222  float _u = _pt.x() * fx / _pt.z() + cx;
223  float _v = _pt.y() * fy / _pt.z() + cy;
224 
225  u = static_cast<int>(_u);
226  v = static_cast<int>(_v);
227  if (u < 0 || imageWidth < u || v < 0 || imageHeight < v || _pt.z() < nearPlane || farPlane < _pt.z())
228  {
229  u = -1, v = -1;
230  return false;
231  }
232 
233  if (useOpenGLCoord)
234  {
235  v = imageHeight - v;
236  }
237 
238  return true;
239 }

254 {
255  // Fit the vector map format into ROS style
256  double signal_pitch_in_map = ConvertDegreeToRadian(vang - 90);
257  double signal_yaw_in_map = ConvertDegreeToRadian(-hang + 90);

260  signal_orientation_in_map_system.setRPY(0, signal_pitch_in_map, signal_yaw_in_map);

263  double signal_roll_in_cam;

274 void echoSignals2(ros::Publisher &pub, bool useOpenGLCoord = false)
275 {
276  int countPoint = 0;
277  autoware_msgs::Signals signalsInFrame;
278 
279  /* Get signals on the path if vecter_map_server is enabled */
280  if (g_use_vector_map_server)
281  {
282  vector_map_server::GetSignal service;
283  /* Set server's request */
284  service.request.pose = g_vector_map_client.pose();
285  service.request.waypoints = g_vector_map_client.waypoints();
286 
287  /* Get server's response*/
288  if (g_ros_client.call(service))
289  {
290  /* Reset signal data container */
291  vmap.signals.clear();
292 
293  /* Newle insert signal data on the path */
294  for (const auto &response: service.response.objects.data)
295  {
296  if (response.id == 0)
297  continue;
298 
299  Signal signal;
300  signal.id = response.id;
301  signal.vid = response.vid;
302  signal.plid = response.plid;
303  signal.type = response.type;
304  signal.linkid = response.linkid;
305 
306  vmap.signals.insert(std::map<int, Signal>::value_type(signal.id, signal));
307  }
308  ROS_INFO("[feat_proj] VectorMapServer available. Publishing only TrafficSignals on the current lane");
309  }
310  }
311 
312  for (const auto& signal_map : vmap.signals)
313  {
314  const Signal signal = signal_map.second;
315  int pid = vmap.vectors[signal.vid].pid;
316 
317  Point3 signalcenter = vmap.getPoint(pid);
318  Point3 signalcenterx(signalcenter.x(), signalcenter.y(), signalcenter.z() + SignalLampRadius);
319 
320  int u, v;
321  if (project2(signalcenter, u, v, useOpenGLCoord) == true)
322  {
323  countPoint++;
324  // std::cout << u << ", " << v << ", " << std::endl;
325 
326  int radius;
327  int ux, vx;
328  project2(signalcenterx, ux, vx, useOpenGLCoord);
329  radius = (int) distance(ux, vx, u, v);
330 
331  autoware_msgs::ExtractedPosition sign;
332  sign.signalId = signal.id;
333 
334  sign.u = u + adjust_proj_x; // shift project position by configuration value from runtime manager
335  sign.v = v + adjust_proj_y; // shift project position by configuration value from runtime manager
336 
337  sign.radius = radius;
338  sign.x = signalcenter.x(), sign.y = signalcenter.y(), sign.z = signalcenter.z();
339  sign.hang = vmap.vectors[signal.vid].hang; // hang is expressed in [0, 360] degree
340  sign.type = signal.type, sign.linkId = signal.linkid;
341  sign.plId = signal.plid;
342 
343  // Get holizontal angle of signal in camera corrdinate system
344  double signal_angle = GetSignalAngleInCameraSystem(vmap.vectors[signal.vid].hang + 180.0f,
345  vmap.vectors[signal.vid].vang + 180.0f);
346 
347  // signal_angle will be zero if signal faces to x-axis
348  // Target signal should be face to -50 <= z-axis (= 90 degree) <= +50
349  if (isRange(-50, 50, signal_angle - 90))
350  {
351  signalsInFrame.Signals.push_back(sign);
352  }
353  }
354  }
355  signalsInFrame.header.stamp = ros::Time::now();
356  pub.publish(signalsInFrame);
357 
358  std::cout << "There are " << signalsInFrame.Signals.size() << " signals in range" << std::endl;
359 }
