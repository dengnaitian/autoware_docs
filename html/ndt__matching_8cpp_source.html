240 static pose convertPoseIntoRelativeCoordinate(const pose &target_pose, const pose &reference_pose)
241 {
242  tf::Quaternion target_q;
243  target_q.setRPY(target_pose.roll, target_pose.pitch, target_pose.yaw);
244  tf::Vector3 target_v(target_pose.x, target_pose.y, target_pose.z);
245  tf::Transform target_tf(target_q, target_v);
246 
247  tf::Quaternion reference_q;
248  reference_q.setRPY(reference_pose.roll, reference_pose.pitch, reference_pose.yaw);
249  tf::Vector3 reference_v(reference_pose.x, reference_pose.y, reference_pose.z);
250  tf::Transform reference_tf(reference_q, reference_v);
251 
252  tf::Transform trans_target_tf = reference_tf.inverse() * target_tf;
253 
254  pose trans_target_pose;
255  trans_target_pose.x = trans_target_tf.getOrigin().getX();
256  trans_target_pose.y = trans_target_tf.getOrigin().getY();
257  trans_target_pose.z = trans_target_tf.getOrigin().getZ();
258  tf::Matrix3x3 tmp_m(trans_target_tf.getRotation());
259  tmp_m.getRPY(trans_target_pose.roll, trans_target_pose.pitch, trans_target_pose.yaw);
260 
261  return trans_target_pose;
262 }

765 static void imu_calc(ros::Time current_time)
766 {

776  current_pose_imu.yaw += diff_imu_yaw;

778  double accX1 = imu.linear_acceleration.x;
779  double accY1 = std::cos(current_pose_imu.roll) * imu.linear_acceleration.y -
780  std::sin(current_pose_imu.roll) * imu.linear_acceleration.z;
781  double accZ1 = std::sin(current_pose_imu.roll) * imu.linear_acceleration.y +
782  std::cos(current_pose_imu.roll) * imu.linear_acceleration.z;

789  double accY = std::sin(current_pose_imu.yaw) * accX2 + std::cos(current_pose_imu.yaw) * accY2;
790  double accZ = accZ2;
791 
792  offset_imu_x += current_velocity_imu_x * diff_time + accX * diff_time * diff_time / 2.0;
793  offset_imu_y += current_velocity_imu_y * diff_time + accY * diff_time * diff_time / 2.0;
794  offset_imu_z += current_velocity_imu_z * diff_time + accZ * diff_time * diff_time / 2.0;
795 
796  current_velocity_imu_x += accX * diff_time;
797  current_velocity_imu_y += accY * diff_time;
798  current_velocity_imu_z += accZ * diff_time;
799 
800  offset_imu_roll += diff_imu_roll;
801  offset_imu_pitch += diff_imu_pitch;
802  offset_imu_yaw += diff_imu_yaw;

806  predict_pose_imu.z = previous_pose.z + offset_imu_z;
807  predict_pose_imu.roll = previous_pose.roll + offset_imu_roll;
808  predict_pose_imu.pitch = previous_pose.pitch + offset_imu_pitch;

810

812 }
