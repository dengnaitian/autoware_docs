110 static void FindResponses(IplImage *img, int startX, int endX, int y, std::vector<int> &list)
111 {
112  /* scans for single response: /^\_ */
113 
114  const int row = y * img->width * img->nChannels;
115  unsigned char *ptr = (unsigned char *)img->imageData;
116 
117  int step = (endX < startX) ? -1 : 1;
118  int range = (endX > startX) ? endX-startX+1 : startX-endX+1;
119 
120  for (int x = startX; range>0; x+=step, range--)
121  {
122  if (ptr[row + x] <= BW_TRESHOLD) continue; // skip black: loop until white pixels show up
123 
124  /* first response found */
125  int idx = x + step;
126 
127  /* skip same response(white) pixels */
128  while (range > 0 && ptr[row+idx] > BW_TRESHOLD) {
129  idx += step;
130  range--;
131  }
132 
133  /* reached black again */
134  if (ptr[row+idx] <= BW_TRESHOLD) {
135  list.push_back(x);
136  }
137 
138  x = idx; // begin from new pos
139  }
140 }

147 static int findSymmetryAxisX(IplImage *half_frame, CvPoint bmin, CvPoint bmax)
148 {
149  float value = 0;
150  int axisX = -1; // not found
151 
152  int xmin = bmin.x;
153  int ymin = bmin.y;
154  int xmax = bmax.x;
155  int ymax = bmax.y;
156  int half_width = half_frame->width/2;
157 
158  for (int x=xmin, j=0; x<xmax; x++, j++)
159  {
160  float HS = 0;
161  for (int y=ymin; y<ymax; y++)
162  {
163  int row = y*half_frame->width*half_frame->nChannels;
164  for (int step=1; step<half_width; step++)
165  {
166  int neg = x-step;
167  int pos = x + step;
168  unsigned char Gneg = (neg < xmin) ? 0 :
169  (unsigned char)half_frame->imageData[row+neg*half_frame->nChannels];
170  unsigned char Gpos = (pos >= xmax) ? 0 :
171  (unsigned char)half_frame->imageData[row+pos*half_frame->nChannels];
172  HS += abs(Gneg - Gpos);
173  }
174  }
175 
176  if (axisX == -1 || value > HS) { // find minimum
177  axisX = x;
178  value = HS;
179  }
180  }
181 
182  return axisX;
183 }

193 static int horizLine(IplImage *edges, int x, int y, CvPoint bmin, CvPoint bmax, int maxHorzGap)
194 {
195  /* scan to right */
196  int right = 0;
197  int gap = maxHorzGap;
198  for (int xx=x; xx<bmax.x; xx++)
199  {
200  if (hasVertResponse(edges, xx, y, bmin.y, bmax.y))
201  {
202  right++;
203  gap = maxHorzGap; // reset
204  }
205  else
206  {
207  gap--;
208  if (gap <= 0)
209  {
210  break;
211  }
212  }
213  }
214 
215  int left = 0;
216  gap = maxHorzGap;
217  for (int xx=x-1; xx>=bmin.x; xx--)
218  {
219  if (hasVertResponse(edges, xx, y, bmin.y, bmax.y))
220  {
221  left++;
222  gap = maxHorzGap; // reset
223  }

226  gap--;
227  if (gap <= 0)
228  {
229  break;

231  }
232  }

237 void processSide(std::vector<Lane> lanes, IplImage *edges, bool right)
238 {
239  Status *side = right ? &laneR : &laneL;
240 
241  /* response search */
242  int w = edges->width;
243  int h = edges->height;
244  const int BEGINY = 0;
245  const int ENDY = h-1;
246  const int ENDX = right ? (w-BORDERX) : BORDERX;
247  int midx = w/2;
248  int midy = edges->height/2;
249 
250  /* show responses */
251  int *votes = new int[lanes.size()];
252  for (std::size_t i = 0; i < lanes.size(); ++i) votes[i++] = 0;
253 
254  for (int y=ENDY; y>=BEGINY; y-=SCAN_STEP)
255  {
256  std::vector<int> rsp;
257  FindResponses(edges, midx, ENDX, y, rsp);
258 
259  if (rsp.size() > 0)
260  {
261  int response_x = rsp[0]; // use first response (closest to screen center)
262 
263  float dmin = 9999999;
264  float xmin = 9999999;
265  int match = -1;
266 
267  for (std::size_t j = 0; j < lanes.size(); ++j)
268  {
269  /* compute response point destance to current line */
270  float d = dist2line (
271  cvPoint2D32f(lanes[j].p0.x, lanes[j].p0.y),
272  cvPoint2D32f(lanes[j].p1.x, lanes[j].p1.y),
273  cvPoint2D32f(response_x, y));
274 
275  /* point on line at current y line */
276  int xline = (y - lanes[j].b) / lanes[j].k;
277  int dist_mid = abs(midx - xline); // distance to midpoint
278 
279  /* pick the best closest match to line & to screen center */
280  if (match == -1 || (d <= dmin && dist_mid < xmin))
281  {
282  dmin = d;
283  match = j;
284  xmin = dist_mid;
285  break;
286  }
287  }
288 
289  /* vote for each line */
290  if (match != -1)
291  {
292  votes[match] += 1;
293  }
294 
295  }
296  }
297 
298  int bestMatch = -1;
299  int mini = 9999999;
300  for (std::size_t i = 0; i<lanes.size(); ++i)
301  {
302  int xline = (midy - lanes[i].b) / lanes[i].k;
303  int dist = abs(midx - xline); // distance to midpoint
304 
305  if (bestMatch == -1 || (votes[i] > votes[bestMatch] && dist < mini))
306  {
307  bestMatch = i;
308  mini = dist;
309  }
310  }
311 
312  if (bestMatch != -1)
313  {
314  Lane *best = &lanes[bestMatch];
315  float k_diff = fabs(best->k - side->k.get());
316  float b_diff = fabs(best->b - side->b.get());
317 
318  bool update_ok = (k_diff <= K_VARY_FACTOR && b_diff <= B_VARY_FACTOR) || side->reset;
319 
320 #ifdef SHOW_DETAIL
321  printf("side: %s, k vary: %.4f, b vary: %.4f, lost: %s\n",
322  (right?"RIGHT":"LEFT"), k_diff, b_diff, (update_ok?"no":"yes"));
323 #endif
324 
325  if (update_ok)
326  {
327  /* update is in valid bounds */
328  side->k.add(best->k);
329  side->b.add(best->b);
330  side->reset = false;
331  side->lost = 0;
332  }
333  else
334  {
335  /* can't update, lanes flicker periodically, start counter for partial reset! */
336  side->lost++;
337  if (side->lost >= MAX_LOST_FRAMES && !side->reset)
338  {
339  side->reset = true;
340  }
341  }
342  }
343  else
344  {
345 #ifdef SHOW_DETAIL
346  printf("no lanes detected - lane tracking lost! counter increased\n");
347 #endif
348  side->lost++;
349  if (side->lost >= MAX_LOST_FRAMES && !side->reset)
350  {
351  /* do full reset when lost for more than N frames */
352  side->reset = true;
353  side->k.clear();
354  side->b.clear();
355  }
356  }
357 
358  delete[] votes;
359 }

362 static void processLanes(CvSeq *lines, IplImage* edges, IplImage *temp_frame, IplImage *org_frame)
363 {
364  /* classify lines to left/right side */
365  std::vector<Lane> left, right;
366 
367  for (int i=0; i<lines->total; i++)
368  {
369  CvPoint *line = (CvPoint *)cvGetSeqElem(lines, i);
370  int dx = line[1].x - line[0].x;
371  int dy = line[1].y - line[0].y;
372  float angle = atan2f(dy, dx) * 180/CV_PI;
373 
374  if (fabs(angle) <= LINE_REJECT_DEGREES) // reject near horizontal lines
375  {
376  continue;
377  }
378 
379  /* assume that vanishing point is close to the image horizontal center
380  calculate line parameters: y = kx + b; */
381  dx = (dx == 0) ? 1 : dx; // prevent DIV/0!
382  float k = dy/(float)dx;
383  float b = line[0].y - k*line[0].x;
384 
385  /* assign lane's side based by its midpoint position */
386  int midx = (line[0].x + line[1].x) / 2;
387  if (midx < temp_frame->width/2)
388  {
389  left.push_back(Lane(line[0], line[1], angle, k, b));
390  }
391  else if (midx > temp_frame->width/2)
392  {
393  right.push_back(Lane(line[0], line[1], angle, k, b));
394  }
395  }
396 
397  /* show Hough lines */
398  int org_offset = temp_frame->height;
399  for (std::size_t i = 0; i < right.size(); ++i)
400  {
401  CvPoint org_p0 = right[i].p0;
402  org_p0.y += org_offset;
403  CvPoint org_p1 = right[i].p1;
404  org_p1.y += org_offset;
405 #ifdef USE_POSIX_SHARED_MEMORY
406 #ifdef SHOW_DETAIL
407  cvLine(temp_frame, right[i].p0, right[i].p1, BLUE, 2);
408 #endif
409  cvLine(org_frame, org_p0, org_p1, BLUE, 2);
410 #endif
411  }
412  for (std::size_t i = 0; i < left.size(); ++i)
413  {
414  CvPoint org_p0 = left[i].p0;
415  org_p0.y += org_offset;
416  CvPoint org_p1 = left[i].p1;
417  org_p1.y += org_offset;
418 #ifdef USE_POSIX_SHARED_MEMORY
419 #ifdef SHOW_DETAIL
420  cvLine(temp_frame, left[i].p0, left[i].p1, RED, 2);
421 #endif
422  cvLine(org_frame, org_p0, org_p1, RED, 2);
423 #endif
424  }
425 
426  processSide(left, edges, false);
427  processSide(right, edges, true);
428 
429  /* show computed lanes */
430  int x = temp_frame->width * 0.55f;
431  int x2 = temp_frame->width;
432 #if defined(USE_POSIX_SHARED_MEMORY)
433 #ifdef SHOW_DETAIL
434  cvLine(temp_frame, cvPoint(x, laneR.k.get()*x + laneR.b.get()),
435  cvPoint(x2, laneR.k.get()*x2 + laneR.b.get()), PURPLE, 2);
436 #endif
437 
438  cvLine(org_frame, cvPoint(x, laneR.k.get()*x + laneR.b.get() + org_offset),
439  cvPoint(x2, laneR.k.get()*x2 + laneR.b.get() + org_offset), PURPLE, 2);
440 #else
441  autoware_msgs::ImageLaneObjects lane_msg;
442  lane_msg.lane_r_x1 = x;
443  lane_msg.lane_r_y1 = laneR.k.get()*x + laneR.b.get() + org_offset;
444  lane_msg.lane_r_x2 = x2;
445  lane_msg.lane_r_y2 = laneR.k.get()*x2 + laneR.b.get() + org_offset;
446 #endif
447 
448  x = temp_frame->width * 0;
449  x2 = temp_frame->width * 0.45f;
450 #if defined(USE_POSIX_SHARED_MEMORY)
451 #ifdef SHOW_DETAIL
452  cvLine(temp_frame, cvPoint(x, laneL.k.get()*x + laneL.b.get()),
453  cvPoint(x2, laneL.k.get()*x2 + laneL.b.get()), PURPLE, 2);
454 #endif
455 
456  cvLine(org_frame, cvPoint(x, laneL.k.get()*x + laneL.b.get() + org_offset),
457  cvPoint(x2, laneL.k.get()*x2 + laneL.b.get() + org_offset), PURPLE, 2);
458 #else

470 static void process_image_common(IplImage *frame)
471 {
472  CvFont font;
473  cvInitFont(&font, CV_FONT_VECTOR0, 0.25f, 0.25f);
474 
475  CvSize video_size;
476 #if defined(USE_POSIX_SHARED_MEMORY)
477  video_size.height = *shrd_ptr_height;
478  video_size.width = *shrd_ptr_width;
479 #else
480  // XXX These parameters should be set ROS parameters
481  video_size.height = frame->height;
482  video_size.width = frame->width;
483 #endif
484  CvSize frame_size = cvSize(video_size.width, video_size.height/2);
485  IplImage *temp_frame = cvCreateImage(frame_size, IPL_DEPTH_8U, 3);
486  IplImage *gray = cvCreateImage(frame_size, IPL_DEPTH_8U, 1);
487  IplImage *edges = cvCreateImage(frame_size, IPL_DEPTH_8U, 1);
488  IplImage *half_frame = cvCreateImage(cvSize(video_size.width/2, video_size.height/2), IPL_DEPTH_8U, 3);
489 
490  CvMemStorage *houghStorage = cvCreateMemStorage(0);
491 
492  cvPyrDown(frame, half_frame, CV_GAUSSIAN_5x5); // Reduce the image by 2
493 
494  /* we're intersted only in road below horizont - so crop top image portion off */
495  crop(frame, temp_frame, cvRect(0, frame_size.height, frame_size.width, frame_size.height));
496  cvCvtColor(temp_frame, gray, CV_BGR2GRAY); // contert to grayscale
497 
498  /* Perform a Gaussian blur & detect edges */
499  // smoothing image more strong than original program
500  cvSmooth(gray, gray, CV_GAUSSIAN, 15, 15);
501  cvCanny(gray, edges, CANNY_MIN_TRESHOLD, CANNY_MAX_TRESHOLD);
502 
503  /* do Hough transform to find lanes */
504  double rho = 1;
505  double theta = CV_PI/180;
506  CvSeq *lines = cvHoughLines2(edges, houghStorage, CV_HOUGH_PROBABILISTIC,
507  rho, theta, HOUGH_TRESHOLD, HOUGH_MIN_LINE_LENGTH, HOUGH_MAX_LINE_GAP);
508 
509  processLanes(lines, edges, temp_frame, frame);
510 
511 #ifdef SHOW_DETAIL
512  /* show middle line */
513  cvLine(temp_frame, cvPoint(frame_size.width/2, 0),
514  cvPoint(frame_size.width/2, frame_size.height), CV_RGB(255, 255, 0), 1);
515 
516  // cvShowImage("Gray", gray);
517  // cvShowImage("Edges", edges);
518  // cvShowImage("Color", temp_frame);
519  // cvShowImage("temp_frame", temp_frame);
520  // cvShowImage("frame", frame);
521 #endif
522 
523 #if defined(USE_POSIX_SHARED_MEMORY)
524  setImage_toSHM(frame);
525 #endif
526 
527 #ifdef SHOW_DETAIL
528  // cvMoveWindow("Gray", 0, 0);
529  // cvMoveWindow("Edges", 0, frame_size.height+25);
530  // cvMoveWindow("Color", 0, 2*(frame_size.height+25));
531 #endif
532 
533  cvReleaseMemStorage(&houghStorage);
534  cvReleaseImage(&gray);
535  cvReleaseImage(&edges);
536  cvReleaseImage(&temp_frame);
537  cvReleaseImage(&half_frame);
538 }
