34 std::pair<uint8_t, double> RateChecker::getErrorLevelAndRate() {

37  if (!rate) {

75 void RateChecker::update() {
76  mtx_.lock();
77  std::vector<ros::Time> buffer;
78  for (auto data_itr = data_.begin(); data_itr != data_.end(); data_itr++) {
79  if (*data_itr > ros::Time::now() - ros::Duration(buffer_length_)) {
80  buffer.push_back(*data_itr);
81  }
82  }
83  data_ = buffer;
84  mtx_.unlock();
85  return;
86 }

88 boost::optional<double> RateChecker::getRate() {
89  boost::optional<double> rate;
90  if (ros::Time::now() - start_time_ < ros::Duration(buffer_length_)) {
91  return boost::none;
92  }
93  update();
94  mtx_.lock();
95  rate = data_.size() / buffer_length_;
96  mtx_.unlock();
97  return rate;
