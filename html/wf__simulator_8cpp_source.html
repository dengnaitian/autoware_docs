61 void CmdCallBack(const autoware_msgs::VehicleCmdConstPtr& msg, double accel_rate)
62 {
63  if (use_ctrl_cmd == true)
64  {
65  linear_acceleration_ = msg->ctrl_cmd.linear_acceleration;
66  steering_angle_ = msg->ctrl_cmd.steering_angle;
67  }
68  else
69  {
70  static double previous_linear_velocity = 0;
71 
72  if (current_velocity_.linear.x < msg->twist_cmd.twist.linear.x)
73  {
74  current_velocity_.linear.x = previous_linear_velocity + accel_rate / (double)LOOP_RATE;
75 
76  if (current_velocity_.linear.x > msg->twist_cmd.twist.linear.x)
77  {
78  current_velocity_.linear.x = msg->twist_cmd.twist.linear.x;
79  }
80  }
81  else
82  {
83  current_velocity_.linear.x = previous_linear_velocity - accel_rate / (double)LOOP_RATE;
84 
85  if (current_velocity_.linear.x < msg->twist_cmd.twist.linear.x)
86  {
87  current_velocity_.linear.x = msg->twist_cmd.twist.linear.x;
88  }
89  }
90 
91  previous_linear_velocity = current_velocity_.linear.x;
92 
93  current_velocity_.angular.z = msg->twist_cmd.twist.angular.z;
94 
95  //current_velocity_ = msg->twist;
96  }
97 }

159 void publishOdometry()
160 {
161  static ros::Time current_time = ros::Time::now();
162  static ros::Time last_time = ros::Time::now();
163  static geometry_msgs::Pose pose;
164  static double th = 0;
165  static tf::TransformBroadcaster tf_broadcaster;
166 
167  if (!pose_set_)
168  {
169  pose.position = initial_pose_.position;
170  pose.orientation = initial_pose_.orientation;
171  th = tf::getYaw(pose.orientation);
172  ROS_INFO_STREAM("pose set : (" << pose.position.x << " " << pose.position.y << " " << pose.position.z << " " << th
173  << ")");
174  pose_set_ = true;
175  }
176 
177  if (waypoint_set_ && is_closest_waypoint_subscribed_)
178  pose.position.z = current_waypoints_.getWaypointPosition(closest_waypoint_).z;
179  double vx = current_velocity_.linear.x;
180  double vth = current_velocity_.angular.z;
181  current_time = ros::Time::now();
182 
183  // compute odometry in a typical way given the velocities of the robot
184  std::random_device rnd;
185  std::mt19937 mt(rnd());
186  std::uniform_real_distribution<double> rnd_dist(0.0, 2.0);
187  double rnd_value_x = rnd_dist(mt) - 1.0;
188  double rnd_value_y = rnd_dist(mt) - 1.0;
189  double rnd_value_th = rnd_dist(mt) - 1.0;
190 
191  double dt = (current_time - last_time).toSec();
192  double delta_x = (vx * cos(th)) * dt + rnd_value_x * position_error_;
193  double delta_y = (vx * sin(th)) * dt + rnd_value_y * position_error_;
194  double delta_th = vth * dt + rnd_value_th * angle_error_ * M_PI / 180;
195 
196  pose.position.x += delta_x;
197  pose.position.y += delta_y;
198  th += delta_th;
199  pose.orientation = tf::createQuaternionMsgFromYaw(th);
200 
201  // first, we'll publish the transform over tf
202  geometry_msgs::TransformStamped odom_trans;
203  odom_trans.header.stamp = current_time;
204  odom_trans.header.frame_id = MAP_FRAME;
205  odom_trans.child_frame_id = SIMULATION_FRAME;
206 
207  odom_trans.transform.translation.x = pose.position.x;
208  odom_trans.transform.translation.y = pose.position.y;
209  odom_trans.transform.translation.z = pose.position.z;
210  odom_trans.transform.rotation = pose.orientation;
211 
212  // send odom transform
213  tf_broadcaster.sendTransform(odom_trans);
214 
215  geometry_msgs::TransformStamped lidar_trans;
216  lidar_trans.header.stamp = odom_trans.header.stamp;
217  lidar_trans.header.frame_id = SIMULATION_FRAME;
218  lidar_trans.child_frame_id = LIDAR_FRAME;
219  lidar_trans.transform.translation.z += lidar_height_;
220  lidar_trans.transform.rotation.w = 1;
221 
222  // send lidar transform
223  tf_broadcaster.sendTransform(lidar_trans);
224 
225  //hui
226  geometry_msgs::TransformStamped camera_trans;
227  camera_trans.header.stamp = current_time;
228  camera_trans.header.frame_id = SIMULATION_FRAME;
229  camera_trans.child_frame_id = CAMERA_FRAME;
230  camera_trans.transform.translation.z += lidar_height_;
231  camera_trans.transform.rotation.w = 1;
232 
233  // send camera transform
234  tf_broadcaster.sendTransform(camera_trans);
235 
236  // next, we'll publish the odometry message over ROS
237  std_msgs::Header h;
238  h.stamp = current_time;
239  h.frame_id = MAP_FRAME;
240 
241  geometry_msgs::PoseStamped ps;
242  ps.header = h;
243  ps.pose = pose;
244 
245  geometry_msgs::TwistStamped ts;
246  ts.header = h;
247  ts.twist.linear.x = vx;
248  ts.twist.angular.z = vth;
249 
250  // publish the message
251  odometry_publisher_.publish(ps);
252  velocity_publisher_.publish(ts);
253 
254  last_time = current_time;
255 }
