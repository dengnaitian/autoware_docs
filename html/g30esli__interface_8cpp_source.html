94 // read can data from the vehicle
95 void readCanData(FILE* fp)
96 {
97  char buf[64];
98  while (fgets(buf, sizeof(buf), fp) != NULL && !g_terminate_thread)
99  {
100  std::string raw_data = std::string(buf);
101  std::cout << "received data: " << raw_data << std::endl;
102 
103  // check if the data is valid
104  if (raw_data.size() > 0)
105  {
106  // split data to strings
107  // data format is like this
108  // can0 200 [8] 08 00 00 00 01 00 01 29
109  std::vector<std::string> parsed_data = ymc::splitString(raw_data);
110 
111  // delete first 3 elements
112  std::vector<std::string> data = parsed_data;
113  data.erase(data.begin(), data.begin() + 3);
114 
115  int id = std::stoi(parsed_data.at(1), nullptr, 10);
116  double _current_vel_mps = ymc::translateCanData(id, data, &g_mode);
117  if (_current_vel_mps != RET_NO_PUBLISH)
118  {
119  geometry_msgs::TwistStamped ts;
120  ts.header.frame_id = "base_link";
121  ts.header.stamp = ros::Time::now();
122  ts.twist.linear.x = _current_vel_mps;
123  g_current_twist_pub.publish(ts);
124  }
125  }
126  }
127 }

131 int main(int argc, char* argv[])

136  ros::NodeHandle private_nh("~");

139  private_nh.param<int>("mode", g_mode, 8);

144

149  ros::Subscriber vehicle_cmd_sub = n.subscribe<autoware_msgs::VehicleCmd>("vehicle_cmd", 1, vehicle_cmd_callback);
150  ros::Subscriber current_vel_sub =
151  n.subscribe<geometry_msgs::TwistStamped>("current_velocity", 1, current_vel_callback);
152 
153  // publisher
154  g_current_twist_pub = n.advertise<geometry_msgs::TwistStamped>("ymc_current_twist", 10);
155 
156  // read can data from candump
157  FILE* fp = popen("candump can0", "r");
158 
159  // create threads
160  std::thread t1(changeMode);
161  std::thread t2(readCanData, fp);
162 
163  ros::Rate loop_rate = g_loop_rate;
164  bool stopping_flag = true;
165  while (ros::ok())
166  {
167  // data
168  unsigned char mode = static_cast<unsigned char>(g_mode);
169  unsigned char shift = 0;
170  uint16_t target_velocity_ui16 = g_target_velocity_ui16;
171  int16_t steering_angle_deg_i16 = g_steering_angle_deg_i16;
172  static unsigned char brake = 1;
173  static unsigned char heart_beat = 0;
174 
175  // change to STOP MODE...
176  if (target_velocity_ui16 == 0 || g_mode == 3)
177  stopping_flag = true;
178 
179  // STOPPING
180  static int stop_count = 0;
181  if (stopping_flag && g_mode == 8)
182  {
183  // TODO: change steering angle according to current velocity ?
184  target_velocity_ui16 = 0;
185  brake = 1;
186 
187  // vehicle is stopping or not
188  double stopping_threshold = 1.0;
189  g_current_vel_kmph < stopping_threshold ? stop_count++ : stop_count = 0;
190 
191  std::cout << "stop count: " << stop_count << std::endl;
192  // vehicle has stopped, so we can restart
193  if (stop_count > g_loop_rate * g_stop_time_sec)
194  {
195  brake = 0;
196  stop_count = 0;
197  stopping_flag = false;
198  }
199  }
200 
201  // Insert data to 8 byte array
202  unsigned char data[8] = {};
203  ymc::setCanData(data, mode, shift, target_velocity_ui16, steering_angle_deg_i16, brake, heart_beat);
204 
205  // send can data
206  std::string send_id("200");
207  size_t data_size = sizeof(data) / sizeof(data[0]);
208  char can_cmd[256];
209  std::strcpy(can_cmd, mycansend::makeCmdArgument(data, data_size, send_id).c_str());
210  g_cansender.send(can_cmd);
211 
212  // wait for callbacks
213  ros::spinOnce();
214  loop_rate.sleep();
215 
216  heart_beat += 1;
217  }
218 
219  g_terminate_thread = true;
220  t1.join();
221  t2.join();
222 
223  pclose(fp);
224 
225  return 0;
226 }
