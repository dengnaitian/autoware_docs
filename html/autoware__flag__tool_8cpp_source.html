104 int AutowareSimObjTool::processMouseEvent( rviz::ViewportMouseEvent& event )
105 {
106  if( !moving_flag_node_ )
107  {
108  return Render;
109  }
110  Ogre::Vector3 intersection;
111  Ogre::Plane ground_plane( Ogre::Vector3::UNIT_Z, 0.0f );
112 
113  if( rviz::getPointOnPlaneFromWindowXY( event.viewport,
114  ground_plane,
115  event.x, event.y, intersection ))
116  {
117  moving_flag_node_->setVisible( true );
118  moving_flag_node_->setPosition( intersection );
119  current_flag_property_->setVector( intersection );
120 
121  if( event.leftDown() )
122  {
123  makeFlag( intersection );
124  current_flag_property_ = NULL; // Drop the reference so that deactivate() won't remove it.
125 
126  geometry_msgs::PointStamped ps;
127  ps.header.stamp = ros::Time::now();
128  ps.header.frame_id = context_->getFrameManager()->getFixedFrame();
129  ps.point.x = intersection.x;//event.x;
130  ps.point.y = intersection.y;//event.y;
131  ps.point.z = intersection.z;//0;
132  pub_.publish(ps);
133 
134  return Render | Finished;
135  }
136  }
137  else
138  {
139  moving_flag_node_->setVisible( false ); // If the mouse is not pointing at the ground plane, don't show the flag.
140  }
141  return Render;
142 }

154 {
155  config.mapSetValue( "Class", getClassId() );
156  rviz::Config flags_config = config.mapMakeChild( "Flags" );
157 
158  rviz::Property* container = getPropertyContainer();
159  int num_children = container->numChildren();

163  rviz::Config flag_config = flags_config.listAppendNew();
164  flag_config.mapSetValue( "Name", position_prop->getName() );
165  position_prop->save( flag_config );

169 void AutowareSimObjTool::load( const rviz::Config& config )
170 {
171  rviz::Config flags_config = config.mapGetChild( "Flags" );
172  int num_flags = flags_config.listLength();
173  for( int i = 0; i < num_flags; i++ )
174  {
175  rviz::Config flag_config = flags_config.listChildAt( i );
176  QString name = "Flag " + QString::number( i + 1 );
177  flag_config.mapGetString( "Name", &name );
178  rviz::VectorProperty* prop = new rviz::VectorProperty( name );
179  prop->load( flag_config );
180  prop->setReadOnly( true );
181  getPropertyContainer()->addChild( prop );
182  makeFlag( prop->getVector() );
183  }
184 }
185 }
