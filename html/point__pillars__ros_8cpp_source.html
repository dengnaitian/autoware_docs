74 void PointPillarsROS::pubDetectedObject(const std::vector<float>& detections, const std_msgs::Header& in_header)
75 {
76  autoware_msgs::DetectedObjectArray objects;
77  objects.header = in_header;
78  int num_objects = detections.size() / OUTPUT_NUM_BOX_FEATURE_;
79  for (size_t i = 0; i < num_objects; i++)
80  {
81  autoware_msgs::DetectedObject object;
82  object.header = in_header;
83  object.valid = true;
84  object.pose_reliable = true;
85 
86  object.pose.position.x = detections[i * OUTPUT_NUM_BOX_FEATURE_ + 0];
87  object.pose.position.y = detections[i * OUTPUT_NUM_BOX_FEATURE_ + 1];
88  object.pose.position.z = detections[i * OUTPUT_NUM_BOX_FEATURE_ + 2];
89 
90  // Trained this way
91  float yaw = detections[i * OUTPUT_NUM_BOX_FEATURE_ + 6];
92  yaw += M_PI/2;
93  yaw = std::atan2(std::sin(yaw), std::cos(yaw));
94  geometry_msgs::Quaternion q = tf::createQuaternionMsgFromYaw(-yaw);
95  object.pose.orientation = q;
96 
97  if (baselink_support_)
98  {
99  object.pose = getTransformedPose(object.pose, angle_transform_inversed_);
100  }
101 
102  // Again: Trained this way
103  object.dimensions.x = detections[i * OUTPUT_NUM_BOX_FEATURE_ + 4];
104  object.dimensions.y = detections[i * OUTPUT_NUM_BOX_FEATURE_ + 3];
105  object.dimensions.z = detections[i * OUTPUT_NUM_BOX_FEATURE_ + 5];
106 
107  //Only detects car in Version 1.0
108  object.label = "car";
109 
110  objects.objects.push_back(object);
111  }
112  pub_objects_.publish(objects);
113 }
