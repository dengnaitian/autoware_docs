290 {
291  u32 entry_pos = 0, i;
292 
293  //Null check the pointer
294  if(buffer == 0)
295  return RING_BUFFER_ERROR;
296 
297  //Check for a full ring buffer
298  if(buffer->current_count >= buffer->max_entries)
299  {
300  buffer->total_overruns++;
301  return RING_BUFFER_FULL;
302  }
303 
304  //Zero byte short circuit
305  if(num_bytes == 0)
306  return RING_BUFFER_OK;
307 
308 
309  //Check entry size
310  if(num_bytes > buffer->entry_size)
311  return RING_BUFFER_MEMORY_ERROR;
312 
313  //Calculate the write position
314  entry_pos = buffer->position + buffer->current_count;
315 
316  if(entry_pos >= buffer->max_entries)
317  entry_pos -= buffer->max_entries;
318 
319  //Write the entry
320  for(i=0;i<num_bytes;i++)
321  buffer->entries[entry_pos*buffer->entry_size + i] = entry[i];
322

326
