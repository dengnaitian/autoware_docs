290  double CalcMatchingPercentage(const std::vector<PlannerHNS::WayPoint>& _path)
291  {
292  if(_path.size() == 0) return 0;
293 
294  if(beh != _path.at(0).beh_state) return 0;
295 
296  int nCount = 0, nIds = 0;
297  int prev_id = -10;
298  int curr_id = -10;
299  std::vector<int> _ids;
300 
301  for(unsigned int i = 0; i < _path.size(); i++)
302  {
303  curr_id = _path.at(i).laneId;
304  if(i < path_ids.size())
305  {
306  nCount++;
307  if(curr_id == path_ids.at(i))
308  nIds++;
309  }
310 
311  if(curr_id != prev_id)
312  {
313  _ids.push_back(curr_id);
314  prev_id = curr_id;
315  }
316  }
317 
318  int nEqualities = 0;
319  for(unsigned int i=0; i < _ids.size(); i++)
320  {
321  for(unsigned int j=0; j < ids.size(); j++)
322  {
323  if(_ids.at(i) == ids.at(j))
324  {
325  nEqualities++;
326  break;
327  }
328  }
329  }
330 
331  double rms_val = 0;
332  for(unsigned int i=0; i < _path.size(); i++)
333  {
334  if(i < trajectory.size())
335  {
336  rms_val += hypot(_path.at(i).pos.y - trajectory.at(i).pos.y, _path.at(i).pos.y - trajectory.at(i).pos.y);
337  }
338  }
339 
340  rms_error = rms_val;
341 
342  if(rms_error < 5.0)
343  return 1;
344 
345  if(_ids.size() == ids.size() && ids.size() == nEqualities && rms_error < 5.0) // perfect match
346  return 1;
347 
348  WayPoint curr_last_pose = _path.at(_path.size()-1);
349  double nMatch = (double)nIds/(double)nCount;
350  double _d = hypot(path_last_pose.pos.y-curr_last_pose.pos.y, path_last_pose.pos.x-curr_last_pose.pos.x);
351 
352  double dCost = _d/FIXED_PLANNING_DISTANCE;
353  if(dCost > 1.0)
354  dCost = 1.0;
355  double dMatch = 1.0 - dCost;
356 
357  double _a_diff = UtilityHNS::UtilityH::AngleBetweenTwoAnglesPositive(path_last_pose.pos.a, curr_last_pose.pos.a);
358  double aCost = _a_diff/M_PI;
359  if(aCost > 1.0)
360  aCost = 1.0;
361  double aMatch = 1.0 - aCost;
362 
363  double totalMatch = (nMatch + dMatch + aMatch)/3.0;
364 
365  return totalMatch;
366  }

712  void MatchTrajectories()
713  {
714  m_TrajectoryTracker_temp.clear();
715  std::vector<LLP> matching_list;
716  std::vector<TrajectoryTracker*> delete_me_track = m_TrajectoryTracker;
717  for(unsigned int t = 0; t < obj.predTrajectories.size();t++)
718  {
719  bool bMatched = false;
720  LLP match_item;
721  match_item.new_index = t;
722 
723  for(int i = 0; i < m_TrajectoryTracker.size(); i++)
724  {
725  TrajectoryTracker* pTracker = m_TrajectoryTracker.at(i);
726 
727  double vMatch = pTracker->CalcMatchingPercentage(obj.predTrajectories.at(t));
728  if(vMatch == 1.0) // perfect match
729  {
730  pTracker->UpdatePathAndIndex(obj.predTrajectories.at(t), t);
731  bool bFound = false;
732  for(unsigned int k=0; k < m_TrajectoryTracker_temp.size(); k++)
733  {
734  if(m_TrajectoryTracker_temp.at(k) == pTracker)
735  {
736  bFound = true;
737  break;
738  }
739  }
740 
741  if(!bFound)
742  m_TrajectoryTracker_temp.push_back(pTracker);
743 
744  DeleteFromList(delete_me_track, pTracker);
745 
746  for(unsigned int k=0; k < matching_list.size(); k++)
747  {
748  if(matching_list.at(k).pTrack == pTracker)
749  {
750  matching_list.erase(matching_list.begin()+k);
751  break;
752  }
753  }
754 
755  m_TrajectoryTracker.erase(m_TrajectoryTracker.begin()+i);
756  bMatched = true;
757  i--;
758  break;
759  }
760  else if(vMatch > 0.5) // any matching less than 50%, the trajectory will be considered new
761  {
762  bMatched = true;
763  match_item.match_percent = vMatch;
764  match_item.pTrack = pTracker;
765  matching_list.push_back(match_item);
766  }
767  }
768 
769  if(!bMatched)
770  {
771  m_TrajectoryTracker_temp.push_back(new TrajectoryTracker(obj.predTrajectories.at(t), t));
772  }
773  }
774 
775  MatchWithMax(matching_list,delete_me_track, m_TrajectoryTracker_temp);
776  m_TrajectoryTracker.clear();
777  DeleteTheRest(delete_me_track);
778  m_TrajectoryTracker = m_TrajectoryTracker_temp;
779  }
