79 // filter data from timestamp and level
80 autoware_system_msgs::DiagnosticStatusArray
81 DiagBuffer::filterBuffer(ros::Time now, uint8_t level) {
82  autoware_system_msgs::DiagnosticStatusArray filterd_data;
83  autoware_system_msgs::DiagnosticStatusArray ret;
84  if (buffer_.count(level) != 0) {
85  filterd_data = buffer_[level];
86  }
87  for (auto data_itr = filterd_data.status.begin();
88  data_itr != filterd_data.status.end(); data_itr++) {
89  if (data_itr->header.stamp > (now - buffer_length_)) {
90  ret.status.push_back(*data_itr);
91  }
92  }
93  return ret;
94 }
