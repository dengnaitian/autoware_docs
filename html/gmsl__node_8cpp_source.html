90 
91  // Gmsl camera instance run

93 
94  // other main thread: waiting for a signal to stop&shutdown
95  while(running & ros::ok())
96  {
97  ros::spinOnce();

99  }
100 
101  // Gmsl camera signal termination- call camera cleanup

104
