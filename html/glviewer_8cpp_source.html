67 void GLViewer::paintGL()
68 {
69  Eigen::Vector4d eye=cameraparameters.transform*Eigen::Vector4d(0,0,0,1);
70  Eigen::Vector4d center=cameraparameters.transform*Eigen::Vector4d(0,0,-1,1);
71  Eigen::Vector4d up=cameraparameters.transform*Eigen::Vector4d(0,1,0,0);
72  makeCurrent();
73  cameraparameters.eye[0]=eye(0);
74  cameraparameters.eye[1]=eye(1);
75  cameraparameters.eye[2]=eye(2);
76  cameraparameters.eye[3]=eye(3);
77  glLightfv(GL_LIGHT0, GL_POSITION,cameraparameters.eye);
78  glLightfv(GL_LIGHT0, GL_AMBIENT, cameraparameters.lightambient);
79  glLightfv(GL_LIGHT0, GL_DIFFUSE,cameraparameters.lightambient);
80  glLightfv(GL_LIGHT0, GL_SPECULAR,cameraparameters.lightambient);
81  glLightModelfv(GL_LIGHT_MODEL_AMBIENT, cameraparameters.lightambient);
82  glLightModeli(GL_LIGHT_MODEL_LOCAL_VIEWER,GL_TRUE);
83  glLightModeli(GL_LIGHT_MODEL_TWO_SIDE,GL_TRUE);
84  glEnable(GL_BLEND);
85  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
86  glClearColor(cameraparameters.background(0),cameraparameters.background(1),cameraparameters.background(2),0.0);
87  glClear(GL_COLOR_BUFFER_BIT|GL_DEPTH_BUFFER_BIT);
88  glMatrixMode(GL_MODELVIEW);
89  glLoadIdentity();
90  gluLookAt(eye(0),eye(1),eye(2),center(0),center(1),center(2),up(0),up(1),up(2));
91  glPushMatrix();
92  int i,n=displaylist.size();
93  glPointSize(cameraparameters.pointsize);
94  for(i=0;i<n;i++)
95  {
96  if(displaylist[i].show)
97  {
98  glPushMatrix();
99  Eigen::Matrix4d matrix=displaylist[i].scale*displaylist[i].transform;
100  glMultMatrixd(matrix.data());
101  glCallList(displaylist[i].listid);
102  glPopMatrix();
103  }
104  }
105  glPopMatrix();
106  glBegin(GL_LINES);
107  glColor4d(1,0,0,1);
108  glVertex3d(0,0,0);glVertex3d(1,0,0);
109  glColor4d(0,1,0,1);
110  glVertex3d(0,0,0);glVertex3d(0,1,0);
111  glColor4d(0,0,1,1);
112  glVertex3d(0,0,0);glVertex3d(0,0,1);
113  glEnd();
114  return;
115 }

124  }
