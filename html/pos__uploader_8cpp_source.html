72 static std::string getTimeStamp(time_t sec, time_t nsec)
73 {
74  char buf[30];
75  int msec = static_cast<int>(nsec / (1000 * 1000));
76 
77  tm *t = gmtime(&sec);
78  sprintf(buf, "%04d-%02d-%02d %02d:%02d:%02d.%03d",
79  t->tm_year + 1900, t->tm_mon + 1, t->tm_mday,
80  t->tm_hour, t->tm_min, t->tm_sec, msec);
81 
82  return std::string(static_cast<const char*>(buf));

145 static std::string makeSendDataDetectedObj(const autoware_msgs::ObjLabel& cp_array, const char *name)
146 {
147  std::string timestamp;
148  if(use_current_time || cp_array.header.stamp.sec == 0) {
149  ros::Time t = ros::Time::now();
150  timestamp = getTimeStamp(t.sec, t.nsec);
151  } else {
152  timestamp = getTimeStamp(cp_array.header.stamp.sec, cp_array.header.stamp.nsec);
153  }
154 
155  std::string ret;
156  for (const auto& point : cp_array.reprojected_pos) {
157  //create sql
158  ret += point_to_insert_statement(point, timestamp, name);
159  ret += "\n";
160  }
161 
162  return ret;
