42 class KeyPair
43 {
44  boost::circular_buffer<ros::Time> sensor;
45  boost::circular_buffer<ros::Time> execution;
46 
47 public:
48  KeyPair() {
49  sensor.resize(1);
50  execution.resize(1);
51  }
52  void resize(int size) {
53  sensor.resize(size);
54  execution.resize(size);
55  }
56  void push_front(ros::Time sensor_time, ros::Time execution_time) {
57  sensor.push_front(sensor_time);
58  execution.push_front(execution_time);
59  }
60  ros::Time front(void) {
61  return execution.front();
62  }
63 
64  ros::Time find(ros::Time sensor_time) {
65  boost::circular_buffer<ros::Time>::iterator it = sensor.begin();
66  for (int i = 0; it != sensor.end(); it++, i++) {
67  if (it->sec == sensor_time.sec && it->nsec == sensor_time.nsec) {
68  return execution.at(i); // find
69  }
70  }
71  ROS_ERROR("error:not found a pair");
72  ros::Time failed;
73  failed.sec = 0;
74  failed.nsec = 0;
75  return failed; // not find
76  }
77 };

147  if (execution_time.sec == 0 && execution_time.nsec == 0) { // not find

154  }
155 
156  double time_diff = ros_time2msec(execution_time) - ros_time2msec(sensor_time);
157  if (time_diff >= 0)
158  return time_diff;
159  else
160  ROS_ERROR("error:time difference is less than 0");
161  return 0.0;
162  }

317 /* time difference */
318 void TimeManager::time_diff_callback(const autoware_msgs::SyncTimeDiff::ConstPtr& time_diff_msg) {
319  ros::Time sensors_time_diff;
320  double lidar = (double)time_diff_msg->lidar.sec + (double)time_diff_msg->lidar.nsec/1000000000.0L;
321  double camera = (double)time_diff_msg->camera.sec + (double)time_diff_msg->camera.nsec/1000000000.0L;
322 
323  if (time_diff_msg->lidar.sec > time_diff_msg->camera.sec) {
324  sensors_time_diff.sec = time_diff_msg->lidar.sec - time_diff_msg->camera.sec;
325  if (time_diff_msg->lidar.nsec >= time_diff_msg->camera.nsec) {
326  sensors_time_diff.nsec = time_diff_msg->lidar.nsec - time_diff_msg->camera.nsec;
327  } else {
328  sensors_time_diff.sec -= 1;
329  sensors_time_diff.nsec = 1000000000L + time_diff_msg->lidar.nsec - time_diff_msg->camera.nsec;
330  }
331  } else if (time_diff_msg->lidar.sec < time_diff_msg->camera.sec) {
332  sensors_time_diff.sec = time_diff_msg->camera.sec - time_diff_msg->lidar.sec;
333  if (time_diff_msg->camera.nsec >= time_diff_msg->lidar.nsec) {
334  sensors_time_diff.nsec = time_diff_msg->camera.nsec - time_diff_msg->lidar.nsec;
335  } else {
336  sensors_time_diff.sec -= 1;
337  sensors_time_diff.nsec = 1000000000L + time_diff_msg->camera.nsec - time_diff_msg->lidar.nsec;
338  }
339  } else if (time_diff_msg->lidar.sec == time_diff_msg->camera.sec) {
340  sensors_time_diff.sec = 0;
341  if (time_diff_msg->lidar.nsec >= time_diff_msg->camera.nsec) {
342  sensors_time_diff.nsec = time_diff_msg->lidar.nsec - time_diff_msg->camera.nsec;
343  } else {
344  sensors_time_diff.nsec = time_diff_msg->camera.nsec - time_diff_msg->lidar.nsec;
345  }
346  } else {
347  // not impl
348  ROS_ERROR("Exception error");
349  }
350  time_diff_.push_front(time_diff_msg->header.stamp, sensors_time_diff);
351 }

353 void TimeManager::obj_pose_callback(const std_msgs::Time::ConstPtr& obj_pose_timestamp_msg) {
354 // ROS_INFO("obj_pose: \t\t\t%d.%d", obj_pose_timestamp_msg->data.sec, obj_pose_timestamp_msg->data.nsec);
355  obj_pose_.push_front(obj_pose_timestamp_msg->data, get_walltime_now());
356  static ros::Time pre_sensor_time;
357 
358  autoware_msgs::SyncTimeMonitor time_monitor_msg;
359  time_monitor_msg.header.frame_id = "0";
360  time_monitor_msg.header.stamp = ros::Time::now();
361 
362  // ROS_INFO("-------------------------------------");
363  // ROS_INFO("image_raw");
364  if (!ros::Time::isSimTime()) {
365  time_monitor_msg.image_raw = time_diff(obj_pose_timestamp_msg->data,
366  image_raw_.find(obj_pose_timestamp_msg->data));
367  } else {
368  time_monitor_msg.image_raw = 0;
369  }
370 
371  // ROS_INFO("points_raw");
372  if (!ros::Time::isSimTime()) {
373  time_monitor_msg.points_raw = time_diff(obj_pose_timestamp_msg->data,
374  points_raw_.find(obj_pose_timestamp_msg->data));
375  } else {
376  time_monitor_msg.points_raw = 0;
377  }
378 
379  if (is_points_image_) {
380  // ROS_INFO("points_image");
381  time_monitor_msg.points_image = time_diff(points_raw_.find(obj_pose_timestamp_msg->data),
382  points_image_.find(obj_pose_timestamp_msg->data));
383  }
384  if (is_vscan_image_) {
385  // ROS_INFO("vscan_points");
386  time_monitor_msg.vscan_points = time_diff(points_raw_.find(obj_pose_timestamp_msg->data),
387  vscan_points_.find(obj_pose_timestamp_msg->data));
388  // ROS_INFO("vscan_image");
389  time_monitor_msg.vscan_image = time_diff(vscan_points_.find(obj_pose_timestamp_msg->data),
390  vscan_image_.find(obj_pose_timestamp_msg->data));
391  }
392 
393  // ROS_INFO("image_obj");
394  time_monitor_msg.image_obj = time_diff(image_raw_.find(obj_pose_timestamp_msg->data),
395  image_obj_.find(obj_pose_timestamp_msg->data));
396  // ROS_INFO("image_obj_ranged");
397  time_monitor_msg.image_obj_ranged = time_diff(sync_image_obj_ranged_.find(obj_pose_timestamp_msg->data),
398  image_obj_ranged_.find(obj_pose_timestamp_msg->data));
399  // ROS_INFO("image_obj_tracked");
400  time_monitor_msg.image_obj_tracked = time_diff(sync_image_obj_tracked_.find(obj_pose_timestamp_msg->data),
401  image_obj_tracked_.find(obj_pose_timestamp_msg->data));
402  // ROS_INFO("current_pose");
403  time_monitor_msg.current_pose = time_diff(points_raw_.find(obj_pose_timestamp_msg->data),
404  current_pose_.find(obj_pose_timestamp_msg->data));
405  // ROS_INFO("obj_label");
406  time_monitor_msg.obj_label = time_diff(sync_obj_label_.find(obj_pose_timestamp_msg->data),
407  obj_label_.find(obj_pose_timestamp_msg->data));
408  // ROS_INFO("cluster_centroids");
409  time_monitor_msg.cluster_centroids = time_diff(points_raw_.find(obj_pose_timestamp_msg->data),
410  cluster_centroids_.find(obj_pose_timestamp_msg->data));
411  // ROS_INFO("obj_pose");
412  time_monitor_msg.obj_pose = time_diff(sync_obj_pose_.find(obj_pose_timestamp_msg->data),
413  obj_pose_.find(obj_pose_timestamp_msg->data));
414  // ROS_INFO("-------------------------------------");
415 
416  if (!ros::Time::isSimTime()) {
417  time_monitor_msg.execution_time = time_diff(obj_pose_timestamp_msg->data, obj_pose_.find(obj_pose_timestamp_msg->data));
418  } else {
419  time_monitor_msg.execution_time = time_diff(points_raw_.find(obj_pose_timestamp_msg->data), obj_pose_.find(obj_pose_timestamp_msg->data));
420  }
421 
422  time_monitor_msg.cycle_time = time_diff(pre_sensor_time, obj_pose_timestamp_msg->data); // cycle time
423  time_monitor_msg.time_diff = ros_time2msec(time_diff_.find(obj_pose_timestamp_msg->data)); // time difference
424  time_monitor_pub.publish(time_monitor_msg);
425 
426  pre_sensor_time = obj_pose_timestamp_msg->data;
427 }
