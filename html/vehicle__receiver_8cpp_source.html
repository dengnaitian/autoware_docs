47 static bool parseCanValue(const std::string &can_data, autoware_can_msgs::CANInfo &msg)
48 {
49  std::istringstream ss(can_data);
50  std::vector<std::string> columns;
51  std::string column;
52  while (std::getline(ss, column, ','))
53  {
54  columns.push_back(column);
55  }
56 
57  for (std::size_t i = 0; i < columns.size(); i += 2)
58  {
59  int key = std::stoi(columns[i]);
60  switch (key)
61  {
62  case CAN_KEY_MODE:
63  mode = std::stoi(columns[i + 1]);
64  if (mode & 0x1)
65  {
66  msg.devmode = 1;
67  }
68  else
69  {
70  msg.devmode = 0;
71  }
72 
73  if (mode & 0x2)
74  {
75  msg.strmode = 1;
76  }
77  else
78  {
79  msg.strmode = 0;
80  }
81  break;
82  case CAN_KEY_TIME:
83  msg.tm = columns[i + 1].substr(1, columns[i + 1].length() - 2); // skip '
84  break;
85  case CAN_KEY_VELOC:
86  msg.speed = std::stod(columns[i + 1]);
87  break;
88  case CAN_KEY_ANGLE:
89  msg.angle = std::stod(columns[i + 1]);
90  break;
91  case CAN_KEY_TORQUE:
92  msg.torque = std::stoi(columns[i + 1]);
93  break;
94  case CAN_KEY_ACCEL:
95  msg.drivepedal = std::stoi(columns[i + 1]);
96  break;
97  case CAN_KEY_BRAKE:
98  msg.brakepedal = std::stoi(columns[i + 1]);
99  break;
100  case CAN_KEY_SHIFT:
101  msg.driveshift = std::stoi(columns[i + 1]);
102  break;
103  default:
104  std::cout << "Warning: unknown key : " << key << std::endl;
105  }
106  }
107 
108  return true;
109 }

184 static void *receiverCaller(void *unused)
185 {

188  int sock = socket(AF_INET, SOCK_STREAM, 0);
189  if (sock == -1)
190  {

195  sockaddr_in client;

197  sockaddr_in addr;
198 
199  std::memset(&addr, 0, sizeof(sockaddr_in));
200  addr.sin_family = PF_INET;

203  // make it available immediately to connect
