104 void OpenPlannerSimulatorSigns::VisualizeTrafficLight(autoware_msgs::Signals& _signals)
105 {
106  std::vector<PlannerHNS::TrafficLight> simulatedLights;
107  for(unsigned int i = 0 ; i < _signals.Signals.size() ; i++)
108  {
109  PlannerHNS::TrafficLight tl;
110  tl.id = _signals.Signals.at(i).signalId;
111 
112  for(unsigned int k = 0; k < m_Map.trafficLights.size(); k++)
113  {
114  if(m_Map.trafficLights.at(k).id == tl.id)
115  {
116  tl.pos = m_Map.trafficLights.at(k).pos;
117  break;
118  }
119  }
120 
121  if(_signals.Signals.at(i).type == 1)
122  {
123  tl.lightState = PlannerHNS::GREEN_LIGHT;
124  }
125  else
126  {
127  tl.lightState = PlannerHNS::RED_LIGHT;
128  }
129 
130  simulatedLights.push_back(tl);
131  }
132 
133  //visualize traffic light
134  visualization_msgs::MarkerArray lights;
135  PlannerHNS::ROSHelpers::GetTrafficLightForVisualization(simulatedLights, lights);
136  pub_TrafficLightsRviz.publish(lights);
137 }

139 void OpenPlannerSimulatorSigns::MainLoop()
140 {
141 
142  ros::Rate loop_rate(10);
143 
144  while (ros::ok())
145  {
146  ros::spinOnce();
147 
148  if(m_MapType == PlannerHNS::MAP_KML_FILE && !bMap)
149  {
150  bMap = true;
151  PlannerHNS::MappingHelpers::LoadKML(m_MapPath, m_Map);
152  }
153  else if (m_MapType == PlannerHNS::MAP_FOLDER && !bMap)
154  {
155  bMap = true;
156  PlannerHNS::MappingHelpers::ConstructRoadNetworkFromDataFiles(m_MapPath, m_Map, true);
157  }
158  else if (m_MapType == PlannerHNS::MAP_AUTOWARE && !bMap)
159  {
160  std::vector<UtilityHNS::AisanDataConnFileReader::DataConn> conn_data;;
161 
162  if(m_MapRaw.GetVersion()==2)
163  {
164  PlannerHNS::MappingHelpers::ConstructRoadNetworkFromROSMessageV2(m_MapRaw.pLanes->m_data_list, m_MapRaw.pPoints->m_data_list,
165  m_MapRaw.pCenterLines->m_data_list, m_MapRaw.pIntersections->m_data_list,m_MapRaw.pAreas->m_data_list,
166  m_MapRaw.pLines->m_data_list, m_MapRaw.pStopLines->m_data_list, m_MapRaw.pSignals->m_data_list,
167  m_MapRaw.pVectors->m_data_list, m_MapRaw.pCurbs->m_data_list, m_MapRaw.pRoadedges->m_data_list, m_MapRaw.pWayAreas->m_data_list,
168  m_MapRaw.pCrossWalks->m_data_list, m_MapRaw.pNodes->m_data_list, conn_data,
169  m_MapRaw.pLanes, m_MapRaw.pPoints, m_MapRaw.pNodes, m_MapRaw.pLines, PlannerHNS::GPSPoint(), m_Map, true);
170 
171  if(m_Map.roadSegments.size() > 0)
172  {
173  bMap = true;
174  std::cout << " ******* Map V2 Is Loaded successfully from the Sign Simulator!! " << std::endl;
175  }
176  }
177  else if(m_MapRaw.GetVersion()==1)
178  {
179  PlannerHNS::MappingHelpers::ConstructRoadNetworkFromROSMessage(m_MapRaw.pLanes->m_data_list, m_MapRaw.pPoints->m_data_list,
180  m_MapRaw.pCenterLines->m_data_list, m_MapRaw.pIntersections->m_data_list,m_MapRaw.pAreas->m_data_list,
181  m_MapRaw.pLines->m_data_list, m_MapRaw.pStopLines->m_data_list, m_MapRaw.pSignals->m_data_list,
182  m_MapRaw.pVectors->m_data_list, m_MapRaw.pCurbs->m_data_list, m_MapRaw.pRoadedges->m_data_list, m_MapRaw.pWayAreas->m_data_list,
183  m_MapRaw.pCrossWalks->m_data_list, m_MapRaw.pNodes->m_data_list, conn_data, PlannerHNS::GPSPoint(), m_Map, true);
184 
185  if(m_Map.roadSegments.size() > 0)
186  {
187  bMap = true;
188  std::cout << " ******* Map V1 Is Loaded successfully from the Sign Simulator !! " << std::endl;
189  }
190  }
191  }
192 
193  if(m_CurrLightState == PlannerHNS::GREEN_LIGHT)
194  {
195  //std::cout << "Greeeeeen" << std::endl;
196  if(UtilityHNS::UtilityH::GetTimeDiffNow(m_Timer) < m_Params.firstGreenTime)
197  {
198  for(unsigned int i =0; i<m_FirstSignals.Signals.size(); i++)
199  m_FirstSignals.Signals.at(i).type = 1;
200 
201  for(unsigned int i =0; i<m_SecondSignals.Signals.size(); i++)
202  m_SecondSignals.Signals.at(i).type = 0;
203  }
204  else
205  {
206  m_CurrLightState = PlannerHNS::YELLOW_LIGHT;
207  UtilityHNS::UtilityH::GetTickCount(m_Timer);
208  }
209  }
210  else if(m_CurrLightState == PlannerHNS::YELLOW_LIGHT)
211  {
212  //std::cout << "Yelowwwwww" << std::endl;
213  if(UtilityHNS::UtilityH::GetTimeDiffNow(m_Timer) < m_Params.firstyellowTime)
214  {
215  for(unsigned int i =0; i< m_FirstSignals.Signals.size(); i++)
216  m_FirstSignals.Signals.at(i).type = 0;
217 
218  for(unsigned int i =0; i<m_SecondSignals.Signals.size(); i++)
219  m_SecondSignals.Signals.at(i).type = 0;
220  }
221  else
222  {
223  m_CurrLightState = PlannerHNS::RED_LIGHT;
224  UtilityHNS::UtilityH::GetTickCount(m_Timer);
225  }
226  }
227  else if(m_CurrLightState == PlannerHNS::RED_LIGHT)
228  {
229  //std::cout << "Reeeeeeed" << std::endl;
230  if(UtilityHNS::UtilityH::GetTimeDiffNow(m_Timer) < m_Params.secondGreenTime)
231  {
232  for(unsigned int i =0; i<m_FirstSignals.Signals.size(); i++)
233  m_FirstSignals.Signals.at(i).type = 0;
234 
235  for(unsigned int i =0; i<m_SecondSignals.Signals.size(); i++)
236  m_SecondSignals.Signals.at(i).type = 1;
237  }
238  else
239  {
240  m_CurrLightState = PlannerHNS::FLASH_YELLOW;
241  UtilityHNS::UtilityH::GetTickCount(m_Timer);
242  }
243  }
244  else if(m_CurrLightState == PlannerHNS::FLASH_YELLOW)
245  {
246  //std::cout << "Yelowwwwww" << std::endl;
247  if(UtilityHNS::UtilityH::GetTimeDiffNow(m_Timer) < m_Params.secondyellowTime)
248  {
249  for(unsigned int i =0; i<m_FirstSignals.Signals.size(); i++)
250  m_FirstSignals.Signals.at(i).type = 0;
251 
252  for(unsigned int i =0; i<m_SecondSignals.Signals.size(); i++)
253  m_SecondSignals.Signals.at(i).type = 0;
254  }
255  else
256  {
257  m_CurrLightState = PlannerHNS::GREEN_LIGHT;
258  UtilityHNS::UtilityH::GetTickCount(m_Timer);
259  }
260  }
261 
262  autoware_msgs::Signals all_signals;
263  all_signals.Signals.insert(all_signals.Signals.end(), m_FirstSignals.Signals.begin(), m_FirstSignals.Signals.end());
264  all_signals.Signals.insert(all_signals.Signals.end(), m_SecondSignals.Signals.begin(), m_SecondSignals.Signals.end());
265 
266  //std::cout << "Number of Signals before send: " << all_signals.Signals.size() << std::endl;
267  pub_trafficLights.publish(all_signals);
268 
269  if(bMap)
270  VisualizeTrafficLight(all_signals);
271 
272  loop_rate.sleep();
273  }
