75 geometry_msgs::Point RequestQueue::dequeue()
76 {
77  std::unique_lock<std::mutex> lock(mtx_);
78  while (queue_.empty() && look_ahead_queue_.empty())
79  cv_.wait(lock);
80  if (!queue_.empty()) {

237 void download_map()
238 {
239  while (true) {
240  geometry_msgs::Point p = request_queue.dequeue();
241 
242  int x = static_cast<int>(p.x);
243  int y = static_cast<int>(p.y);
244  int x_min = static_cast<int>(p.x - margin);
245  int y_min = static_cast<int>(p.y - margin);
246  int x_max = static_cast<int>(p.x + margin);
247  int y_max = static_cast<int>(p.y + margin);
248 
249  std::vector<std::string> locs;
250  locs.push_back(create_location(x, y));
251  locs.push_back(create_location(x_min, y_min));
252  locs.push_back(create_location(x_min, y_max));
253  locs.push_back(create_location(x_max, y_min));
254  locs.push_back(create_location(x_max, y_max));
255  for (const std::string& loc : locs) { // XXX better way?
256  std::string arealist_path = TEMPORARY_DIRNAME + loc + AREALIST_FILENAME;
257 
258  bool cached = false;
259  for (const std::string& path : cached_arealist_paths) {
260  if (path == arealist_path) {
261  cached = true;
262  break;
263  }
264  }
265  if (cached)
266  continue;
267 
268  AreaList areas;
269  if (is_downloaded(arealist_path))
270  areas = read_arealist(arealist_path);
271  else {
272  if (download(gf, TEMPORARY_DIRNAME, loc, AREALIST_FILENAME) != 0)
273  continue;
274  areas = read_arealist(arealist_path);
275  for (Area& area : areas)
276  area.path = TEMPORARY_DIRNAME + loc + basename(area.path.c_str());
277  write_arealist(arealist_path, areas);
278  }
279  for (const Area& area : areas)
280  cache_arealist(area, all_areas);
281  cached_arealist_paths.push_back(arealist_path);
282  }
283 
284  for (const Area& area : all_areas) {
285  if (is_in_area(p.x, p.y, area, margin)) {
286  int x_area = static_cast<int>(area.x_max - MARGIN_UNIT);
287  int y_area = static_cast<int>(area.y_max - MARGIN_UNIT);
288  std::string loc = create_location(x_area, y_area);
289  if (is_downloaded(area.path) ||
290  download(gf, TEMPORARY_DIRNAME, loc, basename(area.path.c_str())) == 0) {
291  std::unique_lock<std::mutex> lock(downloaded_areas_mtx);
292  cache_arealist(area, downloaded_areas);
293  }
294  }
295  }
296  }
297 }

408 void request_lookahead_download(const autoware_msgs::LaneArray& msg)
409 {
410  request_queue.clear_look_ahead();
411 
412  for (const autoware_msgs::Lane& l : msg.lanes) {
413  size_t end = l.waypoints.size() - 1;
414  double distance = 0;
415  double threshold = (MARGIN_UNIT / 2) + margin; // XXX better way?
416  for (size_t i = 0; i <= end; ++i) {
417  if (i == 0 || i == end) {
418  geometry_msgs::Point p;
419  p.x = l.waypoints[i].pose.pose.position.x;
420  p.y = l.waypoints[i].pose.pose.position.y;
421  request_queue.enqueue_look_ahead(p);
422  } else {
423  geometry_msgs::Point p1, p2;
424  p1.x = l.waypoints[i].pose.pose.position.x;
425  p1.y = l.waypoints[i].pose.pose.position.y;
426  p2.x = l.waypoints[i - 1].pose.pose.position.x;
427  p2.y = l.waypoints[i - 1].pose.pose.position.y;
428  distance += hypot(p2.x - p1.x, p2.y - p1.y);
429  if (distance > threshold) {
430  request_queue.enqueue_look_ahead(p1);
431  distance = 0;
432  }
433  }
434  }
435  }

541  if (path == area.path)
