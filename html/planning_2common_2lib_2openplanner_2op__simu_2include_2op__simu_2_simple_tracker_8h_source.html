53  Kalman1dState Update(double measurement)
54  {
55  //prediction update
56  result.p = result.p + result.MovCov;
57 
58  //measurement update
59  result.k = result.p / (result.p + result.MeasureCov);
60  result.x = result.x + result.k * (measurement - result.x);
61  result.p = (1 - result.k) * result.p;
62 
63  return result;
64  }
