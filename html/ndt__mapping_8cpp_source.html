297 static void imu_calc(ros::Time current_time)
298 {
299  static ros::Time previous_time = current_time;
300  double diff_time = (current_time - previous_time).toSec();
301 
302  double diff_imu_roll = imu.angular_velocity.x * diff_time;
303  double diff_imu_pitch = imu.angular_velocity.y * diff_time;
304  double diff_imu_yaw = imu.angular_velocity.z * diff_time;
305 
306  current_pose_imu.roll += diff_imu_roll;
307  current_pose_imu.pitch += diff_imu_pitch;
308  current_pose_imu.yaw += diff_imu_yaw;
309 
310  double accX1 = imu.linear_acceleration.x;
311  double accY1 = std::cos(current_pose_imu.roll) * imu.linear_acceleration.y -
312  std::sin(current_pose_imu.roll) * imu.linear_acceleration.z;
313  double accZ1 = std::sin(current_pose_imu.roll) * imu.linear_acceleration.y +
314  std::cos(current_pose_imu.roll) * imu.linear_acceleration.z;
315 
316  double accX2 = std::sin(current_pose_imu.pitch) * accZ1 + std::cos(current_pose_imu.pitch) * accX1;
317  double accY2 = accY1;
318  double accZ2 = std::cos(current_pose_imu.pitch) * accZ1 - std::sin(current_pose_imu.pitch) * accX1;
319 
320  double accX = std::cos(current_pose_imu.yaw) * accX2 - std::sin(current_pose_imu.yaw) * accY2;
321  double accY = std::sin(current_pose_imu.yaw) * accX2 + std::cos(current_pose_imu.yaw) * accY2;
322  double accZ = accZ2;
323 
324  offset_imu_x += current_velocity_imu_x * diff_time + accX * diff_time * diff_time / 2.0;
325  offset_imu_y += current_velocity_imu_y * diff_time + accY * diff_time * diff_time / 2.0;
326  offset_imu_z += current_velocity_imu_z * diff_time + accZ * diff_time * diff_time / 2.0;
327 
328  current_velocity_imu_x += accX * diff_time;
329  current_velocity_imu_y += accY * diff_time;
330  current_velocity_imu_z += accZ * diff_time;
331 
332  offset_imu_roll += diff_imu_roll;
333  offset_imu_pitch += diff_imu_pitch;
334  offset_imu_yaw += diff_imu_yaw;
335 
336  guess_pose_imu.x = previous_pose.x + offset_imu_x;
337  guess_pose_imu.y = previous_pose.y + offset_imu_y;
338  guess_pose_imu.z = previous_pose.z + offset_imu_z;
339  guess_pose_imu.roll = previous_pose.roll + offset_imu_roll;
340  guess_pose_imu.pitch = previous_pose.pitch + offset_imu_pitch;
341  guess_pose_imu.yaw = previous_pose.yaw + offset_imu_yaw;
342 
343  previous_time = current_time;
344 }

400 static void imu_callback(const sensor_msgs::Imu::Ptr& input)
401 {
402  // std::cout << __func__ << std::endl;
403 
404  if (_imu_upside_down)
405  imuUpsideDown(input);
406 
407  const ros::Time current_time = input->header.stamp;
408  static ros::Time previous_time = current_time;
409  const double diff_time = (current_time - previous_time).toSec();
410 
411  double imu_roll, imu_pitch, imu_yaw;
412  tf::Quaternion imu_orientation;
413  tf::quaternionMsgToTF(input->orientation, imu_orientation);
414  tf::Matrix3x3(imu_orientation).getRPY(imu_roll, imu_pitch, imu_yaw);
415 
416  imu_roll = wrapToPmPi(imu_roll);
417  imu_pitch = wrapToPmPi(imu_pitch);
418  imu_yaw = wrapToPmPi(imu_yaw);
419 
420  static double previous_imu_roll = imu_roll, previous_imu_pitch = imu_pitch, previous_imu_yaw = imu_yaw;
421  const double diff_imu_roll = calcDiffForRadian(imu_roll, previous_imu_roll);
422  const double diff_imu_pitch = calcDiffForRadian(imu_pitch, previous_imu_pitch);
423  const double diff_imu_yaw = calcDiffForRadian(imu_yaw, previous_imu_yaw);
424 
425  imu.header = input->header;
426  imu.linear_acceleration.x = input->linear_acceleration.x;
427  // imu.linear_acceleration.y = input->linear_acceleration.y;
428  // imu.linear_acceleration.z = input->linear_acceleration.z;
429  imu.linear_acceleration.y = 0;
430  imu.linear_acceleration.z = 0;
431 
432  if (diff_time != 0)
433  {
434  imu.angular_velocity.x = diff_imu_roll / diff_time;
435  imu.angular_velocity.y = diff_imu_pitch / diff_time;
436  imu.angular_velocity.z = diff_imu_yaw / diff_time;
437  }
438  else
439  {
440  imu.angular_velocity.x = 0;
441  imu.angular_velocity.y = 0;
442  imu.angular_velocity.z = 0;
443  }
444 
445  imu_calc(input->header.stamp);
446 
447  previous_time = current_time;
448  previous_imu_roll = imu_roll;
449  previous_imu_pitch = imu_pitch;
450  previous_imu_yaw = imu_yaw;
451 }

453 static void points_callback(const sensor_msgs::PointCloud2::ConstPtr& input)
454 {
455  double r;
456  pcl::PointXYZI p;
457  pcl::PointCloud<pcl::PointXYZI> tmp, scan;
458  pcl::PointCloud<pcl::PointXYZI>::Ptr filtered_scan_ptr(new pcl::PointCloud<pcl::PointXYZI>());
459  pcl::PointCloud<pcl::PointXYZI>::Ptr transformed_scan_ptr(new pcl::PointCloud<pcl::PointXYZI>());
460  tf::Quaternion q;
461 
462  Eigen::Matrix4f t_localizer(Eigen::Matrix4f::Identity());
463  Eigen::Matrix4f t_base_link(Eigen::Matrix4f::Identity());
464  static tf::TransformBroadcaster br;
465  tf::Transform transform;
466 
467  current_scan_time = input->header.stamp;
468 
469  pcl::fromROSMsg(*input, tmp);
470 
471  for (pcl::PointCloud<pcl::PointXYZI>::const_iterator item = tmp.begin(); item != tmp.end(); item++)
472  {
473  p.x = (double)item->x;
474  p.y = (double)item->y;
475  p.z = (double)item->z;
476  p.intensity = (double)item->intensity;
477 
478  r = sqrt(pow(p.x, 2.0) + pow(p.y, 2.0));
479  if (min_scan_range < r && r < max_scan_range)
480  {
481  scan.push_back(p);
482  }
483  }
484 
485  pcl::PointCloud<pcl::PointXYZI>::Ptr scan_ptr(new pcl::PointCloud<pcl::PointXYZI>(scan));
486 
487  // Add initial point cloud to velodyne_map
488  if (initial_scan_loaded == 0)
489  {
490  pcl::transformPointCloud(*scan_ptr, *transformed_scan_ptr, tf_btol);
491  map += *transformed_scan_ptr;
492  initial_scan_loaded = 1;
493  }
494 
495  // Apply voxelgrid filter
496  pcl::VoxelGrid<pcl::PointXYZI> voxel_grid_filter;
497  voxel_grid_filter.setLeafSize(voxel_leaf_size, voxel_leaf_size, voxel_leaf_size);
498  voxel_grid_filter.setInputCloud(scan_ptr);
499  voxel_grid_filter.filter(*filtered_scan_ptr);
500 
501  pcl::PointCloud<pcl::PointXYZI>::Ptr map_ptr(new pcl::PointCloud<pcl::PointXYZI>(map));
502 
503  if (_method_type == MethodType::PCL_GENERIC)
504  {
505  ndt.setTransformationEpsilon(trans_eps);
506  ndt.setStepSize(step_size);
507  ndt.setResolution(ndt_res);
508  ndt.setMaximumIterations(max_iter);
509  ndt.setInputSource(filtered_scan_ptr);
510  }
511  else if (_method_type == MethodType::PCL_ANH)
512  {
513  anh_ndt.setTransformationEpsilon(trans_eps);
514  anh_ndt.setStepSize(step_size);
515  anh_ndt.setResolution(ndt_res);
516  anh_ndt.setMaximumIterations(max_iter);
517  anh_ndt.setInputSource(filtered_scan_ptr);
518  }
519 #ifdef CUDA_FOUND
520  else if (_method_type == MethodType::PCL_ANH_GPU)
521  {
522  anh_gpu_ndt.setTransformationEpsilon(trans_eps);
523  anh_gpu_ndt.setStepSize(step_size);
