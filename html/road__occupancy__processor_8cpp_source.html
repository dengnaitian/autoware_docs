25 void ROSRoadOccupancyProcessorApp::ConvertXYZIToRTZ(const pcl::PointCloud<pcl::PointXYZI>::Ptr in_cloud,
26  ROSRoadOccupancyProcessorApp::PointCloudXYZIRTColor &out_organized_points,
27  std::vector<pcl::PointIndices> &out_radial_divided_indices,
28  std::vector<ROSRoadOccupancyProcessorApp::PointCloudXYZIRTColor> &out_radial_ordered_clouds)
29 {
30  out_organized_points.resize(in_cloud->points.size());
31  out_radial_divided_indices.clear();
32  out_radial_divided_indices.resize(radial_dividers_num_);
33  out_radial_ordered_clouds.resize(radial_dividers_num_);
34 
35  for(size_t i=0; i< in_cloud->points.size(); i++)
36  {
37  PointXYZIRT new_point;
38  auto radius = (float) sqrt(
39  in_cloud->points[i].x*in_cloud->points[i].x
40  + in_cloud->points[i].y*in_cloud->points[i].y
41  );
42  auto theta = (float) atan2(in_cloud->points[i].y, in_cloud->points[i].x) * 180 / M_PI;
43  if (theta < 0){ theta+=360; }
44 
45  auto radial_div = (size_t) floor(theta/radial_divider_angle_);
46  auto concentric_div = (size_t) floor(fabs(radius/concentric_divider_distance_));
47 
48  new_point.point = in_cloud->points[i];
49  new_point.radius = radius;
50  new_point.theta = theta;
51  new_point.radial_div = radial_div;
52  new_point.concentric_div = concentric_div;
53  new_point.original_index = i;
54 
55  out_organized_points[i] = new_point;
56 
57  //radial divisions
58  out_radial_divided_indices[radial_div].indices.push_back(i);
59 
60  out_radial_ordered_clouds[radial_div].push_back(new_point);
61 
62  }//end for
63 
64  //order radial points on each division
65 #pragma omp for
66  for(size_t i=0; i< radial_dividers_num_; i++)
67  {
68  std::sort(out_radial_ordered_clouds[i].begin(), out_radial_ordered_clouds[i].end(),
69  [](const PointXYZIRT& a, const PointXYZIRT& b){ return a.radius < b.radius; });
70  }
71 }

111 void ROSRoadOccupancyProcessorApp::Convert3dPointToOccupancy(grid_map::GridMap &in_grid_map,
112  const geometry_msgs::Point &in_point, cv::Point &out_point)
113 {
114  // calculate position
115  grid_map::Position map_pos = in_grid_map.getPosition();
116  double origin_x_offset = in_grid_map.getLength().x() / 2.0 - map_pos.x();
117  double origin_y_offset = in_grid_map.getLength().y() / 2.0 - map_pos.y();
118  // coordinate conversion for cv image
119  out_point.x = (in_grid_map.getLength().y() - origin_y_offset - in_point.y) / in_grid_map.getResolution();
120  out_point.y = (in_grid_map.getLength().x() - origin_x_offset - in_point.x) / in_grid_map.getResolution();
121 }

200 void ROSRoadOccupancyProcessorApp::PointsCallback(const sensor_msgs::PointCloud2::ConstPtr &in_ground_cloud_msg,
201  const sensor_msgs::PointCloud2::ConstPtr &in_no_ground_cloud_msg)
202 {
203  if(road_wayarea_original_mat_.empty())
204  return;
205 
206  // timer start
207  //auto start = std::chrono::system_clock::now();
208 
209  cv::Mat current_road_mat = road_wayarea_original_mat_.clone();
210  cv::Mat original_road_mat = current_road_mat.clone();
211 
212  grid_map::GridMap output_gridmap;
213  output_gridmap.setFrameId(input_gridmap_frame_);
214  output_gridmap.setGeometry(input_gridmap_length_,
215  input_gridmap_resolution_,
216  input_gridmap_position_);
217 
218  //output_gridmap[output_layer_name_].setConstant(OCCUPANCY_NO_ROAD);
219 
220  pcl::PointCloud<pcl::PointXYZI>::Ptr in_ground_cloud(new pcl::PointCloud<pcl::PointXYZI>);
221  pcl::PointCloud<pcl::PointXYZI>::Ptr in_no_ground_cloud(new pcl::PointCloud<pcl::PointXYZI>);
222  pcl::PointCloud<pcl::PointXYZI>::Ptr final_ground_cloud(new pcl::PointCloud<pcl::PointXYZI>);
223  pcl::PointCloud<pcl::PointXYZI>::Ptr final_no_ground_cloud(new pcl::PointCloud<pcl::PointXYZI>);
224 
225  pcl::fromROSMsg(*in_ground_cloud_msg, *in_ground_cloud);
226  pcl::fromROSMsg(*in_no_ground_cloud_msg, *in_no_ground_cloud);
227 
228  //check that both pointcloud and grid are in the same frame, otherwise transform
229  ConvertPointCloud(*in_ground_cloud, output_gridmap.getFrameId(), *final_ground_cloud);
230  ConvertPointCloud(*in_no_ground_cloud, output_gridmap.getFrameId(), *final_no_ground_cloud);
231 
232  //organize pointcloud in cylindrical coords
233  PointCloudXYZIRTColor organized_points;
234  std::vector<pcl::PointIndices> radial_division_indices;
235  std::vector<pcl::PointIndices> closest_indices;
236  std::vector<PointCloudXYZIRTColor> radial_ordered_clouds;
237 
238  ConvertXYZIToRTZ(final_ground_cloud,
239  organized_points,
240  radial_division_indices,
241  radial_ordered_clouds);
242 
243  //draw lines between initial and last point of each ray
244  for (size_t i = 0; i < radial_ordered_clouds.size(); i++)//sweep through each radial division
245  {
246  geometry_msgs::Point prev_point;
247  for (size_t j = 0; j < radial_ordered_clouds[i].size(); j++)//loop through each point
248  {
249  geometry_msgs::Point current_point;
250  current_point.x = radial_ordered_clouds[i][j].point.x;
251  current_point.y = radial_ordered_clouds[i][j].point.y;
252  current_point.z = radial_ordered_clouds[i][j].point.z;
253 
254  DrawLineInGridMap(output_gridmap, current_road_mat, prev_point, current_point, OCCUPANCY_ROAD_FREE);
255  }
256  }
257 
258  //process obstacle points
259  for (const auto &point:final_no_ground_cloud->points)
260  {
261  geometry_msgs::Point sensor_point;
262  sensor_point.x = point.x;
263  sensor_point.y = point.y;
264  sensor_point.z = point.z;
265  SetPointInGridMap(output_gridmap, current_road_mat, sensor_point, OCCUPANCY_ROAD_OCCUPIED);
266  }
267 
268 #pragma omp for
269  for(int row = 0; row < current_road_mat.rows; row++)
270  {
271  for(int col = 0; col < current_road_mat.cols; col++)
272  {
273  if(original_road_mat.at<uchar>(row,col) == OCCUPANCY_NO_ROAD)
274  {
275  current_road_mat.at<uchar>(row,col) = OCCUPANCY_NO_ROAD;
276  }
277  }
278  }
279  //cv::imshow("result", current_road_mat);
280  //cv::waitKey(10);
281  LoadRoadLayerFromMat(output_gridmap, current_road_mat);
282 
283  PublishGridMap(output_gridmap, output_layer_name_);
284 
285  // timer end
286  //auto end = std::chrono::system_clock::now();
287  //auto usec = std::chrono::duration_cast<std::chrono::microseconds>(end - start).count();
288  //std::cout << "time: " << usec / 1000.0 << " [msec]" << std::endl;
289 
290 }
