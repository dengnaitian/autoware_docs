319 // always ocurring consecutively
320 inline const char * getCudaDrvErrorString(CUresult error_id)
321 {
322  int index = 0;
323  while (sCudaDrvErrorString[index].error_id != error_id &&
324  sCudaDrvErrorString[index].error_id != -1)
325  {
326  index++;
327  }
328 
329  if(sCudaDrvErrorString[index].error_id == error_id)
330  return (const char *)sCudaDrvErrorString[index].error_string;
331  else
332  return (const char *)"CUDA_ERROR not found!";
333 }
