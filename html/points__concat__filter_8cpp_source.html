64  private_node_handle_.param("input_topics", input_topics_, std::string("[/points_alpha, /points_beta]"));

86  }

95 void PointsConcatFilter::pointcloud_callback(const PointCloudMsgT::ConstPtr &msg1, const PointCloudMsgT::ConstPtr &msg2,
96  const PointCloudMsgT::ConstPtr &msg3, const PointCloudMsgT::ConstPtr &msg4,
97  const PointCloudMsgT::ConstPtr &msg5, const PointCloudMsgT::ConstPtr &msg6,
98  const PointCloudMsgT::ConstPtr &msg7, const PointCloudMsgT::ConstPtr &msg8)
99 {
100  assert(2 <= input_topics_size_ && input_topics_size_ <= 8);
101 
102  PointCloudMsgT::ConstPtr msgs[8] = { msg1, msg2, msg3, msg4, msg5, msg6, msg7, msg8 };
103  PointCloudT::Ptr cloud_sources[8];
104  PointCloudT::Ptr cloud_concatenated(new PointCloudT);
105 
106  // transform points
107  try
108  {
109  for (size_t i = 0; i < input_topics_size_; ++i)
110  {
111  // Note: If you use kinetic, you can directly receive messages as
112  // PointCloutT.
113  cloud_sources[i] = PointCloudT().makeShared();
114  pcl::fromROSMsg(*msgs[i], *cloud_sources[i]);
115  tf_listener_.waitForTransform(output_frame_id_, msgs[i]->header.frame_id, ros::Time(0), ros::Duration(1.0));
116  pcl_ros::transformPointCloud(output_frame_id_, *cloud_sources[i], *cloud_sources[i], tf_listener_);
117  }
118  }
119  catch (tf::TransformException &ex)
120  {
121  ROS_ERROR("%s", ex.what());
122  return;
123  }
124 
125  // merge points
126  for (size_t i = 0; i < input_topics_size_; ++i)
127  {
128  *cloud_concatenated += *cloud_sources[i];
129  }
130 
131  // publsh points
132  cloud_concatenated->header = pcl_conversions::toPCL(msgs[0]->header);
133  cloud_concatenated->header.frame_id = output_frame_id_;
134  cloud_publisher_.publish(cloud_concatenated);
135 }
