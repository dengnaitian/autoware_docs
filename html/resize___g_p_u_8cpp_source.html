434 void dpm_ttic_gpu_calc_resized_image_size(int *org_image_size,
435  int *resized_image_size,
436  int interval,
437  FLOAT sc,
438  int max_scale,
439  FLOAT *scale_array)
440 {
441  const int org_height = org_image_size[0];
442  const int org_width = org_image_size[1];
443  const int org_nChannels = org_image_size[2];
444 
445  for (int ii=0; ii<interval; ii++)
446  {
447  /* resizing rate */
448  FLOAT scale = 1.0/pow(sc, ii);
449 
450  /* calculte and assign resized image size */
451  if (scale==1.0)
452  {
453  resized_image_size[ii*3] = org_height;
454  resized_image_size[ii*3 + 1] = org_width;
455  resized_image_size[ii*3 + 2] = org_nChannels;
456  }
457  else
458  {
459  resized_image_size[ii*3] = (int)((FLOAT)org_height*scale + 0.5);
460  resized_image_size[ii*3 + 1] = (int)((FLOAT)org_width*scale + 0.5);
461  resized_image_size[ii*3 + 2] = org_nChannels;
462  }
463 
464  memcpy(resized_image_size + (ii+interval)*3, resized_image_size + (ii)*3, 3*sizeof(int));
465 
466  /* save scale */
467  scale_array[ii] = scale*2;
468  scale_array[ii+interval] = scale;
469 
470  /* extra resizing */
471  const FLOAT extra_scale = 0.5;
472  for (int jj=ii+interval; jj<max_scale; jj+=interval)
473  {
474  resized_image_size[(jj+interval)*3] = (int)((FLOAT)resized_image_size[jj*3]*extra_scale + 0.5);
475  resized_image_size[(jj+interval)*3 + 1] = (int)((FLOAT)resized_image_size[jj*3 + 1]*extra_scale + 0.5);
476  resized_image_size[(jj+interval)*3 + 2] = resized_image_size[jj*3 + 2];
477 
478  /* save scale */
479  scale_array[jj+interval] = 0.5*scale_array[jj];
480  }
481  }
482 
483  return;
