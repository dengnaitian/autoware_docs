577 void segmentByDistance(const pcl::PointCloud<pcl::PointXYZ>::Ptr in_cloud_ptr,
578  pcl::PointCloud<pcl::PointXYZRGB>::Ptr out_cloud_ptr,
579  autoware_msgs::Centroids &in_out_centroids, autoware_msgs::CloudClusterArray &in_out_clusters)
580 {
581  // cluster the pointcloud according to the distance of the points using different thresholds (not only one for the
582  // entire pc)
583  // in this way, the points farther in the pc will also be clustered
584 
585  // 0 => 0-15m d=0.5
586  // 1 => 15-30 d=1
587  // 2 => 30-45 d=1.6
588  // 3 => 45-60 d=2.1
589  // 4 => >60 d=2.6
590 
591  std::vector<ClusterPtr> all_clusters;
592 
593  if (!_use_multiple_thres)
594  {
595  pcl::PointCloud<pcl::PointXYZ>::Ptr cloud_ptr(new pcl::PointCloud<pcl::PointXYZ>);
596 
597  for (unsigned int i = 0; i < in_cloud_ptr->points.size(); i++)
598  {
599  pcl::PointXYZ current_point;
600  current_point.x = in_cloud_ptr->points[i].x;
601  current_point.y = in_cloud_ptr->points[i].y;
602  current_point.z = in_cloud_ptr->points[i].z;
603 
604  cloud_ptr->points.push_back(current_point);
605  }
606 #ifdef GPU_CLUSTERING
607  if (_use_gpu)
608  {
609  all_clusters = clusterAndColorGpu(cloud_ptr, out_cloud_ptr, in_out_centroids,
610  _clustering_distance);
611  } else
612  {
613  all_clusters =
614  clusterAndColor(cloud_ptr, out_cloud_ptr, in_out_centroids, _clustering_distance);
615  }
616 #else
617  all_clusters =
618  clusterAndColor(cloud_ptr, out_cloud_ptr, in_out_centroids, _clustering_distance);
619 #endif
620  } else
621  {
622  std::vector<pcl::PointCloud<pcl::PointXYZ>::Ptr> cloud_segments_array(5);
623  for (unsigned int i = 0; i < cloud_segments_array.size(); i++)
624  {
625  pcl::PointCloud<pcl::PointXYZ>::Ptr tmp_cloud(new pcl::PointCloud<pcl::PointXYZ>);
626  cloud_segments_array[i] = tmp_cloud;
627  }
628 
629  for (unsigned int i = 0; i < in_cloud_ptr->points.size(); i++)
630  {
631  pcl::PointXYZ current_point;
632  current_point.x = in_cloud_ptr->points[i].x;
633  current_point.y = in_cloud_ptr->points[i].y;
634  current_point.z = in_cloud_ptr->points[i].z;

637

639  {

641  }
642  else if (origin_distance < _clustering_ranges[1])
643  {
644  cloud_segments_array[1]->points.push_back(current_point);
645 
646  }else if (origin_distance < _clustering_ranges[2])
647  {
648  cloud_segments_array[2]->points.push_back(current_point);
649 
650  }else if (origin_distance < _clustering_ranges[3])
651  {
652  cloud_segments_array[3]->points.push_back(current_point);
653 
654  }else
655  {
656  cloud_segments_array[4]->points.push_back(current_point);
657  }
658  }
659 
660  std::vector<ClusterPtr> local_clusters;
661  for (unsigned int i = 0; i < cloud_segments_array.size(); i++)
662  {
663 #ifdef GPU_CLUSTERING
664  if (_use_gpu)
665  {
666  local_clusters = clusterAndColorGpu(cloud_segments_array[i], out_cloud_ptr,

669  {
670  local_clusters = clusterAndColor(cloud_segments_array[i], out_cloud_ptr,
671  in_out_centroids, _clustering_distances[i]);

673 #else
674  local_clusters = clusterAndColor(
675  cloud_segments_array[i], out_cloud_ptr, in_out_centroids, _clustering_distances[i]);

679  }

682  //....
683  // check for mergable clusters
684  std::vector<ClusterPtr> mid_clusters;
685  std::vector<ClusterPtr> final_clusters;
686 
687  if (all_clusters.size() > 0)
688  checkAllForMerge(all_clusters, mid_clusters, _cluster_merge_threshold);
689  else
690  mid_clusters = all_clusters;
691 
692  if (mid_clusters.size() > 0)
693  checkAllForMerge(mid_clusters, final_clusters, _cluster_merge_threshold);

716  color.a = 1;
717  color.r = 1;
718  color.g = 1;
719  color.b = 1;
720  pictogram_cluster.color = color;
721  pictogram_cluster.character = std::to_string(i);
722  // PICTO
723 
724  // pcl::PointXYZ min_point = final_clusters[i]->GetMinPoint();
725  // pcl::PointXYZ max_point = final_clusters[i]->GetMaxPoint();
726  pcl::PointXYZ center_point = final_clusters[i]->GetCentroid();
727  geometry_msgs::Point centroid;

746 void removeFloor(const pcl::PointCloud<pcl::PointXYZ>::Ptr in_cloud_ptr,
747  pcl::PointCloud<pcl::PointXYZ>::Ptr out_nofloor_cloud_ptr,
748  pcl::PointCloud<pcl::PointXYZ>::Ptr out_onlyfloor_cloud_ptr, float in_max_height = 0.2,
749  float in_floor_max_angle = 0.1)
750 {
751 
752  pcl::SACSegmentation<pcl::PointXYZ> seg;
753  pcl::PointIndices::Ptr inliers(new pcl::PointIndices);
754  pcl::ModelCoefficients::Ptr coefficients(new pcl::ModelCoefficients);
755 
756  seg.setOptimizeCoefficients(true);
757  seg.setModelType(pcl::SACMODEL_PERPENDICULAR_PLANE);
758  seg.setMethodType(pcl::SAC_RANSAC);
759  seg.setMaxIterations(100);
760  seg.setAxis(Eigen::Vector3f(0, 0, 1));
761  seg.setEpsAngle(in_floor_max_angle);
762 
763  seg.setDistanceThreshold(in_max_height); // floor distance
764  seg.setOptimizeCoefficients(true);
765  seg.setInputCloud(in_cloud_ptr);
766  seg.segment(*inliers, *coefficients);
767  if (inliers->indices.size() == 0)
768  {
769  std::cout << "Could not estimate a planar model for the given dataset." << std::endl;
770  }
771 
772  // REMOVE THE FLOOR FROM THE CLOUD
773  pcl::ExtractIndices<pcl::PointXYZ> extract;
774  extract.setInputCloud(in_cloud_ptr);
775  extract.setIndices(inliers);
776  extract.setNegative(true); // true removes the indices, false leaves only the indices
777  extract.filter(*out_nofloor_cloud_ptr);
778 
779  // EXTRACT THE FLOOR FROM THE CLOUD
780  extract.setNegative(false); // true removes the indices, false leaves only the indices
781  extract.filter(*out_onlyfloor_cloud_ptr);
782 }

806 void differenceNormalsSegmentation(const pcl::PointCloud<pcl::PointXYZ>::Ptr in_cloud_ptr,
807  pcl::PointCloud<pcl::PointXYZ>::Ptr out_cloud_ptr)
808 {
809  float small_scale = 0.5;
810  float large_scale = 2.0;
811  float angle_threshold = 0.5;
812  pcl::search::Search<pcl::PointXYZ>::Ptr tree;
813  if (in_cloud_ptr->isOrganized())
814  {
815  tree.reset(new pcl::search::OrganizedNeighbor<pcl::PointXYZ>());
816  } else
817  {
818  tree.reset(new pcl::search::KdTree<pcl::PointXYZ>(false));
819  }
820 
821  // Set the input pointcloud for the search tree
822  tree->setInputCloud(in_cloud_ptr);
823 
824  pcl::NormalEstimationOMP<pcl::PointXYZ, pcl::PointNormal> normal_estimation;
825  // pcl::gpu::NormalEstimation<pcl::PointXYZ, pcl::PointNormal> normal_estimation;
826  normal_estimation.setInputCloud(in_cloud_ptr);
827  normal_estimation.setSearchMethod(tree);
828 
829  normal_estimation.setViewPoint(std::numeric_limits<float>::max(), std::numeric_limits<float>::max(),
830  std::numeric_limits<float>::max());
831 
832  pcl::PointCloud<pcl::PointNormal>::Ptr normals_small_scale(new pcl::PointCloud<pcl::PointNormal>);
833  pcl::PointCloud<pcl::PointNormal>::Ptr normals_large_scale(new pcl::PointCloud<pcl::PointNormal>);
834 
835  normal_estimation.setRadiusSearch(small_scale);
836  normal_estimation.compute(*normals_small_scale);
837 
838  normal_estimation.setRadiusSearch(large_scale);
839  normal_estimation.compute(*normals_large_scale);
840 
841  pcl::PointCloud<pcl::PointNormal>::Ptr diffnormals_cloud(new pcl::PointCloud<pcl::PointNormal>);
842  pcl::copyPointCloud<pcl::PointXYZ, pcl::PointNormal>(*in_cloud_ptr, *diffnormals_cloud);
843 
844  // Create DoN operator
845  pcl::DifferenceOfNormalsEstimation<pcl::PointXYZ, pcl::PointNormal, pcl::PointNormal> diffnormals_estimator;
846  diffnormals_estimator.setInputCloud(in_cloud_ptr);
847  diffnormals_estimator.setNormalScaleLarge(normals_large_scale);
848  diffnormals_estimator.setNormalScaleSmall(normals_small_scale);
849 
850  diffnormals_estimator.initCompute();
851 
852  diffnormals_estimator.computeFeature(*diffnormals_cloud);
853 
854  pcl::ConditionOr<pcl::PointNormal>::Ptr range_cond(new pcl::ConditionOr<pcl::PointNormal>());
855  range_cond->addComparison(pcl::FieldComparison<pcl::PointNormal>::ConstPtr(
856  new pcl::FieldComparison<pcl::PointNormal>("curvature", pcl::ComparisonOps::GT, angle_threshold)));
857  // Build the filter
858  pcl::ConditionalRemoval<pcl::PointNormal> cond_removal;
859  cond_removal.setCondition(range_cond);
860  cond_removal.setInputCloud(diffnormals_cloud);
861 
862  pcl::PointCloud<pcl::PointNormal>::Ptr diffnormals_cloud_filtered(new pcl::PointCloud<pcl::PointNormal>);
863 
864  // Apply filter
865  cond_removal.filter(*diffnormals_cloud_filtered);
866 
867  pcl::copyPointCloud<pcl::PointNormal, pcl::PointXYZ>(*diffnormals_cloud, *out_cloud_ptr);
868 }
