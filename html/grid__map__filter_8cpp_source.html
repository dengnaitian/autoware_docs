99  void GridMapFilter::CreateDistanceTransformLayer(grid_map::GridMap &out_grid_map, const std::string &in_layer)
100  {
101  cv::Mat original_image;
102  if (!out_grid_map.exists(in_layer))
103  {
104  ROS_INFO("%s layer not yet available", in_layer.c_str());
105  return;
106  }
107  grid_map::GridMapCvConverter::toImage<unsigned char, 1>(out_grid_map,
108  in_layer,
109  CV_8UC1,
110  original_image);
111 
112  cv::Mat binary_image;
113  cv::threshold(original_image,
114  binary_image,
115  fill_circle_cost_thresh_,
116  grid_max_value_,
117  cv::THRESH_BINARY_INV);
118 
119  // distance transform method
120  // 3: fast
121  // 5: slow but accurate
122  cv::Mat dt_image;
123  cv::distanceTransform(binary_image, dt_image, CV_DIST_L2, 5);
124 
125  // Convert to int...
126  cv::Mat dt_int_image(dt_image.size(), CV_8UC1);
127  cv::Mat dt_int_inv_image(dt_image.size(), CV_8UC1);
128 
129  // max distance for cost propagation
130  double max_dist = dist_transform_distance_; // meter
131  double resolution = out_grid_map.getResolution();
132 
133  for (int y = 0; y < dt_image.rows; y++)
134  {
135  for (int x = 0; x < dt_image.cols; x++)
136  {
137  // actual distance [meter]
138  double dist = dt_image.at<float>(y, x) * resolution;
139  if (dist > max_dist)
140  dist = max_dist;
141 
142  // Make value range 0 ~ 255
143  int round_dist = dist / max_dist * grid_max_value_;
144  int inv_round_dist = grid_max_value_ - round_dist;
145 
146  dt_int_image.at<unsigned char>(y, x) = round_dist;
147  dt_int_inv_image.at<unsigned char>(y, x) = inv_round_dist;
148  }
149  }
150 
151  // convert to ROS msg
152  grid_map::GridMapCvConverter::addLayerFromImage<unsigned char, 1>(dt_int_inv_image,
153  "distance_transform",
154  out_grid_map,
155  grid_min_value_,
156  grid_max_value_);
157  }
