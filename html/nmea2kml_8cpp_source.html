79 bool OpenROSBag(const std::string& bag_Name, rosbag::Bag & _bag, rosbag::View& _bagView, rosbag::View::iterator& _viewIterator)
80 {
81  std::set<std::string> bagTopics;
82 
83  try
84  {
85  _bag.open(bag_Name, rosbag::bagmode::Read);
86 
87  rosbag::View view(_bag);
88  std::vector<const rosbag::ConnectionInfo *> connection_infos = view.getConnections();
89  std::vector<std::string> topics_to_subscribe;
90 
91  bagTopics.clear();
92 
93  BOOST_FOREACH(const rosbag::ConnectionInfo *info, connection_infos)
94  {
95  bagTopics.insert(info->topic);
96  }
97 
98  if (bagTopics.find(nmea_topic_name) == bagTopics.end())
99  {
100  std::cout << "Can't Find NMEA Sentence Topic in ROSBag File :" << nmea_topic_name << std::endl;
101  return false;
102  }
103  else
104  {
105  topics_to_subscribe.push_back(std::string(nmea_topic_name));
106  }
107 
108  _bagView.addQuery(_bag, rosbag::TopicQuery(topics_to_subscribe));
109  _viewIterator = _bagView.begin();
110  return true;
111 
112  }
113  catch (rosbag::BagIOException& e)
114  {
115  std::cout << "Can't Open ROSbag with path: " << bag_Name << std::endl;
116  return false;
117  }

193 void InitializeStyleForSateliteNumbers(TiXmlElement* pElem, int max_sat_nu)
194 {
195  for(int i=max_sat_nu; i >= 0; i--)
196  {
197  double r=0, g=0, b=0;
198 
199  float norm_cost = (float)i / (float)max_sat_nu;
200  if(norm_cost <= 0.5)
201  {
202  g = (0.5 - norm_cost)*2.0;
203  r = 0;
204  }
205  else if(norm_cost > 0.5)
206  {
207  g = 0;
208  r = (norm_cost - 0.5)*2.0;
209  }
210 
211  std::ostringstream style_id;
212  style_id<< "stl_" << i;
213  InsertNewStyle(pElem,style_id.str(), r*255.0,g*255.0,b*255.0);
214  }

263 void ParseAndWrite(const nmea_msgs::SentenceConstPtr& msg, std::ofstream& _ofs)
264 {
265  if(_ofs.is_open())
266  {
267  _ofs << msg->sentence << std::endl;
268  }
269 
270  NMEA_PARSER::ReadNMEASentence _parser;
271  for(unsigned int i=0; i < msg->sentence.size(); i++)
272  {
273  _parser.Parse(msg->sentence.at(i));
274  }
275 
276  if(_parser.gpgga.count == 1)
277  {
278  if(_parser.gpgga.satellites > MAX_SAT_NUM)
279  _parser.gpgga.satellites = MAX_SAT_NUM;
280 
281  if(_parser.gpgga.satellites > g_max_sat)
282  g_max_sat = _parser.gpgga.satellites;
283 
284  if(_parser.gpgga.satellites < g_min_sat)
285  g_min_sat = _parser.gpgga.satellites;
286 
287  if(_parser.gpgga.satellites != g_prev_sat_number)
288  {
289  if(GROUND_RELATIVE_HRIGHT > 0)
290  g_line_points.push_back(GPS_Point(_parser.gpgga.latitude, _parser.gpgga.longitude, GROUND_RELATIVE_HRIGHT));
291  else
292  g_line_points.push_back(GPS_Point(_parser.gpgga.latitude, _parser.gpgga.longitude, _parser.gpgga.altitude));
293 
294  CreateLinePlaceMark(g_line_points, g_prev_sat_number);
295  g_prev_sat_number = _parser.gpgga.satellites;
296  g_line_points.clear();
297  }
298 
299 
300  if(_parser.gpgga.satellites == g_prev_sat_number)
301  {
302  if(GROUND_RELATIVE_HRIGHT > 0)
303  g_line_points.push_back(GPS_Point(_parser.gpgga.latitude, _parser.gpgga.longitude, GROUND_RELATIVE_HRIGHT));
304  else
305  g_line_points.push_back(GPS_Point(_parser.gpgga.latitude, _parser.gpgga.longitude, _parser.gpgga.altitude));
306  }
307 
308  std::cout << "GPGGA: " << _parser.gpgga.longitude << ", " << _parser.gpgga.latitude <<", " << _parser.gpgga.altitude << ", " << g_max_sat << ", " << g_min_sat << std::endl;
309  }

329 void ConvertOneFile(const std::string& fileName)
330 {
331  path_id = 0;
332  g_prev_sat_number = 0;
333  g_line_points.clear();
334  std::string output_kml = fileName;
335  output_kml.replace(fileName.size() - 4, fileName.size() , ".kml");
336  std::string output_csv = fileName;
337  output_csv.replace(fileName.size() - 4, fileName.size() , ".csv");
338 
339  rosbag::Bag bag;
340  rosbag::View bagView;
341  rosbag::View::iterator viewIterator;
342  TiXmlDocument xml_doc;
343 
344  CreateTemplateDocument(xml_doc);
345  pHeadElem = xml_doc.FirstChildElement()->FirstChildElement("Document");
346  InitializeStyleForSateliteNumbers(pHeadElem, MAX_SAT_NUM);
347 
348  std::ofstream ofs;
349  ofs.open(output_csv.c_str(), std::ios::app);
350 
351  OpenROSBag(fileName, bag, bagView, viewIterator);
352 
353  while(1)
354  {
355  nmea_msgs::SentenceConstPtr pOutSentence;
356  int ret = GetNextSentence(pOutSentence, bagView, viewIterator);
357  if(ret == 1)
358  ParseAndWrite(pOutSentence, ofs);
359  else if(ret == -1)
360  break;
361  }
362  CreateLinePlaceMark(g_line_points, g_prev_sat_number);
363  xml_doc.SaveFile(output_kml);
364 }
