126  double a12 = input(1, 2);

128

131  double max2 = (fabs(a12) > fabs(a22)) ? fabs(a12) : fabs(a22);
132

134 
135  maxAbsElement = (maxAbsElement > max2) ? maxAbsElement : max2;
136 
137  if (maxAbsElement == 0.0) {
138  MatrixDevice evec(3, 3, offset_, eigenvectors_ + tid);
139 
140  evec(0, 0) = 1.0;
141  evec(1, 1) = 1.0;
142  evec(2, 2) = 1.0;
143 
144  norm_[tid] = 0.0;
145  return;
146  }
147 
148  double invMaxAbsElement = 1.0 / maxAbsElement;
149 
150  a00 *= invMaxAbsElement;

152  a02 *= invMaxAbsElement;
153  a11 *= invMaxAbsElement;
154  a12 *= invMaxAbsElement;

172 CUDAH void SymmetricEigensolver3x3::computeEigenvalues(int tid)
173 {
174  MatrixDevice input(3, 3, offset_, input_matrices_ + tid);
175  MatrixDevice eval(3, 1, offset_, eigenvalues_ + tid);
176 
177  double a00 = input(0, 0);
178  double a01 = input(0, 1);
179  double a02 = input(0, 2);
180  double a11 = input(1, 1);
181  double a12 = input(1, 2);
182  double a22 = input(2, 2);
183 
184  double norm = norm_[tid];
185 
186  if (norm > 0.0) {
187  double traceDiv3 = (a00 + a11 + a22) / 3.0;
188  double b00 = a00 - traceDiv3;
189  double b11 = a11 - traceDiv3;
190  double b22 = a22 - traceDiv3;
191  double denom = sqrt((b00 * b00 + b11 * b11 + b22 * b22 + norm * 2.0) / 6.0);
192  double c00 = b11 * b22 - a12 * a12;
193  double c01 = a01 * b22 - a12 * a02;
194  double c02 = a01 * a12 - b11 * a02;
195  double det = (b00 * c00 - a01 * c01 + a02 * c02) / (denom * denom * denom);
196  double halfDet = det * 0.5;
197 
198  halfDet = (halfDet > -1.0) ? halfDet : -1.0;
199  halfDet = (halfDet < 1.0) ? halfDet : 1.0;
200 
201  double angle = acos(halfDet) / 3.0;
202  double beta2 = cos(angle) * 2.0;
203  double beta0 = cos(angle + M_PI * 2.0 / 3.0) * 2.0;
204  double beta1 = -(beta0 + beta2);
205 
206  eval(0) = traceDiv3 + denom * beta0;
207  eval(1) = traceDiv3 + denom * beta1;
208  eval(2) = traceDiv3 + denom * beta2;
209 
210  i02_[tid] = (halfDet >= 0) ? 2 : 0;
211  i02_[tid + offset_] = (halfDet >= 0) ? 0 : 2;
212 
213  } else {
214  eval(0) = a00;
215  eval(1) = a11;
216  eval(2) = a22;
217  }
218 }

315 CUDAH void SymmetricEigensolver3x3::computeEigenvector11(int tid)
316 {
317  if (norm_[tid] > 0.0) {
318  MatrixDevice evec1(3, 1, offset_ * 3, eigenvectors_ + tid + offset_);
319 
320  MatrixDevice u(3, 1, offset_, buffer_ + tid);
321  MatrixDevice v(3, 1, offset_, buffer_ + 3 * offset_ + tid);
322 
323  MatrixDevice au(3, 1, offset_, buffer_ + 6 * offset_ + tid);
324  MatrixDevice av(3, 1, offset_, buffer_ + 9 * offset_ + tid);
325 
326  double m00 = u(0) * au(0) + u(1) * au(1) + u(2) * au(2);
327  double m01 = u(0) * av(0) + u(1) * av(1) + u(2) * av(2);
328  double m11 = v(0) * av(0) + v(1) * av(1) + v(2) * av(2);
329 
330  double abs_m00 = fabs(m00);
331  double abs_m01 = fabs(m01);
332  double abs_m11 = fabs(m11);
333 
334  if (abs_m00 > 0 || abs_m01 > 0 || abs_m11 > 0) {
335  double u_mult = (abs_m00 >= abs_m11) ? m01 : m11;
336  double v_mult = (abs_m00 >= abs_m11) ? m00 : m01;
337 
338  bool res = fabs(u_mult) >= fabs(v_mult);
339  double *large = (res) ? &u_mult : &v_mult;
340  double *small = (res) ? &v_mult : &u_mult;
341 
342  *small /= (*large);
343  *large = 1.0 / sqrt(1.0 + (*small) * (*small));
344  *small *= (*large);
345 
346  multiply(u, u_mult, u);
347  multiply(v, v_mult, v);
348  subtract(u, v, evec1);
349 
350  } else {
351  u.copy(evec1);
352  }
353  }
