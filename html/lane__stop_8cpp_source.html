36 void select_current_lane(const autoware_msgs::TrafficLight& msg)
37 {
38  const autoware_msgs::LaneArray *current;
39  switch (msg.traffic_light) {
40  case lane_planner::vmap::TRAFFIC_LIGHT_RED:
41  current = &current_red_lane;
42  break;
43  case lane_planner::vmap::TRAFFIC_LIGHT_GREEN:
44  current = &current_green_lane;
45  break;
46  case lane_planner::vmap::TRAFFIC_LIGHT_UNKNOWN:
47  current = previous_lane; // if traffic light state is unknown, keep previous state
48  break;
49  default:
50  ROS_ERROR_STREAM("undefined traffic light");
51  return;
52  }
53 
54  if (current->lanes.empty()) {
55  ROS_ERROR_STREAM("empty lanes");
56  return;
57  }

59  traffic_pub.publish(*current);

61  previous_lane = current;
62 }
