117 void TwistGate::watchdog_timer()
118 {
119  while (is_alive)
120  {
121  ros::Time now_time = ros::Time::now();
122  bool emergency_flag = false;
123 
124  // check command mode
125  if (previous_command_mode_ != command_mode_)
126  {
127  if (command_mode_ == CommandMode::AUTO)
128  {
129  command_mode_topic_.data = "AUTO";
130  }
131  else if (command_mode_ == CommandMode::REMOTE)
132  {
133  command_mode_topic_.data = "REMOTE";
134  }
135  else
136  {
137  command_mode_topic_.data = "UNDEFINED";
138  }
139 
140  control_command_pub_.publish(command_mode_topic_);
141  previous_command_mode_ = command_mode_;
142  }
143 
144  // if lost Communication
145  if (command_mode_ == CommandMode::REMOTE && now_time - remote_cmd_time_ > timeout_period_)
146  {
147  emergency_flag = true;
148  ROS_WARN("Lost Communication!");
149  }
150 
151  // if push emergency stop button
152  if (emergency_stop_msg_.data == true)
153  {
154  emergency_flag = true;
155  ROS_WARN("Emergency Mode!");
156  }
157 
158  // Emergency
159  if (emergency_flag)
160  {
161  // Change Auto Mode
162  command_mode_ = CommandMode::AUTO;
163  if (send_emergency_cmd == false)
164  {
165  // Change State to Stop
166  std_msgs::String state_cmd;
167  state_cmd.data = "emergency";
168  state_cmd_pub_.publish(state_cmd);
169  send_emergency_cmd = true;
170  }
171  // Set Emergency Stop
172  emergency_stop_pub_.publish(emergency_stop_msg_);
173  ROS_WARN("Emergency Stop!");
174  }
175  else {
176  send_emergency_cmd = false;
177  }
178 
179  std::this_thread::sleep_for(std::chrono::milliseconds(10));
180  }
181 }

183 void TwistGate::remote_cmd_callback(const remote_msgs_t::ConstPtr& input_msg)
184 {
185  remote_cmd_time_ = ros::Time::now();
186  command_mode_ = static_cast<CommandMode>(input_msg->control_mode);
187  emergency_stop_msg_.data = static_cast<bool>(input_msg->vehicle_cmd.emergency);
188 
189  // Update Emergency Mode
190  twist_gate_msg_.emergency = input_msg->vehicle_cmd.emergency;
191 
192  if (command_mode_ == CommandMode::REMOTE && emergency_stop_msg_.data == false)
193  {
194  twist_gate_msg_.header.frame_id = input_msg->vehicle_cmd.header.frame_id;
195  twist_gate_msg_.header.stamp = input_msg->vehicle_cmd.header.stamp;
196  twist_gate_msg_.header.seq++;
197  twist_gate_msg_.twist_cmd.twist = input_msg->vehicle_cmd.twist_cmd.twist;
198  twist_gate_msg_.ctrl_cmd = input_msg->vehicle_cmd.ctrl_cmd;
199  //added by hui, enforce lateral_accel_limit
200  twist_gate_msg_.ctrl_cmd.linear_velocity = twist_gate_msg_.twist_cmd.twist.linear.x;
201  twist_gate_msg_.accel_cmd = input_msg->vehicle_cmd.accel_cmd;
202  twist_gate_msg_.brake_cmd = input_msg->vehicle_cmd.brake_cmd;

317 void TwistGate::state_callback(const std_msgs::StringConstPtr& input_msg)
318 {
319  if (command_mode_ == CommandMode::AUTO)
320  {
321  // Set Parking Gear
322  if (input_msg->data.find("WaitOrder") != std::string::npos)
323  {
324  twist_gate_msg_.gear = CMD_GEAR_P;
325  }
326  // Set Drive Gear
327  else
328  {
329  twist_gate_msg_.gear = CMD_GEAR_D;
330  }
331 
332  // get drive state
333  if (input_msg->data.find("Drive\n") != std::string::npos && input_msg->data.find("VehicleReady\n") != std::string::npos)
334  {
335  is_state_drive_ = true;
336  }
337  else
338  {
339  is_state_drive_ = false;
340  }
341  vehicle_cmd_pub_.publish(twist_gate_msg_);
342 
343  // reset emergency flags
344  if (input_msg->data.find("VehicleReady") != std::string::npos)
345  {
346  emergency_stop_msg_.data = false;
347  send_emergency_cmd = false;
348  }
349  }
